69 void GMU::SendSetMsgToAmp(string action, string mcast_addr, string amp_addr)
70 {
71 
72  // Connect to the AMP.
73  struct sockaddr_in amp_sock_addr;
74  ::memset(&amp_sock_addr, 0, sizeof(amp_sock_addr));
75  amp_sock_addr.sin_family = AF_INET;
76  amp_sock_addr.sin_addr.s_addr = inet_addr(amp_addr.c_str());
77  amp_sock_addr.sin_port = htons(kDefaultAmpCtlPort);
78  uint32_t amp_ep = 0;
79  while (amp_ep ==0)
80  {
81  LogD(kClassName, __func__, "Connecting to AMP\n");
82  amp_ep = rc_client_.Connect(amp_sock_addr);
83  if (amp_ep != 0)
84  {
85  LogD(kClassName, __func__, "Connected to AMP\n");
86  break;
87  }
88  sleep(2);
89  }
90 
91  LogD(kClassName, __func__,
92  "Sending message to AMP: %s multicast group %s\n",
93  action.c_str(), mcast_addr.c_str());
94 
95  rc_client_.SendSetMessage(amp_ep, "bpf", "parameter;mcast_group;action;" + action +
96  ";addr;" + mcast_addr);
97 
98 }
