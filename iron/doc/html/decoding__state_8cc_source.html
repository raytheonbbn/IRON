152 bool DecodingState::CreateReleaseController(const string utility_def)
153 {
154  // First, get the utility function type for the flow.
155  size_t type_str_pos = utility_def.find("type=");
156  if (type_str_pos == string::npos)
157  {
158  LogF(kClassName, __func__, "fid: %" PRIu32 ", invalid utility "
159  "definition.\n", flow_tag_);
160  return false;
161  }
162 
163  size_t type_str_end_pos = utility_def.find(":", type_str_pos);
164  if (type_str_end_pos == string::npos)
165  {
166  LogF(kClassName, __func__, "fid: %" PRIu32 ", invalid utility "
167  "definition.\n", flow_tag_);
168  return false;
169  }
170 
171  string utility_def_type = utility_def.substr(type_str_pos + 5,
172  type_str_end_pos - 5);
173 
174  if (release_controller_ != NULL)
175  {
176  delete release_controller_;
177  release_controller_ = NULL;
178  }
179 
180  // Create the appropriate packet release controller.
181  if ((utility_def_type == "TRAP") || (utility_def_type == "STRAP") ||
182  (utility_def_type == "FLOG"))
183  {
184  LogC(kClassName, __func__, "fid: %" PRIu32 ", Creating "
185  "ThrottledReleaseControl.\n", flow_tag_);
186  release_controller_ = new (std::nothrow)
187  ThrottledReleaseController(*this, packet_pool_);
188  }
189  else
190  {
191  LogC(kClassName, __func__, "fid: %" PRIu32 ", Creating "
192  "UnthrottledReleaseControl.\n", flow_tag_);
193  release_controller_ = new (std::nothrow)
194  UnthrottledReleaseController(*this);
195  }
196 
197  if (release_controller_ == NULL)
198  {
199  LogF(kClassName, __func__, "fid: %" PRIu32 ", error allocating decoding "
200  "state packet release controller.\n", flow_tag_);
201  return false;
202  }
203 
204  return true;
205 }

208 void DecodingState::HandlePkt(Packet* pkt)
209 {
210  // Remove the FEC control trailer.
211  FECControlTrailer fecTrlr;
212  memset(&fecTrlr, 0, sizeof(FECControlTrailer));
213 
214  if (!pkt->RemoveBlockFromEnd((unsigned char*)&fecTrlr, sizeof(fecTrlr)))
215  {
216  LogE(kClassName, __func__, "Failed to remove FEC control trailer from "
217  "packet\n");
218  TRACK_UNEXPECTED_DROP(kClassName, packet_pool_);
219  packet_pool_.Recycle(pkt);
220  return;
221  }
222 
223  BinId pkt_bin_id = pkt->bin_id();
224  BinIndex pkt_bin_idx = bin_map_.GetPhyBinIndex(pkt_bin_id);
225 
226  // Check the bin id to make sure that it is valid.
227  if (pkt_bin_idx == iron::kInvalidBinIndex)
228  {
229  LogE(kClassName, __func__, "Attempting to handle a packet without a "
230  "source bin id, which will not work with the decoding state.\n");
231  TRACK_UNEXPECTED_DROP(kClassName, packet_pool_);
232  packet_pool_.Recycle(pkt);
233  return;
234  }
235 
236  uint64_t pkt_total_bytes_sent = fecTrlr.get_total_bytes_sent();
237  uint32_t pkt_seq_num = fecTrlr.get_seq_number();
238  uint8_t pkt_priority = fecTrlr.priority;
239  uint8_t pkt_loss_thresh = fecTrlr.loss_thresh;
240  Time pkt_reorder_time =
241  Time(static_cast<double>(fecTrlr.reorder_time_ms) / 1000);
242  int32_t pkt_delay_ms = 0;
243 
244 #ifdef LAT_MEASURE
245  struct timeval now_tval;
246  gettimeofday(&now_tval, 0);
247 
248  Time now(now_tval);
249  uint16_t now_ms = now.GetTimeInMsec() & 0x7FFF;
250 
251  if (now_ms < pkt->origin_ts_ms())
252  {
253  // The origin timestamp wrapped. It is a 15 bit number, so we add
254  // 0x7FFF to the difference.
255  pkt_delay_ms += 0x7FFF + now_ms - pkt->origin_ts_ms();
256  }
257  else
258  {
259  pkt_delay_ms = now_ms - pkt->origin_ts_ms();
260  }
261 
262  LogD(kClassName, __func__, "fid: %" PRIu32 ", got FEC pkt %s with sn %" PRIu32
263  ", bytes sent: %" PRIu64 "B, ttg: %s, delay: %" PRId32 "ms, "
264  "priority: %" PRIu8 ", loss threshold: %" PRIu8 "\%, reorder time: %"
265  PRId64 "ms.\n",
266  flow_tag_,
267  pkt->GetPacketMetadataString().c_str(),
268  pkt_seq_num, pkt_total_bytes_sent, pkt->GetTimeToGo().ToString().c_str(),
269  pkt_delay_ms,
270  pkt_priority, pkt_loss_thresh, pkt_reorder_time.GetTimeInMsec());
271 #else
272  Time now = Time::Now();
273 
274 
275  LogD(kClassName, __func__, "fid: %" PRIu32 ", got FEC pkt %s with sn %" PRIu32
276  ", bytes sent: %" PRIu64 " bytes, ttg: %s, priority: %" PRIu8
277  ", loss threshold: %" PRIu8 "\%, reorder time: %" PRId64 "ms.\n",
278  flow_tag_,
279  pkt->GetPacketMetadataString().c_str(),
280  pkt_seq_num, pkt_total_bytes_sent, pkt->GetTimeToGo().ToString().c_str(),
281  pkt_priority, pkt_loss_thresh, pkt_reorder_time.GetTimeInMsec());
282 #endif
283 
284  last_time_ = now.GetTimeInSec();
285 
286  UdpProxy::ReleaseRecord* release_record = NULL;
287  if (max_reorder_time_.GetTimeInUsec() > 0)
288  {
289  udp_proxy_.GetReleaseRecord(pkt_bin_idx, four_tuple_, release_record);
290 
291  if (release_record == NULL)
292  {
293  LogD(kClassName, __func__, "fid: %" PRIu32 ", creating ReleaseRecord "
294  "for flow %s for src bin %" PRIBinId ", total num bytes sent %"
295  PRIu32 "bytes, SeqNum: %" PRIu32".\n", flow_tag_,
296  four_tuple_.ToString().c_str(), pkt_bin_id, pkt_total_bytes_sent,
297  pkt_seq_num);
298  // TODO: Delete
299  // TODO: Delete upon flush?
300  if (!udp_proxy_.CreateReleaseRecord(pkt_bin_idx, four_tuple_,
301  pkt_total_bytes_sent, pkt_seq_num,
302  pkt_priority))
303  {
304  LogW(kClassName, __func__, "fid: %" PRIu32 ", error creating "
305  "ReleaseRecord for flow %s.\n", flow_tag_,
306  four_tuple_.ToString().c_str());
307  }
308  }
309  }
310 
311  // Check if the priority and the loss threshold needs updating.
312  if (pkt_seq_num_ < pkt_seq_num)
313  {
314  pkt_seq_num_ = pkt_seq_num;
315  if (pkt_priority != priority_)
316  {
317  priority_ = pkt_priority;
318  if (release_record != NULL)
319  {
320  release_record->HandlePriorityChange(priority_);
321  }
322  }
323  if (pkt_loss_thresh != loss_thresh_pct_)
324  {
325  loss_thresh_pct_ = pkt_loss_thresh;
326  }
327  if (pkt_reorder_time != max_reorder_time_)
328  {
329  max_reorder_time_ = pkt_reorder_time;
330  }
331  }
332 
333  AccumulatePacketInfo(pkt->GetLengthInBytes(), Time::FromMsec(pkt_delay_ms));
334 
335  // In the case of the very first packet we must set the next group_id
336  if (next_grp_id_ == -1)
337  {
338  next_grp_id_ = fecTrlr.group_id;
339 
340  // and the source bin_idx.
341  bin_idx_ = pkt_bin_idx;
342  }
343 
344  // First see if we have any clean up work to do this happens when the
345  // group_id changes in ways we are not expecting, primarily due to dropped
346  // packets.
347  //
348  // Check to see if this is a late arriving packet
349  if (IsLate(fecTrlr.group_id))
350  {
351  if (max_reorder_time_.GetTimeInUsec() > 0)
352  {
353  LogW(kClassName, __func__, "fid: %" PRIu32 ", pktcount: Late pkt "
354  "group: %" PRIu32 ", slot: %" PRIu32 " %s.\n", flow_tag_,
355  fecTrlr.group_id, fecTrlr.slot_id,
356  pkt->GetPacketMetadataString().c_str());
357  }
358 
359  // We have a late packet. Since we have already moved on we don't put it
360  // in cache. If the order flag is not set, there is no harm in sending it
361  // as long as it is complete (i.e., not chunked). If the order flag IS
362  // set, we can only discard it
363  if (!fecTrlr.in_order)
364  {
365  if (fecTrlr.type == FEC_ORIGINAL)
366  {
367  // We also need to make sure that this is a self-contained packet --
368  // i.e., there is only one chunk *OR* this is a blob (i.e., it
369  // contains multiple smaller packets)
370  FECChunkTrailer chunkTrlr;
371  if (!pkt->RemoveBlockFromEnd((uint8_t*)&chunkTrlr,
372  sizeof(chunkTrlr)))
373  {
374  LogE(kClassName, __func__, "fid: %" PRIu32 ", Failed to remove "
375  "chunk trailer, recycling packet.\n", flow_tag_);
376  TRACK_UNEXPECTED_DROP(kClassName, packet_pool_);
377  packet_pool_.Recycle(pkt);
378  return;
379  }
380 
381  if ((chunkTrlr.n_chunks == 1) || (chunkTrlr.is_blob == 1))
382  {
383  LogD(kClassName, __func__, "fid: %" PRIu32 ", SEND: UDP proxy to "
384  "LAN IF late packet %s (groupdID %d, slotID " "%d)\n", flow_tag_,
385  pkt->GetPacketMetadataString().c_str(),
386  fecTrlr.group_id, fecTrlr.slot_id);
387 
388  // If we have a blob, we need to pull it apart
389  if (chunkTrlr.is_blob)
390  {
391  unsigned short pktlen;
392  unsigned char* pdata;
393 
394  pdata = pkt->GetBuffer();
395  pdata += pkt->GetIpPayloadOffset();
396 
397  for (int i = 0; i < chunkTrlr.n_chunks; i++)
398  {
399  memcpy((void*)&pktlen, pdata, sizeof(pktlen));
400  pdata += sizeof(pktlen);
401 
402  Packet* rpkt = packet_pool_.CloneHeaderOnly(
403  pkt, iron::PACKET_NO_TIMESTAMP);
404  if (!rpkt)
405  {
406  LogE(kClassName, __func__, "fid: %" PRIu32 ", Received "
407  "packet cannot be cloned.\n", flow_tag_);
408  return;
409  }
410  else
411  {
412  rpkt->AppendBlockToEnd(pdata, pktlen);
413  rpkt->UpdateChecksums();
414 
415  size_t bytes_sent = ReleasePkt(pkt);
416  if (bytes_sent == 0)
417  {
418  packet_pool_.Recycle(pkt);
419  return;
420  }
421  pdata += pktlen;
422  }
423  }
424  packet_pool_.Recycle(pkt);
425  }
426  else
427  {
428  // Otherwise, we can just update the checksums and send it.
429  // TESTING: checksum shouldn't need to be updated if not a blob
430  pkt->UpdateChecksums();
431  size_t bytes_sent = ReleasePkt(pkt);
432  if (bytes_sent == 0)
433  {
434  packet_pool_.Recycle(pkt);
435  return;
436  }
437  }
438  }
439  }
440  else
441  {
442  LogD(kClassName, __func__, "fid: %" PRIu32 ", Discarding late repair "
443  "packet (group_id %d, slotID %d)\n", flow_tag_, fecTrlr.group_id,
444  fecTrlr.slot_id);
445  TRACK_EXPECTED_DROP(kClassName, packet_pool_);
446  packet_pool_.Recycle(pkt);
447  }
448  }
449  else
450  {
451  packet_pool_.Recycle(pkt);
452  }
453 
454  // Make sure to skip over the remainder of the transmit processing - there
455  // is nothing more to do.
456  return;
457  }
458 
459  LogD(kClassName, __func__, "fid: %" PRIu32 ", Processing packet for group "
460  "%d\n", flow_tag_, fecTrlr.group_id);
461 
462  // Get/create the FecState for this group Id.
463  FecState* fec_state = NULL;
464  if (HasFecState(fecTrlr.group_id))
465  {
466  GetFecState(fecTrlr.group_id, fec_state);
467  }
468  else
469  {
470  fec_state = fecstate_pool_.Get();
471  fec_state_map_[fecTrlr.group_id] = fec_state;
472  fec_state->set_group_id(fecTrlr.group_id);
473  fec_state->set_decoding_state(this);
474  }
475 
476  // Add the packet we just received to the cache of the appropriate FecState.
477  //
478  // The operation of adding the packet to the FECState hands off ownership of
479  // the packet to the FECState. We need to extract everything required from
480  // the packet prior to adding it to the cache so that the values can be used
481  // following a successful addition to the cache.
482  Time ttg = pkt->GetTimeToGo();
483  bool pkt_ttg_valid = pkt->time_to_go_valid();
484  Time pkt_rcv_time = pkt->recv_time();
485  FECRepairTrailer rep_trlr;
486  if (fecTrlr.type == FEC_REPAIR)
487  {
488  // If its a repair packet save the encoding rate for this group Note that
489  // we don't remove the block at this point because we need the FECed
490  // length embedded in the repair header for the FEC processing.
491  memset(&rep_trlr, 0, sizeof(FECRepairTrailer));
492  pkt->CopyBlockFromEnd((unsigned char*)&rep_trlr, sizeof(rep_trlr));
493  }
494 
495  if (fec_state->AddToCache(fecTrlr.type, pkt, fecTrlr.slot_id,
496  fecTrlr.fec_used, fecTrlr.total_bytes_sent,
497  fecTrlr.seq_number) != FECSTATE_OKAY)
498  {
499  LogW(kClassName, __func__, "fid: %" PRIu32 ", Failed to add pkt of len "
500  "%d with group_id of %d and type %d to cache at slot %d\n", flow_tag_,
501  pkt->GetLengthInBytes(), fecTrlr.group_id, fecTrlr.type,
502  fecTrlr.slot_id);
503  TRACK_UNEXPECTED_DROP(kClassName, packet_pool_);
504  packet_pool_.Recycle(pkt);
505  return;
506  }
507 
508  // Get the expiration time of this packet.
509  Time pkt_exp_time = Time::Infinite();
510  if (pkt_ttg_valid && (ttg < max_reorder_time_))
511  {
512  pkt_exp_time = ttg + pkt_rcv_time;
513  }
514  else
515  {
516  pkt_exp_time = pkt_rcv_time + max_reorder_time_;
517  }
518 
519  // Make adjustments to ensure this expires before higher numbered packets in
520  // this group or the next group to expire. However, This can only lower the
521  // packet expiration time, so only do it if the packet has not already
522  // expired.
523  if (pkt_exp_time > Time::Now())
524  {
525  Time next_pkt_exp = GetNextExpTime(fecTrlr.slot_id, fecTrlr.group_id);
526  if (pkt_exp_time > next_pkt_exp)
527  {
528  pkt_exp_time = next_pkt_exp;
529  }
530  }
531 
533  if (fec_grp_ready_time_ > pkt_exp_time)
534  {
535  fec_grp_ready_time_ = pkt_exp_time;
536  }
537 
539  fec_state->set_pkt_expiration_time(fecTrlr.slot_id, pkt_exp_time);
540 
541  if (fecTrlr.type == FEC_REPAIR)
542  {
543  fec_state->setRates(rep_trlr.base_rate, rep_trlr.fec_rate);
544  }
545 }

572 void DecodingState::WriteStats(Time& now, string& log_str,
573  Writer<StringBuffer>* writer)
574 {
575  // The collected statistics for inbound flows are reported via the following
576  // name/value pairs.
577  //
578  // "flow_id" : "a.b.c.d:eph -> e.f.g.h:svc",
579  // "pkts" : xxxxxx,
580  // "bytes" : xxxxxx,
581  // "rate_bps" : xxxx.xxx,
582  // "rate_pps" : xxxx.xxx,
583  // "avg_delay_ms" : xxx.xxx,
584  // "max_delay_ms" : xxx,
585 
586  double rate_bps = 0.0;
587  double pps = 0.0;
588  double avg_delay_ms = 0.0;
589 
590  if (now > last_report_time_)
591  {
592  double delta_usec = static_cast<double>(
593  (now - last_report_time_).GetTimeInUsec());
594 
595  rate_bps = static_cast<double>((dump_byte_number_ * 8000000.0) /
596  delta_usec);
597  pps = static_cast<double>((dump_pkt_number_ * 1000000.0) / delta_usec);
598 
599  if (dump_pkt_number_ > 0)
600  {
601  avg_delay_ms = static_cast<double>(cum_pkt_delay_.GetTimeInMsec());
602  avg_delay_ms /= static_cast<double>(dump_pkt_number_);
603  }
604  }
605 
606  string flow_id_str = (Ipv4Endpoint(four_tuple_.src_addr_nbo(),
607  four_tuple_.src_port_nbo()).ToString() +
608  " -> " +
609  Ipv4Endpoint(four_tuple_.dst_addr_nbo(),
610  four_tuple_.dst_port_nbo()).ToString());
611 
612 
613  if (udp_proxy_.log_stats())
614  {
615  log_str.append(
616  StringUtils::FormatString(256, "'%s':{", flow_id_str.c_str()));
617 
618  log_str.append(
619  StringUtils::FormatString(256, "'rcvd_pkts':'%" PRIu32 "', ",
620  total_pkt_number_));
621 
622  log_str.append(
623  StringUtils::FormatString(256, "'rcvd_bytes':'%" PRIu64 "', ",
624  total_byte_number_));
625 
626  log_str.append(

678 void DecodingState::ReleaseInOrderPackets()
679 {
680  FecState* fec_state = NULL;
681 
682  // Starting with next_grp_id, release as many grps that are ready and in
683  // order until there is a group that is not yet ready.
684  while (true)
685  {
686  int next_fec_grp = GetNextFecGrp(next_grp_id_);
687  Time next_exp = Time::Infinite();
688  if (next_fec_grp != -1)
689  {
690  next_exp = grp_exp_time(next_fec_grp);
691  }
692 
693  if(!HasFecState(next_grp_id_))
694  {
695  fec_grp_ready_time_ = next_exp;
696  break;
697  }
698 
699  GetFecState(next_grp_id_, fec_state);
700  fec_state->UpdateFEC();
701  LogD(kClassName, __func__, "Releasing sequential packet (group_id %d)\n",
702  fec_state->group_id());
703 
704  if (SendToReleaseController(fec_state, next_exp))
705  {
706  DeleteFecState(next_grp_id_);
707  ++next_grp_id_;
708  LogD(kClassName, __func__, "continue rel\n");
709  }
710  else // The next FEC group is not complete, so we wait.
711  {
712  fec_grp_ready_time_ = fec_state->expiration_time();
713  LogD(kClassName, __func__, "Done rel\n");
714  break;
715  }
716  }
717 }

719 int DecodingState::GetNextFecGrp(int cur_group)
720 {
721  if (fec_state_map_.size() == 0)
722  {
723  return -1;
724  }
725 
726  // The smallest FEC group ID that is larger than current group ID.
727  int next_grp_id = cur_group + FEC_GROUPID_ROLLOVER;
728 
729  std::map<int,FecState*>::iterator it;
730  for(it = fec_state_map_.begin(); it != fec_state_map_.end(); ++it)
731  {
732  if (IsLate(it->first, cur_group))
733  {
734  continue;
735  }
736 
737  int grp_id = 0;
738  if (it->first <= cur_group)
739  {
740  grp_id = it->first + FEC_GROUPID_ROLLOVER;
741  }
742  else
743  {
744  grp_id = it->first;
745  }
746 
747 
748  //LogD(kClassName, __func__, "has grp %d vs %d\n", it->first,next_grp_id);
749  if (grp_id < next_grp_id)
750  {
751  next_grp_id = grp_id;
752  }
753  }
754 
755  next_grp_id = next_grp_id & FEC_GROUPID_MASK;
756 
757  if (next_grp_id == cur_group)
758  {
759  return -1;
760  }
761  return next_grp_id;
762 }

895 //============================================================================
896 bool DecodingState::SendToReleaseController(FecState* fec_state,
897  Time& next_exp)
898 {
899  Time next_pkt_exp = next_exp;
900 
901  // Do repairs, if possible.
902  if (fec_state->base_rate() > 0)
903  {
904  fec_state->UpdateFEC();
905  }
906 
907  // Send whatever packets we can.
908  int lower_limit = fec_state->getFirstUnsentPktID();
909  int upper_limit = fec_state->max_pkt_id();
910  Time now = Time::Now();
911 
912  LogD(kClassName, __func__, "limits: %d, %d\n", lower_limit, upper_limit);
913  if (upper_limit >= 0)
914  {
915  for (int i = lower_limit; i <= upper_limit; i++)
916  {
917  // Retrieve and send all valid packets that haven't been sent yet
918  // Checking for valid packets that haven't been sent is done in the
919  // reassembly code.
920  Packet* rpkt = NULL;
921  if ((rpkt = fec_state->ReassembleFromCache(i)) != NULL)
922  {
923  udp_proxy_.IncrementTotalPktsSent();
924 
925  LogD(kClassName, __func__, "SEND: FECState to ReleaseController, "
926  "original packet (group_id %d, pktID %d)\n", fec_state->group_id(),
927  i);
928 
929  if (release_controller_ != NULL)
930  {
931  release_controller_->HandlePkt(rpkt);
932  fec_state->set_pkt_sent(i);
933  }
934  else
935  {
936  packet_pool_.Recycle(rpkt);
937  }
938  }
939  else
940  {
941  // Could not reassemble packet, check if we should skip.
942  next_pkt_exp = Time::Min(next_exp, fec_state->next_pkt_exp(i));
943  if (next_pkt_exp < now)
944  {
945  LogA(kClassName, __func__, "pktcount: Missing FECState group: %"
946  PRIu32 " slot: %" PRIu32 " \n.", fec_state->group_id(), i);
947  fec_state->set_pkt_sent(i);
948  }
949  else
950  {
951  // We still have time, wait until the next packet expires.
952  break;
953  }
954  }
955  }
956  }
957 
958  // Check if there is still packets to send in this state.
959  if (fec_state->fec_used() && next_pkt_exp > now)
960  {
961  if (fec_state->getFirstUnsentPktID() < fec_state->base_rate() ||
962  fec_state->base_rate() == 0)
963  {
964  // We haven't received all the original packets or a repair pkt.
965  fec_state->set_expiration_time(next_pkt_exp);
966  return false;
967  }
968  }
969 
970  // Only do error recording and RRMs if there is reordering.
971  if (max_reorder_time_.GetTimeInUsec() > 0)
972  {
973  // Advance the counter of bytes/packets sourced to this FEC group.
974  UdpProxy::ReleaseRecord* release_record = NULL;
975  udp_proxy_.GetReleaseRecord(bin_idx_, four_tuple_, release_record);
976 
977  if (release_record != NULL)
978  {
979  double curr_loss_rate = release_record->ReleaseFecState(*fec_state);
980 
981  // Send an RRM to the source if the loss rate cross the threshold.
982  if ((curr_loss_rate * 100 > loss_thresh_pct_) &&
983  (now - release_record->last_rrm_sent_) >
984  Time::FromMsec(kMinEventRrmIntervalMsec))
985  {
986  LogD(kClassName, __func__, "RRM triggered by flow %s.\n",
987  fec_state->decoding_state()->four_tuple().ToString().c_str());
988  udp_proxy_.SendRRMs(now);
989  release_record->last_rrm_sent_ = now;
990  }
991  }
992  else
993  {
994  LogE(kClassName, __func__, "Release record not found for %u\n",
995  four_tuple_.ToString().c_str());
996  }
997  }
998 
999  return true;
1000 }
