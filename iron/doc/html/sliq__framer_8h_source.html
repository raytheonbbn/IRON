127  enum HeaderType
128  {
129  // Connection establishment, reset, and close headers. Cannot be
130  // concatenated.
131  CONNECTION_HANDSHAKE_HEADER = 0, // 0x00
132  RESET_CONNECTION_HEADER = 1, // 0x01
133  CLOSE_CONNECTION_HEADER = 2, // 0x02
134 
135  // Stream creation and reset headers. Cannot be concatenated.
136  CREATE_STREAM_HEADER = 3, // 0x03
137  RESET_STREAM_HEADER = 4, // 0x04
138 
139  // Data transfer headers. Can be concatenated, but any data header must
140  // be last.
141  DATA_HEADER = 32, // 0x20
142  ACK_HEADER = 33, // 0x21
143  CC_SYNC_HEADER = 34, // 0x22
144  RCVD_PKT_CNT_HEADER = 35, // 0x23
145  CONN_MEAS_HEADER = 36, // 0x24
146 
147  // Specialized stand-alone headers. Cannot be concatenated.
148  CC_PKT_TRAIN_HEADER = 40, // 0x28
149 
150  // Special header type value for an unknown header.
151  UNKNOWN_HEADER = 255
152  };
