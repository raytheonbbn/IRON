145 int FecState::AddToCache(unsigned long type, Packet* qpkt, int index,
146  bool fec_used, uint64_t bytes_sourced,
147  uint32_t pkts_sourced)
148 {
149  if (bytes_sourced > bytes_sourced_)
150  {
151  bytes_sourced_ = bytes_sourced;
152  max_pkt_sn_ = pkts_sourced;
153  }
154 
155  if (min_pkt_sn_ == 0)
156  {
157  min_pkt_sn_ = pkts_sourced - index;
158  bin_id_ = qpkt->bin_id();
159  }
160 
161  if (type == FEC_ORIGINAL)
162  {
163  if ((index < 0) || (index >= MAX_FEC_RATE) ||
164  (orig_valid_[index] == true))
165  {
166  return FECSTATE_OUTOFBOUNDS;
167  }
168 
169  orig_cache_[index] = qpkt;
170  orig_valid_[index] = true;
171  orig_count_++;
172  fec_used_ = fec_used;
173 
174  // Do some bookeeping to help reassembly functions
175  UpdateLookupInfo(index);
176  }
177 
178  else // if (type == FEC_REPAIR)
179  {
180  if ((index < 0) || (index >= MAX_FEC_RATE) || (fec_valid_[index] == true))
181  {
182  return FECSTATE_OUTOFBOUNDS;
183  }
184 
185  fec_cache_[index] = qpkt;
186  fec_valid_[index] = true;
187  fec_count_++;
188  }
189 
190  return FECSTATE_OKAY;
191 }

194 Packet* FecState::ReassembleFromCache(int pktID)
195 {
196  FECChunkTrailer* chunkTrlr;
197  Packet* cpkt = NULL;
198  Packet* qpkt = NULL;
199 
200  unsigned char* bffr;
201  int start;
202  int len;
203  int initPktID;
204  int nChunks;
205  int index;
206  int i;
207  bool failed = false;
208 
209  // Sanity check for a call against a pktID for which we have no data
210  if ((pktID < 0) || (pktID > max_pkt_id_))
211  {
212  LogW(kClassName, __func__, " pktID %d out of range\n", pktID);
213  return NULL;
214  }
215 
216  // Retrieve the index into the cache for this packet ID and do another
217  // sanity check.
218  index = pkt_lookup_[pktID];
219 
220  if ((index < 0) || (index >= MAX_FEC_RATE) || (!orig_valid_[index]))
221  {
222  LogD(kClassName,__func__, " Packet index %d out of range or not "
223  "valid\n", index);
224  return NULL;
225  }
226 
227  // If we have already sent this packet, we just return.
228  if ( pkt_sent_[pktID])
229  {
230  LogD(kClassName,__func__, " Packet %d (index %d) already sent.\n",
231  pktID, index);
232  return NULL;
233  }
234 
235  // We have a valid ID and some supporting data for a packet that has not yet
236  // been sent. Looks like we might actually have something to do. See
237  // if we have all the chunks
238 
239  cpkt = orig_cache_[index];
240  len = cpkt->GetLengthInBytes() - sizeof(FECChunkTrailer);
241 
242  chunkTrlr = (FECChunkTrailer *)(cpkt->GetBuffer() + len);
243  initPktID = chunkTrlr->pkt_id;
244  nChunks = chunkTrlr->n_chunks;
245 
246  // Please note that although we could check the chunk trailers for
247  // consistency (i.e., they are in sequence and all from the same packet) we
248  // assume that the FEC control trailer causes us to correctly place the
249  // chunks into the right slots -- so all we really need to test for is
250  // whether the packet is in the slot.
251  if (chunkTrlr->is_blob)
252  {
253  if ((pktID < initPktID) || (pktID >= initPktID + nChunks))
254  {
255  LogW(kClassName, __func__, " pktID out of range.\n");
256  return NULL;
257  }
258  }
259  else
260  {
261  for (i = 0; i < nChunks; i++)
262  {
263  if (!orig_valid_[index+i])
264  {
265  // Looks like we are still missing chunks.
266  return NULL;
267  }
268  }
269  }
270 
271  // Looks like we have them all. Set up a new packet object to hold the
272  // rebuilt packet.
273  qpkt = packet_pool_.CloneHeaderOnly(cpkt, iron::PACKET_NO_TIMESTAMP);
274  qpkt->set_bin_id(cpkt->bin_id());
275  qpkt->set_origin_ts_ms(cpkt->origin_ts_ms());
276  if (!qpkt)
277  {
278  LogF(kClassName, __func__, "Failed to clone packet\n");
279  }
280 
281  // Now reassemble the original packet from its chunks
282  if (chunkTrlr->is_blob)
283  {
284  unsigned char *payld;
285  unsigned short plen;
286  unsigned int chkLen;
287 
288  // Grab key values, pointers
289  bffr = cpkt->GetBuffer();
290  start = cpkt->GetIpPayloadOffset();
291  payld = &bffr[start];
292 
293  memcpy(&plen,payld,sizeof(plen));
294  payld += sizeof(plen);
295  chkLen = sizeof(plen) + plen;
296 
297  while ((initPktID < pktID) && !failed)
298  {
299  if ((chkLen + sizeof(plen)) > (unsigned)len)
300  {
301  // We are about to read past the end of the packet
302  LogW(kClassName, __func__, " Reconstruction failure: request to "
303  "read past the end of muliple packet chunk.\n");
304  failed = true;
305  }
306  else
307  {
308  payld += plen;
309  memcpy(&plen,payld,sizeof(plen));
310  payld += sizeof(plen);;
311  chkLen += plen + sizeof(plen);
312 
313  initPktID ++;
314  }
315  }
316 
317  // Append this chunk to the end of our reassembly packet.
318  if (!failed)
319  {
320  qpkt->AppendBlockToEnd((uint8_t*)payld,plen);
321  }
322  }
323  else
324  {
325  failed = false;
326  for (i = 0; i < nChunks; i++)
327  {
328  // Get a pointer to the next chunk
329  cpkt = orig_cache_[index+i];
330 
331  // Grab key values, pointers
332  bffr = cpkt->GetBuffer();
333  start = cpkt->GetIpPayloadOffset();
334  len = cpkt->GetLengthInBytes() - (start + sizeof(FECChunkTrailer));
335 
336  chunkTrlr = (FECChunkTrailer *)&bffr[cpkt->GetLengthInBytes() -
337  sizeof(FECChunkTrailer)];
338  if ((chunkTrlr->chunk_id != i) || (chunkTrlr->pkt_id != pktID))
339  {
340  LogW(kClassName, __func__, "Mismatch in reconstruction parameters: "
341  "expected chunkID %u got %u; expected pktID %u got %u\n", i,
342  chunkTrlr->chunk_id, pktID, chunkTrlr->pkt_id);
343  failed = true;
344  }
345 
346  // Append this chunk to the end of our reassembly packet
347  if (!failed)
348  {
349  qpkt->AppendBlockToEnd((uint8_t*)&bffr[start],len);
350  }
351  }
352  }
353 
354  // Throw the packet away if reconstruction fails
355  if (failed)
356  {
357  LogW(kClassName, __func__, " failed reassembly\n");
358  TRACK_UNEXPECTED_DROP(kClassName, packet_pool_);
359  packet_pool_.Recycle(qpkt);
360  return NULL;
361  }
362 
363  // Recompute the checksums, and we're good to go
364  qpkt->UpdateChecksums();
365 
366  // Mark that we have sent this packet
367  pkt_sent_[pktID] = true;
368  bytes_released_ += qpkt->GetLengthInBytes();
369 
370  return qpkt;
371 }

374 bool FecState::UpdateFEC()
375 {
376  Packet *qpkt = (Packet *)NULL;
377  Packet *rpkt = (Packet *)NULL;
378  Packet *lastPkt = (Packet *)NULL;
379 
380  FECRepairTrailer repTrlr;
381  ::memset(&repTrlr, 0, sizeof(repTrlr));
382 
383  unsigned char *qptr;
384  unsigned char *qdata;
385  int qlen;
386 
387  unsigned char *rptr;
388  unsigned char *rdata;
389  int rlen;
390 
391  unsigned short fecLen;
392 
393  int rprID = 0;
394  int hole;
395  int i;
396  int j;
397  int rc;
398 
399  // Check for benign condition (we have all the orig packets)
400 
401  if (orig_count_ == base_rate_)
402  return true;
403 
404 
405  // Return false if we don't have enough combined original
406  // and repair packets to do anything
407 
408  if ((orig_count_ + fec_count_) < base_rate_)
409  return false;
410 
411 
412  // If here we have enough packets to do a repair.
413 
414 
415  // We support two special modes: rate 1/N, and rate N/(N+1)
416  // in addition to the more general N/(N+K) Vandermond matrix
417  // based FEC encoder
418 
419  if (base_rate_ == 1) // rate 1/N mode
420  {
421  // If we are here, we haven't sent the original packet
422  // but we have at least one repair packet in the cache.
423  // which is good enough for this coding rate
424 
425  for (i=0; i<(int)fec_rate_; i++)
426  {
427  if (fec_valid_[i])
428  {
429  rpkt = fec_cache_[i];
430  rprID = i;
431  break;
432  }
433  }
434 
435  if (!rpkt)
436  {
437  return(false);
438  }
439 
440  // Note that we store repair packets with the additional
441  // trailer so we can conveniently retain the FECed length
442  // Hence we need to remove it before sending it
443 
444  if (!rpkt->RemoveBlockFromEnd((uint8_t*)&repTrlr,(int)sizeof(repTrlr)))
445  {
446  LogW(kClassName, __func__, "Failed to remove block from end\n");
447  }
448 
449  // We used the repair packet to form the original
450  // packet. So we need to fixup the various caches
451 
452  orig_cache_[0] = rpkt;
453  orig_valid_[0] = true;
454  pkt_sent_[0] = false;
455  orig_count_++;
456 
457  UpdateLookupInfo(0);
458 
459  fec_cache_[rprID] = NULL;
460  fec_valid_[rprID] = false;
461  fec_count_--;
462  }
463 
464  else if (fec_rate_ == 1) // rate N/(N+1) mode
465  {
466  // If we are here, we have exactly one repair packet in
467  // the cache, so we don't need to search
468 
469  rpkt = fec_cache_[0];
470 
471  // Again we store repair packets with the additional
472  // trailer, so we need to remove it
473 
474  if (!rpkt->RemoveBlockFromEnd((uint8_t*)&repTrlr,(int)sizeof(repTrlr)))
475  {
476  LogW(kClassName, __func__, "Failed to remove block from end\n");
477  }
478 
479  // Now perform the FEC processing
480 
481  rptr = rpkt->GetBuffer();
482  rdata = rptr + rpkt->GetIpPayloadOffset();
483 
484  fecLen = repTrlr.fec_len;
485 
486  // Pull remaining packets in sequence from the cache
487  // and use them to compute the single FEC block
488 
489  hole = -1;
490 
491  for (i=0; i<base_rate_; i++)
492  {
493  if (orig_valid_[i])
494  {
495  qpkt = orig_cache_[i];
496  qptr = qpkt->GetBuffer();
497  qdata = qptr + qpkt->GetIpPayloadOffset();
498  qlen = qpkt->GetLengthInBytes() - (qdata - qptr);
499 
500  for (j=0; j<qlen; j++)
501  {
502  rdata[j] ^= qdata[j];
503  }
504 
505  // Also reconstruct the length of the missing
506  // packet
507 
508  fecLen ^= (unsigned short)qlen;
509  }
510 
511  else // Found the hole
512  {
513  hole = i;
514  }
515  }
516 
517  if (hole == -1)
518  {
519  LogF(kClassName, __func__, "Could not find a hole\n");
520  TRACK_UNEXPECTED_DROP(kClassName, packet_pool_);
521  packet_pool_.Recycle(qpkt);
522  packet_pool_.Recycle(rpkt);
523  return(false);
524  }
525 
526  rpkt->UpdateIpLen(fecLen + (rdata-rptr));
527 
528  // We used the repair packet to form the missing original
529  // packet. So we need to fixup the various caches
530 
531  orig_cache_[hole] = rpkt;
532  orig_valid_[hole] = true;
533  pkt_sent_[hole] = false;
534 
535  UpdateLookupInfo(hole);
536 
537  orig_count_++;
538 
539  fec_cache_[0] = NULL;
540  fec_valid_[0] = false;
541  fec_count_--;
542  }
543 
544  else // Must be we used the VDM FEC code
545  {
546  unsigned char *psrc [MAX_FEC_RATE];
547  unsigned char *pdst [MAX_FEC_RATE];
548  int index [MAX_FEC_RATE];
549  unsigned short szArray[MAX_FEC_RATE];
550  unsigned short fecSz [MAX_FEC_RATE];
551  unsigned short recSz [MAX_FEC_RATE];
552 
553  memset(psrc, 0, sizeof(psrc));
554  memset(pdst, 0, sizeof(pdst));
555  memset(index, 0, sizeof(index));
556  memset(szArray, 0, sizeof(szArray));
557  memset(fecSz, 0, sizeof(fecSz));
558  memset(recSz, 0, sizeof(recSz));
559 
560  for (i=0,j=0; i<base_rate_; i++)
561  {
562  if (orig_valid_[i])
563  {
564  qpkt = orig_cache_[i];
565  qptr = qpkt->GetBuffer();
566  qdata = qptr + qpkt->GetIpPayloadOffset();
567  qlen = qpkt->GetLengthInBytes() - (qdata - qptr);
568 
569  psrc[j] = qdata;
570  szArray[j] = qlen;
571  fecSz[j] = qlen;
572  index[j++] = i;
573 
574  pdst[i] = qdata;
575  lastPkt = qpkt;
576  }
577  }
578 
579  for (i=0; i<fec_rate_; i++)
580  {
581  if (fec_valid_[i])
582  {
583  rpkt = fec_cache_[i];
584 
585  if (!rpkt->RemoveBlockFromEnd((uint8_t*)&repTrlr,(int)sizeof(repTrlr)))
586  {
587  LogW(kClassName, __func__, "Failed to remove block from end\n");
588  }
589 
590  rptr = rpkt->GetBuffer();
591  rdata = rptr + rpkt->GetIpPayloadOffset();
592  rlen = rpkt->GetLengthInBytes() - (rdata - rptr);
593 
594  psrc[j] = rdata;
595  szArray[j] = rlen;
596  fecSz[j] = repTrlr.fec_len;
597  index[j++] = i + base_rate_;
598  lastPkt = rpkt;
599  }
600  }
601 
602  if (j != base_rate_)
603  {
604  LogW(kClassName, __func__, "Corrupted state in FEC decoder\n");
605  return false;
606  }
607 
608  // We finish setting up the call by creating empty packets
609  // with the correct IP headers as targets for the reconstruction
610  // process
611  qpkt = lastPkt;
612  for (i=0; i<base_rate_; i++)
613  {
614  if (!orig_valid_[i])
615  {
616  rpkt = packet_pool_.Clone(qpkt, false, iron::PACKET_NO_TIMESTAMP);
617  rptr = rpkt->GetBuffer();
618  rdata = rptr + rpkt->GetIpPayloadOffset();
619 
620  pdst[i] = rdata;
621 
622  orig_cache_[i] = rpkt;
623  }
624  }
625 
626  if ((rc = decode_vdmfec (psrc, pdst, index, base_rate_, szArray, fecSz, recSz)) != 0)
627  {
628  LogW(kClassName, __func__, "FEC decoding error: decoder returned %d)\n", rc);
629 
630  // If we hit a decoding error, we need to back out the newly allocated
631  // packets intended to hold the repairs
632  for (i=0; i<base_rate_; i++)
633  {
634  if (!orig_valid_[i])
635  {
636  TRACK_UNEXPECTED_DROP(kClassName, packet_pool_);
637  packet_pool_.Recycle(orig_cache_[i]);
638  }
639  }
640 
641  // and abort
642  return false;
643  }
644 
645  LogD(kClassName, __func__, "Decode vdm success\n");
646 
647  // If we are here, we successfully performed a reconstruction
648  // Now we just need to assign the packet lengths and mark them as valid
649  for (i=0; i<base_rate_; i++)
650  {
651  if (!orig_valid_[i])
652  {
653  rpkt = orig_cache_[i];
654  rptr = rpkt->GetBuffer();
655  rdata = rptr + rpkt->GetIpPayloadOffset();
656 
657  rpkt->UpdateIpLen(recSz[i] + (rdata-rptr));
658 
659  // Can now declare the repaired packet as valid
660 
661  orig_valid_[i] = true;
662  orig_count_++;
663 
664  UpdateLookupInfo(i);
665  }
666  }
667  }
668 
669  return true;
670 }

672 //============================================================================
673 int FecState::FlushCache()
674 {
675  int i;
676 
677  for (i=0; i<MAX_FEC_RATE; i++)
678  {
679  if (orig_valid_[i])
680  {
681  packet_pool_.Recycle(orig_cache_[i]);
682  orig_valid_[i] = false;
683  }
684  pkt_sent_[i] = false;
685  pkt_lookup_[i] = -1;
686  }
687 
688  orig_count_ = 0;
689  max_pkt_id_ = -1;
690 
691  for (i=0; i<MAX_FEC_RATE; i++)
692  {
693  if (fec_valid_[i])
694  {
695  packet_pool_.Recycle(fec_cache_[i]);
696  fec_valid_[i] = false;
697  }
698  }
699 
700  fec_count_ = 0;
701 
702  // Also reset the rates.
703  base_rate_ = 0;
704  fec_rate_ = 0;
705 
706  return FECSTATE_OKAY;
707 }

739 Packet* FecState::FetchFromCache(unsigned long type, int index)
740 {
741  if (type == FEC_ORIGINAL)
742  {
743  if ((index < 0) || (index >= MAX_FEC_RATE) || (!orig_valid_[index]))
744  {
745  return NULL;
746  }
747  return orig_cache_[index];
748  }
749  else // if (type == FEC_REPAIR)
750  {
751  if ((index < 0) || (index >= MAX_FEC_RATE) || (!fec_valid_[index]))
752  {

792 
793  }
794 
795  // Remember the maximum packet index seen so far
796  if (max_pkt_id_ < (int)(pktID + nChunks - 1))
797  {
798  max_pkt_id_ = (int)(pktID + nChunks - 1);
799  }
800  }
