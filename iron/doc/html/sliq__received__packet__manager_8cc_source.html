199 bool RcvdPktManager::Initialize(
200  const Reliability& rel, DeliveryMode del_mode, PktSeqNumber init_seq_num)
201 {
202  // Prevent multiple initializations.
203  if (rcvd_pkts_ != NULL)
204  {
205  LogE(kClassName, __func__, "Conn %" PRIEndptId " Stream %" PRIStreamId
206  ": Error, already initialized.\n", conn_id_, stream_id_);
207  return false;
208  }
209 
210  // Initialize the FEC encoder.
211  if (rel.mode == SEMI_RELIABLE_ARQ_FEC)
212  {
213  if (!rel.fec_del_time_flag)
214  {
215  // Check that the target number of rounds is within limits.
216  if ((rel.fec_target_pkt_del_rounds < 1) ||
217  (rel.fec_target_pkt_del_rounds > kMaxTgtPktDelRnds))
218  {
219  LogE(kClassName, __func__, "Conn %" PRIEndptId " Stream %" PRIStreamId
220  ": Error, FEC target number of rounds %" PRIRexmitRounds
221  " exceeds limits of 1 to %zu.\n", conn_id_, stream_id_,
222  rel.fec_target_pkt_del_rounds, kMaxTgtPktDelRnds);
223  return false;
224  }
225  }
226 
227  VdmFec::Initialize();
228  }
229 
230  // Set the packet pool for the RcvdPktInfo objects to use.
231  RcvdPktInfo::SetPacketPool(&packet_pool_);
232 
233  // Allocate the arrays of FEC information.
234  if (rel.mode == SEMI_RELIABLE_ARQ_FEC)
235  {
236  fec_grp_info_ = new (std::nothrow) FecGroupInfo[kFecGroupInfoSize];
237  fec_src_pkts_ = new (std::nothrow) RcvdPktInfo[kMaxFecGroupLengthPkts];
238 
239  if ((fec_grp_info_ == NULL) || (fec_src_pkts_ == NULL))
240  {
241  LogE(kClassName, __func__, "Conn %" PRIEndptId " Stream %" PRIStreamId
242  ": Error allocating FEC arrays.\n", conn_id_, stream_id_);
243  return false;
244  }
245 
246  // Invalidate the first entry in the FEC group array. The others are
247  // already invalid, as the FEC group IDs are initialized to zero.
248  fec_grp_info_[0].fec_grp_id_ = 1;
249  }
250 
251  // Allocate the circular array of received packet information.
252  rcvd_pkts_ = new (std::nothrow) RcvdPktInfo[kFlowCtrlWindowPkts];
253 
254  if (rcvd_pkts_ == NULL)
255  {
256  LogE(kClassName, __func__, "Conn %" PRIEndptId " Stream %" PRIStreamId
257  ": Error allocating received packet array.\n", conn_id_, stream_id_);
258  return false;
259  }
260 
261  // Store the settings.
262  rel_ = rel;
263  del_mode_ = del_mode;
264  rcv_wnd_ = kFlowCtrlWindowPkts;
265  rcv_min_ = init_seq_num;
266  rcv_nxt_ = init_seq_num;
267  rcv_max_ = (init_seq_num - 1);
268 
269 #ifdef SLIQ_DEBUG
270  LogD(kClassName, __func__, "Conn %" PRIEndptId " Stream %" PRIStreamId
271  ": Initialize receive packet manager: rel_mode %d rexmit_limit %"
272  PRIRexmitLimit " del_time %d tgt_rnds %" PRIFecRound " tgt_time %f "
273  "tgt_prob %f del_mode %d rcv_wnd_ %" PRIWindowSize " rcv_min_ %"
274  PRIPktSeqNumber " rcv_nxt_ %" PRIPktSeqNumber " rcv_max_ %"
275  PRIPktSeqNumber ".\n", conn_id_, stream_id_, rel_.mode,
276  rel_.rexmit_limit, static_cast<int>(rel_.fec_del_time_flag),
277  rel_.fec_target_pkt_del_rounds, rel_.fec_target_pkt_del_time_sec,
278  rel_.fec_target_pkt_recv_prob, del_mode_, rcv_wnd_, rcv_min_, rcv_nxt_,
279  rcv_max_);
280 #endif
281 
282  return true;
283 }

286 bool RcvdPktManager::IsGoodDataPacket(DataHeader& pkt)
287 {
288  // The manager must have been initialized.
289  if (rcvd_pkts_ == NULL)
290  {
291  LogF(kClassName, __func__, "Conn %" PRIEndptId " Stream %" PRIStreamId
292  ": Error, not initialized.\n", conn_id_, stream_id_);
293  }
294 
295  // If the packet is not to the right of rcv_max_, then look closer.
296  if (SEQ_LEQ(pkt.sequence_number, rcv_max_))
297  {
298  RcvdPktInfo& pkt_info = rcvd_pkts_[(pkt.sequence_number %
299  kFlowCtrlWindowPkts)];
300 
301  // Check if this is a duplicate packet.
302  if ((pkt.sequence_number == pkt_info.seq_num_) &&
303  (IS_RECEIVED(pkt_info)) &&
304  (pkt.retransmission_count <= pkt_info.rexmit_cnt_))
305  {
306  LogE(kClassName, __func__, "Conn %" PRIEndptId " Stream %" PRIStreamId
307  ": Error, duplicate packet seq %" PRIPktSeqNumber " detected (%"
308  PRIRetransCount " <= %" PRIRetransCount ").\n", conn_id_,
309  stream_id_, pkt_info.seq_num_, pkt.retransmission_count,
310  pkt_info.rexmit_cnt_);
311  return false;
312  }
313  }
314 
315  return true;
316 }

319 bool RcvdPktManager::AddPkt(DataHeader& pkt, const Time& rcv_time)
320 {
321  bool ack_now = false;
322 
323  // Update the packet receive statistics.
324  stats_pkts_.Update(rel_, pkt);
325 
326  // Make sure that the current window has not already moved beyond this
327  // packet.
328  if (SEQ_LT(pkt.sequence_number, rcv_min_))
329  {
330 #ifdef SLIQ_DEBUG
331  if (pkt.fec_flag)
332  {
333  LogD(kClassName, __func__, "Conn %" PRIEndptId " Stream %" PRIStreamId
334  ": Redundant packet, received seq %" PRIPktSeqNumber " grp %"
335  PRIFecGroupId " idx %" PRIFecSize " precedes the current window."
336  "\n", conn_id_, stream_id_, pkt.sequence_number, pkt.fec_group_id,
337  pkt.fec_group_index);
338  }
339  else
340  {
341  LogD(kClassName, __func__, "Conn %" PRIEndptId " Stream %" PRIStreamId
342  ": Redundant packet, received seq %" PRIPktSeqNumber " precedes "
343  "the current window.\n", conn_id_, stream_id_,
344  pkt.sequence_number);
345  }
346 #endif
347 
348  // Report the packet's timestamp in the next ACK packet to be sent.
349  obs_times_.StoreObsTime(pkt.sequence_number, pkt.timestamp, rcv_time);
350 
351  // Record the packet's sequence number as a recently received data packet.
352  rct_rcvs_.RecordSeqNum(pkt.sequence_number);
353 
354  // Release the packet.
355  packet_pool_.Recycle(pkt.payload);
356  pkt.payload = NULL;
357 
358  // Update the packet receive statistics for this redundant FEC packet.
359  if ((pkt.fec_flag) && (rel_.mode == SEMI_RELIABLE_ARQ_FEC) &&
360  (pkt.fec_round > 0) &&
361  (((rel_.fec_del_time_flag) && (pkt.fec_round < kOutOfRounds)) ||
362  ((!rel_.fec_del_time_flag) &&
363  (pkt.fec_round <= rel_.fec_target_pkt_del_rounds))))
364  {
365  // This value may be decremented if FEC encoded packets are used to
366  // regenerate FEC source packets.
367  ++stats_pkts_.fec_total_ext_rcvd_;
368  }
369 
370  // Attempt to update the flags and retransmission count for the packet.
371  // This can help detect duplicate data packets so that they can be
372  // ignored.
373  RcvdPktInfo& old_rpi = rcvd_pkts_[(pkt.sequence_number %
374  kFlowCtrlWindowPkts)];
375 
376  if ((pkt.sequence_number == old_rpi.seq_num_) &&
377  (pkt.retransmission_count > old_rpi.rexmit_cnt_))
378  {
379  SET_RECEIVED(old_rpi);
380  old_rpi.rexmit_cnt_ = pkt.retransmission_count;
381  }
382 
383  // This packet is out of order, so send an ACK packet immediately.
384  return true;
385  }
386 
387  // Make sure that there will be room for this packet in the circular receive
388  // window.
389  if (SEQ_GT(pkt.sequence_number, rcv_max_) &&
390  (((pkt.sequence_number - rcv_min_) + 1) > kFlowCtrlWindowPkts))
391  {
392  LogE(kClassName, __func__, "Conn %" PRIEndptId " Stream %" PRIStreamId
393  ": Dropping seq %" PRIPktSeqNumber ", circular array size %"
394  PRIPktSeqNumber " would exceed max %" PRIWindowSize ".\n", conn_id_,
395  stream_id_, pkt.sequence_number,
396  ((pkt.sequence_number - rcv_min_) + 1), kFlowCtrlWindowPkts);
397 
398  // Release the packet.
399  packet_pool_.Recycle(pkt.payload);
400  pkt.payload = NULL;
401 
402  // The sender should not overrun the receive window, so send an ACK packet
403  // immediately in order to hopefully correct the situation.
404  return true;
405  }
406 
407  // Report the packet's timestamp in the next ACK packet to be sent.
408  obs_times_.StoreObsTime(pkt.sequence_number, pkt.timestamp, rcv_time);
409 
410  // Record the packet's sequence number as a recently received data packet.
411  rct_rcvs_.RecordSeqNum(pkt.sequence_number);
412 
413  // Get access to the packet information for the received packet.
414  RcvdPktInfo& pkt_info = rcvd_pkts_[(pkt.sequence_number %
415  kFlowCtrlWindowPkts)];
416 
417  // Check if this packet is beyond the current window.
418  if (SEQ_GT(pkt.sequence_number, rcv_max_))
419  {
420  // Check that a FIN has not already been received.
421  if (rcv_max_fin_flag_)
422  {
423  LogF(kClassName, __func__, "Conn %" PRIEndptId " Stream %" PRIStreamId
424  ": Received seq %" PRIPktSeqNumber " that is greater than FIN seq "
425  "%" PRIPktSeqNumber ".\n", conn_id_, stream_id_,
426  pkt.sequence_number, rcv_max_);
427  }
428 
429  // Add "holes" for the missing packets.
430  size_t num_holes_to_add = (pkt.sequence_number - rcv_max_ - 1);
431 
432  if (num_holes_to_add > 0)
433  {
434  // This packet is out of order, so send an ACK packet immediately.
435  ack_now = true;
436  }
437 
438  for (PktSeqNumber seq_num = (rcv_max_ + 1);
439  SEQ_LT(seq_num, pkt.sequence_number); ++seq_num)
440  {
441  RcvdPktInfo& rpi = rcvd_pkts_[(seq_num % kFlowCtrlWindowPkts)];
442 
443  rpi.Clear();
444 
445 #ifdef SLIQ_DEBUG
446  LogD(kClassName, __func__, "Conn %" PRIEndptId " Stream %" PRIStreamId
447  ": Adding hole for seq %" PRIPktSeqNumber ".\n", conn_id_,
448  stream_id_, seq_num);
449 #endif
450  }
451 
452  // Now add the packet.
453  StorePkt(pkt, pkt_info);
454 
455  // Update the upper edge of the window.
456  rcv_max_ = pkt.sequence_number;
457  rcv_max_fin_flag_ = pkt.fin_flag;
458  }
459  else
460  {
461  // This packet is within the current window. Check if the packet has
462  // already been received and/or regenerated.
463  if ((!IS_RECEIVED(pkt_info)) && (!IS_REGENERATED(pkt_info)))
464  {
465  // The packet has not been received or regenerated yet. Add the packet.
466  StorePkt(pkt, pkt_info);
467 
468  // This packet is out of order, so send an ACK packet immediately.
469  ack_now = true;
470  }
471  else
472  {
473  // The packet has already been received or regenerated. Mark the packet
474  // as being received, update the retransmission count, then recycle the
475  // packet.
476  SET_RECEIVED(pkt_info);
477 
478  if (pkt.retransmission_count > pkt_info.rexmit_cnt_)
479  {
480  pkt_info.rexmit_cnt_ = pkt.retransmission_count;
481  }
482 
483  packet_pool_.Recycle(pkt.payload);
484  pkt.payload = NULL;
485 
486 #ifdef SLIQ_DEBUG
487  if (pkt.fec_flag)
488  {
489  LogD(kClassName, __func__, "Conn %" PRIEndptId " Stream %" PRIStreamId
490  ": Redundant packet, received seq %" PRIPktSeqNumber " grp %"
491  PRIFecGroupId " idx %" PRIFecSize " already present.\n",
492  conn_id_, stream_id_, pkt.sequence_number, pkt.fec_group_id,
493  pkt.fec_group_index);
494  }
495  else
496  {
497  LogD(kClassName, __func__, "Conn %" PRIEndptId " Stream %" PRIStreamId
498  ": Redundant packet, received seq %" PRIPktSeqNumber " already "
499  "present.\n", conn_id_, stream_id_, pkt.sequence_number);
500  }
501 #endif
502 
503  // Update the packet receive statistics for this redundant FEC packet.
504  if ((pkt.fec_flag) && (rel_.mode == SEMI_RELIABLE_ARQ_FEC) &&
505  (pkt.fec_round > 0) &&
506  (((rel_.fec_del_time_flag) && (pkt.fec_round < kOutOfRounds)) ||
507  ((!rel_.fec_del_time_flag) &&
508  (pkt.fec_round <= rel_.fec_target_pkt_del_rounds))))
509  {
510  // This value may be decremented if FEC encoded packets are used to
511  // regenerate FEC source packets.
512  ++stats_pkts_.fec_total_ext_rcvd_;
513  }
514 
515  // Since we have not added a new packet to the window, there is nothing
516  // left to do but return. This packet is out of order, so send an ACK
517  // packet immediately.
518  return true;
519  }
520  }
521 
522 #ifdef SLIQ_DEBUG
523  LogD(kClassName, __func__, "Conn %" PRIEndptId " Stream %" PRIStreamId
524  ": Adding packet, seq %" PRIPktSeqNumber " fec %d fin %d rcv_time %s."
525  "\n", conn_id_, stream_id_, pkt.sequence_number,
526  static_cast<int>(pkt.fec_flag), static_cast<int>(pkt.fin_flag),
527  rcv_time.ToString().c_str());
528 #endif
529 
530  // If the packet just added to the window is an FEC packet and FEC packets
531  // are expected, then update the packet receive statistics for this new FEC
532  // packet and attempt to regenerate packets within the same FEC group.
533  if ((pkt.fec_flag) && (rel_.mode == SEMI_RELIABLE_ARQ_FEC))
534  {
535  if ((pkt.fec_round > 0) &&
536  (((rel_.fec_del_time_flag) && (pkt.fec_round < kOutOfRounds)) ||
537  ((!rel_.fec_del_time_flag) &&
538  (pkt.fec_round <= rel_.fec_target_pkt_del_rounds))))
539  {
540  if (pkt.fec_pkt_type == FEC_SRC_PKT)
541  {
542  ++stats_pkts_.fec_total_src_rcvd_;
543  }
544  else
545  {
546  // This value may be decremented if FEC encoded packets are used to
547  // regenerate FEC source packets.
548  ++stats_pkts_.fec_total_ext_rcvd_;
549  }
550  }
551 
552  RegeneratePkts(pkt, rcv_time);
553  }
554 
555  // Now that the received packet has been added to the window and any FEC
556  // packets have been regenerated into the window, update the next expected
557  // sequence number.
558  UpdateNextExpected(false);
559 
560  // If there are any packets missing or if this packet has the FIN flag set,
561  // then send an ACK packet immediately.
562  if ((SEQ_GT(rcv_max_, rcv_nxt_)) || (pkt.fin_flag))
563  {
564  ack_now = true;
565  }
566 
567 #ifdef SLIQ_DEBUG
568  LogD(kClassName, __func__, "Conn %" PRIEndptId " Stream %" PRIStreamId
569  ": Updated receive window, rcv_min_ %" PRIPktSeqNumber " rcv_nxt_ %"
570  PRIPktSeqNumber " rcv_max_ %" PRIPktSeqNumber ".\n", conn_id_,
571  stream_id_, rcv_min_, rcv_nxt_, rcv_max_);
572 #endif
573 
574  return ack_now;
575 }

578 bool RcvdPktManager::GetNextAppPkt(Packet*& pkt, size_t& payload_offset,
579  size_t& payload_len, bool& fin_flag)
580 {
581  // Check if there is no possible way of delivering a packet.
582  if ((rcvd_pkts_ == NULL) || (SEQ_GT(rcv_min_, rcv_max_)))
583  {
584  return false;
585  }
586 
587  PktSeqNumber del_seq_num = rcv_min_;
588 
589  // First, find a candidate for delivery to the application. Ordered
590  // delivery will always use a sequence number of rcv_min_, which was set
591  // above.
592  if (del_mode_ == UNORDERED_DELIVERY)
593  {
594  // Unordered packet delivery to the application. Loop over the receive
595  // window to find a packet to deliver.
596  for (del_seq_num = rcv_min_;
597  SEQ_LEQ(del_seq_num, rcv_max_); ++del_seq_num)
598  {
599  RcvdPktInfo& rpi = rcvd_pkts_[(del_seq_num % kFlowCtrlWindowPkts)];
600 
601  // Be careful not to deliver FEC encoded packets.
602  if ((rpi.packet_ != NULL) &&
603  (IS_RECEIVED(rpi) || IS_REGENERATED(rpi)) &&
604  ((!IS_FEC(rpi)) || (rpi.fec_pkt_type_ == FEC_SRC_PKT)) &&
605  (!IS_DELIVERED(rpi)))
606  {
607  break;
608  }
609  }
610 
611  if (SEQ_GT(del_seq_num, rcv_max_))
612  {
613  // Attempt to move the left edge of the window to the right.
614  MoveWindowRight();
615 
616  return false;
617  }
618  }
619 
620  // Get the candidate for delivery using the sequence number identified
621  // above.
622  RcvdPktInfo& pkt_info = rcvd_pkts_[(del_seq_num % kFlowCtrlWindowPkts)];
623 
624  // Determine if the candidate for delivery meets the criteria for delivery.
625  if ((pkt_info.packet_ == NULL) || IS_DELIVERED(pkt_info))
626  {
627  // Attempt to move the left edge of the window to the right.
628  MoveWindowRight();
629 
630  return false;
631  }
632 
633  // Mark that the packet was received on time (i.e., not "late").
634  pkt_info.packet_->set_recv_late(false);
635 
636  // Update the packet receive counts and the packet's received "late" flag.
637  if (rel_.mode == SEMI_RELIABLE_ARQ_FEC)
638  {
639  if ((pkt_info.fec_round_ > 0) &&
640  (((rel_.fec_del_time_flag) && (pkt_info.fec_round_ < kOutOfRounds)) ||
641  ((!rel_.fec_del_time_flag) &&
642  (pkt_info.fec_round_ <= rel_.fec_target_pkt_del_rounds))))
643  {
644  if (pkt_info.payload_len_ > 0)
645  {
646  ++stats_pkts_.target_app_rcvd_;
647  }
648  }
649  else
650  {
651  pkt_info.packet_->set_recv_late(true);
652  }
653  }
654 
655  // Update the raw goodput statistics.
656  if (pkt_info.payload_len_ > 0)
657  {
658  Time now = Time::Now();
659 
660  if (stats_pkts_.raw_goodput_bytes_ == 0)
661  {
662  stats_pkts_.start_time_ = now;
663  }
664 
665  stats_pkts_.raw_goodput_bytes_ += (kRawGpHdrSizeBytes +
666  pkt_info.payload_len_);
667  stats_pkts_.end_time_ = now;
668  }
669 
670  // Deliver this packet to the application.
671  if (IS_FEC(pkt_info) && (pkt_info.fec_pkt_type_ == FEC_SRC_PKT))
672  {
673  // The packet must be cloned in order to keep a copy for FEC decoding.
674  //
675  // NOTE: This code makes a deep copy of all FEC source data packets. It
676  // would be more efficient to do a shallow copy (a simple reference count
677  // increment) of the packet. However, the packet might be modified by one
678  // of the proxies while SLIQ is still holding onto it for FEC decoding,
679  // and any changes to the packet will corrupt the decoding.
680  pkt = packet_pool_.Clone(pkt_info.packet_, true,
681  iron::PACKET_COPY_TIMESTAMP);
682 
683  if (pkt == NULL)
684  {
685  LogF(kClassName, __func__, "Conn %" PRIEndptId " Stream %" PRIStreamId
686  ": Error cloning packet.\n", conn_id_, stream_id_);
687  }
688 
689  // Update the FEC group delivered packet count.
690  FecGroupInfo& grp_info = fec_grp_info_[(pkt_info.fec_grp_id_ %
691  kFecGroupInfoSize)];
692 
693  if (grp_info.fec_grp_id_ == pkt_info.fec_grp_id_)
694  {
695  ++(grp_info.delivered_cnt_);
696  }
697  }
698  else
699  {
700  // Simply hand off the packet since SLIQ no longer needs it.
701  pkt = pkt_info.packet_;
702  pkt_info.packet_ = NULL;
703  }
704 
705  payload_offset = pkt_info.payload_offset_;
706  payload_len = pkt_info.payload_len_;
707  fin_flag = IS_FIN(pkt_info);
708 
709  SET_DELIVERED(pkt_info);
710 
711 #ifdef SLIQ_DEBUG
712  LogD(kClassName, __func__, "Conn %" PRIEndptId " Stream %" PRIStreamId
713  ": Deliver seq %" PRIPktSeqNumber " to app.\n", conn_id_, stream_id_,
714  del_seq_num);
715 #endif
716 
717  // Attempt to move the left edge of the window to the right.
718  MoveWindowRight();
719 
720 #ifdef SLIQ_DEBUG
721  LogD(kClassName, __func__, "Conn %" PRIEndptId " Stream %" PRIStreamId
722  ": Updated receive window, rcv_min_ %" PRIPktSeqNumber " rcv_nxt_ %"
723  PRIPktSeqNumber " rcv_max_ %" PRIPktSeqNumber ".\n", conn_id_,
724  stream_id_, rcv_min_, rcv_nxt_, rcv_max_);
725 #endif
726 
727  return true;
728 }

730 //============================================================================
731 bool RcvdPktManager::MoveForward(PktSeqNumber ne_seq_num)
732 {
733  // Note that this method is called only if the reliability mode is not
734  // RELIABLE_ARQ.
735  if (rcvd_pkts_ == NULL)
736  {
737  return false;
738  }
739 
740  // Check if the specified next expected sequence number is greater than
741  // rcv_min_. If this is not the case, then there is nothing to do here.
742  if (SEQ_GT(ne_seq_num, rcv_min_))
743  {
744 #ifdef SLIQ_DEBUG
745  LogD(kClassName, __func__, "Conn %" PRIEndptId " Stream %" PRIStreamId
746  ": Moving forward to seq %" PRIPktSeqNumber ", rcv_min_ %"
747  PRIPktSeqNumber " rcv_nxt_ %" PRIPktSeqNumber " rcv_max_ %"
748  PRIPktSeqNumber ".\n", conn_id_, stream_id_, ne_seq_num, rcv_min_,
749  rcv_nxt_, rcv_max_);
750 #endif
751 
752  // First, update rcv_min_. Move the window forward until the new next
753  // expected sequence number is reached, then attempt to move it forward as
754  // far as possible. Stop if the receive window becomes empty.
755  while (SEQ_LEQ(rcv_min_, rcv_max_))
756  {
757  RcvdPktInfo& rpi = rcvd_pkts_[(rcv_min_ % kFlowCtrlWindowPkts)];
758 
759  // Once ne_seq_num is reached, then rcv_min_ is moved to the right only
760  // if the packet has been delivered or if the packet is an FEC encoded
761  // packet.
762  if (SEQ_GEQ(rcv_min_, ne_seq_num))
763  {
764  bool drop = ((IS_RECEIVED(rpi) || IS_REGENERATED(rpi)) &&
765  (IS_DELIVERED(rpi) ||
766  (IS_FEC(rpi) && (rpi.fec_pkt_type_ == FEC_ENC_PKT))));
767 
768  if (!drop)
769  {
770  break;
771  }
772  }
773 
774 #ifdef SLIQ_DEBUG
775  if (IS_DELIVERED(rpi))
776  {
777  LogD(kClassName, __func__, "Conn %" PRIEndptId " Stream %" PRIStreamId
778  ": Pkt seq %" PRIPktSeqNumber " already delivered, being "
779  "dropped.\n", conn_id_, stream_id_, rcv_min_);
780  }
781  else
782  {
783  LogD(kClassName, __func__, "Conn %" PRIEndptId " Stream %" PRIStreamId
784  ": Pkt seq %" PRIPktSeqNumber " being dropped.\n", conn_id_,
785  stream_id_, rcv_min_);
786  }
787 #endif
788 
789  // If this is an FEC source data packet, then add it to the FEC source
790  // data packet array.
791  if (IS_FEC(rpi) && (rpi.fec_pkt_type_ == FEC_SRC_PKT))
792  {
793 #ifdef SLIQ_DEBUG
794  LogD(kClassName, __func__, "Conn %" PRIEndptId " Stream %" PRIStreamId
795  ": Storing FEC SRC pkt seq %" PRIPktSeqNumber " for grp %"
796  PRIFecGroupId " idx %" PRIFecSize ".\n", conn_id_, stream_id_,
797  rcv_min_, rpi.fec_grp_id_, rpi.fec_grp_idx_);
798 #endif
799 
800  fec_src_pkts_[rpi.fec_grp_idx_].MoveFecInfo(rpi);
801  }
802 
803  // Drop the element.
804  if (rpi.packet_ != NULL)
805  {
806  packet_pool_.Recycle(rpi.packet_);
807  rpi.packet_ = NULL;
808  }
809 
810  ++rcv_min_;
811  }
812 
813  // If the receive window is now empty and the new next expected sequence
814  // was not reached, then simply adjust rcv_min_.
815  if (SEQ_GT(rcv_min_, rcv_max_) && SEQ_LT(rcv_min_, ne_seq_num))
816  {
817  rcv_min_ = ne_seq_num;
818  }
819 
820  // Next, update rcv_nxt_ as needed.
821  if (SEQ_LT(rcv_nxt_, rcv_min_))
822  {
823  UpdateNextExpected(true);
824  }
825 
826  // Finally, update rcv_max_ as needed.
827  if (SEQ_LT(rcv_max_, (rcv_min_ - 1)))
828  {
829  rcv_max_ = (rcv_min_ - 1);
830  }
831 
832 #ifdef SLIQ_DEBUG
833  LogD(kClassName, __func__, "Conn %" PRIEndptId " Stream %" PRIStreamId
834  ": Moved forward to seq %" PRIPktSeqNumber ", rcv_min_ %"
835  PRIPktSeqNumber " rcv_nxt_ %" PRIPktSeqNumber " rcv_max_ %"
836  PRIPktSeqNumber ".\n", conn_id_, stream_id_, ne_seq_num, rcv_min_,
837  rcv_nxt_, rcv_max_);
838 #endif
839 
840  // An ACK should be sent when rcv_min_ is updated, since the advertised
841  // window size might change.
842  return true;
843  }
844 
845  return false;
846 }

848 //============================================================================
849 size_t RcvdPktManager::PrepareNextAckHdr()
850 {
851  size_t len = kAckHdrBaseSize;
852 
853  if (rcvd_pkts_ == NULL)
854  {
855  return len;
856  }
857 
858  // Add the length of any observed times.
859  len += (obs_times_.cnt_ * kAckHdrObsTimeSize);
860 
861  // If this stream is currently in semi-reliable ARQ+FEC mode and there are
862  // no new observed times, then attempt to add the latest observed time.
863  if ((rel_.mode == SEMI_RELIABLE_ARQ_FEC) && (obs_times_.cnt_ == 0) &&
864  obs_times_.has_latest_)
865  {
866  len += kAckHdrObsTimeSize;
867  }
868 
869  // Add the length of any ACK block offsets. If there are no missing data
870  // packets, then there are no ACK block offsets.
871  ack_blk_.Clear();
872 
873  if (SEQ_GT(rcv_max_, rcv_nxt_))
874  {
875  // Build the ACK blocks necessary for the ACK header. The ACK blocks must
876  // include the most recently received data packet and largest observed
877  // data packet. This is done as follows:
878  // - As data packets are received, their sequence numbers are recorded in
879  // an ordered, circular list.
880  // - Working from newest to oldest sequence number in the circular list:
881  // > Walk window forward and backward from the next most recent
882  // sequence number to generate an ACK block that does not repeat an
883  // existing ACK block.
884  // > If this is the second time through the loop, then walk the window
885  // forward and backward from the largest observed sequence number to
886  // generate an ACK block that does not repeat an existing ACK block.
887  PktSeqNumber seq_num = 0;
888 
889  for (size_t i = 0; i < kAckHistorySize; ++i)
890  {
891  // After the most recently received data packet has been processed (when
892  // i = 0), generate an ACK block for the largest observed sequence
893  // number (rcv_max_).
894  if (i == 1)
895  {
896  GenerateAckBlock(rcv_max_);
897  }
898 
899  // Attempt to get the next most recently received data packet sequence
900  // number and generate an ACK block for it.
901  if (!rct_rcvs_.GetSeqNum(i, seq_num))
902  {
903  break;
904  }
905 
906  GenerateAckBlock(seq_num);
907 
908  // If the target number of ACK block offsets has been reached, then
909  // stop.
910  if (ack_blk_.hdr_cnt_ >= kTargetAckBlockOffsets)
911  {
912  break;
913  }
914  }
915 
916  // Add the length of the resulting ACK block offsets.
917  len += (ack_blk_.hdr_cnt_ * kAckHdrAckBlockOffsetSize);
918  }
919 
920  return len;
921 }

923 //============================================================================
924 bool RcvdPktManager::BuildNextAckHdr(AckHeader& ack_hdr, const Time& now)
925 {
926  if (rcvd_pkts_ == NULL)
927  {
928  return false;
929  }
930 
931  // This is a check just to warn if the element at the left edge of the
932  // receive window still has a packet that has not been delivered to the
933  // application.
934  if (SEQ_LEQ(rcv_min_, rcv_max_))
935  {
936  RcvdPktInfo& rpi = rcvd_pkts_[(rcv_min_ % kFlowCtrlWindowPkts)];
937 
938  if ((rpi.packet_ != NULL) && (IS_RECEIVED(rpi) || IS_REGENERATED(rpi)) &&
939  ((!IS_FEC(rpi)) || (rpi.fec_pkt_type_ == FEC_SRC_PKT)) &&
940  (!IS_DELIVERED(rpi)))
941  {
942  LogW(kClassName, __func__, "Conn %" PRIEndptId " Stream %" PRIStreamId
943  ": Warning, packet at rcv_min_ %" PRIPktSeqNumber " is here but "
944  "not delivered.\n", conn_id_, stream_id_, rcv_min_);
945  }
946  }
947 
948  // Populate the ACK header.
949  ack_hdr.stream_id = stream_id_;
950  ack_hdr.num_observed_times = 0;
951  ack_hdr.num_ack_block_offsets = 0;
952  ack_hdr.next_expected_seq_num = rcv_nxt_;
953 
954 #ifdef SLIQ_DEBUG
955  LogD(kClassName, __func__, "Conn %" PRIEndptId " Stream %" PRIStreamId
956  ": ACK header: ne_seq %" PRIPktSeqNumber ".\n", conn_id_, stream_id_,
957  ack_hdr.next_expected_seq_num);
958 #endif
959 
960  // Add any observed times to the ACK header.
961  obs_times_.AddObsTimesToAckHdr(ack_hdr, now);
962 
963  // If this stream is currently in semi-reliable ARQ+FEC mode and there are
964  // no new observed times, then attempt to add the latest observed time.
965  // This is necessary in order to end the FEC group rounds quickly in the
966  // presence of data and ACK packet loss.
967  if ((rel_.mode == SEMI_RELIABLE_ARQ_FEC) &&
968  (ack_hdr.num_observed_times == 0))
969  {
970  obs_times_.AddLatestObsTimeToAckHdr(ack_hdr, now);
971  }
972 
973  // Add the ACK block offsets prepared in PrepareNextAckHdr() to the ACK
974  // header.
975  if (ack_blk_.cnt_ > 0)
976  {
977  ack_blk_.AddAckBlocksToAckHdr(ack_hdr, rcv_nxt_);
978  }
979 
980  return true;
981 }

1088 //============================================================================
1089 void RcvdPktManager::GenerateAckBlock(PktSeqNumber seq_num)
1090 {
1091  // Make sure that the packet is still within the window.
1092  if (SEQ_LT(seq_num, rcv_nxt_))
1093  {
1094 #ifdef SLIQ_DEBUG
1095  LogD(kClassName, __func__, " Packet seq %" PRIPktSeqNumber " no longer "
1096  "in window.\n", seq_num);
1097 #endif
1098 
1099  return;
1100  }
1101 
1102  // Check if the packet is already covered by an ACK block.
1103  if ((ack_blk_.cnt_ > 0) && ack_blk_.IsAlreadyInAckBlock(seq_num))
1104  {
1105 #ifdef SLIQ_DEBUG
1106  LogD(kClassName, __func__, " Packet seq %" PRIPktSeqNumber " already "
1107  "covered by an ACK block.\n", seq_num);
1108 #endif
1109 
1110  return;
1111  }
1112 
1113  // The packet is not already covered by an ACK block. Walk the window
1114  // backward and forward from the specified sequence number to generate the
1115  // ACK block.
1116  PktSeqNumber ack_lo = seq_num;
1117  PktSeqNumber ack_hi = seq_num;
1118 
1119  for (PktSeqNumber sn = (seq_num - 1); SEQ_GEQ(sn, rcv_nxt_); --sn)
1120  {
1121  RcvdPktInfo& rpi = rcvd_pkts_[(sn % kFlowCtrlWindowPkts)];
1122 
1123  if ((!IS_RECEIVED(rpi)) && (!IS_REGENERATED(rpi)))
1124  {
1125  break;
1126  }
1127 
1128  ack_lo = sn;
1129  }
1130 
1131  for (PktSeqNumber sn = (seq_num + 1); SEQ_LEQ(sn, rcv_max_); ++sn)
1132  {
1133  RcvdPktInfo& rpi = rcvd_pkts_[(sn % kFlowCtrlWindowPkts)];
1134 
1135  if ((!IS_RECEIVED(rpi)) && (!IS_REGENERATED(rpi)))
1136  {
1137  break;
1138  }
1139 
1140  ack_hi = sn;
1141  }
1142 
1143  ack_blk_.AddAckBlock(ack_lo, ack_hi);
1144 
1145 #ifdef SLIQ_DEBUG
1146  LogD(kClassName, __func__, " Packet seq %" PRIPktSeqNumber " ACK block %"
1147  PRIPktSeqNumber "-%" PRIPktSeqNumber ".\n", seq_num, ack_lo, ack_hi);
1148 #endif
1149 }

1151 //============================================================================
1152 void RcvdPktManager::StorePkt(DataHeader& pkt, RcvdPktInfo& pkt_info)
1153 {
1154  // Release any existing Packet object in the RcvdPktInfo.
1155  if (pkt_info.packet_ != NULL)
1156  {
1157  // There might be an FEC regenerated packet, which is OK. Otherwise, this
1158  // is unexpected.
1159  if (!IS_REGENERATED(pkt_info))
1160  {
1161  LogW(kClassName, __func__, "Conn %" PRIEndptId " Stream %" PRIStreamId
1162  ": Warning, seq %" PRIPktSeqNumber " was not received but has a "
1163  "pkt.\n", conn_id_, stream_id_, pkt.sequence_number);
1164  TRACK_UNEXPECTED_DROP(kClassName, packet_pool_);
1165  }
1166 
1167  packet_pool_.Recycle(pkt_info.packet_);
1168  pkt_info.packet_ = NULL;
1169  }
1170 
1171  // If needed, get the packet's time-to-go (TTG) from the data header, update
1172  // it for the current one-way delay (OWD) estimate, and store it in the
1173  // payload Packet object.
1174  if (((!pkt.fec_flag) || (pkt.fec_pkt_type == FEC_SRC_PKT)) &&
1175  (pkt.payload != NULL) && (pkt.num_ttg == 1))
1176  {
1177  Time rcv_time(pkt.payload->recv_time());
1178  double owd_est_sec = conn_.GetRtlOwdEst(pkt.timestamp, rcv_time);
1179  double new_ttg_sec = (pkt.ttg[0] - owd_est_sec);
1180 
1181  if (new_ttg_sec < 0.0)
1182  {
1183  new_ttg_sec = 0.0;
1184  }
1185 
1186  Time nttg(new_ttg_sec);
1187 
1188  pkt.payload->set_track_ttg(true);
1189  pkt.payload->SetTimeToGo(nttg, true);
1190 
1191 #ifdef SLIQ_DEBUG
1192  LogD(kClassName, __func__, "Conn %" PRIEndptId " Stream %" PRIStreamId

1389 #endif

1394  LogE(kClassName, __func__, "Conn %" PRIEndptId " Stream %" PRIStreamId
1395  ": No FEC group info found for grp %" PRIFecGroupId ".\n", conn_id_,
1396  stream_id_, grp_id);
1397  return;
1398  }
1399 
1400  // If no FEC encoded data packets have been received for the FEC group or
1401  // the number of FEC source data packets for the FEC group is not known,
1402  // then regeneration cannot be done yet.
1403  if ((grp_info.fec_enc_rcvd_cnt_ == 0) || (grp_info.fec_num_src_ == 0))
1404  {
1405  return;
1406  }
1407

1409  // correct total number of FEC source and encoded data packets, then
1410  // regeneration is not needed.
1411  if ((grp_info.fec_src_rcvd_cnt_ == grp_info.fec_num_src_) ||
1412  ((grp_info.fec_src_rcvd_cnt_ + grp_info.fec_enc_rcvd_cnt_) !=
1413  grp_info.fec_num_src_))
1414  {
1415  return;
1416  }

1419  FecSize src_cnt = 0;
1420  PktSeqNumber seq_num = 0;
1421  RetransCount max_rnd = 0;
1422 
1423  // Clear the VDM decoder information.
1424  memset(&vdm_info_, 0, sizeof(vdm_info_));
1425

1428  ((grp_info.fec_src_rcvd_cnt_ > 0) &&
1429  (src_cnt < grp_info.fec_src_rcvd_cnt_) &&

1434  ": Considering seq %" PRIPktSeqNumber ".\n", conn_id_, stream_id_,
1435  seq_num);
1436 #endif

1439  bool in_wnd = SEQ_GEQ(seq_num, rcv_min_);
1440 
1441  // If the packet is not in the current window, then search for it in the
1442  // FEC source packet array.
1443  if (!in_wnd)
1444  {
1445  bool pkt_found = false;
1446

1448  {
1449  if ((fec_src_pkts_[src_idx].seq_num_ == seq_num) &&
1450  (fec_src_pkts_[src_idx].fec_grp_id_ == grp_id) &&
1451  (fec_src_pkts_[src_idx].packet_ != NULL))
1452  {
1453 #ifdef SLIQ_DEBUG
1454  LogD(kClassName, __func__, "Conn %" PRIEndptId " Stream %"
1455  PRIStreamId ": Found FEC SRC seq %" PRIPktSeqNumber " grp %"
1456  PRIFecGroupId " at index %" PRIFecSize " outside of window."

1461  break;
1462  }
1463  }

1469  ": Could not find FEC SRC seq %" PRIPktSeqNumber " grp %"
1470  PRIFecGroupId " outside of window.\n", conn_id_, stream_id_,
1471  seq_num, grp_id);
1472 #endif
1473

1475  }
1476  }
1477 
1478  RcvdPktInfo& pkt_info = (in_wnd ?
1479  rcvd_pkts_[(seq_num % kFlowCtrlWindowPkts)] :
1480  fec_src_pkts_[(src_idx %
1481  kMaxFecGroupLengthPkts)]);

1484  // regeneration work is already done.

1490  conn_id_, stream_id_, seq_num);
1491 #endif
1492  return;
1493  }
1494

1496  if (IS_RECEIVED(pkt_info) && IS_FEC(pkt_info) &&
1497  (pkt_info.fec_pkt_type_ == FEC_SRC_PKT))
1498  {
1499  // If this FEC source packet is from another FEC group, then stop.
1500  if (pkt_info.fec_grp_id_ != grp_id)
1501  {
1502  break;
1503  }
1504 
1505  // The FEC source data packet must still be available.
1506  if (pkt_info.packet_ == NULL)
1507  {

1509  ": Error, NULL packet pointer for received FEC SRC data packet."
1510  "\n", conn_id_, stream_id_);
1511  }
1512

1514  LogD(kClassName, __func__, "Conn %" PRIEndptId " Stream %" PRIStreamId
1515  ": Found received FEC SRC pkt seq %" PRIPktSeqNumber " idx %"

1531  }
1532  else

1562

1600

1660 #endif
1661

1685  vdm_info_.in_pkt_index_[in_idx] = pkt_info.fec_grp_idx_;

1687  ++enc_cnt;
1688  }
1689  }

1691  // Verify the number of FEC encoded data packets found.

1693  {

1695  ": Error, only found %" PRIFecSize " of %" PRIFecSize " FEC ENC "
1696  "pkts.\n", conn_id_, stream_id_, enc_cnt,
1697  grp_info.fec_enc_rcvd_cnt_);
1698  }
1699 
1700  // Verify the number of FEC source and encoded data packets found.
1701  if (vdm_info_.num_src_pkt_ != static_cast<int>(grp_info.fec_num_src_))
1702  {
1703  LogF(kClassName, __func__, "Conn %" PRIEndptId " Stream %" PRIStreamId
1704  ": Error, only found %d of %" PRIFecSize " FEC pkts.\n", conn_id_,
1705  stream_id_, vdm_info_.num_src_pkt_, grp_info.fec_num_src_);
1706  }
1707 
1708 #ifdef SLIQ_DEBUG
1709  LogD(kClassName, __func__, "Conn %" PRIEndptId " Stream %" PRIStreamId
1710  ": Results src %" PRIFecSize " enc %" PRIFecSize " num_src %"
1711  PRIFecSize ".\n", conn_id_, stream_id_, src_cnt, enc_cnt,
1712  grp_info.fec_num_src_);
1713 #endif
1714 
1715  // Set the output packet information for the FEC source data packets that
1716  // will be regenerated.
1717  int out_idx = 0;
1718 
1719  for (out_idx = 0; out_idx < grp_info.fec_num_src_; ++out_idx)
1720  {
1721  if (vdm_info_.out_pkt_data_[out_idx] == NULL)
1722  {
1723  vdm_info_.out_pkt_[out_idx] = packet_pool_.Get();
1724 
1725  if (vdm_info_.out_pkt_[out_idx] == NULL)
1726  {
1727  LogF(kClassName, __func__, "Conn %" PRIEndptId " Stream %" PRIStreamId
1728  ": Error getting packet from pool.\n", conn_id_, stream_id_);
1729  }
1730 
1731  vdm_info_.out_pkt_data_[out_idx] =
1732  vdm_info_.out_pkt_[out_idx]->GetBuffer(0);
1733  }
1734  }
1735 
1736  // Decode the packets.
1737  if (VdmFec::DecodePackets(vdm_info_.num_src_pkt_, vdm_info_.in_pkt_data_,
1738  vdm_info_.in_pkt_size_,
1739  vdm_info_.in_enc_pkt_size_,
1740  vdm_info_.in_pkt_index_, vdm_info_.out_pkt_data_,
1741  vdm_info_.out_pkt_size_) != 0)
1742  {
1743  LogE(kClassName, __func__, "Conn %" PRIEndptId " Stream %" PRIStreamId
1744  ": Error decoding FEC packets for grp %" PRIFecGroupId ".\n",
1745  conn_id_, stream_id_, grp_id);
1746 
1747  // Release the allocated Packet objects that were to be used to hold the
1748  // decoding data packets.
1749  for (out_idx = 0; out_idx < grp_info.fec_num_src_; ++out_idx)
1750  {
1751  if (vdm_info_.out_pkt_[out_idx] != NULL)
1752  {
1753  packet_pool_.Recycle(vdm_info_.out_pkt_[out_idx]);
1754  vdm_info_.out_pkt_[out_idx] = NULL;
1755  }
1756  }
1757 
1758  return;
1759  }

1761  // Compute the time-to-go (TTG) correction, if any. This is necessary if
1762  // the last packet received in the group is an FEC source packet, it has a
1763  // TTG, and a TTG vector was received for the group previously in an FEC
1764  // encoded packet.
1765  double ttg_corr = 0.0;
1766 
1767  if ((fec_pkt.fec_pkt_type == FEC_SRC_PKT) &&
1768  (fec_pkt.num_ttg == 1) && (grp_info.ttg_cnt_ >= grp_info.fec_num_src_))
1769  {
1770  ttg_corr = (grp_info.ttg_[fec_pkt.fec_group_index] - fec_pkt.ttg[0]);
1771 
1772 #ifdef SLIQ_DEBUG
1773  LogD(kClassName, __func__, "Conn %" PRIEndptId " Stream %" PRIStreamId
1774  ": Packet TTG correction (%f - %f) = %f\n", conn_id_, stream_id_,
1775  static_cast<double>(grp_info.ttg_[fec_pkt.fec_group_index]),
1776  fec_pkt.ttg[0], ttg_corr);
1777 #endif
1778  }
1779 
1780  // Store the regenerated FEC source data packets while updating their sizes.
1781 
1782  // The position of the FEC source data packets can vary. Use the decoded
1783  // sequence numbers in the regenerated FEC source data packets to position
1784  // them correctly.
1785  for (out_idx = 0; out_idx < grp_info.fec_num_src_; ++out_idx)
1786  {
1787  if (vdm_info_.out_pkt_[out_idx] != NULL)
1788  {
1789  // This is a missing FEC source data packet. Get the sequence number.
1790  Packet* pkt = vdm_info_.out_pkt_[out_idx];
1791  uint16_t pkt_len = vdm_info_.out_pkt_size_[out_idx];
1792  uint32_t seq_num_nbo = 0;
1793 
1794  if (pkt_len < sizeof(seq_num_nbo))
1795  {
1796  LogF(kClassName, __func__, "Conn %" PRIEndptId " Stream %" PRIStreamId
1797  ": Error, FEC SRC pkt for grp %" PRIFecGroupId " too small for "
1798  "sequence number.\n", conn_id_, stream_id_, grp_id);
1799  }
1800 
1801  pkt_len -= sizeof(seq_num_nbo);
1802  ::memcpy(&seq_num_nbo, reinterpret_cast<void*>(pkt->GetBuffer(pkt_len)),
1803  sizeof(seq_num_nbo));
1804  seq_num = ntohl(seq_num_nbo);
1805 
1806  // Make sure that the regenerated FEC source data packet is still
1807  // within the window.
1808  if (SEQ_LT(seq_num, rcv_min_))
1809  {
1810  LogW(kClassName, __func__, "Conn %" PRIEndptId " Stream %" PRIStreamId
1811  ": Warning, no place for FEC SRC pkt seq %" PRIPktSeqNumber
1812  " in grp %" PRIFecGroupId " idx %d.\n", conn_id_, stream_id_,
1813  seq_num, grp_id, out_idx);
1814  }
1815  else
1816  {
1817  RcvdPktInfo& pkt_info = rcvd_pkts_[(seq_num % kFlowCtrlWindowPkts)];
1818 
1819  if (IS_RECEIVED(pkt_info))
1820  {
1821  LogF(kClassName, __func__, "Conn %" PRIEndptId " Stream %"
1822  PRIStreamId ": Error, FEC SRC pkt seq %" PRIPktSeqNumber
1823  " in grp %" PRIFecGroupId " idx %d already received.\n",
1824  conn_id_, stream_id_, seq_num, grp_id, out_idx);
1825  }
1826 
1827  if (pkt_info.packet_ != NULL)
1828  {
1829  packet_pool_.Recycle(pkt_info.packet_);
1830  }
1831 
1832  pkt_info.packet_ = pkt;
1833  vdm_info_.out_pkt_[out_idx] = NULL;
1834 
1835  pkt_info.seq_num_ = seq_num;
1836  pkt_info.flags_ = 0;
1837 
1838  SET_FEC(pkt_info);
1839 
1840  pkt_info.fec_pkt_type_ = static_cast<uint8_t>(FEC_SRC_PKT);
1841  pkt_info.fec_grp_id_ = grp_id;
1842  pkt_info.fec_enc_pkt_len_ = 0;
1843  pkt_info.fec_grp_idx_ = static_cast<FecSize>(out_idx);
1844  pkt_info.fec_num_src_ = grp_info.fec_num_src_;
1845  pkt_info.fec_round_ = max_rnd;
1846 
1847  SET_REGENERATED(pkt_info);
1848 
1849  pkt_info.packet_->SetLengthInBytes(pkt_len);
1850  pkt_info.packet_->set_recv_time(rcv_time);
1851  pkt_info.payload_offset_ = 0;
1852  pkt_info.payload_len_ = pkt_len;
1853  pkt_info.rexmit_cnt_ = 0;
1854 
1855  // Determine the TTG for the regenerated FEC source packet, if
1856  // possible.
1857  double new_ttg_sec = 0.0;
1858 
1859  if (grp_info.ttg_cnt_ >= grp_info.fec_num_src_)
1860  {
1861  double owd_est_sec = conn_.GetRtlOwdEst(0, rcv_time);
1862 
1863  new_ttg_sec = (grp_info.ttg_[out_idx] - owd_est_sec - ttg_corr);
1864 
1865  if (new_ttg_sec < 0.0)
1866  {
1867  new_ttg_sec = 0.0;
1868  }
1869 
1870  Time nttg(new_ttg_sec);
1871 
1872  pkt_info.packet_->set_track_ttg(true);
1873  pkt_info.packet_->SetTimeToGo(nttg, true);
1874 
1875 #ifdef SLIQ_DEBUG
1876  LogD(kClassName, __func__, "Conn %" PRIEndptId " Stream %"
1877  PRIStreamId ": Latency-sensitive pkt seq %" PRIPktSeqNumber
1878  " enc_ttg %f owd_est %f ttg_corr %f new_ttg %f.\n", conn_id_,
1879  stream_id_, seq_num,
1880  static_cast<double>(grp_info.ttg_[out_idx]), owd_est_sec,
1881  ttg_corr, new_ttg_sec);
1882 #endif
1883 
1884 #ifdef TTG_TRACKING

1886  // Format: PLT_OWD <seq_num> <ttg_delta> <final_ttg>
1887  LogC(kClassName, __func__, "Conn %" PRIEndptId ": PLT_OWD %"
1888  PRIPktSeqNumber " %f %f\n", conn_id_, seq_num, owd_est_sec,
1889  new_ttg_sec);
1890 #endif // TTG_TRACKING
1891  }
1892

1894  // data packet.
1895  rct_rcvs_.RecordSeqNum(seq_num);
1896 
1897  // Update the packet regeneration statistics.
1898  ++stats_pkts_.fec_src_regen_;
1899 
1900 #ifdef SLIQ_DEBUG
1901  if (pkt_info.packet_->track_ttg())
1902  {
1903  LogD(kClassName, __func__, "Conn %" PRIEndptId " Stream %"
1904  PRIStreamId ": Regenerated grp %" PRIFecGroupId " idx %d seq %"
1905  PRIPktSeqNumber " len %" PRIu16 " ttg %f.\n", conn_id_,
1906  stream_id_, grp_id, out_idx, seq_num, pkt_len, new_ttg_sec);
1907  }
1908  else
1909  {
1910  LogD(kClassName, __func__, "Conn %" PRIEndptId " Stream %"
1911  PRIStreamId ": Regenerated grp %" PRIFecGroupId " idx %d seq %"
1912  PRIPktSeqNumber " len %" PRIu16 ".\n", conn_id_, stream_id_,
1913  grp_id, out_idx, seq_num, pkt_len);
1914  }
1915 #endif
1916  }
1917  }
1918  }
1919 
1920 #ifdef SLIQ_DEBUG
1921  LogD(kClassName, __func__, "Conn %" PRIEndptId " Stream %" PRIStreamId
1922  ": Completed grp %" PRIFecGroupId " via regeneration.\n", conn_id_,
1923  stream_id_, grp_id);
1924 #endif
1925 
1926  // Update the packet receive statistics for the regenerated FEC source
1927  // packets.
1928  if (((rel_.fec_del_time_flag) && (max_rnd < kOutOfRounds)) ||
1929  ((!rel_.fec_del_time_flag) &&
1930  (max_rnd <= rel_.fec_target_pkt_del_rounds)))
1931  {
1932  // Increment the number of FEC source packets received, and decrement the
1933  // number of "extra" FEC encoded packets used to do the regeneration.
1934  stats_pkts_.fec_total_src_rcvd_ += static_cast<size_t>(enc_cnt);
1935  stats_pkts_.fec_total_ext_rcvd_ -= static_cast<size_t>(enc_cnt);
1936  }
1937 
1938  // Release any allocated Packet objects that were not transferred to
1939  // received packet information entries.
1940  for (out_idx = 0; out_idx < grp_info.fec_num_src_; ++out_idx)
1941  {
1942  if (vdm_info_.out_pkt_[out_idx] != NULL)
1943  {
1944  LogW(kClassName, __func__, "Conn %" PRIEndptId " Stream %" PRIStreamId
1945  ": Warning, FEC SRC Packet object for grp %" PRIFecGroupId " idx "
1946  "%d not used.\n", conn_id_, stream_id_, grp_id, out_idx);
1947 
1948  packet_pool_.Recycle(vdm_info_.out_pkt_[out_idx]);
1949  vdm_info_.out_pkt_[out_idx] = NULL;
1950  }
1951  }
