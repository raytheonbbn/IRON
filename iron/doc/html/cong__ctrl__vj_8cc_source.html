75 void VJCongCtrlAlg::AckRcvd(uint32_t ack_num, int bytes_acked)
76 {
77  if (!selected_)
78  {
79  return;
80  }
81 
82  // Credit our cwnd for the value of the data acked.
83  //
84  // This should only be called if you are not in a fast recovery epoch.
85  if (!(socket_->funct_flags() & FUNCT_HIGH_SEQ))
86  {
87  if ((socket_->funct_flags() & FUNCT_HIGH_CONGESTION_SEQ) &&
88  (SEQ_LEQ(ack_num, socket_->high_cong_seq())))
89  {
90  // No soup for you!
91  }
92  else
93  {
94  // [VF010.2]
95  // Correct condition: removed diff from calculation
96  // Simplified if-else
97  // Added cast to int to take care of sequence wrap
98  // Replaced maxseg with maxdata
99 
100  if ((int)(socket_->seq_sent() - ack_num) <=
101  (int)socket_->snd_prev_cwnd())
102  {
103  socket_->snd_cwnd() += MAX(socket_->max_data(), bytes_acked);
104  }
105 
106  //
107  // Always cap snd_cwnd to snd_prevcwnd when you are NOT in an epoch.
108  //
109 
110  socket_->set_snd_cwnd(MIN(socket_->snd_cwnd(),
111  socket_->snd_prev_cwnd()));
112  }
113  }
114 
115  // Moving sendUna forward gets us out of fast retransmit if sendUna moves
116  // across sock->highSeq. Otherwise, we remain in fast retransmit mode,
117  // trying to recover 1 lost packet per RTT.
118  if ((socket_->funct_flags() & FUNCT_HIGH_SEQ) &&
119  SEQ_GEQ(ack_num, socket_->high_seq()))
120  {
121  // We're transitioning out of Fast Recovery. Give ourselves a full
122  // (1/2 of previous) sendCwnd bucket of credit. Since ssthresh gets
123  // cut later in this function we have to fake it. If you clip
124  // tmp_prevcwnd to the CURRENT window, then if, when exiting the
125  // congestion eopoch the offered window is small, you get no prevcwnd
126  // credit and essentially go into very slow growth.
127  //
128  // Behavior on exiting fast retransmit:
129  //
130  // If the number of packets in flight is less than prevcwnd, (which is
131  // the maximum number of packets you may have when exiting a
132  // congestion epoch) then you may emit as many packets as necessary so
133  // the number of packets in flight equal to the prevcwnd.
134  socket_->set_snd_cwnd(MAX(0, socket_->snd_prev_cwnd() -
135  (socket_->seq_sent() - ack_num)));
136  socket_->funct_flags() &= ~FUNCT_HIGH_CONGESTION_SEQ;
137  socket_->set_high_cong_seq(0);
138 
139  LogD(kClassName, __func__, "Out of FR, snd_cwnd(%ld) relative ackHBO "
140  "(%ld) prevcwnd (%ld) \n", socket_->snd_cwnd(),
141  ack_num - socket_->initial_seq_num(), socket_->snd_prev_cwnd());
142 
143  socket_->set_funct_flags(socket_->funct_flags() & (~FUNCT_HIGH_SEQ));
144  socket_->set_high_seq(0);
145  socket_->set_pkts_ack_in_epoch(0);
146  }
147  else if (!(socket_->funct_flags() & FUNCT_HIGH_SEQ))
148  {
149  // Get sendCwnd credit
150  if (socket_->snd_prev_cwnd() < socket_->snd_ssthresh())
151  {
152  // We're in exponential VJ mode.
153  socket_->snd_prev_cwnd() += socket_->max_data();
154  socket_->snd_cwnd() += socket_->max_data();
155  }
156  else
157  {
158  // Linear VJ mode.
159  //
160  // We give ourselves credit here if ack_num is greater than
161  // highCongSeq.
162  if ((!(socket_->funct_flags() & FUNCT_HIGH_CONGESTION_SEQ)) ||
163  ((socket_->funct_flags() & FUNCT_HIGH_CONGESTION_SEQ) &&
164  SEQ_GT(ack_num, socket_->high_cong_seq())))
165  {
166  // [VF021] Corrected: if condition was negated by mistake
167  if (socket_->snd_prev_cwnd())
168  {
169  socket_->snd_prev_cwnd() += ((socket_->max_data() *
170  socket_->max_data()) /
171  socket_->snd_prev_cwnd());
172  socket_->snd_cwnd() += ((socket_->max_data() *
173  socket_->max_data()) /
174  socket_->snd_prev_cwnd());
175  }
176  else
177  {
178  socket_->snd_prev_cwnd() += socket_->max_data();
179  socket_->snd_cwnd() += ((socket_->max_data() *
180  socket_->max_data()) /
181  socket_->snd_prev_cwnd());
182  }
183  }
184  }
185 
186  // Clip socket_->sendPrevCwnd to the offered window.
187  socket_->set_snd_prev_cwnd(MIN(socket_->snd_prev_cwnd(),
188  TCP_MAXWIN << socket_->snd_scale()));
189  }
190 
191  if (socket_->funct_flags() & FUNCT_HIGH_SEQ)
192  {
193  socket_->set_pkts_ack_in_epoch(bytes_acked / socket_->max_data());
194  }
195 }

198 void VJCongCtrlAlg::SnackRcvd(const struct tcphdr* tcp_hdr, int data_len,
199  int bytes_acked)
200 {
201  if (!selected_)
202  {
203  return;
204  }
205 
206  uint32_t ack_num = ntohl(tcp_hdr->th_ack);
207 
208  // Clip CWND to the amount of data in flight first.
209  socket_->set_snd_prev_cwnd(MIN(socket_->snd_prev_cwnd(),
210  socket_->seq_sent() - socket_->snd_una()));
211  socket_->set_snd_prev_cwnd(MIN(socket_->snd_prev_cwnd(),
212  (ntohs(tcp_hdr->th_win) <<
213  socket_->snd_scale())));
214  socket_->set_snd_ssthresh(MAX((socket_->max_data() << 1),
215  (socket_->snd_prev_cwnd() >> 1)));
216 
217  // Round down prevcwnd to a multiple of maxdata. This allows VJ
218  // congestion control algorithm to perform a bit better.
219  if ((int)socket_->snd_ssthresh() ==
220  ((int)((socket_->snd_ssthresh()) / socket_->max_data())) *
221  socket_->max_data())
222  {
223  socket_->set_snd_prev_cwnd(
224  (((int)((socket_->snd_ssthresh()) / socket_->max_data())) *
225  socket_->max_data()));
226  }
227  else
228  {
229  socket_->set_snd_prev_cwnd(
230  (((int)((socket_->snd_ssthresh()) / socket_->max_data())) *
231  socket_->max_data()) + socket_->max_data());
232  }
233  socket_->set_snd_ssthresh(MIN(socket_->snd_ssthresh(),
234  socket_->snd_prev_cwnd()));
235 
236  // When we enter a congestion epoch, we cut ssthresh in half AND set
237  // cwnd to 1 packet. If this dup ack does not increase the
238  // advertised window then we will set cwnd to 1 packet. If this is a
239  // pure dup ack, then the cwnd will be incremented a little later.
240  if (((socket_->last_uwe_in() !=
241  (ack_num + (ntohs(tcp_hdr->th_win) << socket_->snd_scale()))) ||
242  (bytes_acked > 0)) &&
243  (data_len == 0))
244  {
245  socket_->set_snd_cwnd(socket_->max_data());
246  }
247  else
248  {
249  socket_->set_snd_cwnd(0);
250  }
251 }

254 void VJCongCtrlAlg::DupAckRcvd(const struct tcphdr* tcp_hdr, int data_len)
255 {
256  bool enter_fast_rexmit_from_dup_ack = false;
257 
258  if (!selected_)
259  {
260  return;
261  }
262 
263  uint32_t ack_num = ntohl(tcp_hdr->th_ack);
264 
265  // Increment cwnd by one mss regardless, a segment has left the network.
266  // But, if this has a window update, it is *not* a duplicate ACK,
267  if ((socket_->last_uwe_in() ==
268  (ack_num + (ntohs(tcp_hdr->th_win) << socket_->snd_scale()))) &&
269  (data_len == 0))
270  {
271  // CWND INFLATION
272  //
273  // Don't clip s->sendCwnd to s->sendPrevCwnd. Chances of this actually being
274  // a problem are low.
275  socket_->pkts_ack_in_epoch()--;
276  if ((socket_->pkts_ack_in_epoch() * socket_->max_data() <
277  socket_->snd_ssthresh() + (DUPACK_THRESH * socket_->max_data())) ||
278  (socket_->t_dupacks() < DUPACK_THRESH))
279  {
280  socket_->pkts_ack_in_epoch()++;
281  socket_->snd_cwnd() += socket_->max_data();
282  }
283  }
284 
285  PktInfo* send_buf_snd_una = socket_->send_buf()->snd_una();
286  if ((socket_->funct_flags() & FUNCT_HIGH_SEQ) &&
287  (socket_->t_dupacks() <= DUPACK_THRESH) &&
288  (send_buf_snd_una))
289  {
290  // Round down prevcwnd to a multiple of maxdata. This allows VJ
291  // congestion control algorithm to perform a bit better.
292  socket_->set_snd_ssthresh(
293  MAX(socket_->snd_ssthresh(),
294  (socket_->seq_sent() - send_buf_snd_una->seq_num +
295  socket_->max_data()) >> 1));
296  if (socket_->snd_ssthresh() ==
297  (((socket_->snd_ssthresh()) / socket_->max_data())) *
298  socket_->max_data())
299  {
300  socket_->set_snd_prev_cwnd((((int)((socket_->snd_ssthresh()) /
301  socket_->max_data())) *
302  socket_->max_data()));
303  }
304  else
305  {
306  socket_->set_snd_prev_cwnd((((int)((socket_->snd_ssthresh()) /
307  socket_->max_data())) *
308  socket_->max_data()) + socket_->max_data());
309  }
310  socket_->set_snd_ssthresh(MAX(socket_->snd_ssthresh(),
311  socket_->snd_prev_cwnd()));
312  }
313 
314  if (send_buf_snd_una && (socket_->t_dupacks() >= DUPACK_THRESH))
315  {
316  // Need to modify this so that we only send a single Fast-Retransmit per
317  // rtt, otherwise noisy channels will really guber us up!
318  //
319  // How to do this?
320  //
321  // Add a rxmit_last value to the tpcb structure (ugh!) that has the
322  // clock value for the last retrans of this packet. This is set to 0
323  // when snduna moves forward and dup_ack_cnt is cleared; When the nth
324  // duplicate ack arrives and current_time > (rtt_curr + rxmit_last) we
325  // do a fast retransmit, clear the dup_ack_cnt and that's that.
326  if (!(socket_->funct_flags() & FUNCT_HIGH_SEQ))
327  {
328  enter_fast_rexmit_from_dup_ack = true;
329 
330  socket_->set_pkts_ack_in_epoch(
331  (socket_->seq_sent() - send_buf_snd_una->seq_num) /
332  socket_->max_data());
333  socket_->set_funct_flags(socket_->funct_flags() | FUNCT_HIGH_SEQ);
334  socket_->set_high_seq(socket_->snd_max());
335  socket_->set_snd_cwnd(socket_->snd_prev_cwnd() + ack_num -
336  socket_->snd_max());
337 
338  // [VF011]
339  //
340  // cwnd can get negative and thus a big unsigned number.
341  //
342 
343  socket_->set_snd_cwnd(MAX(socket_->snd_cwnd(), 0));
344  socket_->set_snd_prev_cwnd(MIN(socket_->snd_prev_cwnd(),
345  (ntohs(tcp_hdr->th_win) <<
346  socket_->snd_scale())));
347  socket_->set_snd_ssthresh(MAX(socket_->max_data() << 1,
348  (socket_->snd_prev_cwnd() >> 1)));
349 
350  // Round down prevcwnd to a multiple of maxdata. This allows
351  // VJ congestion control algorithm to perform a bit better.
352  if (socket_->snd_ssthresh() ==
353  (((socket_->snd_ssthresh()) / socket_->max_data())) *
354  socket_->max_data())
355  {
356  socket_->set_snd_prev_cwnd(
357  (((int)((socket_->snd_ssthresh()) / socket_->max_data())) *
358  socket_->max_data()));
359  }
360  else
361  {
362  socket_->snd_prev_cwnd() =
363  (((int)((socket_->snd_ssthresh()) / socket_->max_data())) *
364  socket_->max_data()) + socket_->max_data();
365  }
366  socket_->set_snd_ssthresh(MIN(socket_->snd_ssthresh(),
367  socket_->snd_prev_cwnd()));
368 
369  LogD(kClassName, __func__, "Entering FR, high_seq(%lu) snduna(%lu) "
370  "cwnd(%lu) ssthresh(%lu), prevcwnd(%lu)\n",
371  socket_->high_seq() - socket_->initial_seq_num(),
372  socket_->snd_una() - socket_->initial_seq_num(),
373  socket_->snd_cwnd(), socket_->snd_ssthresh(),
374  socket_->snd_prev_cwnd());
375  }
376 
377  // Force out the retransmission by making it a hole and calling
378  // TpProcessor::Send().
379 
380  if (send_buf_snd_una)
381  {
382  send_buf_snd_una->rexmit_time = Time::Now();
383  }
384 
385  // XXX
386  // socket_->sendBuf->holes =
387  // socket_->sendBuf->addHole(socket_->sendBuf->holes, socket_->sendBuf->snd_una_,
388  // socket_->sendBuf->snd_una_->m_ext.len,
389  // socket_->maxSeqSent, snackDelay);
390 
391  // On the third dupack, force out the retransmission. Make sure that we
392  // have enough sendCwnd credit to send the packet.
393  if ((socket_->t_dupacks() == DUPACK_THRESH) &&
394  (enter_fast_rexmit_from_dup_ack) &&
395  ((socket_->funct_flags() & FUNCT_HIGH_SEQ)))
396  {
397  int oldCwnd = socket_->snd_cwnd();
398 
399  socket_->set_snd_cwnd(socket_->max_data());
400  socket_->Send(NULL, false);
401  socket_->set_snd_cwnd(oldCwnd);
402  }
403  }
404 }

407 void VJCongCtrlAlg::Timeout()
408 {
409  // We had a retransmission timeout, so knock down ssThresh and reenter
410  // slow-start.
411  //
412  // On an RTO set sendSsThresh to the max of half the congestion window
413  // sendPrevCwnd (for VJ) and 2 segments.
414  socket_->set_snd_ssthresh(MAX((socket_->snd_prev_cwnd() >> 1),
415  (socket_->max_data() << 1)));
416  socket_->set_snd_cwnd(0);
417  socket_->set_snd_prev_cwnd(socket_->max_data());
418 }
