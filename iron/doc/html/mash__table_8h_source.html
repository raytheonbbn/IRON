197  bool Insert(const K& k, const V& v)
198  {
199  typename List<V>::LLElem* lle = linked_list_.GetLle(v);
200  if (lle == NULL)
201  {
202  return false;
203  }
204 
205  MTElem* mte = NULL;
206 
207  if (mte_pool_ == NULL)
208  {
209  mte = new (std::nothrow) MTElem(lle, v);
210 
211  if (mte == NULL)
212  {
213  linked_list_.Recycle(lle);
214  return false;
215  }
216  }
217  else
218  {
219  mte = mte_pool_;
220  mte_pool_ = mte->next;
221  mte->llelem = lle;
222  mte->val = v;
223  mte->next = NULL;
224  }
225 
226  if (hash_table_.Insert(k, mte))
227  {
228  linked_list_.Push(lle);
229  return true;
230  }
231  linked_list_.Recycle(lle);
232  mte->next = mte_pool_;
233  mte_pool_ = mte;
234  return false;
235  }
