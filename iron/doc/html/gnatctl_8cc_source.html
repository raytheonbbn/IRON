72 void Usage(const std::string& prog_name)
73 {
74  fprintf(stderr,"\n");
75  fprintf(stderr,"Usage:\n");
76  fprintf(stderr," %s [options]\n", prog_name.c_str());
77  fprintf(stderr,"\n");
78  fprintf(stderr,"Options:\n");
79  fprintf(stderr," -S <IP:port> The source IP address and port of the flow.\n");
80  fprintf(stderr," -D <IP:port> The dest IP address and port of the flow.\n");
81  fprintf(stderr," -A <IP> The IP address of the ingress IRON node.\n");
82  fprintf(stderr," -P <udp/tcp> The type of flow (udp or tcp).\n");
83  fprintf(stderr," -V Service definition to be added.\n");
84  fprintf(stderr," -R remove flow with the specified tuple.\n");
85  fprintf(stderr," -f Flag to indicate if the flow is a file transfer with deadline.\n");
86  fprintf(stderr," -s The size of the file, if it is a file transfer, in bytes.\n");
87  fprintf(stderr," -d The deadline of the filetransfer, if it is a file transfer.\n");
88  fprintf(stderr," -p The priority of the flow. \n");
89  fprintf(stderr," -u The utility function to be assigned to the flow.\n");
90 
91  exit(2);
92 }

97  string source_info = "";

100  bool is_file_transfer = false;

110  int c=0;

112  {
113  switch (c)
114  {
115  case 'S':
116  source_info = optarg;
117  break;
118  case 'D':
119  dest_info = optarg;
120  break;
121  case 'A':
122  amp_addr_str = optarg;
123  break;
124  case 'P':
125  protocol = optarg;
126  break;
127  case 'R':
128  delete_flow = true;
129  break;
130  case 'V':
131  service_defn = optarg;
132  break;
133  case 'U':
134  is_file_transfer_update = true;
135  break;
136  case 'f':
137  is_file_transfer = true;

139  case 's':
140  file_size_bytes = optarg;
141  break;

146  priority = optarg;

151  default:
152  Usage(argv[0]);
153  }
154  }
155 
156  string message = "";
157 
158  if (amp_addr_str == "")
159  {
160  LogE(kClassName, __func__, "AMP IP address is required.\n");
161  Usage(argv[0]);
162  }
163 
164  if (service_defn != "")

166  string search = ";";

168  iron::StringUtils::Replace(service_defn, search, replace);

174  else if (source_info == "")
175  {
176  LogE(kClassName, __func__, "Source info is required.\n");
177  Usage(argv[0]);
178  }
179  else if (dest_info == "")
180  {
181  LogE(kClassName, __func__, "Destination info is required.\n");
182  Usage(argv[0]);
183  }
184  else if (delete_flow)
185  {
186  message = "parameter;del_flow;flow_tuple;" + source_info +
187  " -> " + dest_info;
188  }
189  else if (is_file_transfer)
190  {
191  if (file_size_bytes == "0")
192  {
193  LogE(kClassName, __func__, "File transfers must have a size.\n");
194  return 1;
195  }
196  if (transfer_deadline_sec == "0")
197  {
198  LogE(kClassName, __func__, "File transfers must have a deadline.\n");
199  return 1;
200  }
201  if (priority == "0")
202  {
203  LogE(kClassName, __func__, "File transfers must have a priority.\n");
204  return 1;
205  }
206  message = "parameter;ft_params;flow_tuple;" + source_info +
207  " -> " + dest_info + ";deadline;" +
208  transfer_deadline_sec + ";size;" + file_size_bytes +
209  ";priority;" + priority;
210  }
211  else if (is_file_transfer_update)
212  {
213  message = "parameter;ft_params;flow_tuple;" + source_info +
214  " -> " + dest_info + ";deadline;" +
215  transfer_deadline_sec + ";size;" + file_size_bytes +
216  ";priority;" + priority;
217  }
218  else if (utility_type != "")
219  {
220  if (protocol == "")
221  {
222  LogE(kClassName, __func__,
223  "Utility function configuration requires the protocol type.\n");
224  exit(2);
225  }
226  message = "parameter;utility_fn;flow_tuple;" + source_info +
227  " -> " + dest_info + ";utility;" + utility_type;
228  if (priority != "0")
229  {
230  message = message + ";priority;" + priority;
231  }
232  }
233  else if (priority != "0")
234  {
235  if (protocol == "")
236  {
237  LogE(kClassName, __func__,
238  "Utility function configuration requires the protocol type.\n");
239  exit(2);
240  }
241  message = message = "parameter;priority;flow_tuple;" + source_info +
242  " -> " + dest_info + ";priority;" + priority + ";protocol;" + protocol;
243  }
244  else
245  {
246  LogE(kClassName, __func__, "Unsupported configuration request.\n");
247  Usage(argv[0]);
248  }
249 
250  // Connect to the AMP.
251  iron::RemoteControlClient rc_client_;
252  struct sockaddr_in amp_addr;
253  ::memset(&amp_addr, 0, sizeof(amp_addr));

256  amp_addr.sin_port = htons(kDefaultAmpCtlPort);
257  uint32_t amp_ep = 0;
258 
259  string target = "amp";
260  if (protocol == "udp")
261  {
262  target = "udp_proxy";
263  }
264  else if (protocol == "tcp")
265  {
266  target = "tcp_proxy";
267  }
268 
269  while (amp_ep ==0)
270  {
271  LogD(kClassName, __func__, "Connecting to AMP\n");
272  amp_ep = rc_client_.Connect(amp_addr);
273  if (amp_ep != 0)
274  {
275  LogD(kClassName, __func__, "Connected to AMP\n");
276  break;
277  }
278  sleep(2);
279  }
280 
281  rc_client_.SendSetMessage(amp_ep, target, message);
282 
283  exit(0);
284 }
