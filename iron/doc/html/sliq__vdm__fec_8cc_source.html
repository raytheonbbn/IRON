89 #define P_KMAX MAX_FEC_RATE

91 #define MM 16 // code over GF(2**MM)
92 #define NN ((1 << MM) - 1) // powers of @
93 
94 // This defines the type used to store an element of the Galois Field
95 // used by the code.
96 typedef uint16_t gf; // Galois Field 65536
97 
98 // To speed up computations, we have tables for logarithm, exponent
99 // and inverse of a number.
100 // There is also a macro for multiplications.
101 #define A0 (NN) // log(0) is not defined, use a special value
102 
103 static bool init_flag = false;
104 static gf gf_exp [NN + 1]; // index->polynomial form conversion table
105 static int gf_log [NN + 1]; // Polynomial->index form conversion table
106 static gf inverse[NN + 1]; // inverse of a number inv[@^i] = @^[NN-i-1]
107

109 //============================================================================
110 // Compute x % NN, where NN is 2**MM - 1, without a slow divide. Many calls
111 // (about 1/8) are done with a small x < 2 NN.
112 //
113 // Note: modnn() does the following, only much faster:
114 // #define modnn(x) ((x) % 0xffff)
115 static inline gf modnn(int x)
116 {
117  while (x >= NN)
118  {
119  x -= NN;
120  x = (x >> MM) + (x & NN);
121  }
122  return x;
123 }
124 
125 //============================================================================
126 // gf_mul(x,y) multiplies two numbers.
127 static inline gf gf_mul(gf x, gf y)
128 {
129  if (((x) == 0) || ((y) == 0))
130  {
131  return 0;
132  }
133  return gf_exp[modnn(gf_log[x] + gf_log[y])] ;
134 }

136 //============================================================================
137 // Generate GF(2**m) from the irreducible polynomial p(X) in p[0]..p[m].
138 // Lookup tables:
139 // index->polynomial form gf_exp[] contains j=alpha**i;
140 // polynomial form -> index form gf_log[j=alpha**i] = i
141 // alpha=2 is the primitive element of GF(2**m)
142 void VdmFec::Initialize()
143 {
144  // Primitive polynomials - see Lin & Costello, Appendix A,
145  // and Lee & Messerschmitt, p. 453.
146  char primPoly[] =
147  {
148  "11010000000010001" // MM=16: 1+x+x^3+x^12+x^16
149  };
150  int i = 0;
151  int mask = 1;
152 
153  if (init_flag)
154  {
155  return;
156  }
157 
158  init_flag = true;
159  gf_exp[MM] = 0; // Will be updated at the end of the 1st loop.
160 
161  for (i = 0; i < MM; i++)
162  {
163  gf_exp[i] = mask;
164  gf_log[gf_exp[i]] = i;
165 
166  // If primPoly[i] == 1 then, term @^i occurs in poly-repr of @^MM.
167  if (primPoly[i] == '1') // primPoly[i] != 0
168  {
169  gf_exp[MM] ^= mask; // Bit-wise XOR operation.
170  }
171 
172  mask <<= 1; // Single left-shift.
173  }
174 
175  gf_log[gf_exp[MM]] = MM;
176 
177  // Have obtained poly-repr of @^MM. Poly-repr of @^(i+1) is given by
178  // poly-repr of @^i shifted left one-bit and accounting for any @^MM term
179  // that may occur when poly-repr of @^i is shifted.
180  mask >>= 1;
181 
182  for (i = (MM + 1); i < NN; i++)
183  {
184  if (gf_exp[i - 1] >= mask)
185  {
186  gf_exp[i] = gf_exp[MM] ^ ((gf_exp[i - 1] ^ mask) << 1);
187  }
188  else
189  {
190  gf_exp[i] = gf_exp[i - 1] << 1;
191  }
192  gf_log[gf_exp[i]] = i;
193  }
194 
195  gf_log[0] = A0;
196  gf_exp[NN] = 0;
197 
198  inverse[0] = NN; // Invalid!
199  inverse[1] = 1;
200 
201  for (i = 2; i <= NN; i++)
202  {
203  inverse[i] = gf_exp[NN-gf_log[i]];
204  }
205 }

208 void VdmFec::EncodePackets(int num_src_pkt, // n
209  uint8_t** src_pkt_data, // pdata
210  uint16_t* src_pkt_size, // szArray
211  int num_enc_pkt, // k
212  uint8_t** enc_pkt_data, // pfec
213  uint16_t* enc_pkt_size) // fecSz
214 {
215  int i = 0;
216  int j = 0;
217  int max_size = 0;
218  int item = 0;
219  gf** data = (gf**)src_pkt_data;
220  gf** fec = (gf**)enc_pkt_data;
221 
222  for (i = 0; i < num_src_pkt; i++)
223  {
224  if (src_pkt_size[i] > max_size)
225  {
226  max_size = src_pkt_size[i];
227  }
228  }
229 
230  // Need to make sure we clear enough of the repair buffer by ensuring the
231  // length is an even number of bytes.
232  if (max_size & 0x1)
233  {
234  max_size++;
235  }
236 
237  for (j = 0; j < num_enc_pkt; j++)
238  {
239  if (fec[j] == NULL)
240  {
241  continue;
242  }
243 
244  gf* fp = fec[j];
245 
246  bzero(fp, max_size * sizeof(uint8_t));
247  enc_pkt_size[j] = 0;
248 
249  for (i = 0; i < num_src_pkt; i++)
250  {
251  int ix = gf_exp[modnn(i * j)]; // This is the encoding matrix.
252  gf* g = data[i];
253  int sz = src_pkt_size[i] >> 1;
254 
255  for (item = 0; item < sz; item++)
256  {
257  fp[item] ^= gf_mul(ix, *g++);
258  }
259 
260  if (src_pkt_size[i] & 0x1)
261  {
262  uint16_t tmp;
263 
264 #if (__BYTE_ORDER == __LITTLE_ENDIAN)
265  tmp = (uint16_t)(*(uint8_t*)g);
266 #else // (__BYTE_ORDER == __BIG_ENDIAN)
267  tmp = (uint16_t)((*(uint8_t*)g) << 8);
268 #endif
269 
270  fp[item] ^= gf_mul(ix, tmp);
271  }
272 
273  enc_pkt_size[j] ^= gf_mul(ix, src_pkt_size[i]);
274  }
275  }
276 }

279 int VdmFec::DecodePackets(int num_src_pkt, // n
280  uint8_t** in_pkt_data, // psrc
281  uint16_t* in_pkt_size, // szArray
282  uint16_t* in_enc_pkt_size, // fecSz
283  int* in_pkt_index, // index
284  uint8_t** out_pkt_data, // pdst
285  uint16_t* out_pkt_size) // recSz
286 {
287  int i = 0;
288  int missing = 0;
289  int max_size = 0;
290  gf** src = (gf**)in_pkt_data;
291  gf** dst = (gf**)out_pkt_data;
292  gf b [P_KMAX][P_KMAX];
293  gf a1[P_KMAX][P_KMAX];
294 
295  for (i = 0; i < num_src_pkt; i++)
296  {
297  if (in_pkt_size[i] > max_size)
298  {
299  max_size = in_pkt_size[i];
300  }
301  }
302 
303  // Need to make sure we clear enough of the repair buffer by ensuring the
304  // length is an even number of bytes.
305  if (max_size & 0x1)
306  {
307  max_size++;
308  }
309 
310 #define SWAP(a,b,t) { t tmp; tmp = a; a = b; b = tmp; }
311  {
312  int i = 0;
313  int v = 0;
314 
315  again:
316 
317  // A word about the following swapping business:
318  //
319  // If it finds a packet index that is less than n, it is an original
320  // packet -- so the orginal packet is moved to its "correct" position
321  // within the pointer array. When we are done, the pointer array will
322  // consist either of packets in the correct position *or* repair
323  // packets. No original packets will be out of place.
324  //
325  // This is done so that the later in the processing, if the index value is
326  // equal to the loop index, it just copies the original packet into the
327  // destination array.
328  for (i = 0; i < num_src_pkt; i++)
329  {
330  v = in_pkt_index[i];
331 
332  if ((v < num_src_pkt) && (v != i))
333  {
334  SWAP(src[i], src[v], gf*);
335  SWAP(in_pkt_index[i], in_pkt_index[v], int);
336  SWAP(in_pkt_size[i], in_pkt_size[v], uint16_t);
337  SWAP(in_enc_pkt_size[i], in_enc_pkt_size[v], uint16_t);
338  goto again;
339  }
340  }
341  }
342 
343  // Build matrix b.
344  {
345  int i = 0;
346 
347  bzero(b, sizeof(b));
348  bzero(a1, sizeof(a1));
349 
350  for (i = 0; i < num_src_pkt; i++)
351  {
352  b[i][i] = 1; // Initialize.
353 
354  if (in_pkt_index[i] < num_src_pkt)
355  {
356  if (in_pkt_index[i] != i)
357  {
358  // fprintf(stderr,"ouch, %d should not be at %d\n",
359  // in_pkt_index[i], i);
360  return -1;
361  }
362  else
363  {
364  a1[i][i] = 1;
365  }
366  }
367  else
368  {
369  int pow = (in_pkt_index[i] - num_src_pkt);
370  int j = 0;
371 
372  for (j = 0; j < num_src_pkt; j++)
373  {
374  a1[i][j] = gf_exp[modnn(j * pow)];
375  }
376  }
377  }
378  }
379 
380  // Invert matrix, using a crude method.
381  {
382  int row = 0;
383 
384  for (row = 0; row < num_src_pkt; row++)
385  {
386  int mul = 0;
387  int r = 0;
388  int col = 0;
389 
390  if (a1[row][row] == 0) // Pivot...
391  {
392  int t = 0;
393 
394  for (t = (row + 1); t < num_src_pkt; t++)
395  {
396  if (a1[row][t] != 0) // Found a good one.
397  {
398  int i = 0;
399 
400  SWAP(src[row], src[t], gf*);
401  SWAP(in_pkt_index[row], in_pkt_index[t], int);
402  SWAP(in_pkt_size[row], in_pkt_size[t], uint16_t);
403  SWAP(in_enc_pkt_size[row], in_enc_pkt_size[t], uint16_t);
404 
405  for (i = 0; i < num_src_pkt; i++)
406  {
407  SWAP(a1[i][row], a1[i][t], int);
408  SWAP(b [i][row], b [i][t], int);
409  }
410  break;
411  }
412  }
413 
414  if (t == num_src_pkt)
415  {
416  // fprintf(stderr,"ouch, diagonal element %d = 0\n", row);
417  return -2;
418  }
419  }
420 
421  if ((mul = inverse[a1[row][row]]) != 1)
422  {
423  for (col = 0; col < num_src_pkt; col++)
424  {
425  b [row][col] = gf_mul(mul, b[row][col]);
426  a1[row][col] = gf_mul(mul, a1[row][col]);
427  }
428  }
429 
430  for (r = 0; r < num_src_pkt; r++)
431  {
432  if ((r == row) || ((mul = a1[r][row]) == 0))
433  {
434  continue;
435  }
436 
437  if (in_pkt_index[row] == row) // Source, only a1[row][row] != 0.
438  {
439  b [r][row] ^= gf_mul(mul, b[row][row]);
440  a1[r][row] ^= gf_mul(mul, a1[row][row]);
441  }
442  else
443  {
444  for (col = 0; col < num_src_pkt; col++)
445  {
446  b [r][col] ^= gf_mul(mul, b[row][col]);
447  a1[r][col] ^= gf_mul(mul, a1[row][col]);
448  }
449  }
450  }
451  }
452  }
453 
454  // Do the actual decoding.
455  {
456  int item = 0;
457  int row = 0;
458  int col = 0;
459 
460  for (row = 0; row < num_src_pkt; row++)
461  {
462  // Per above discussion, if index[row] == row then this is an original
463  // packet. Just copy it to the destination array.
464  if (in_pkt_index[row] == row)
465  {
466  // We have set this up so it does the repair in place hence the
467  // following is commented out:
468  // bcopy(src[row], dst[row], in_pkt_size[row]*sizeof(uint8_t));
469  out_pkt_size[row] = in_pkt_size[row];
470  }
471 
472  // Othwerwise, we do a reconstruction for this position.
473  else
474  {
475  // Set up a pointer to the reconstruction buffer.
476  gf* d = dst[row];
477 
478  // Increment our "number of missing packets" counter.
479  missing++;
480 
481  // Clear the reconstruction buffer.
482  bzero(d, max_size * sizeof(uint8_t));
483  out_pkt_size[row] = 0;
484 
485  // Loop over the available packets to reconstruct the missing packet.
486  for (col = 0; col < num_src_pkt; col++)
487  {
488  gf x = b[row][col];
489  gf* s = src[col];
490  int sz = in_pkt_size[col] >> 1;
491 
492  for (item = 0; item < sz; item++)
493  {
494  d[item] ^= gf_mul(x, *s++);
495  }
496 
497  if (in_pkt_size[col] & 0x1)
498  {
499  uint16_t tmp;
500 
501 #if (__BYTE_ORDER == __LITTLE_ENDIAN)
502  tmp = (uint16_t)(*(uint8_t *)s);
503 #else // (__BYTE_ORDER == __BIG_ENDIAN)
504  tmp = (uint16_t)((*(uint8_t *)s) << 8);
505 #endif
506 
507  d[item] ^= gf_mul(x, tmp);
508  }
509 
510  out_pkt_size[row] ^= gf_mul(x, in_enc_pkt_size[col]);
511  }
512  }
513  }
514  }
515 
516  return 0;
517 }
