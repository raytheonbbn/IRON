84 Packet* ZombieQueue::Dequeue(uint32_t max_size_bytes, DstVec dst_vec)
85 {
86  uint32_t zombie_len = max_size_bytes;
87  if (zombie_len > kMaxZombieLenBytes)
88  {
89  zombie_len = kMaxZombieLenBytes;
90  }
91 
92  DstVec new_dst_vec = 0;
93  if (!is_multicast_)
94  {
95  LogD(kClassName, __func__, "Attempting to dequeue a zombie of size %"
96  PRIu32 "B. Queue length is %" PRIu32 "B.\n",
97  max_size_bytes, queue_size_);
98  if (queue_size_ < kMinZombieLenBytes)
99  {
100  LogW(kClassName, __func__, "Unable to dequeue a zombie because there "
101  "are only %" PRIu32 " Bytes in the queue, and the min zombie size "
102  "is %zu.\n", queue_size_, kMinZombieLenBytes);
103  return NULL;
104  }
105  if (zombie_len >= queue_size_)
106  {
107  // Don't dequeue more than what's left in the queue.
108  zombie_len = queue_size_;
109  }
110  }
111  else // is_multicast
112  {
113  LogD(kClassName, __func__, "Attempting to dequeue a zombie of size %"
114  PRIu32 "B with dst vec 0x%x.\n", max_size_bytes, dst_vec);
115 
116  // Check the lengths for each destination. Find the min non-zero length
117  // so that we can include all non-zero destinations in the packet.
118  uint32_t dst_depth = 0;
119  BinIndex dst_bidx = 0;
120  bool all_zero = true;
121 
122  for (bool valid = bin_map_.GetFirstUcastBinIndex(dst_bidx);
123  valid;
124  valid = bin_map_.GetNextUcastBinIndex(dst_bidx))
125  {
126  if (bin_map_.IsBinInDstVec(dst_vec, dst_bidx))
127  {
128  dst_depth = zombie_counts_.GetBinDepthByIdx(dst_bidx);
129  LogD(kClassName, __func__, "dst %s, depth = %" PRIu32 "\n",
130  bin_map_.GetIdToLog(dst_bidx).c_str(), dst_depth);
131  if (dst_depth > 0)
132  {
133  new_dst_vec = bin_map_.AddBinToDstVec(new_dst_vec, dst_bidx);
134  all_zero = false;
135  if (dst_depth <= zombie_len)
136  {
137  zombie_len = dst_depth;
138  }
139  }
140  }
141  }
142  if (all_zero)
143  {
144  LogD(kClassName, __func__, "No destinations have the minimum zombie "
145  "size available to dequeue.\n");
146  return NULL;
147  }
148  LogD(kClassName, __func__, "Actually dequeuing a zombie of size %"
149  PRIu32 "\n", zombie_len);
150  }
151 
152  if (zombie_len < kMinZombieLenBytes)
153  {
154  zombie_len = kMinZombieLenBytes;
155  }
156 
157  // Encode the local node's BinId in a fake IPv4 source address.
158  BinId node_bin_id = bin_map_.GetPhyBinId(node_bin_index_);
159  in_addr_t src_addr_nbo = htonl((static_cast<in_addr_t>(10 << 24) |
160  node_bin_id));
161 
162  Packet* zombie = Zombie::CreateNewZombie(
163  packet_pool_,
164  src_addr_nbo,
165  dst_addr_nbo_,
166  zombie_len,
167  lat_class_);
168  if (zombie != NULL)
169  {
170  zombie->set_dst_vec(new_dst_vec);
171  if (!is_multicast_)
172  {
173  if (queue_size_ > zombie->virtual_length())
174  {
175  queue_size_ -= zombie->virtual_length();
176  }
177  else
178  {
179  queue_size_ = 0;
180  }
181  }
182  else // is_multicast
183  {
184  // Decrease the counts for appropriate destinations.
185  BinIndex dst_bidx = 0;
186  // Multicast bins can't be in the destination bit vector.
187  for (bool valid = bin_map_.GetFirstUcastBinIndex(dst_bidx);
188  valid;
189  valid = bin_map_.GetNextUcastBinIndex(dst_bidx))
190  {
191  if (bin_map_.IsBinInDstVec(new_dst_vec, dst_bidx))
192  {
193  zombie_counts_.Decrement(dst_bidx, zombie_len);
194  queue_size_ -= zombie_len;
195  }
196  }
197  }
198  }
199  return zombie;
200 }

203 bool ZombieQueue::Enqueue(Packet* pkt)
204 {
205  if (pkt == NULL)
206  {
207  LogF(kClassName, __func__,
208  "Attempting to enqueue a NULL zombie packet.\n");
209  return false;
210  }
211  if (pkt->GetLatencyClass() != lat_class_)
212  {
213  LogE(kClassName, __func__, "Attempting to enqueue packet with latency "
214  "class %s into ZombieQueue of latency class %s\n",
215  LatencyClass_Name[pkt->GetLatencyClass()].c_str(),
216  LatencyClass_Name[lat_class_].c_str());
217  }
218 
219  uint32_t UNUSED(previous) = queue_size_;
220  size_t pkt_size = pkt->virtual_length();
221  if (!is_multicast_)
222  {
223  if (UINT32_MAX - pkt_size < queue_size_)
224  {
225  LogE(kClassName, __func__,
226  "Attempting to enqueue too many zombies. Have %" PRIu32
227  "B, trying to add %zu.\n", queue_size_, pkt_size);
228  queue_size_ = UINT32_MAX;
229  }
230  else
231  {
232  queue_size_ += pkt_size;
233  }
234  LogD(kClassName, __func__, "Enqueued a zombie of size %"
235  PRIu32 "B. Queue length changed from %" PRIu32 " to %" PRIu32 "B.\n",
236  pkt->virtual_length(), previous, queue_size_);
237  }
238  else // is_multicast
239  {
240  DstVec dst_vec = pkt->dst_vec();
241 
242  BinIndex dst_bidx = 0;
243  // All multicast bins should be empty.
244  for (bool valid = bin_map_.GetFirstUcastBinIndex(dst_bidx);
245  valid;
246  valid = bin_map_.GetNextUcastBinIndex(dst_bidx))
247  {
248  if (bin_map_.IsBinInDstVec(dst_vec, dst_bidx))
249  {
250  // Increment checks internally for overflow.
251  zombie_counts_.Increment(dst_bidx, pkt_size);
252  queue_size_ += pkt_size;
253  }
254  }
255  }
256  packet_pool_.Recycle(pkt);
257  pkt = NULL;
258  return true;
259 }

302 uint32_t ZombieQueue::DropPacket(uint32_t max_size_bytes, DstVec dst_vec)
303 {
304  uint32_t dropped_bytes = max_size_bytes;
305  if (!is_multicast_)
306  {
307  LogD(kClassName, __func__, "Attempting to drop %" PRIu32
308  "B from a zombie queue. Queue length is %" PRIu32 "B.\n",
309  max_size_bytes, queue_size_);
310  if (queue_size_ == 0)
311  {
312  return 0;
313  }
314  if (queue_size_ < dropped_bytes)
315  {
316  dropped_bytes = queue_size_;
317  }
318  // Now decrement the overall queue size.
319  queue_size_ -= dropped_bytes;
320  }
321  else // is_multicast
322  {
323  LogD(kClassName, __func__, "Attempting to dequeue a zombie of size %"
324  PRIu32 "B.\n", max_size_bytes);
325 
326  // If there's only one destination in the destination bit vector, we can
327  // make sure the zombie we dequeue won't leave a dribble of less than the
328  // minimum zombie size in the queue. If there are multiple destinations,
329  // that's much harder (and still TBD how to do it).
330 
331  // Also make sure we aren't dropping more than what exists for
332  // any of the specified destinations.
333  bool one_dst = (BinMap::GetNumBinsInDstVec(dst_vec) == 1);
334  bool all_zero = true;
335  BinIndex dst_bidx = 0;
336  for (bool valid = bin_map_.GetFirstUcastBinIndex(dst_bidx);
337  valid;
338  valid = bin_map_.GetNextUcastBinIndex(dst_bidx))
339  {
340  if (bin_map_.IsBinInDstVec(dst_vec, dst_bidx))
341  {
342  uint32_t avail = zombie_counts_.GetBinDepthByIdx(dst_bidx);
343  if (avail > 0)
344  {
345  all_zero = false;
346  if (avail <= dropped_bytes)
347  {
348  dropped_bytes = avail;
349  }
350  else if (one_dst &&
351  (avail - dropped_bytes < kMinZombieLenBytes))
352  {
353  LogW(kClassName, __func__,
354  "Increasing zombie size so we don't leave a dribble.\n");
355  // If there's only one destination, and the remaining zombies in
356  // the queue after this dequeue would be too small to legally
357  // drain later, we should just clear out the queue now.
358  dropped_bytes = avail;
359  }
360  }
361  else
362  {
363  dst_vec = bin_map_.RemoveBinFromDstVec(dst_vec, dst_bidx);
364  }
365  }
366  }
367  if (all_zero)
368  {
369  return 0;
370  }
371  LogD(kClassName, __func__, "Actually dequeuing a zombie of size %"
372  PRIu32 "\n", dropped_bytes);
373  // Now we know how many bytes to drop from each bin. Go ahead and drop
374  // them.
375  for (bool valid = bin_map_.GetFirstUcastBinIndex(dst_bidx);
376  valid;
377  valid = bin_map_.GetNextUcastBinIndex(dst_bidx))
378  {
379  if (bin_map_.IsBinInDstVec(dst_vec, dst_bidx))
380  {
381  zombie_counts_.Decrement(dst_bidx, dropped_bytes);
382  queue_size_ -= dropped_bytes;
383  }
384  }
385  }
386 
387  return dropped_bytes;
388 }

422 size_t ZombieQueue::GetNextDequeueSize()
423 {
424  if (is_multicast_)
425  {
426  LogF(kClassName, __func__, "GetNextDequeueSize MUST take a "
427  "BinIndex when called on a multicast queue.\n");
428  return 0;
429  }
430  if (queue_size_ <= 0)
431  {
432  // If the queue size is 0, we cannot dequeue anything.
433  return 0;
434  }
435 
436  size_t dequeue_size = queue_size_;
437 
438  if (queue_size_ > kMaxZombieLenBytes)
439  {
440  // Cap the dequeue to maximum.
441  dequeue_size = kMaxZombieLenBytes;
442  }
443 
444  size_t min_size = sizeof(struct iphdr);
445 
446  if (dequeue_size < min_size)
447  {
448  // If the queue size is less than the header, we will dequeue a header size
449  // worth.
450  return min_size;
451  }
452  else
453  {
454  return dequeue_size;
455  }
456 }
