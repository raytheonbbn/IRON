113 bool SendBuffer::Enqueue(PktInfo* pkt_info)
114 {
115  LogD(kClassName, __func__, "%s, enqueuing packet into send buffer: "
116  "seq (%" PRIu32 ") data len (%" PRIu32 ").\n",
117  socket_->flow_id_str(), pkt_info->seq_num, pkt_info->data_len);
118 
119  if (pkt_info == NULL)
120  {
121  LogW(kClassName, __func__, "%s, Invalid argument: pkt_info is NULL.\n",
122  socket_->flow_id_str());
123  return false;
124  }
125 
126  // There is always room to enqueue a zero-length packet. If non-zero length,
127  // max sure it will fit. Also we need to bypass this step on SYN packets
128  // since nxt_seq_ may not yet be initialized.
129  if (pkt_info->data_len > 0)
130  {
131  if (SEQ_GT((pkt_info->seq_num + pkt_info->data_len), win_hwm_))
132  {
133  return false;
134  }
135  }
136 
137  if (tail_)
138  {
139  const struct tcphdr* pkt_tcp_hdr = pkt_info->pkt->GetTcpHdr();
140  if ((tail_->flags & TH_FIN) && (pkt_tcp_hdr->th_flags & TH_FIN))
141  {
142  LogW(kClassName, __func__, "%s, Unable to enqueue packet, TH_FIN set "
143  "for buffer tail packet.\n", socket_->flow_id_str());
144  return false;
145  }
146  }
147 
148  if ((snd_una_ != NULL) && (tail_ == NULL))
149  {
150  // There is a head but no tail. Something isn't right.
151  LogW(kClassName, __func__, "%s, Something is wrong. Send packet buffer "
152  "has a head but no tail.\n", socket_->flow_id_str());
153  return false;
154  }
155  else if ((snd_una_ == NULL) && (tail_ != NULL))
156  {
157  // There is a tail but no head. Something isn't right.
158  LogW(kClassName, __func__, "%s, Something is wrong. Send packet buffer "
159  "has a tail but no head.\n", socket_->flow_id_str());
160  return false;
161  }
162  else if (tail_ == NULL)
163  {
164  // There isn't anything in the send buffer. This is the first packet in
165  // the send buffer.
166  pkt_info->prev = NULL;
167  pkt_info->next = NULL;
168  snd_una_ = pkt_info;
169  snd_nxt_ = pkt_info;
170  tail_ = pkt_info;
171  una_seq_ = snd_una_->seq_num;
172  nxt_seq_ = tail_->seq_num + tail_->data_len;
173  if (!una_seq_initialized_)
174  {
175  win_hwm_ = una_seq_ + max_size_bytes_;
176  una_seq_initialized_ = true;
177  }
178  }
179  else
180  {
181  // Append the packet to the tail of the send buffer.
182  tail_->next = pkt_info;
183  pkt_info->prev = tail_;
184  pkt_info->next = NULL;
185  tail_ = pkt_info;
186 
187  if (SEQ_LT(nxt_seq_,tail_->seq_num + tail_->data_len))
188  {
189  nxt_seq_ = tail_->seq_num + tail_->data_len;
190  }
191  }
192 
193  if (!snd_nxt_)
194  {
195  snd_nxt_ = tail_;
196  }
197 
198  socket_->Send(NULL, false);
199 
200  return true;
201 }

204 bool SendBuffer::EnqueuePackets(PktInfo* pkt_info_list)
205 {
206  // For now, we will simply unlink each element to be added from the provided
207  // list and invoke the Enqueue() method. This will take care of adjusting
208  // all of the necessary class member variables. A future optimization might
209  // include investigating the appending of this list as a single operation.
210  PktInfo* pkt_to_enqueue = pkt_info_list;
211  while (pkt_to_enqueue)
212  {
213  PktInfo* next_pkt_info = pkt_to_enqueue->next;
214 
215  pkt_to_enqueue->next = NULL;
216  if (next_pkt_info)
217  {
218  next_pkt_info->prev = NULL;
219  }
220 
221  if (!Enqueue(pkt_to_enqueue))
222  {
223  // This should never fail as we should have figured out how many packets
224  // can be moved prior to moving them. If this fails something is wrong,
225  // so abort.
226  LogF(kClassName, __func__, "%s, Error enqueuing packets.\n",
227  socket_->flow_id_str());
228  return false;
229  }
230 
231  pkt_to_enqueue = next_pkt_info;
232  }
233 
234  socket_->Send(NULL, false);
235 
236  return true;
237 }

240 PktInfo* SendBuffer::GetNextTransmission(Time& now, uint32_t sock_uwe,
241  ProxyIfType cfg_if_id)
242 {
243  // This method will first search for a packet to be retransmitted, a packet
244  // whose rexmit time has expired. If no such packet exists, the packet at
245  // snd_nxt_ is returned.
246  if (rexmit_head_ &&
247  (rexmit_head_->rexmit_time < now + socket_->min_burst_usec()))
248  // The following additional check further limits the number of
249  // retransmissions. However, there is the potential that a packet gets
250  // stuck in a CAT that never recovers and we will never retransmit the
251  // packet, essentially deadlocking the flow.
252  // &&
253  // (rexmit_head_->pkt->ref_cnt() < 2))
254  {
255  if (adaptive_buffers_)
256  {
257  xmit_bytes_since_last_size_update_ +=
258  rexmit_head_->pkt->GetLengthInBytes();
259  }
260 
261  return rexmit_head_;
262  }
263  else
264  {
265  if (adaptive_buffers_ && snd_nxt_)
266  {
267  xmit_bytes_since_last_size_update_ += snd_nxt_->pkt->GetLengthInBytes();
268  }
269 
270  if ((cfg_if_id == WAN) &&
271  ((snd_nxt_ == NULL) ||
272  (SEQ_GT(snd_nxt_->seq_num + snd_nxt_->data_len, sock_uwe) &&
273  snd_nxt_->data_len != 0)))
274  {
275  // snd_nxt_ is window blocked or NULL.
276  if ((rexmit_head_ != NULL) &&
277  (rexmit_head_->pkt->ref_cnt() < 2))
278  {
279  // The proxy is the only component that has a reference to the
280  // packet.
281  return rexmit_head_;
282  }
283  else
284  {
285  return NULL;
286  }
287  }
288  else
289  {
290  return snd_nxt_;
291  }
292  }

311 void SendBuffer::ProcessPlugs(OutSeqBuffer::PlugInfo* plugs,
312  uint32_t num_plugs, bool& buf_changed)
313 {
314  if ((plugs == NULL) || (num_plugs == 0))
315  {
316  // Invalid parameter provided.
317  return;
318  }
319 
320  if (snd_una_ == NULL)
321  {
322  // The send buffer is empty, do nothing.
323  LogD(kClassName, __func__, "%s, Cannot record any plugs, send buffer is "
324  "empty.\n", socket_->flow_id_str());
325  return;
326  }
327 
328  bool done = false;
329  Time now = Time::Now();
330  PktInfo* cur_pkt_info = snd_una_;
331  uint32_t i = 0;
332  while ((i < num_plugs) && !done)
333  {
334  // Mark all packets in the buffer whose sequence numbers are less than
335  // the sequence number of the current plug as holes, if necessary.
336  while (cur_pkt_info && SEQ_LT(cur_pkt_info->seq_num, plugs[i].lower_seq))
337  {
338  if (cur_pkt_info->rexmit_time.IsInfinite())
339  {
340  // The current packet does not have a retransmission time set yet, so
341  // we mark it as a hole.
342  MarkHole(cur_pkt_info, now);
343 
344  buf_changed = true;
345  }
346 
347  cur_pkt_info = cur_pkt_info->next;
348  }
349 
350  PktInfo* pkt_info_at_plug_start = cur_pkt_info;
351 
352  // A data length of 0 in the following loop is meant to cover SYN and
353  // FIN packets.
354  bool snd_nxt_plugged = false;
355  bool snd_una_plugged = false;
356  bool found_plugs = false;
357  while (cur_pkt_info &&
358  SEQ_LT(cur_pkt_info->seq_num, plugs[i].upper_seq))
359  {
360  found_plugs = true;
361 
362  if (cur_pkt_info == snd_una_)
363  {
364  LogD(kClassName, __func__, "%s, snd_una_ is plugged. This should not "
365  "happen.\n", socket_->flow_id_str());
366 
367  snd_una_plugged = true;
368  }
369  if (cur_pkt_info == snd_nxt_)
370  {
371  snd_nxt_plugged = true;
372  }
373 
374  cur_pkt_info = cur_pkt_info->next;
375  }
376 
377  if (found_plugs)
378  {
379  // If snd_una_ or snd_nxt_ has been plugged, move them.
380  if (snd_una_plugged)
381  {
382  snd_una_ = cur_pkt_info;
383  }
384  if (snd_nxt_plugged)
385  {
386  snd_nxt_ = cur_pkt_info;
387  }
388 
389  // At this point, we have a list of packets that have been plugged. We can
390  // now remove this list of packets from the send buffer and recycle them.
391  if (pkt_info_at_plug_start->prev)
392  {
393  pkt_info_at_plug_start->prev->next = cur_pkt_info;
394  }
395 
396  if (cur_pkt_info != NULL)
397  {
398  // There are packets in the send buffer following the plug.
399  cur_pkt_info->prev->next = NULL;
400  cur_pkt_info->prev = pkt_info_at_plug_start->prev;
401  }
402  else
403  {
404  // The plug is at the end of the send buffer. If and when this becomes
405  // the case, we can return as we know there are no additional packets
406  // that can be plugged.
407  tail_ = pkt_info_at_plug_start->prev;
408  done = true;
409  }
410  pkt_info_at_plug_start->prev = NULL;
411 
412  ReleasePkts(pkt_info_at_plug_start);
413 
414  buf_changed = true;
415  }
416 
417  ++i;
418  }
419 
420  if (snd_una_ != NULL)
421  {
422  una_seq_ = snd_una_->seq_num;
423  }
424  else
425  {
426  una_seq_ = nxt_seq_;
427  }
428 }

431 void SendBuffer::MoveToHeadOfRexmitList(PktInfo* pkt_info)
432 {
433  if (rexmit_head_ == pkt_info)
434  {
435  // Packet is already at the head of the retransmission list.
436  return;
437  }
438 
439  // If the packet to be moved is in the retransmission list, unlink it.
440  if (pkt_info->rexmit_prev)
441  {
442  pkt_info->rexmit_prev->rexmit_next = pkt_info->rexmit_next;
443  }
444 
445  if (pkt_info->rexmit_next)
446  {
447  pkt_info->rexmit_next->rexmit_prev = pkt_info->rexmit_prev;
448  }
449 
450  if (rexmit_tail_ == pkt_info)
451  {
452  rexmit_tail_ = pkt_info->rexmit_prev;
453  }
454 
455  // Add packet to the head of the retransmission list.
456  if (rexmit_head_ == NULL)
457  {
458  rexmit_head_ = pkt_info;
459  rexmit_tail_ = pkt_info;
460  pkt_info->rexmit_next = NULL;
461  pkt_info->rexmit_prev = NULL;
462  }
463  else
464  {
465  rexmit_head_->rexmit_prev = pkt_info;
466  pkt_info->rexmit_prev = NULL;
467  pkt_info->rexmit_next = rexmit_head_;
468  rexmit_head_ = pkt_info;
469  }
470 }

472 //============================================================================
473 void SendBuffer::MoveToEndOfRexmitList(PktInfo* pkt_info)
474 {
475  if (pkt_info->rexmit_next == NULL)
476  {
477  // Only 1 packet in retransmission list or the packet is already at the
478  // end of the retransmission list.
479  return;
480  }
481 
482  // Remove from head. The packet to be moved will always be at the head of
483  // the retransmission list because we only move to the end of the
484  // retransmission list when we reset the rexmit_time of a packet. We only do
485  // this when we actually do a retransmission of a packet. Retransmissions
486  // are ALWAYS pulled from the head of the retransmission list.
487  rexmit_head_ = pkt_info->rexmit_next;
488  pkt_info->rexmit_next = NULL;
489  rexmit_head_->rexmit_prev = NULL;
490 
491  // Append to tail.
492  rexmit_tail_->rexmit_next = pkt_info;
493  pkt_info->rexmit_prev = rexmit_tail_;
494  pkt_info->rexmit_next = NULL;
495  rexmit_tail_ = pkt_info;
496 
497  // The retransmission time of the new tail of the retransmission list is
498  // based on the current RTT calculation and may be scheduled to occur before
499  // elements in the list that are to be transmitted prior to the tail. This
500  // can occur when there is a decrease in the RTT. We will adjust the time of
501  // the retransmission list elements ahead of the new tail whose
502  // retransmission time is greater than the retransmission time of the new
503  // tail. The retransmission time of the elements for which this is the case
504  // will be set to the retransmission time of the new tail. This should
505  // increase the efficiency of repairing "holes".
506  PktInfo* cur_pkt_info = rexmit_tail_->rexmit_prev;
507  while ((cur_pkt_info != NULL) &&
508  (cur_pkt_info->rexmit_time > rexmit_tail_->rexmit_time))
509  {
510  LogD(kClassName, __func__, "%s, resetting retransmission time (%s) to "
511  "new tail retransmission time (%s).\n", socket_->flow_id_str(),
512  cur_pkt_info->rexmit_time.ToString().c_str(),
513  rexmit_tail_->rexmit_time.ToString().c_str());
514 
515  cur_pkt_info->rexmit_time = rexmit_tail_->rexmit_time;
516  cur_pkt_info = cur_pkt_info->rexmit_prev;
517  }
518 }

540 void SendBuffer::Trim(uint32_t seq_num)
541 {
542  LogD(kClassName, __func__, "%s, Trimming packets from send buffer to seq (%"
543  PRIu32 ").\n", socket_->flow_id_str(), seq_num);
544 
545  // Trim the packets from the send buffer.
546  if (snd_una_ && SEQ_GT(seq_num, snd_una_->seq_num))
547  {
548  bool snd_nxt_trimmed = false;
549  PktInfo* cur_pkt_info = snd_una_;
550  PktInfo* pkts_to_trim = snd_una_;
551 
552  while (cur_pkt_info &&
553  SEQ_LT(cur_pkt_info->seq_num, seq_num))
554  {
555  if (cur_pkt_info == snd_nxt_)
556  {
557  snd_nxt_trimmed = true;
558  }
559  una_seq_ = cur_pkt_info->seq_num + cur_pkt_info->data_len;
560 
561  cur_pkt_info = cur_pkt_info->next;
562  }
563 
564  snd_una_ = cur_pkt_info;
565  if (snd_una_ == NULL)
566  {
567  // The new snd_una_ is NULL, which means there are no more packets in
568  // the send buffer. So, set tail_ to NULL also.
569  tail_ = NULL;
570  }
571  else
572  {
573  if (snd_una_->prev != NULL)
574  {
575  snd_una_->prev->next = NULL;
576  snd_una_->prev = NULL;
577  }
578  }
579 
580  if (snd_nxt_trimmed)
581  {
582  snd_nxt_ = cur_pkt_info;
583  }
584 
585  ReleasePkts(pkts_to_trim);
586  }
587 
588  if (snd_una_ != NULL)
589  {
590  una_seq_ = snd_una_->seq_num;
591  }
592  else
593  {
594  una_seq_ = nxt_seq_;
595  }
596 }

599 bool SendBuffer::RexmitSanityCheck()
600 {
601  if ((snd_una_ == NULL) && ((rexmit_head_ != NULL) || (rexmit_tail_ != NULL)))
602  {
603  LogF(kClassName, __func__, "%s, bug found in rexmit list: "
604  "NULL snd_una, non-NULL rexmit_head or rexmit_tail\n",
605  socket_->flow_id_str());
606  }
607 
608  if ((rexmit_head_ == NULL) && (rexmit_tail_ != NULL))
609  {
610  LogF(kClassName, __func__, "%s, head, tail mismatch\n",
611  socket_->flow_id_str());
612  }
613 
614  if ((rexmit_head_ != NULL) && (rexmit_tail_ == NULL))
615  {
616  LogF(kClassName, __func__, "%s, non-NULL rexmit head, NULL rexmit tail\n",
617  socket_->flow_id_str());
618  }
619 
620  if (snd_una_ == NULL)
621  {
622  return true;
623  }
624 
625  uint32_t seq_num = snd_una_->seq_num;
626  PktInfo* pkt_info = rexmit_head_;
627 
628  while (pkt_info != NULL)
629  {
630  if (SEQ_GT(seq_num, pkt_info->seq_num))
631  {
632  LogF(kClassName, __func__, "%s, bug found in rexmit list: ACK seq is %",
633  PRIu32 "; packet sequence in rexmit list is %" PRIu32 "\n",
634  socket_->flow_id_str(), seq_num, pkt_info->seq_num);
635  }
636  pkt_info = pkt_info->rexmit_next;
637  }
638 
639  return true;
640 }

655 void SendBuffer::UpdateBufferSize(uint32_t rtt_us, double send_rate_bps,
656  uint32_t queue_depth)
657 {
658  if (!adaptive_buffers_)
659  {
660  return;
661  }
662 
663  Time now = Time::Now();
664  if ((now - last_size_update_time_) > Time::FromUsec(rtt_us))
665  {
666  if (queue_depth > 0)
667  {
668  size_t new_size_bytes =
669  static_cast<size_t>(2.0 * rtt_us * send_rate_bps / 8000000.0);
670 
671  if (new_size_bytes > 2 * max_size_bytes_)
672  {
673  max_size_bytes_ *= 2;
674  }
675  else
676  {
677  max_size_bytes_ = new_size_bytes;
678  }
679 
680  if (max_size_bytes_ < adaptive_buffer_min_size_)
681  {
682  LogD(kClassName, __func__, "%s, desired max_size_bytes_ %"
683  PRIu32 ", limited to %" PRIu32 " bytes.\n",
684  socket_->flow_id_str(), max_size_bytes_,

761 void SendBuffer::MarkHole(PktInfo* pkt_info, Time& now, bool force)
762 {
763  if (pkt_info == NULL)
764  {
765  return;
766  }
767 
768  bool add_to_rexmit_list = pkt_info->rexmit_time.IsInfinite();
769 
770  if (!pkt_info->rexmit_time.IsInfinite() && !force)
771  {
772  LogW(kClassName, __func__, "%s, packet with sequence number %" PRIu32
773  " has already been marked as a hole.\n", socket_->flow_id_str(),
774  pkt_info->seq_num);
775  return;
776  }
777 
778  if (socket_->t_srtt())
779  {
780  pkt_info->rexmit_time = now +
781  Time::FromUsec((MIN(socket_->max_rto_us(),
782  ((MAX(0, socket_->t_srtt()) >> TCP_RTT_SHIFT) +
783  socket_->t_rttvar()))));
784  }
785  else
786  {
787  pkt_info->rexmit_time = now +
788  Time::FromUsec(MIN(socket_->max_rto_us(),
789  MAX(0, socket_->initial_rto())));
790  }
791 
792  if (add_to_rexmit_list)
793  {
794  // Add packet to the end of the retransmission list.
795  if (rexmit_head_ == NULL)
796  {
797  rexmit_head_ = pkt_info;
798  rexmit_tail_ = pkt_info;
799  pkt_info->rexmit_next = NULL;
800  pkt_info->rexmit_prev = NULL;
801  }
802  else
803  {
804  rexmit_tail_->rexmit_next = pkt_info;
805  pkt_info->rexmit_prev = rexmit_tail_;
806  pkt_info->rexmit_next = NULL;
807  rexmit_tail_ = pkt_info;
808  }
809  }

813 void SendBuffer::ReleasePkts(PktInfo* pkt_info)
814 {
815  while (pkt_info)
816  {
817  PktInfo* next_pkt_info = pkt_info->next;
818 
819  // Remove the packet from the retransmission list.
820  if (pkt_info->rexmit_prev)
821  {
822  pkt_info->rexmit_prev->rexmit_next = pkt_info->rexmit_next;
823  }
824 
825  if (pkt_info->rexmit_next)
826  {
827  pkt_info->rexmit_next->rexmit_prev = pkt_info->rexmit_prev;
828  }
829 
830  if (rexmit_head_ == pkt_info)
831  {
832  rexmit_head_ = pkt_info->rexmit_next;
833  }
834 
835  if (rexmit_tail_ == pkt_info)
836  {
837  rexmit_tail_ = pkt_info->rexmit_prev;
838  }
839 
840  // Track the number of bytes acked.
841  if (socket_->cfg_if_id() == WAN)
842  {
843  cum_acked_bytes_ += pkt_info->data_len;
844  }
845 
846  // Now we can recycle the packet.
847  pkt_info_pool_.Recycle(pkt_info);
848 
849  pkt_info = next_pkt_info;
850  }
