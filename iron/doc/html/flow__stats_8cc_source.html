79 bool FlowFilter::Configure(const string& filter_spec)
80 {
81  // Following is the format of the flow statistics filter specification:
82  //
83  // field1=value1,field2=value2,...
84  //
85  // Valid field/value pairs are:
86  //
87  // saddr=<source address>
88  // sport=<source port>
89  // daddr=<destination address>
90  // dport=<destination port>
91  // proto=<protocol number>
92  // dscp=<dscp value>
93 
94  LogD(kClassName, __func__, "Received filter specification: %s\n",
95  filter_spec.c_str());
96 
97  List<string> filter_spec_tokens;
98  StringUtils::Tokenize(filter_spec, ",", filter_spec_tokens);
99 
100  if (filter_spec_tokens.size() == 0)
101  {
102  LogE(kClassName, __func__, "Invalid flow statistics filter spec "
103  "received.\n");
104  return false;
105  }
106 
107  while (filter_spec_tokens.size() > 0)
108  {
109  string filter_spec_token;
110  if (!filter_spec_tokens.Pop(filter_spec_token))
111  {
112  LogE(kClassName, __func__, "Error getting filter spec token.\n");
113  return false;
114  }
115 
116  List<string> key_value_pair;
117  StringUtils::Tokenize(filter_spec_token, "=", key_value_pair);
118 
119  string filter_key;
120  if (!key_value_pair.Pop(filter_key))
121  {
122  LogE(kClassName, __func__, "Error getting filter spec token key.\n");
123  return false;
124  }
125 
126  string filter_value;
127  if (!key_value_pair.Pop(filter_value))
128  {
129  LogE(kClassName, __func__, "Error getting filter spec token value.\n");
130  return false;
131  }
132 
133  if (filter_key == "saddr")
134  {
135  LogD(kClassName, __func__, "Source address is part of the filter.\n");
136 
137  Ipv4Address addr = StringUtils::GetIpAddr(filter_value);
138 
139  set_saddr(addr.address());
140  }
141  else if (filter_key == "sport")
142  {
143  LogD(kClassName, __func__, "Source port is part of the filter.\n");
144 
145  set_sport(htons(static_cast<uint16_t>(StringUtils::GetUint(filter_value))));
146  }
147  else if (filter_key == "daddr")
148  {
149  LogD(kClassName, __func__, "Destination address is part of the "
150  "filter.\n");
151 
152  Ipv4Address addr = StringUtils::GetIpAddr(filter_value);
153 
154  set_daddr(addr.address());
155  }
156  else if (filter_key == "dport")
157  {
158  LogD(kClassName, __func__, "Destination port is part of the filter.\n");
159 
160  set_dport(htons(static_cast<uint16_t>(StringUtils::GetUint(filter_value))));
161  }
162  else if (filter_key == "proto")
163  {
164  LogD(kClassName, __func__, "Protocol is part of the filter.\n");
165 
166  set_proto(static_cast<uint8_t>(StringUtils::GetUint(filter_value)));
167  }
168  else if (filter_key == "dscp")
169  {
170  LogD(kClassName, __func__, "DSCP is part of the filter.\n");
171 
172  set_dscp(static_cast<uint8_t>(StringUtils::GetUint(filter_value)));
173  }
174  else
175  {
176  LogW(kClassName, __func__, "Received improperly formatted flow "
177  "statistics filter spec.: %s\n", filter_spec.c_str());
178 
179  // Since we have received an improperly formatted filter spec., ensure
180  // that the filter is "empty".
181  saddr_set_ = false;
182  sport_set_ = false;
183  daddr_set_ = false;
184  dport_set_ = false;
185  proto_set_ = false;
186  dscp_set_ = false;
187 
188  return false;
189  }
190  }
191 
192  return true;
193 }

196 bool FlowFilter::Matches(const Packet* packet) const
197 {
198  bool get_five_tuple = (saddr_set_ || sport_set_ || daddr_set_ ||
199  dport_set_ || proto_set_);
200 
201  if (get_five_tuple || dscp_set_)
202  {
203  if (get_five_tuple)
204  {
205  uint32_t saddr;
206  uint32_t daddr;
207  uint16_t sport;
208  uint16_t dport;
209  uint32_t proto;
210 
211  if (!packet->GetFiveTuple(saddr, daddr, sport, dport, proto))
212  {
213  return false;
214  }
215 
216  if (saddr_set_ && (saddr_ != saddr))
217  {
218  return false;
219  }
220 
221  if (sport_set_ && (sport_ != sport))
222  {
223  return false;
224  }
225 
226  if (daddr_set_ && (daddr_ != daddr))
227  {
228  return false;
229  }
230 
231  if (dport_set_ && (dport_ != dport))
232  {
233  return false;
234  }
235 
236  if (proto_set_ && (proto_ != proto))
237  {
238  return false;
239  }
240  }
241 
242  if (dscp_set_)
243  {
244  uint8_t dscp;
245 
246  if ((!packet->GetIpDscp(dscp)) || (dscp_ != dscp))
247  {
248  return false;
249  }
250  }
251 
252  return true;
253  }
254 
255  return false;
256 }
