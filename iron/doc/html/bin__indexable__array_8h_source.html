91  virtual bool Initialize(BinMap& bin_map)
92  {
93  if (init_flag_)
94  {
95  LogE("BinIndexableArray", __func__, "Already initialized.\n");
96  return false;
97  }
98 
99  idx_offset_0_ = bin_map.ucast_bin_idx_offset();
100  size_0_ = bin_map.max_num_ucast_bin_idxs();
101  idx_offset_1_ = bin_map.int_bin_idx_offset();
102  size_1_ = bin_map.max_num_int_bin_idxs();
103  idx_offset_2_ = bin_map.mcast_bin_idx_offset();
104  size_2_ = bin_map.max_num_mcast_bin_idxs();
105 
106  if (((size_0_ + size_1_ + size_2_) < 1) ||
107  (idx_offset_1_ < (idx_offset_0_ + size_0_)) ||
108  (idx_offset_2_ < (idx_offset_1_ + size_1_)))
109  {
110  LogE("BinIndexableArray", __func__, "Invalid ranges: %" PRIBinIndex
111  " - %zu, %" PRIBinIndex " - %zu, %" PRIBinIndex " - %zu\n",
112  idx_offset_0_, (idx_offset_0_ + size_0_ - 1), idx_offset_1_,
113  (idx_offset_1_ + size_1_ - 1), idx_offset_2_,
114  (idx_offset_2_ + size_2_ - 1));
115  return false;
116  }
117 
118  if (!AllocateArray())
119  {
120  LogE("BinIndexableArray", __func__, "Array allocation error, size "
121  "%zu elements.\n", (size_0_ + size_1_ + size_2_));
122  return false;
123  }
124 
125  init_flag_ = true;
126 
127  return true;
128  }

172  virtual C& operator [](BinIndex index)
173  {
174  if ((init_flag_) && (array_ != NULL))
175  {
176  if ((index >= idx_offset_0_) && (index < (idx_offset_0_ + size_0_)))
177  {
178  return array_[index - idx_offset_0_];
179  }
180  else if ((index >= idx_offset_1_) &&
181  (index < (idx_offset_1_ + size_1_)))
182  {
183  return array_[size_0_ + index - idx_offset_1_];
184  }
185  else if ((index >= idx_offset_2_) &&
186  (index < (idx_offset_2_ + size_2_)))
187  {
188  return array_[size_0_ + size_1_ + index - idx_offset_2_];
189  }
190  else
191  {
192  LogW("BinIndexableArray", __func__, "Index (%" PRIBinIndex
193  ") is out of bounds. Valid index ranges: %" PRIBinIndex
194  " - %zu, %" PRIBinIndex " - %zu, %" PRIBinIndex " - %zu\n",
195  index, idx_offset_0_, (idx_offset_0_ + size_0_ - 1),
196  idx_offset_1_, (idx_offset_1_ + size_1_ - 1), idx_offset_2_,
197  (idx_offset_2_ + size_2_ - 1));
198  }
199  }
200  else
201  {
202  LogE("BinIndexableArray", __func__, "Not initialized yet.\n");
203  }
204 
205  return null_elem_;
206  }
