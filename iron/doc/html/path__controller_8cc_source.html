66 bool PathController::NeedsMetadataHeaders(Packet* pkt)
67 {
68  if (pkt == NULL)
69  {
70  return false;
71  }
72 
73  // Note: The latency header is needed only if there is an origin timestamp
74  // to send. This is because the SLIQ data header captures the TTG
75  // information from the Packet object.
76  return ((pkt->origin_ts_ms() != kUnsetOriginTs) ||
77  pkt->send_packet_history() || pkt->send_packet_id() ||
78  pkt->send_packet_dst_vec());
79 }

82 bool PathController::AddMetadataHeaders(Packet* pkt)
83 {
84  if (pkt == NULL)
85  {
86  return false;
87  }
88 
89  // Clear any existing metadata headers in the packet.
90  size_t mdh_len = 0;
91 
92  if (!pkt->SetMetadataHeaderLengthInBytes(mdh_len))
93  {
94  return false;
95  }
96 
97  // Add a latency header only if there is an origin timestamp to send.
98  if (pkt->origin_ts_ms() != kUnsetOriginTs)
99  {
100  PktLatencyHeader lat_hdr;
101 
102  // Only store the origin timestamp.
103  lat_hdr.type = CAT_PKT_LATENCY_HEADER;
104  lat_hdr.flags = 0;
105  lat_hdr.origin_ts = htons(pkt->origin_ts_ms());
106  lat_hdr.time_to_go = 0;
107 
108  mdh_len += kPktLatHdrSize;
109 
110  if (!pkt->SetMetadataHeaderLengthInBytes(mdh_len))
111  {
112  return false;
113  }
114 
115  ::memcpy(reinterpret_cast<void*>(pkt->GetMetadataHeaderBuffer()),
116  &lat_hdr, kPktLatHdrSize);
117 
118  LogD(kClassName, __func__, "Path controller %" PRIu32 " added latency "
119  "header: origin_ts %" PRIu16 "\n", path_controller_number_,
120  pkt->origin_ts_ms());
121  }
122 
123  // Add the CAT packet history header if needed.
124  if (pkt->send_packet_history())
125  {
126  PktHistoryHeader hst_hdr;
127 
128  hst_hdr.type = CAT_PKT_HISTORY_HEADER;
129  ::memcpy(reinterpret_cast<void*>(hst_hdr.history), pkt->history(),
130  sizeof(hst_hdr.history));
131 
132  mdh_len += kPktHistHdrSize;
133 
134  if (!pkt->SetMetadataHeaderLengthInBytes(mdh_len))
135  {
136  return false;
137  }
138 
139  ::memcpy(reinterpret_cast<void*>(pkt->GetMetadataHeaderBuffer()),
140  &hst_hdr, kPktHistHdrSize);
141 
142  LogD(kClassName, __func__, "Path controller %" PRIu32 " added history "
143  "header: %" PRIu8 " %" PRIu8 " %" PRIu8 " %" PRIu8 " %" PRIu8 " %"
144  PRIu8 " %" PRIu8 " %" PRIu8 " %" PRIu8 " %" PRIu8 " %" PRIu8 "\n",
145  path_controller_number_, hst_hdr.history[0], hst_hdr.history[1],
146  hst_hdr.history[2], hst_hdr.history[3], hst_hdr.history[4],
147  hst_hdr.history[5], hst_hdr.history[6], hst_hdr.history[7],
148  hst_hdr.history[8], hst_hdr.history[9], hst_hdr.history[10]);
149  }
150 
151  // Add the CAT packet ID header if needed.
152  if (pkt->send_packet_id())
153  {
154  PktIdHeader id_hdr;
155 
156  id_hdr.type_bin_id_pkt_id = htonl(
157  (static_cast<uint32_t>(CAT_PKT_ID_HEADER) << 24) |
158  ((static_cast<uint32_t>(pkt->bin_id()) & 0x0f) << 20) |
159  (pkt->packet_id() & 0x000fffff));
160 
161  mdh_len += kPktIdHdrSize;
162 
163  if (!pkt->SetMetadataHeaderLengthInBytes(mdh_len))
164  {
165  return false;
166  }
167 
168  ::memcpy(reinterpret_cast<void*>(pkt->GetMetadataHeaderBuffer()), &id_hdr,
169  kPktIdHdrSize);
170 
171  LogD(kClassName, __func__, "Path controller %" PRIu32 " added packet ID "
172  "header: bin_id %" PRIBinId " pkt_id %" PRIu32 "\n",
173  path_controller_number_, pkt->bin_id(), pkt->packet_id());
174  }
175 
176  // Add the CAT packet destination vector header if needed.
177  if (pkt->send_packet_dst_vec())
178  {
179  PktDstVecHeader dst_hdr;
180 
181  dst_hdr.type_dst_vec = htonl(
182  (static_cast<uint32_t>(CAT_PKT_DST_VEC_HEADER) << 24) |
183  (pkt->dst_vec() & 0x00ffffff));
184 
185  mdh_len += kPktDstVecHdrSize;
186 
187  if (!pkt->SetMetadataHeaderLengthInBytes(mdh_len))
188  {
189  return false;
190  }
191 
192  ::memcpy(reinterpret_cast<void*>(pkt->GetMetadataHeaderBuffer()),
193  &dst_hdr, kPktDstVecHdrSize);
194 
195  LogD(kClassName, __func__, "Path controller %" PRIu32 " added "
196  "destination vector header: dst vec %" PRIu32 "\n",
197  path_controller_number_, pkt->dst_vec());
198  }
199 
200  return true;
201 }

204 bool PathController::ProcessMetadataHeaders(Packet* pkt)
205 {
206  if (pkt == NULL)
207  {
208  return false;
209  }
210 
211  // Loop over the packet headers, processing and removing each of the Packet
212  // object metadata headers that are understood by the path controller.
213  bool stop = false;
214 
215  while ((!stop) && (pkt->GetLengthInBytes() > 0))
216  {
217  // Get the next packet header's type.
218  int hdr_type = pkt->GetRawType();
219 
220  // Process the Packet object metadata headers.
221  switch (hdr_type)
222  {
223  case CAT_PKT_DST_VEC_HEADER:
224  {
225  if (pkt->GetLengthInBytes() < kPktDstVecHdrSize)
226  {
227  return false;
228  }
229 
230  PktDstVecHeader dst_hdr;
231 
232  ::memcpy(&dst_hdr, reinterpret_cast<void*>(pkt->GetBuffer()),
233  kPktDstVecHdrSize);
234 
235  if (!pkt->RemoveBytesFromBeginning(kPktDstVecHdrSize))
236  {
237  return false;
238  }
239 
240  uint32_t dst_mask = (ntohl(dst_hdr.type_dst_vec) & 0x00ffffff);
241 
242  pkt->set_dst_vec(dst_mask);
243  pkt->set_send_packet_dst_vec(true);
244 
245  LogD(kClassName, __func__, "Path controller %" PRIu32 " processed "
246  "destination vector header: dst vec %" PRIu32 "\n",
247  path_controller_number_, pkt->dst_vec());
248 
249  break;
250  }
251 
252  case CAT_PKT_ID_HEADER:
253  {
254  if (pkt->GetLengthInBytes() < kPktIdHdrSize)
255  {
256  return false;
257  }
258 
259  PktIdHeader id_hdr;
260 
261  ::memcpy(&id_hdr, reinterpret_cast<void*>(pkt->GetBuffer()),
262  kPktIdHdrSize);
263 
264  if (!pkt->RemoveBytesFromBeginning(kPktIdHdrSize))
265  {
266  return false;
267  }
268 
269  uint32_t hdr_hbo = ntohl(id_hdr.type_bin_id_pkt_id);
270  uint8_t bin_id = ((hdr_hbo >> 20) & 0x0f);
271  uint32_t pkt_id = (hdr_hbo & 0x000fffff);
272 
273  pkt->set_bin_id(bin_id);
274  pkt->set_packet_id(pkt_id);
275  pkt->set_send_packet_id(true);
276 
277  LogD(kClassName, __func__, "Path controller %" PRIu32 " processed "
278  "packet ID header: bin_id %" PRIBinId " pkt_id %" PRIu32 "\n",
279  path_controller_number_, pkt->bin_id(), pkt->packet_id());
280 
281  break;
282  }
283 
284  case CAT_PKT_HISTORY_HEADER:
285  {
286  if (pkt->GetLengthInBytes() < kPktHistHdrSize)
287  {
288  return false;
289  }
290 
291  PktHistoryHeader hst_hdr;
292 
293  ::memcpy(&hst_hdr, reinterpret_cast<void*>(pkt->GetBuffer()),
294  kPktHistHdrSize);
295 
296  if (!pkt->RemoveBytesFromBeginning(kPktHistHdrSize))
297  {
298  return false;
299  }
300 
301  pkt->set_history(hst_hdr.history);
302  pkt->set_send_packet_history(true);
303 
304  LogD(kClassName, __func__, "Path controller %" PRIu32 " processed "
305  "history header: %" PRIu8 " %" PRIu8 " %" PRIu8 " %" PRIu8 " %"
306  PRIu8 " %" PRIu8 " %" PRIu8 " %" PRIu8 " %" PRIu8 " %" PRIu8 " %"
307  PRIu8 "\n", path_controller_number_, hst_hdr.history[0],
308  hst_hdr.history[1], hst_hdr.history[2], hst_hdr.history[3],
309  hst_hdr.history[4], hst_hdr.history[5], hst_hdr.history[6],
310  hst_hdr.history[7], hst_hdr.history[8], hst_hdr.history[9],
311  hst_hdr.history[10]);
312 
313  break;
314  }
315 
316  case CAT_PKT_LATENCY_HEADER:
317  {
318  if (pkt->GetLengthInBytes() < kPktLatHdrSize)
319  {
320  return false;
321  }
322 
323  PktLatencyHeader lat_hdr;
324 
325  ::memcpy(&lat_hdr, reinterpret_cast<void*>(pkt->GetBuffer()),
326  kPktLatHdrSize);
327 
328  if (!pkt->RemoveBytesFromBeginning(kPktLatHdrSize))
329  {
330  return false;
331  }
332 
333  // Only extract the origin timestamp.
334  uint16_t origin_ts = ntohs(lat_hdr.origin_ts);
335 
336  pkt->set_origin_ts_ms(origin_ts);
337 
338  LogD(kClassName, __func__, "Path controller %" PRIu32 " processed "
339  "latency header: origin_ts %" PRIu16 "\n",
340  path_controller_number_, pkt->origin_ts_ms());
341 
342  break;
343  }
344 
345  default:
346  stop = true;
347  }
348  }
349 
350  return true;
351 }
