150  size_t Hash() const
151  {
152  // Compute a 16-bit sum, similar to an IP header checksum but without
153  // the unnecessary operations, to obtain an index into a hash table.
154  uint32_t sum = ((src_addr_nbo_ >> 16) + (src_addr_nbo_ & 0xffff) +
155  (dst_addr_nbo_ >> 16) + (dst_addr_nbo_ & 0xffff) +
156  (src_dst_ports_nbo_ >> 16) +
157  (src_dst_ports_nbo_ & 0xffff));
158 
159  return static_cast<size_t>((sum >> 16) + (sum & 0xffff));
160  }
