160 void QueueDepthDynamics::ProcessBytesAdded(
161  uint16_t bytes, LatencyClass lat, uint32_t new_depth)
162 {
163  CheckReset();
164  if (std::numeric_limits<int32_t>::max() - bytes <
165  net_bytes_[current_idx_net_])
166  {
167  // Just leave it at the max rate to avoid overflow.
168  net_bytes_[current_idx_net_] = std::numeric_limits<int32_t>::max();
169  }
170  else
171  {
172  net_bytes_[current_idx_net_] += bytes;
173  }
174  // This can happen if we haven't had a dequeue since the last reset, so
175  // min_bytes is still at its initial maximum value.
176  if (new_depth < min_bytes_[current_idx_min_])
177  {
178  min_bytes_[current_idx_min_] = new_depth;
179  }
180 }

200 void QueueDepthDynamics::BytesRemoved(
201  uint16_t bytes, LatencyClass lat, uint32_t new_depth)
202 {
203  CheckReset();
204  if (std::numeric_limits<int32_t>::min() + bytes >
205  net_bytes_[current_idx_net_])
206  {
207  // Just leave it at the min rate to avoid overflow.
208  net_bytes_[current_idx_net_] = std::numeric_limits<int32_t>::min();
209  }
210  else
211  {
212  net_bytes_[current_idx_net_] -= bytes;
213  }
214  if (new_depth < min_bytes_[current_idx_min_])
215  {
216  min_bytes_[current_idx_min_] = new_depth;
217  }
218 }

221 int32_t QueueDepthDynamics::GetChangeRateBytesPerSec()
222 {
223  if (initializing_net_)
224  {
225  // Still starting up. Return the max allowed, since we're likely quickly
226  // building up a queue during system start up.
227  return std::numeric_limits<int32_t>::max();
228  }
229  // Compute the change rate, which is the net bytes over all segments (all
230  // except current is already cached in net_sum_) divided by the time over
231  // which these values are valid.
232  Time time_diff = Time::Now();
233  // The next index in the buffer is currently the oldest.
234  Time oldest = last_reset_net_[NEXT_QDD_INDEX(current_idx_net_)];
235  if (time_diff <= oldest)
236  {
237  // This is highly unlikely, if not impossible, since we will be in
238  // "intializing_net_" period for a full kChangeRateResetPeriod. However,
239  // this extra check removes any possibility of a divide by 0 error.
240  return std::numeric_limits<int32_t>::max();
241  }
242  time_diff = time_diff.Subtract(oldest);
243  double rate = static_cast<double>(net_sum_ + net_bytes_[current_idx_net_])
244  / time_diff.ToDouble();
245  // Truncating is fine. We're not being precise enough to care about a
246  // difference of less than one byte.
247  return static_cast<int32_t>(rate);
248 }

251 uint32_t QueueDepthDynamics::GetMinQueueDepthBytes()
252 {
253  // NOTE: we don't need an initialization period for min bytes the way we do
254  // for change rate, because a minimum makes sense (and isn't skewed) from
255  // being a minimum over a short period of time.
256 
257  // overall_min_ caches the minimum across all segments except the current
258  // one.
259  uint32_t min = overall_min_;
260  if (min_bytes_[current_idx_min_] < overall_min_)
261  {
262  min = min_bytes_[current_idx_min_];
263  }
264 
265  if (min < total_zombies_added_ + zombie_bytes_added_[current_idx_min_])
266  {
267  // This could be possible if the minimum queue depth decreased AND we
268  // added zombies during this period.
269  min = 0;
270  }
271  else
272  {
273  min -= total_zombies_added_;
274  min -= zombie_bytes_added_[current_idx_min_];
275  }
276  return min;
277 }

280 void QueueDepthDynamics::IncrementMinBytesResetPeriod()
281 {
282  if (!dynamic_min_depths_window_ ||
283  (Time::Now() - last_changed_min_bytes_period_ < kIncrMinBytesFrequency))
284  {
285  return;
286  }
287  if (min_bytes_reset_period_ + kMinBytesResetPeriodDelta >=
288  min_bytes_reset_period_upper_bound_)
289  {
290  min_bytes_reset_period_ = min_bytes_reset_period_upper_bound_;
291  }
292  else
293  {
294  min_bytes_reset_period_ += kMinBytesResetPeriodDelta;
295  }
296  min_bytes_rotate_period_ = min_bytes_reset_period_.Multiply(
297  (double)1/kNumQDDSegments);
298  last_changed_min_bytes_period_.GetNow();
299 }

302 void QueueDepthDynamics::DecrementMinBytesResetPeriod()
303 {
304  if (!dynamic_min_depths_window_)
305  {
306  return;
307  }
308  Time now = Time::Now();
309  if ((now - last_changed_min_bytes_period_ < kDecrMinBytesFrequency)
310  || (now - zombie_bytes_last_added_ < kDecrMinBytesTimeSinceZombieSent))
311  {
312  return;
313  }
314  if (min_bytes_reset_period_ - kMinBytesResetPeriodDelta <=
315  min_bytes_reset_period_lower_bound_)
316  {
317  min_bytes_reset_period_ = min_bytes_reset_period_lower_bound_;
318  }
319  else
320  {
321  min_bytes_reset_period_ = min_bytes_reset_period_ -
322  kMinBytesResetPeriodDelta;
323  }
324  min_bytes_rotate_period_ = min_bytes_reset_period_.Multiply(
325  (double)1/kNumQDDSegments);
326  last_changed_min_bytes_period_.GetNow();
327 }

330 void QueueDepthDynamics::CheckReset()
331 {
332  Time now = Time::Now();
333  // Reset and move the circular buffer along once every 1/kNumQDDSegments
334  // seconds.
335  if (now - last_reset_net_[current_idx_net_] > kChangeRateRotatePeriod)
336  {
337  if (current_idx_net_ == kNumQDDSegments - 1)
338  {
339  // We've now filled up an entire buffer. We have sufficient data to
340  // start returning it.
341  initializing_net_ = false;
342  }
343  uint8_t next_idx = NEXT_QDD_INDEX(current_idx_net_);
344  // Update the cached sum, since we'll have a new current segment to be
345  // excluded.
346  net_sum_ += (net_bytes_[current_idx_net_] - net_bytes_[next_idx]);
347  current_idx_net_ = next_idx;
348  net_bytes_[current_idx_net_] = 0;
349  last_reset_net_[current_idx_net_] = now;
350  }
351 
352  if (now - last_reset_min_ > min_bytes_rotate_period_)
353  {
354  uint8_t next_idx = NEXT_QDD_INDEX(current_idx_min_);
355  // Update the cached sum, since we'll have a new current segment to be
356  // excluded.
357  total_zombies_added_ +=
358  (zombie_bytes_added_[current_idx_min_] - zombie_bytes_added_[next_idx]);
359  current_idx_min_ = next_idx;
360  min_bytes_[current_idx_min_] = std::numeric_limits<uint32_t>::max();
361  zombie_bytes_added_[current_idx_min_] = 0;
362 
363  // Update the cached minimum, since we'll have a new current segment to be
364  // excluded.
365  overall_min_ = std::numeric_limits<uint32_t>::max();
366  for (uint8_t idx = NEXT_QDD_INDEX(current_idx_min_);
367  idx != current_idx_min_;
368  idx = NEXT_QDD_INDEX(idx))
369  {
370  if (min_bytes_[idx] < overall_min_)
371  {
372  overall_min_ = min_bytes_[idx];
373  }
374  }
375  last_reset_min_ = now;
376  }
