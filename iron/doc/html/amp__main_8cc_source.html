165 int main(int argc, char** argv)
166 {
167  extern char* optarg;
168  int c;
169  ConfigInfo config_info;
170  bool debug = false;
171  string cmd_file = "";
172 
173  while ((c = getopt(argc, argv, "c:f:l:dh")) != -1)
174  {
175  switch (c)
176  {
177  case 'c':
178  if (!config_info.LoadFromFile(optarg))
179  {
180  LogE(kClassName, __func__, "Error loading configuration file %s.\n",
181  optarg);
182  Usage(argv[0]);
183  exit(1);
184  }
185  break;
186  case 'f':
187  cmd_file = optarg;
188  break;
189 
190  case 'l':
191  if(!Log::SetOutputFile(optarg, false))
192  {
193  LogF(kClassName, __func__, "Unable to set log file %s,\n", optarg);
194  exit(1);
195  }
196  break;
197 
198  case 'd':
199  debug = true;
200  break;
201 
202  case 'h':
203  default:
204  Usage(argv[0]);
205  }
206  }
207 
208  //
209  // Set the signal handlers for this process right from the beginning.
210  //
211  SetSignalHandler();
212 
213  //
214  // Set logging options based on properties.
215  //
216  if (debug)
217  {
218  Log::SetDefaultLevel("FEWIAD");
219  }
220  else
221  {
222  Log::SetDefaultLevel(config_info.Get("Log.DefaultLevel", "All", false));
223  }
224 
225  // Set class level logging.
226  std::string class_levels = config_info.Get("Log.ClassLevels", "", false);
227  List<string> tokens;
228  StringUtils::Tokenize(class_levels, ";", tokens);
229  List<string>::WalkState token_ws;
230  token_ws.PrepareForWalk();
231 
232  string token;
233  while (tokens.GetNextItem(token_ws, token))
234  {
235  if (token.find("=") == string::npos)
236  {
237  continue;
238  }
239 
240  List<string> token_values;
241  StringUtils::Tokenize(token, "=", token_values);
242 
243  string token_name;
244  string token_value;
245 
246  token_values.Pop(token_name);
247  token_values.Peek(token_value);
248 
249  LogI(kClassName, __func__,
250  "Setting class %s logging to %s.\n",
251  token_name.c_str(), token_value.c_str());
252  Log::SetClassLevel(token_name, token_value);
253  }
254 
255  timer = new (std::nothrow) Timer();
256  if (!timer)
257  {
258  LogF(kClassName, __func__,
259  "Failed to allocate timer.\n");
260  exit(0);
261  }
262 
263  if (cmd_file != "")
264  {
265  amp = new (std::nothrow) Amp(*timer, cmd_file);
266  }
267  else
268  {
269  amp = new (std::nothrow) Amp(*timer);
270  }
271 
272  if (!amp)
273  {
274  LogF(kClassName, __func__, "Unable to allocate memory for AMP.\n");
275  }
276 
277  //
278  // initialize it, and
279  //
280 
281  if (!amp->Initialize(config_info))
282  {
283  LogF(kClassName, __func__, "Error initializing Amp. Aborting...\n");
284  exit(1);
285  }
286 
287  //
288  // start it.
289  //
290 
291  amp->Start();
292 
293  CleanUp();
294 
295  exit(0);
296 }
