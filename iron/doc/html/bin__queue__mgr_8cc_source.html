193 bool BinQueueMgr::Initialize(const ConfigInfo& config_info,
194  BinIndex node_bin_idx)
195 {
196  if (!bin_map_.BinIndexIsAssigned(my_bin_index_))
197  {
198  LogF(kClassName, __func__,
199  "My bin id %s is invalid.\n",
200  bin_map_.GetIdToLog(my_bin_index_).c_str());
201  return false;
202  }
203 
204  // Store the node's bin index for use in generating Zombie packets.
205  node_bin_idx_ = node_bin_idx;
206 
207  // Set the queue depth computation for proxy admission.
208  max_dst_admission_ = config_info.GetBool("Bpf.Alg.Mcast.MaxAdmission",
209  kDefaultMaxDestinationProxyAdmission);
210 
211  // Set the drop policy.
212  string drop_policy_str = config_info.Get(
213  "Bpf.BinQueueMgr.DropPolicy",
214  kDefaultBpfDropPolicy);
215 
216  if (drop_policy_str == "HEAD")
217  {
218  SetDefaultDropPolicy(iron::HEAD);
219  }
220  else if (drop_policy_str == "TAIL")
221  {
222  SetDefaultDropPolicy(iron::TAIL);
223  }
224  else if (drop_policy_str == "NO_DROP")
225  {
226  SetDefaultDropPolicy(iron::NO_DROP);
227  }
228  else
229  {
230  LogE(kClassName, __func__, "Invalid BinQueueMgr.DropPolicy %s.\n",
231  drop_policy_str.c_str());
232  return false;
233  }
234 
235  uint32_t max_bin_depth_pkts =
236  config_info.GetUint("Bpf.BinQueueMgr.MaxBinDepthPkts",
237  kDefaultBpfMaxBinDepthPkts);
238  set_max_bin_depth_pkts(max_bin_depth_pkts);
239 
240  std::string ef_ordering_str = config_info.Get("Bpf.Alg.EFOrdering", "");
241  EFOrdering ef_ordering = kDefaultEFOrdering;
242 
243  if (ef_ordering_str == "DeliveryMargin")
244  {
245  ef_ordering = EF_ORDERING_DELIVERY_MARGIN;
246  }
247  else if (ef_ordering_str == "Ttg")
248  {
249  ef_ordering = EF_ORDERING_TTG;
250  }
251 
252  // Initialize the physical queue for the node's bin index.
253  Ipv4Address dst_addr = bin_map_.GetViableDestAddr(my_bin_index_);
254 
255  for (uint8_t lat = 0; lat < NUM_LATENCY_DEF; ++lat)
256  {
257  // Free any existing queues to allow re-Initialization without a memory
258  // leak. This is used in unit tests.
259  Queue* queue = phy_queue_.lat_queues[lat];
260  if (queue)
261  {
262  delete queue;
263  }
264  queue = NULL;
265  if (!IS_PKTLESS_Z_QUEUE[lat])
266  {
267  queue = new (std::nothrow) PacketQueue(
268  packet_pool_, max_bin_depth_pkts_, drop_policy_,
269  ((ef_ordering == EF_ORDERING_DELIVERY_MARGIN) ||
270  (ef_ordering == EF_ORDERING_TTG)) &&
271  (lat == LOW_LATENCY));
272  }
273  else
274  {
275  queue = new (std::nothrow) ZombieQueue(
276  packet_pool_, bin_map_, is_multicast_, static_cast<LatencyClass>(lat),
277  node_bin_idx, dst_addr);
278  }
279 
280  if (!queue)
281  {
282  LogF(kClassName, __func__, "Error allocating Queue object.\n");
283  return false;
284  }
285  phy_queue_.lat_queues[lat] = queue;
286 
287  }
288 
289  // Set up the parameters for anti-starvation
290  use_anti_starvation_zombies_ =
291  config_info.GetBool("Bpf.UseAntiStarvationZombies",
292  kDefaultUseAntiStarvationZombies);
293  if (use_anti_starvation_zombies_)
294  {
295  asap_mgr_ = new (std::nothrow) ASAP(
296  packet_pool_, bin_map_, *this, my_bin_index_, node_bin_idx_);
297  if (!asap_mgr_)
298  {
299  LogF(kClassName, __func__, "Error allocating ASAP manager\n");
300  return false;
301  }
302  if (!asap_mgr_->Initialize(config_info))
303  {
304  LogF(kClassName, __func__, "Initialization error for ASAP\n");
305  return false;
306  }
307  }
308 
309  // Set up the parameters for latency reduction via zombies
310  do_zombie_latency_reduction_ = config_info.GetBool(
311  "Bpf.ZombieLatencyReduction", kZombieLatencyReduction);
312 
313  if (do_zombie_latency_reduction_)
314  {
315  zlr_manager_.Initialize(config_info);
316  }
317 
318  // Set up the byte count storage.
319  for (int i = 0; i < NUM_LATENCY_DEF; ++i)
320  {
321  if (!per_dst_per_lat_class_bytes_[i].Initialize(bin_map_))
322  {
323  LogF(kClassName, __func__, "Unable to initialize byte count array "
324  "%d.\n", i);
325  return false;
326  }
327  per_dst_per_lat_class_bytes_[i].Clear(0);
328  }
329 
330  // Set up the queue depths plotting array.
331  if (!queue_depths_xplot_.Initialize(bin_map_))
332  {
333  LogF(kClassName, __func__, "Unable to initialize queue depths plotting "
334  "array.\n");
335  return false;
336  }
337  queue_depths_xplot_.Clear(NULL);
338 
339 #ifdef XPLOT
340  bool do_qd_xplot =
341  config_info.GetBool("Bpf.GenerateQueueDepthsGraphs",
342  kDefaultGenerateQueueDepthGraphs);
343  if (do_qd_xplot)
344  {
345  if (is_multicast_)
346  {
347  // Set up xplot objects for unicast and multicast destination bin
348  // indexes.
349  DstVec my_dst_vec = bin_map_.GetMcastDst(my_bin_index_);
350  BinIndex bin_idx = 0;
351 
352  for (bool valid = bin_map_.GetFirstDstBinIndex(bin_idx);
353  valid;
354  valid = bin_map_.GetNextDstBinIndex(bin_idx))
355  {
356  if (bin_map_.IsBinInDstVec(my_dst_vec, bin_idx))
357  {
358  SetUpQueueDepthsXplot(bin_idx);
359  }
360  }
361  }
362  SetUpQueueDepthsXplot(my_bin_index_);
363  }
364 #endif // XPLOT
365 
366  // Set up the last dequeue time array.
367  if (!last_dequeue_time_.Initialize(bin_map_))
368  {
369  LogF(kClassName, __func__, "Unable to initialize last dequeue time "
370  "array.\n");
371  return false;
372  }
373  Time zero_time;
374  zero_time.Zero();
375  last_dequeue_time_.Clear(zero_time);
376 
377  // Set up the non-zombie queue depths array
378  if (!non_zombie_queue_depth_bytes_.Initialize(bin_map_))
379  {
380  LogF(kClassName, __func__, "Unable to initialize non-zombie queue depth "
381  "array.\n");
382  return false;
383  }
384  non_zombie_queue_depth_bytes_.Clear(0);
385 
386  LogC(kClassName, __func__, "BinQueueMgr configuration:\n");
387  LogC(kClassName, __func__, "Bpf.Alg.Mcast.MaxAdmission: %s\n",
388  max_dst_admission_ ? "Max" : "Sum");
389  LogC(kClassName, __func__, "Drop Policy: %s\n",
390  drop_policy_str.c_str());
391  LogC(kClassName, __func__, "Bpf.BinQueueMgr.MaxBinDepthPkts: %" PRIu32
392  " packets\n", max_bin_depth_pkts);
393  LogC(kClassName, __func__, "Anti-starvation zombies (ASAP): %s\n",
394  (use_anti_starvation_zombies_ ? "ON" : "OFF"));
395  LogC(kClassName, __func__, "kDefaultZombieCompression: %s\n",
396  (kDefaultZombieCompression ? "ON" : "OFF"));
397  LogC(kClassName, __func__, "Zombie-based latency reduction: %s\n",
398  (do_zombie_latency_reduction_ ? "ON" : "OFF"));
399  LogC(kClassName, __func__, "Bin Id: %s\n",
400  bin_map_.GetIdToLog(my_bin_index_).c_str());
401 
402  initialized_ = true;
403  return true;
404 }

542 bool BinQueueMgr::Enqueue(Packet* pkt)
543 {
544  if (pkt == NULL)
545  {
546  LogE(kClassName, __func__, "Error, adding NULL packet to bin with ID %s.\n",
547  bin_map_.GetIdToLog(my_bin_index_).c_str());
548  return false;
549  }
550 
551  LatencyClass lat = pkt->GetLatencyClass();
552 
553  if (!support_ef_ && (lat == LOW_LATENCY))
554  {
555  pkt->SetIpDscp(NORMAL_LATENCY);
556  lat = NORMAL_LATENCY;
557  }
558 
559  Queue* queue = phy_queue_.lat_queues[lat];
560 
561  if (!queue)
562  {
563  LogF(kClassName, __func__,
564  "Latency %" PRIu8 " queue for bin id %s is NULL. "
565  "Cannot enqueue packet.\n",
566  lat, bin_map_.GetIdToLog(my_bin_index_).c_str());
567  return false;
568  }
569 
570  Time UNUSED(ttg) = pkt->GetTimeToGo();
571  size_t pkt_size = pkt->virtual_length();
572  DstVec dst_vec = pkt->dst_vec();
573 
574  if (is_multicast_ && dst_vec == 0)
575  {
576  LogE(kClassName, __func__,
577  "Attempt to enqueue multicast packet with no destinations\n");
578  packet_pool_.Recycle(pkt);
579  return false;
580  }
581 
582  // Attempt to enqueue the packet.
583  bool rv = queue->Enqueue(pkt);
584 
585  if (rv)
586  {
587  OnEnqueue(pkt_size, lat, dst_vec);
588  if (WouldLogD(kClassName))
589  {
590  LogD(kClassName, __func__,
591  "Enqueued pkt %p w/ deadline %s in latency queue %s for bin id %s"
592  ": lat size %" PRIu32 "PB and total size now %" PRIu32 "B.\n",
593  pkt,
594  ttg.ToString().c_str(),
595  LatencyClass_Name[lat].c_str(),
596  bin_map_.GetIdToLog(my_bin_index_).c_str(),
597  phy_queue_.lat_queues[lat]->GetSize(),
598  queue_depths_.GetBinDepthByIdx(my_bin_index_));
599  }
600  }
601  else
602  {
603  LogD(kClassName, __func__,
604  "Failed to enqueue pkt %p w/ deadline %s in latency queue %s for "
605  "bin id %s: lat size %" PRIu32 "PB and total size now %" PRIu32
606  "B.\n", pkt,
607  ttg.ToString().c_str(),
608  LatencyClass_Name[lat].c_str(),
609  bin_map_.GetIdToLog(my_bin_index_).c_str(),
610  phy_queue_.lat_queues[lat]->GetSize(),
611  queue_depths_.GetBinDepthByIdx(my_bin_index_));
612  }
613  return rv;
614 }

838 //============================================================================
839 Packet* BinQueueMgr::DequeueAtIterator(LatencyClass lat,
840  PacketQueue::QueueWalkState& qws,
841  DstVec send_to)
842 {
843  if (IS_PKTLESS_Z_QUEUE[lat])
844  {
845  LogE(kClassName, __func__,
846  "Attempting to DequeueAtIterator from a Zombie Queue.\n");
847  return NULL;
848  }
849 
850  Queue* queue = FindQueue(lat);
851  Packet* orig_pkt = NULL;
852  Packet* pkt = NULL;
853  bool cloned = false;
854 
855  if (queue)
856  {
857  // Check if we've been passed a send-to list, since send_to = 0 is a code
858  // for "dequeue entire packet."
859  if (send_to)
860  {
861  orig_pkt = static_cast<PacketQueue*>(queue)->PeekAtIterator(qws);
862 
863  if (orig_pkt && (orig_pkt->dst_vec() != send_to))
864  {
865  pkt = packet_pool_.Clone(orig_pkt, true, iron::PACKET_COPY_TIMESTAMP);
866  cloned = true;
867 
868  if (!pkt)
869  {
870  LogF(kClassName, __func__,
871  "Packet %p clone operation failed.\n",
872  orig_pkt);
873  return NULL;
874  }
875 
876  pkt->set_dst_vec(send_to);
877  // Subtract function will LogF if the list we are subtracting is not a
878  // subset of the list we are subtracting from.
879  orig_pkt->set_dst_vec(
880  bin_map_.DstVecSubtract(orig_pkt->dst_vec(), send_to));
881  LogA(kClassName, __func__,
882  "Cloned packet %p->%p and sending to destinations 0x%X, leaving "
883  "0x%X in orig pkt.\n",
884  orig_pkt, pkt, pkt->dst_vec(), orig_pkt->dst_vec());
885  }
886  }
887 
888  // We didn't need to remove a subset of the destinations, so just dequeue
889  // the entire packet.
890  if (!pkt)
891  {
892  pkt = static_cast<PacketQueue*>(queue)->DequeueAtIterator(qws);
893  }
894  }
895 
896  if (pkt)
897  {
898  DequeuedInfo info(pkt, send_to);
899  OnDequeue(info, cloned);
900  }
901  else
902  {
903  LogD(kClassName, __func__,
904  "Found no packet to dequeue for latency %s.\n",
905  LatencyClass_Name[lat].c_str());
906  }
907 
908  return pkt;
909 }

983 //============================================================================
984 void BinQueueMgr::AddNewZombie(
985  uint32_t dst_addr_nbo,
986  uint32_t total_zombie_bytes,
987  LatencyClass zombie_class,
988  DstVec dst_vec)
989 {
990  // Note: we allow adding packetless zombies of less than kMinZombieLenBytes
991  // - the minimum packet size requirement will only be enforced when we
992  // remove zombies.
993  if (IS_PKTLESS_Z_QUEUE[zombie_class])
994  {
995  (static_cast<ZombieQueue*>(
996  phy_queue_.lat_queues[zombie_class]))->AddZombieBytes(
997  total_zombie_bytes, dst_vec);
998  OnEnqueue(total_zombie_bytes, zombie_class, dst_vec);
999  }
1000  else
1001  {
1002  // Encode the local node's BinId in a fake IPv4 source address.
1003  BinId my_bin_id = bin_map_.GetPhyBinId(my_bin_index_);
1004  in_addr_t src_addr_nbo = htonl((static_cast<in_addr_t>(10) << 24) |
1005  static_cast<in_addr_t>(my_bin_id));
1006 
1007  Packet* zombie = NULL;
1008  while (total_zombie_bytes > 0)
1009  {
1010  size_t size = total_zombie_bytes;
1011  if (size > kMaxZombieLenBytes)
1012  {
1013  size = kMaxZombieLenBytes;
1014  }
1015  zombie = iron::Zombie::CreateNewZombie(
1016  packet_pool_, src_addr_nbo, dst_addr_nbo, size,
1017  zombie_class);
1018  if (!zombie)
1019  {
1020  LogF(kClassName, __func__, "Unable to generate zombie to enqueue.\n");
1021  return;
1022  }
1023  if (dst_vec != 0)
1024  {
1025  zombie->set_dst_vec(dst_vec);
1026  }
1027  if (zombie->GetLengthInBytes() > size)
1028  {
1029  // In case we tried to create a zombie smaller than the minimum size
1030  // packet, make sure our accounting is correct after creating the
1031  // packet.
1032  size = zombie->GetLengthInBytes();
1033  }
1034  if (!Enqueue(zombie))
1035  {
1036  LogF(kClassName, __func__, "Unable to enqueue zombie\n");
1037  TRACK_UNEXPECTED_DROP(kClassName, packet_pool_);
1038  packet_pool_.Recycle(zombie);
1039  zombie = NULL;
1040  return;
1041  }
1042  total_zombie_bytes -= size;
1043  }
1044  }
