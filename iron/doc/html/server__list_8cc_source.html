80 bool ServerList::Initialize(const ConfigInfo& config_info,
81  PacketPool& packet_pool, BinMap& bin_map,
82  uint8_t server_list_index)
83 {
84  string config_prefix = "ServerList.";
85  config_prefix.append(
86  StringUtils::ToString(static_cast<int>(server_list_index)));
87 
88  LogC(kClassName, __func__, "%s configuration:\n", config_prefix.c_str());
89 
90  // Extract the client configured server for the server list.
91  string config_name = config_prefix;
92  config_name.append(".ClientConfiguredServer");
93  string ccs_str = config_info.Get(config_name, "");
94 
95  if (ccs_str == "")
96  {
97  LogE(kClassName, __func__, "Required client configured server not "
98  "provided for server list %" PRIu8 " configuration.\n",
99  server_list_index);
100  return false;
101  }
102 
103  client_configured_server_.SetEndpoint(ccs_str);
104 
105  LogC(kClassName, __func__, "%s : %s\n", config_name.c_str(),
106  ccs_str.c_str());
107 
108  // Extract the number of servers for the server list.
109  config_name = config_prefix;
110  config_name.append(".NumAlternateServers");
111 
112  num_alternate_servers_ = static_cast<uint8_t>(
113  config_info.GetUint(config_name, 0));
114 
115  LogC(kClassName, __func__, "%s : %" PRIu8 "\n",
116  config_name.c_str(), num_alternate_servers_);
117 
118  // Extract the server information.
119  for (uint8_t i = 0; i < num_alternate_servers_; i++)
120  {
121  config_name = config_prefix;
122  config_name.append(".AlternateServer.");
123  config_name.append(StringUtils::ToString(static_cast<int>(i)));
124 
125  string server_str = config_info.Get(config_name, "");
126 
127  if (server_str == "")
128  {
129  LogE(kClassName, __func__, "Missing required alternate server "
130  "'address:port' pair for alternate server %" PRIu8 ".\n", i);
131  return false;
132  }
133 
134  alternate_servers_[i].server.SetEndpoint(server_str);
135 
136  // This will LogF if there is no valid bin for this address.
137  alternate_servers_[i].bin_idx = bin_map.GetDstBinIndexFromAddress(
138  alternate_servers_[i].server);
139  alternate_servers_[i].reachable = true;
140 
141  LogC(kClassName, __func__, "%s : %s\n",
142  config_name.c_str(), server_str.c_str());
143  }
144 
145  LogC(kClassName, __func__, "%s configuration complete.\n",
146  config_prefix.c_str());
147 
148  return true;
149 }

152 bool ServerList::GetServer(Ipv4Endpoint& server) const
153 {
154  bool found_server = false;
155  uint8_t chosen_index = 0;
156  uint32_t min_bin_depth = numeric_limits<uint32_t>::max();
157 
158  for (uint8_t i = 0; i < num_alternate_servers_; i++)
159  {
160  if (alternate_servers_[i].reachable)
161  {
162  uint32_t cur_bin_depth =
163  tcp_proxy_.GetBinDepth(alternate_servers_[i].bin_idx);
164 
165  LogD(kClassName, __func__, "Current bin depth is %" PRIu32 ", minimum "
166  "bin depth is %" PRIu32 ".\n", cur_bin_depth, min_bin_depth);
167 
168  if (cur_bin_depth < min_bin_depth)
169  {
170  min_bin_depth = cur_bin_depth;
171  chosen_index = i;
172  found_server = true;
173  }
174  }
175  else
176  {
177  LogD(kClassName, __func__, "Server %s is unreachable.\n",
178  alternate_servers_[i].server.ToString().c_str());
179  }
180  }
181 
182  if (found_server)
183  {
184  server = alternate_servers_[chosen_index].server;
185  return true;
186  }
187 
188  LogD(kClassName, __func__, "Did not find a suitable server.\n");
189 
190  return false;
191 }

194 void ServerList::MarkAsUnreachable(Ipv4Endpoint& server)
195 {
196  for (uint8_t i = 0; i < num_alternate_servers_; i++)
197  {
198  if (server == alternate_servers_[i].server)
199  {
200  alternate_servers_[i].reachable = false;
201  LogD(kClassName, __func__, "Marking server %s as unreachable.\n",
202  alternate_servers_[i].server.ToString().c_str());
203  break;
204  }
205  }
206 }
