199

202 
203  // Cancel the service sockets update timer.
204  if (timer_.IsTimerSet(svc_sockets_timer_))
205  {
206  LogD(kClassName, __func__, "Canceling service sockets timer %" PRIu64
207  ".\n", svc_sockets_timer_.id());

209  }
210 
211  // Delete the seamless server handoff lists.
212  for (uint8_t i = 0; i < num_server_lists_; i++)
213  {
214  delete server_lists_[i];
215  server_lists_[i] = NULL;
216  }
217 
218  // Detach the shared memory segments.
219  weight_qd_shared_memory_.Detach();
220 
221  // Cancel the statistics timer.
222  if (timer_.IsTimerSet(tcp_stats_push_.timer_handle))
223  {
224  LogD(kClassName, __func__, "Canceling timer %" PRIu64 ".\n",
225  tcp_stats_push_.timer_handle.id());
226  timer_.CancelTimer(tcp_stats_push_.timer_handle);
227  }
228 
229  // Clean up the timer callback object pool.
230  CallbackNoArg<TcpProxy>::EmptyPool();
231 }

234 bool TcpProxy::Initialize(const ConfigInfo& config_info)
235 {
236  // Create the edge interface and insert the iptables rules and attach the
237  // Berkeley Packet Filter that will divert packets into the TCP Proxy.
238  if (!edge_if_.Open())
239  {
240  LogF(kClassName, __func__, "Error creating edge interface.\n");
241  return false;
242  }
243 
244  // Inialize the inter-process communications between the TCP Proxy and the
245  // Backpressure Forwarder.
246  if (!bpf_to_tcp_pkt_fifo_.OpenReceiver())
247  {
248  LogW(kClassName, __func__, "Unable to open backpressure forwarder packet "
249  "FIFO.\n");
250  return false;
251  }
252 
253  if (!tcp_to_bpf_pkt_fifo_.OpenSender())
254  {
255  LogD(kClassName, __func__, "Backpressure forwarder packet FIFO not ready "
256  "yet.\n");
257  }
258 
259  // Get the default utility function definition
260  default_utility_def_ = config_info.Get("DefaultUtilityDef", "");
261 
262  // Inialize k
263  double double_k = config_info.GetDouble("KVal", iron::kDefaultK);
264  if (double_k > std::numeric_limits<uint64_t>::max())
265  {
266  LogE(kClassName, __func__, "k val is too large.\n");
267  k_val_.set_k_current(static_cast<uint64_t>(::iron::kDefaultK));
268  }
269  else
270  {
271  k_val_.set_k_current(static_cast<uint64_t>(double_k));
272  }
273 
274  // Initialize the hash tables.
275  if ((!flow_utility_def_cache_.Initialize(kUtilDefHashTableBuckets)) ||
276  (!context_dscp_cache_.Initialize(kContextDscpHashTableBuckets)))
277  {
278  LogF(kClassName, __func__, "Unable to initialize hash tables.\n");
279  return false;
280  }
281 
282  // Initialize the remote control communications module.
283  uint16_t remote_control_port = static_cast<uint16_t>(
284  config_info.GetUint("Tcp.RemoteControl.Port", kDefaultRemoteControlPort));
285 
286  if (!remote_control_.Initialize(remote_control_port))
287  {
288  LogF(kClassName, __func__, "Unable to initialize remote control "
289  "communications module.\n");
290  return false;
291  }
292 
293  // Log the configuration information.
294  LogC(kClassName, __func__, "TCP Proxy configuration:\n");
295  LogC(kClassName, __func__, "DefaultUtilityFn : %s\n",
296  default_utility_def_.c_str());
297  LogC(kClassName, __func__, "RemoteControlPort : %" PRIu16 "\n",
298  remote_control_port);
299  LogC(kClassName, __func__, "K : %.2e\n",
300  static_cast<double>(k_val_.GetValue()));
301  LogC(kClassName, __func__, "DirectAccess : %s\n",
302  qd_direct_access_ ? "On" : "Off");
303 
304  // Retrieve zero or more service configurations.
305  for (int i = 0; i < 16; i++)
306  {
307  char serv_name[100];
308 
309  snprintf(&serv_name[0], sizeof(serv_name)-1, "Service%d", i);
310 
311  char parm[300];
312  string pvar;
313  if ((pvar = config_info.Get(serv_name, "")) != "")
314  {
315  strncpy(&parm[0], pvar.c_str(), sizeof(parm)-1);
316 
317  TcpContext* context;
318  if ((context = ParseService(&parm[0], TcpModAction)) != NULL)
319  {
320  // Enable this service.
321  if (ModService(context) == false)
322  {
323  LogE(kClassName, __func__, "Addition of service %s failed.\n",
324  pvar.c_str());
325 
326  delete context;
327  return false;
328  }
329 
330  delete context;
331 
332  LogC(kClassName, __func__, "Service : %s\n",
333  pvar.c_str());
334  }
335  }
336  }
337 
338  // Initialize any configured server lists.
339 
340  // Extract the number of server lists.
341  num_server_lists_ =
342  static_cast<uint8_t>(config_info.GetUint("NumServerLists", 0));
343 
344  if (num_server_lists_ > kMaxServerLists)
345  {
346  LogE(kClassName, __func__, "Too many server lists (%" PRIu32
347  ") secified.\n", num_server_lists_);
348  return false;
349  }
350 
351  // Extract the server lists.
352  for (uint8_t i = 0; i < num_server_lists_; i++)
353  {
354  if (server_lists_[i] != NULL)
355  {
356  LogE(kClassName, __func__, "Server list %" PRIu32 " already "
357  "created.\n", i);
358  return false;
359  }
360 
361  // Create and initialize the server list.
362  server_lists_[i] = new (std::nothrow) ServerList(*this);
363  if (server_lists_[i] == NULL)
364  {
365  LogF(kClassName, __func__, "Error allocating new ServerList.\n");
366  return false;
367  }
368 
369  server_lists_[i]->Initialize(config_info, packet_pool_, bin_map_shm_, i);
370  }
371 
372  // Extract the statistics collection interval.
373  stats_interval_ms_ =
374  config_info.GetUint("StatsCollectionIntervalMs",
375  kDefaultStatsCollectionIntervalMs);
376 
377  LogC(kClassName, __func__, "StatsCollectionIntervalMs : %" PRIu32 "\n",
378  stats_interval_ms_);
379 
380  // Extract the directive that controls whether the statistics will be
381  // logged.
382  log_stats_ = config_info.GetBool("LogStatistics", true);
383 
384  LogC(kClassName, __func__, "LogStatistics : %s\n", log_stats_ ?
385  "true" : "false");
386 
387  if (!AttachSharedMemory(config_info))
388  {
389  LogE(kClassName, __func__, "TCP Proxy failed to attach to required "
390  "shared memory segments.");
391  return false;
392  }
393 
394  LogC(kClassName, __func__, "TCP Proxy configuration complete.\n");
395 
396  return true;
397 }

431 void TcpProxy::MainLoop()
432 {
433  int max_fd = 0;
434  fd_set read_fds;
435  FD_ZERO(&read_fds);
436 
437  edge_if_.AddFileDescriptors(max_fd, read_fds);
438  bpf_to_tcp_pkt_fifo_.AddFileDescriptors(max_fd, read_fds);
439 
440  // Add the fd for the remote control to the set of read fds.
441  remote_control_.AddFileDescriptors(max_fd, read_fds);
442 
443  // Get the next backstop time.
444  Time next_timer_expiration = timer_.GetNextExpirationTime();
445  struct timeval tv = next_timer_expiration.ToTval();
446 
447  int rv = Select(max_fd + 1, &read_fds, &tv);
448 
449  if (rv < 0)
450  {
451  LogE(kClassName, __func__, "select() error %s.\n", strerror(errno));
452  }
453  else if (rv > 0)
454  {
455  LogD(kClassName, __func__, "Servicing LAN side file descriptor.\n");
456 
457  // For now, limit the number of packets that are read from the LAN IF.
458  //
459  // if (edge_if_.InSet(&read_fds))
460  // {
461  ssize_t bytes_read = 0;
462  uint32_t num_lan_rcvs = 0;
463  do
464  {
465  Packet* pkt = packet_pool_.Get();
466  bytes_read = edge_if_.Recv(pkt, kMaxTcpOptLen);
467  if (bytes_read < 0)
468  {
469  packet_pool_.Recycle(pkt);
470  continue;
471  }
472 
473  if ((size_t)bytes_read > kMaxPktSizeBytes)
474  {
475  LogF(kClassName, __func__, "Packet size of %" PRIu32 " is too "
476  "large for proxy.\n", bytes_read);
477  }
478  else if ((size_t)bytes_read < kMinPktSizeBytes)
479  {
480  LogF(kClassName, __func__, "Packet size of %" PRIu32 " is too "
481  "small for proxy.\n", bytes_read);
482  }
483 
484  pkt->SetLengthInBytes(bytes_read + kMaxTcpOptLen);
485  pkt->RemoveBytesFromBeginning(kMaxTcpOptLen);
486  ProcessRcvdPkt(pkt, LAN);
487  num_lan_rcvs++;
488  } while (bytes_read > 0 && num_lan_rcvs < kMaxLanRecvs);
489  // }
490 
491  if (bpf_to_tcp_pkt_fifo_.InSet(&read_fds))
492  {
493  ReceivePktsFromBpf();
494  }
495 
496  // Process any messages received from the remote control communications.
497  if (remote_control_.ServiceFileDescriptors(read_fds))
498  {
499  ProcessRemoteControlMessage();
500  }
501  }
502 
503  socket_mgr_.RemoveMarkedSockets();
504  timer_.DoCallbacks();
505 
506  if (num_recon_reqs_ > 0)
507  {
508  socket_mgr_.RemoveMarkedSockets();
509 
510  for (uint8_t i = 0; i < num_recon_reqs_; i++)
511  {
512  ProcessRcvdPkt(recon_reqs_[i], LAN);
513  }
514 
515  num_recon_reqs_ = 0;
516  }
517 }

572 uint32_t TcpProxy::SimpleSendPkt(ProxyIfType out_if, PktInfo* pkt_info)
573 {
574  if (pkt_info->pkt == NULL)
575  {
576  LogW(kClassName, __func__, "Invalid argument. Discarding...\n");
577  pkt_info_pool_.Recycle(pkt_info);
578  return 0;
579  }
580 
581  if (out_if == LAN)
582  {
583  pkt_info->pkt->UpdateChecksums();
584 
585  uint32_t bytes_written = edge_if_.Send(pkt_info->pkt);
586 
587  // Recycle the packet and delete its container.
588  pkt_info_pool_.Recycle(pkt_info);
589 
590  return bytes_written;
591  }
592 
593  packet_pool_.AssignPacketId(pkt_info->pkt);
594 
595  bool sent_pkt = false;
596  bool fifo_is_open = tcp_to_bpf_pkt_fifo_.IsOpen();
597 
598  if (!fifo_is_open)
599  {
600  fifo_is_open = tcp_to_bpf_pkt_fifo_.OpenSender();
601 
602  if (!fifo_is_open)
603  {
604  LogW(kClassName, __func__, "Backpressure forwarder packet FIFO not "
605  "ready yet, dropping packet.\n");
606  }
607  }
608 
609  if (fifo_is_open)
610  {
611  sent_pkt = tcp_to_bpf_pkt_fifo_.Send(pkt_info->pkt);
612  }
613 
614  if (sent_pkt)
615  {
616  // Remove the packet so only the PktInfo is recycled.
617  // When the packet is processed by the bpf, the packet will be recycled.
618  uint32_t length = pkt_info->pkt->GetLengthInBytes();
619  pkt_info->pkt = NULL;
620  pkt_info_pool_.Recycle(pkt_info);
621  return length;
622  }
623  TRACK_UNEXPECTED_DROP(kClassName, packet_pool_);
624 
625  pkt_info_pool_.Recycle(pkt_info);
626  return 0;
627 }

630 void TcpProxy::SvcSocketsTimeout()
631 {
632  LogD(kClassName, __func__, "Service sockets timeout fired for handle %"
633  PRIu64 ".\n", svc_sockets_timer_.id());
634 
635  Time now = Time::Now();
636 
637  // If we aren't configured to do direct access to the queue depths, update
638  // them now.
639  if (!qd_direct_access_)
640  {
641  local_queue_depths_.CopyFromShm(weight_qd_shared_memory_);
642  }
643 
644  LogD(kClassName, __func__, "Servicing sockets, Queue depths are: %s.\n",
645  local_queue_depths_.ToString().c_str());
646 
647  // Service all of the sockets.
648  Socket* iter = socket_mgr_.GetSocketList();
649  while (iter != NULL)
650  {
651  iter->SvcEvents(now);
652 
653  iter = iter->next();
654  }
655 
656  // Schedule the next service sockets timer.
657  Time end_time = Time::Now();
658  Time duration = Time::FromUsec(kDefaultSvcSocketsIntervalUs) -
659  (end_time - now);
660  if (duration.GetTimeInUsec() < 0)
661  {
662  duration = 0.0;
663  }
664  CallbackNoArg<TcpProxy> callback(this, &TcpProxy::SvcSocketsTimeout);
665 
666  if (!timer_.StartTimer(duration, &callback, svc_sockets_timer_))
667  {
668  LogE(kClassName, __func__, "Error starting service sockets timer.\n");
669  }
670 
671  next_sched_socket_svc_time_ = now + duration;
672 
673  LogD(kClassName, __func__, "Started service sockets timer with duration %s "
674  "for handle %" PRId64 ".\n", duration.ToString().c_str(),
675  svc_sockets_timer_.id());
676 
677  LogD(kClassName, __func__, "Finished servicing sockets.\n");
678 }

925 //============================================================================
926 void TcpProxy::ProcessRcvdPkt(Packet* packet, ProxyIfType in_if)
927 {
928  // We first need to determine if the received packet is an tunneled TCP
929  // packet. If so, the flow's TCP and IP headers are encapsulated and we need
930  // to take this into account when extracting them.
931  uint8_t protocol;
932  if (!packet->GetIpProtocol(protocol))
933  {
934  LogE(kClassName, __func__, "Unable to get packet protocol from received "
935  "packet.\n");
936  packet_pool_.Recycle(packet);
937  return;
938  }
939 
940  bool is_tunnel = false;
941  struct iphdr* ip_hdr = NULL;
942  struct tcphdr* tcp_hdr = NULL;
943  if (protocol == IPPROTO_UDP)
944  {
945  // This is a TCP packet encapsulated in a UDP tunnel. Verify that the
946  // destination port for the encapsulated packet is supported by the
947  // proxy. Currently, we only support VXLAN tunnels which have a
948  // destination port of 4789.
949  uint16_t dport;
950  if (!packet->GetDstPort(dport))
951  {
952  LogE(kClassName, __func__, "Unable to get destination port from "
953  "received packet.\n");
954  packet_pool_.Recycle(packet);
955  return;
956  }
957 
958  if (ntohs(dport) != iron::kVxlanTunnelDstPort)
959  {
960  LogE(kClassName, __func__, "Received tunneled packet to unsupported "
961  "destination port: %" PRIu16 ".\n", dport);
962  packet_pool_.Recycle(packet);
963  return;
964  }
965 
966  LogD(kClassName, __func__, "Rcvd. VXLAN tunneled packet.\n");
967 
968  is_tunnel = true;
969 
970  // Skip over the encapsulating tunnel headers to get to the IP and TCP
971  // headers for the received TCP packet.
972  ip_hdr = reinterpret_cast<struct iphdr*>(
973  packet->GetBuffer(iron::kVxlanTunnelHdrLen));
974  tcp_hdr = reinterpret_cast<struct tcphdr*>(
975  packet->GetBuffer(iron::kVxlanTunnelHdrLen + (ip_hdr->ihl * 4)));
976  }
977  else
978  {
979  // Normal, non-tunneled TCP packet.
980  ip_hdr = packet->GetIpHdr();
981  tcp_hdr = packet->GetTcpHdr();
982  }
983 
984  if ((tcp_hdr == NULL) || (ip_hdr == NULL))
985  {
986  LogE(kClassName, __func__, "Packet is not an IPv4 TCP packet. Dumping "
987  "contents:\n%s\n", packet->ToHexString().c_str());
988  LogF(kClassName, __func__, "TCP proxy is misconfigured.\n");
989  }
990 
991  // Verify that the destination address of the packet received on the
992  // LAN-facing interface has a mapping in the BinMap. If no such mapping is
993  // found, simply recycle the packet.
994  BinIndex index = iron::kInvalidBinIndex;
995  if ((in_if == LAN) &&
996  (!GetBinIndex(Ipv4Address(ip_hdr->daddr), index)))
997  {
998  LogW(kClassName, __func__, "No mapping for destination address %s found "
999  "in BinMap.\n", Ipv4Address(ip_hdr->daddr).ToString().c_str());
1000  packet_pool_.Recycle(packet);
1001  packet = NULL;
1002  return;
1003  }
1004 
1005  // Try demuxing to an existing socket.
1006  FourTuple ft(ip_hdr->saddr, tcp_hdr->th_sport,
1007  ip_hdr->daddr, tcp_hdr->th_dport);
1008 
1009  Socket* sock = socket_mgr_.GetExistingSocket(ft);
1010 
1011  bool created_active_socket = false;
1012 
1013  if (sock)
1014  {
1015  LogD(kClassName, __func__, "Demuxed packet to initialized socket on %s "
1016  "IF with flow myport(%" PRIu32 ") hisport(%" PRIu32 ") seq(%" PRIu32
1017  ") ack(%" PRIu32 ")\n", sock->cfg_if_id() == WAN ? "WAN" : "LAN",
1018  ntohs(sock->my_port()), ntohs(sock->his_port()), ntohl(tcp_hdr->th_seq),
1019  htonl(tcp_hdr->th_ack));
1020 
1021  if ((in_if == LAN) && sock->is_tunneled())
1022  {
1023  // We have recevied an encapsulated TCP packet for a known
1024  // flow. "Remove" the tunnel headers.
1025  packet->RemoveBytesFromBeginning(iron::kVxlanTunnelHdrLen);
1026  }
1027  }
1028  else
1029  {
1030  if (tcp_hdr->th_flags & TH_RST)
1031  {
1032  // We have no state for the received packet, so just recycle it.
1033  LogD(kClassName, __func__, "Discarding RST packet with no state.\n");
1034 
1035  packet_pool_.Recycle(packet);
1036 
1037  return;
1038  }
1039 
1040  if ((tcp_hdr->th_flags & TH_SYN) && (tcp_hdr->th_flags & TH_ACK))
1041  {
1042  // If we are here, we have received a SYN/ACK packet for which we don't
1043  // have any existing flow state information. This means that the proxy
1044  // never "saw" the SYN packet. We'll log an error and recycle the
1045  // packet.
1046  LogE(kClassName, __func__, "Received a SYN/ACK packet on %s IF for "
1047  "flow %s for which there is no existing internal proxy flow "
1048  "information. Discarding packet with no state.\n",
1049  in_if == LAN ? "LAN" : "WAN", ft.ToString().c_str());
1050 
1051  packet_pool_.Recycle(packet);
1052 
1053  return;
1054  }
1055 
1056  if (!(tcp_hdr->th_flags & TH_SYN))
1057  {
1058  if (in_if == LAN)
1059  {
1060  // We have no state and the receive packet is not a SYN, so generate a
1061  // RST and send it out the interface the packet was received on.
1062  PktInfo* smpl_pkt_info = pkt_info_pool_.Get(packet);
1063  if (smpl_pkt_info == NULL)
1064  {
1065  LogF(kClassName, __func__, "Error allocating new PktInfo.\n");
1066  return;
1067  }
1068 
1069  SimpleReset(smpl_pkt_info, in_if, is_tunnel);
1070  }
1071  else
1072  {
1073  // If we are here, we don't have any existing state, and the packet
1074  // was received on the WAN side and doesn't have a SYN. This can
1075  // happen with multipath delayed packets after the socket is
1076  // closed. We simply recycle the packet.
1077  LogD(kClassName, __func__, "Discarding packet with no state.\n");
1078 
1079  packet_pool_.Recycle(packet);
1080  }
1081 
1082  return;
1083  }
1084 
1085  sock = CreateSocketPair(packet, in_if, is_tunnel);
1086  if (sock == NULL)
1087  {
1088  LogW(kClassName, __func__, "Error creating socket pair.\n");
1089 
1090  return;
1091  }
1092 
1093  created_active_socket = true;
1094  }
1095 
1096  if (sock->state() == TCP_CLOSE)
1097  {
1098  // The socket is in a CLOSED state, so simply recycle the packet.
1099  packet_pool_.Recycle(packet);
1100  return;
1101  }
1102 
1103  if (sock->desired_dscp() != -1)
1104  {
1105  if (!packet->SetIpDscp(sock->desired_dscp()))
1106  {
1107  LogW(kClassName, __func__, "Failed to set DSCP value %" PRIu8
1108  " in packet.\n", sock->desired_dscp());
1109  }
1110  }
1111  else
1112  {
1113  if (sock->tos() != ip_hdr->tos)
1114  {
1115  sock->set_tos(ip_hdr->tos);
1116  }
1117  }
1118 
1119  if ((sock->peer()) && (sock->peer()->tos() != ip_hdr->tos))
1120  {
1121  sock->peer()->set_tos(ip_hdr->tos);
1122  }
1123 
1124  // Record the reception of a packet. We only record statistics for WAN side
1125  // sockets.
1126  if (sock->cfg_if_id() == WAN)
1127  {
1128  int len = (int)(ntohs(ip_hdr->tot_len) - (ip_hdr->ihl * 4) -
1129  (tcp_hdr->th_off << 2));
1130 
1131  LogD(kClassName, __func__, "Recording %d bytes rcvd for on %s IF for "
1132  "flow myport(%" PRIu16 ") hisport(%" PRIu16 ")\n",
1133  len, sock->cfg_if_id() == WAN ? "WAN" : "LAN", ntohs(sock->my_port()),
1134  ntohs(sock->his_port()));
1135 
1136  sock->IncrementRcvdBytes(len);
1137  }
1138 
1139  PktInfo* pkt_info = pkt_info_pool_.Get(packet);
1140  if (pkt_info == NULL)
1141  {
1142  LogF(kClassName, __func__, "Error allocating new PktInfo.\n");
1143  return;
1144  }
1145 
1146  pkt_info->seq_num = ntohl(tcp_hdr->th_seq);
1147  pkt_info->data_len = ntohs(ip_hdr->tot_len) - (ip_hdr->ihl * 4) -
1148  (tcp_hdr->th_off * 4);
1149  pkt_info->flags = tcp_hdr->th_flags;
1150 
1151  LogD(kClassName, __func__, "Created PktInfo for packet rcvd on %s IF: seq "
1152  "(%" PRIu32 ") data len (%" PRIu32 ").\n", sock->cfg_if_id() == WAN ?
1153  "WAN" : "LAN", pkt_info->seq_num, pkt_info->data_len);
1154 
1155  int rc = sock->ProcessPkt(pkt_info, tcp_hdr, ip_hdr);
1156 
1157  // Need to pull these out to make sure s1 and s2 are non-NULL, otherwise
1158  // this will generate a core dump
1159  Socket* s1 = sock;
1160  Socket* s2 = sock->peer();
1161 
1162  switch (rc)
1163  {
1164  case 0:
1165  socket_mgr_.CloseSocket(s1);
1166  socket_mgr_.CloseSocket(s2);
1167  break;
1168 
1169  case -2:
1170  s1->Abort();
1171  s2->Abort();
1172  break;
1173 
1174  case -1:
1175  break;
1176 
1177  default:
1178  if ((sock->gw_flags() & PROXY_SEND_SYN) && (s2 > 0)) // <- UGGGH!
1179  {
1180  sock->gw_flags() &= (~PROXY_SEND_SYN);
1181 
1182  LogD(kClassName, __func__, "Invoking connect...\n");
1183 
1184  if (!s2->Connect())
1185  {
1186  // If you can't complete the connection you must abort.
1187  s1->Abort();
1188  s2->Abort();
1189 
1190  return;
1191  }
1192 
1193  // If we created a socket pair, we would have set
1194  // created_active_socket to true. If so, the passive socket is s1 and
1195  // the active socket is s2.
1196  if (created_active_socket)
1197  {
1198  Socket* UNUSED(passive_socket) = s1;
1199  Socket* active_socket = s2;
1200  LogI(kClassName, __func__, "Active side %d %d %d passive side %d "
1201  "%d %d\n", htons(active_socket->my_port()),
1202  htons(active_socket->his_port()), active_socket->cfg_if_id(),
1203  htons(passive_socket->my_port()), htons(passive_socket->his_port()),
1204  passive_socket->cfg_if_id());
1205 
1206  // Setup the src and destination endpoints for statistics
1207  // reporting. We will report statistics on the WAN side socket but
1208  // the reported flow id will be the corresponding LAN side
1209  // addresses/ports. See the class level comment in TpSocket.hh for
1210  // an example of why the values are as they are below.
1211  if (active_socket->cfg_if_id() == WAN)
1212  {
1213  Ipv4Endpoint& stats_src_endpt = active_socket->stats_src_endpt();
1214  Ipv4Endpoint& stats_dst_endpt = active_socket->stats_dst_endpt();
1215 
1216  stats_src_endpt.set_address(active_socket->peer()->his_addr().s_addr);
1217  stats_src_endpt.set_port(active_socket->peer()->his_port());
1218  stats_dst_endpt.set_address(active_socket->peer()->my_addr().s_addr);
1219  stats_dst_endpt.set_port(active_socket->peer()->my_port());
1220  }
1221  else
1222  {
1223  // We can't use tpSock here as it may have been cloned. Instead we
1224  // will use the activeSock addresses/ports as it is on the LAN
1225  // side.
1226  Ipv4Endpoint& stats_src_endpt =
1227  active_socket->peer()->stats_src_endpt();
1228  Ipv4Endpoint& stats_dst_endpt =
1229  active_socket->peer()->stats_dst_endpt();
1230 
1231  stats_src_endpt.set_address(active_socket->my_addr().s_addr);
1232  stats_src_endpt.set_port(active_socket->my_port());
1233  stats_dst_endpt.set_address(active_socket->his_addr().s_addr);
1234  stats_dst_endpt.set_port(active_socket->his_port());
1235  }
1236  }
1237  }
1238 
1239  break;
1240  }
1241 }

1244 void TcpProxy::SimpleReset(PktInfo* rst_pkt_info, ProxyIfType rcv_if,
1245  bool is_tunnel)
1246 {
1247  if (is_tunnel)
1248  {
1249  // The received packet is a VXLAN tunneled packet. We need to swap:
1250  //
1251  // - the source and destination addresses in the outer IP header
1252  // - the destination and source MAC addresses in the inner Ethernet header
1253  struct iphdr* ip_hdr = rst_pkt_info->pkt->GetIpHdr();
1254  uint32_t tmp_addr = ip_hdr->saddr;
1255 
1256  ip_hdr->saddr = ip_hdr->daddr;
1257  ip_hdr->daddr = tmp_addr;
1258 
1259  uint8_t tmp_eth_addr[ETH_ALEN];
1260  struct ethhdr* eth_hdr = reinterpret_cast<struct ethhdr*>(
1261  rst_pkt_info->pkt->GetBuffer(sizeof(struct iphdr) +
1262  sizeof(struct udphdr) + 8));
1263 
1264  memcpy(tmp_eth_addr, eth_hdr->h_dest, ETH_ALEN);
1265  memcpy(eth_hdr->h_dest, eth_hdr->h_source, ETH_ALEN);
1266  memcpy(eth_hdr->h_source, tmp_eth_addr, ETH_ALEN);
1267 
1268  // Now, we temporarily "remove" the VXLAN headers from the packet so that
1269  // we can create the RST packet to send back to the source of the received
1270  // packet.
1271  rst_pkt_info->pkt->RemoveBytesFromBeginning(iron::kVxlanTunnelHdrLen);
1272  }
1273 
1274 
1275  // Swap src/dst addrs and ports, seq and ack numbers, and set RST flag. Then
1276  // send RST packet back out interface it was received on. Additionally, trim
1277  // off all TCP header options and any data in the received packet and set
1278  // the window size to 0.
1279  struct iphdr* ip_hdr = rst_pkt_info->pkt->GetIpHdr();
1280  struct tcphdr* tcp_hdr = rst_pkt_info->pkt->GetTcpHdr();
1281 
1282  uint32_t tmp_addr = ip_hdr->saddr;
1283  ip_hdr->saddr = ip_hdr->daddr;
1284  ip_hdr->daddr = tmp_addr;
1285 
1286  uint16_t tmp_port = tcp_hdr->th_sport;
1287  tcp_hdr->th_sport = tcp_hdr->th_dport;
1288  tcp_hdr->th_dport = tmp_port;
1289 
1290  uint32_t tmp_seq = tcp_hdr->th_seq;
1291  tcp_hdr->th_seq = tcp_hdr->th_ack;
1292  tcp_hdr->th_ack = tmp_seq;
1293 
1294  tcp_hdr->th_flags |= TH_RST;
1295 
1296  tcp_hdr->th_win = 0;
1297 
1298  ip_hdr->tot_len = htons(sizeof(struct iphdr) + sizeof(struct tcphdr));
1299  tcp_hdr->th_off = sizeof(struct tcphdr) >> 2;
1300  rst_pkt_info->pkt->SetLengthInBytes(sizeof(struct iphdr) +
1301  sizeof(struct tcphdr));
1302 
1303  if (is_tunnel)
1304  {
1305  // Make sure that the checksums are computed on the inner packet.
1306  rst_pkt_info->pkt->UpdateChecksums();
1307 
1308  // "Add" back the VXLAN headers. The bytes were never removed from the
1309  // received packet so we simply need to add bytes to the beginning of the
1310  // packet.
1311  rst_pkt_info->pkt->AddBytesToBeginning(iron::kVxlanTunnelHdrLen);
1312  rst_pkt_info->pkt->SetLengthInBytes(iron::kVxlanTunnelHdrLen +
1313  sizeof(struct iphdr) +
1314  sizeof(struct tcphdr));
1315  }
1316 
1317  if (SimpleSendPkt(rcv_if, rst_pkt_info) < 0)
1318  {
1319  LogW(kClassName, __func__, "Error sending RST.\n");
1320  pkt_info_pool_.Recycle(rst_pkt_info);
1321  }
1322 }

1331  struct tcphdr* handoff_tcp_hdr = NULL;
1332  struct iphdr* ip_hdr = NULL;
1333  struct iphdr* handoff_ip_hdr = NULL;
1334

1338  // headers for the received TCP packet.
1339  ip_hdr = reinterpret_cast<struct iphdr*>(
1340  packet->GetBuffer(iron::kVxlanTunnelHdrLen));
1341  tcp_hdr = reinterpret_cast<struct tcphdr*>(
1342  packet->GetBuffer(iron::kVxlanTunnelHdrLen + (ip_hdr->ihl * 4)));
1343  }

1345  {

1354  "configured server is %s.\n", num_server_lists_,

1359  // doing seamless handoffs. We only need to worry about this for packets
1360  // that were received on the LAN-facing interface.
1361  //
1362  // NOTE: For now, we don't support seamless handoffs for tunneled packets.

1364  {

1367  if (client_configured_server ==
1368  server_lists_[i]->client_configured_server())
1369  {

1372  seamless_handoff = true;

1375  "configured server %s to server %s.\n",
1376  client_configured_server.ToString().c_str(),
1377  handoff_server.ToString().c_str());

1380  handoff_ip_hdr = new (std::nothrow) struct iphdr;

1382  {

1384  "header.\n");

1386  }

1388  memcpy(handoff_ip_hdr, ip_hdr, sizeof(struct iphdr));
1389  handoff_ip_hdr->daddr = handoff_server.address();

1392  }
1393  else
1394  {
1395  LogI(kClassName, __func__, "There are no reachable servers for "
1396  "client configured server %s. Generating and sending TCP RST "
1397  "packet.\n", client_configured_server.ToString().c_str());
1398

1400  // destination address. Generate a TCP RST and send it out the
1401  // LAN-facing interface.
1402  GenerateAndSendReset(tcp_hdr, ip_hdr);
1403  return NULL;
1404  }
1405  }
1406  }
1407  }
1408 
1409  // Create passive socket.
1410  LogD(kClassName, __func__, "Attempting to create passive socket...\n");
1411

1413  if ((passive_socket = CreatePassiveSocket(tcp_hdr, ip_hdr)) == NULL)
1414  {
1415  LogE(kClassName, __func__, "Error creating passive socket.\n");
1416  return NULL;
1417  }

1420

1422  passive_socket->set_flow_tag(tag);
1423  passive_socket->set_cfg_if_id(in_if);
1424

1426  passive_socket->SetMss(0);
1427 
1428  passive_socket->set_prev_state(TCP_CLOSE);
1429  passive_socket->set_state(TCP_LISTEN);
1430  passive_socket->set_timeout(0x7ffffff);
1431

1433  LogD(kClassName, __func__, "Attempting to create active socket...\n");
1434 
1435  Socket* active_socket = NULL;
1436  struct tcphdr* active_socket_tcp_hdr =
1437  (handoff_tcp_hdr == NULL) ? tcp_hdr : handoff_tcp_hdr;
1438  struct iphdr* active_socket_ip_hdr =
1439  (handoff_ip_hdr == NULL) ? ip_hdr : handoff_ip_hdr;
1440

1442  active_socket_ip_hdr)) == NULL)
1443  {
1444  LogE(kClassName, __func__, "Error creating active socket. "
1445  "Closing passive socket.\n");

1447 
1448  return NULL;
1449  }
1450

1452

1454  // sockets, if required.
1455  if (seamless_handoff)
1456  {

1458  passive_socket->set_seamless_handoff_endpoint(handoff_server);
1459  active_socket->set_do_seamless_handoff();

1461  client_configured_server);
1462  }
1463

1465  {

1467  // socket so that they can be prepended to the tranmitted SYN packets. In
1468  // the passive socket we remember the tunnel headers and then invert them
1469  // so they are ready for transmissions to the LAN-facing interface. Then
1470  // "remove" the tunnel headers from the received packet.

1477

1481  }
1482

1484  active_socket->set_flow_tag(tag);
1485 
1486  if (in_if == LAN)
1487  {
1488  active_socket->set_cfg_if_id(WAN);
1489  }
1490  else
1491  {
1492  active_socket->set_cfg_if_id(LAN);
1493  }

1533  }
1534 
1535  if (active_socket->cfg_if_id() == WAN)
1536  {
1537  // Since the active socket is on the WAN side, we find the bin index by
1538  // matching the destination address of the original packet when not doing

1561  // matching the source address of the original packet.
1562  BinIndex idx = iron::kInvalidBinIndex;

1564  {
1565  LogF(kClassName, __func__, "No bin defined for address %" PRIu32 "\n",
1566  ip_hdr->saddr);
1567  }
1568  passive_socket->set_bin_idx(idx);
1569

1571  // the passive_socket. We must do this after the socket's bin index has
1572  // been set.
1573  passive_socket->ConfigureUtilityFn(utility_fn_def, local_queue_depths_);
1574  LogI(kClassName, __func__, "Flow tag: %" PRIu32 " <==> %s\n", tag,
1575  four_tuple.ToString().c_str());
1576  }
1577 
1578  if (passive_socket->cfg_if_id() == LAN)
1579  {
1580  int8_t context_dscp = -1;
1581  if (!GetFlowDscpDef(four_tuple, context_dscp))
1582  {
1583  context_dscp = GetContextDscp(ntohs(four_tuple.dst_port_nbo()));
1584  }
1585

1587  if (context_dscp != -1)
1588  {
1589  packet->SetIpDscp(context_dscp);
1590  }
1591  }
1592 
1593  LogI(kClassName, __func__, "Flow tag: %" PRIu32 " <==> %s (%s)\n", tag,
1594  four_tuple.ToString().c_str(), is_tunnel ? "is tunnel" :
1595  "is not tunnel");
1596 
1597  if (!timer_.IsTimerSet(svc_sockets_timer_))
1598  {
1599  // We have created sockets and the service sockets timer is not started,
1600  // so start it now.
1601  Time duration =
1602  Time::FromUsec(kDefaultSvcSocketsIntervalUs);
1603  next_sched_socket_svc_time_ = Time::Now() + duration;
1604  CallbackNoArg<TcpProxy> callback(this, &TcpProxy::SvcSocketsTimeout);
1605 
1606  if (!timer_.StartTimer(duration, &callback, svc_sockets_timer_))
1607  {
1608  LogE(kClassName, __func__, "Error setting service sockets timer.\n");
1609  }
1610 
1611  LogD(kClassName, __func__, "Started service sockets timer with duration "
1612  "%s for handle %" PRIu64 ".\n", duration.ToString().c_str(),
1613  svc_sockets_timer_.id());
1614  }
1615 
1616  // If we are doing seamless handoff for the socket pair that was just
1617  // created, be sure to delete the temporary copies of the original TCP and
1618  // IP headers.
1619  if (handoff_tcp_hdr != NULL)
1620  {
1621  delete handoff_tcp_hdr;
1622  handoff_tcp_hdr = NULL;
1623  }
1624 
1625  if (handoff_ip_hdr != NULL)
1626  {
1627  delete handoff_ip_hdr;
1628  handoff_ip_hdr = NULL;
1629  }
1630

1632 }
