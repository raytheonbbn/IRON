82 string AdmissionController::GetUtilityFnType(string& utility_def,
83  uint32_t flow_id) const
84 {
85  size_t type_str_pos = utility_def.find("type=");
86  if (type_str_pos == string::npos)
87  {
88  LogF(kClassName, __func__, "fid: %" PRIu32 ", invalid utility definition "
89  "string, type not provided.\n", flow_id);
90  return "";
91  }
92 
93  size_t type_str_end_pos = utility_def.find(":", type_str_pos);
94  if (type_str_end_pos == string::npos)
95  {
96  LogF(kClassName, __func__, "fid: %" PRIu32 ", invalid utility definition "
97  "string format.\n", flow_id);
98  return "";
99  }
100 
101  return utility_def.substr(type_str_pos + 5, type_str_end_pos -
102  (type_str_pos + 5));
103 }

106 bool AdmissionController::ConfigureUtilityFn(UtilityFn* utility_fn,
107  string& utility_def)
108 {
109  ConfigInfo ci;
110  List<string> tokens;
111 
112  StringUtils::Tokenize(utility_def, ":", tokens);
113 
114  List<string>::WalkState tokens_ws;
115  tokens_ws.PrepareForWalk();
116 
117  string token;
118  while (tokens.GetNextItem(tokens_ws, token))
119  {
120  if (token.find("=") == string::npos)
121  {
122  continue;
123  }
124 
125  List<string> token_values;
126  StringUtils::Tokenize(token, "=", token_values);
127 
128  if (token_values.size() == 2)
129  {
130  string name;
131  string value;
132 
133  token_values.Pop(name);
134  token_values.Peek(value);
135  ci.Add(name, value);
136  }
137  }
138 
139  ci.Add("ttg", StringUtils::ToString(
140  static_cast<uint64_t>(encoding_state_.time_to_go().GetTimeInUsec())));
141  if (!utility_fn->Initialize(ci))
142  {
143  return false;
144  }
145 
146  return true;
147 }

168 void AdmissionController::AdmitPkts(Time& now)
169 {
170  // last_send_rate_ is updated in the UpdateScheduledAdmissionEvent method,
171  // which is always called before this method. So using last_send_rate_
172  // instead of calling GetSendRate in the utility function is OK.
173  LogD(kClassName, __func__, "fid: %" PRIu32 ", send rate is %f.\n",
174  encoding_state_.flow_tag(), last_send_rate_);
175 
176  if (last_send_rate_ > 0)
177  {
178  // Make sure we don't allow for an indefinite amount of catch up time.
179  Time low_adm_time = now.Subtract(bpf_min_burst_);
180  if (next_admission_time_ < low_adm_time)
181  {
182  next_admission_time_ = low_adm_time;
183  }
184 
185  Time burst_time = now.Add(bpf_min_burst_ +
186  Time(0,rng_.GetInt(bpf_min_burst_.GetTimeInUsec()/2)));
187 
188  while (next_admission_time_ <= burst_time)
189  {
190  size_t bytes_sent = AdmitPkt();
191 
192  if (bytes_sent == 0)
193  {
194  // XXX DO WE REALLY SET THE FLOW IDLE IN THIS CASE?
195  flow_is_idle_ = true;
196  break;
197  }
198 
199  // Compute the serialization time for the transmission.
200  Time serialization_time(static_cast<double>(bytes_sent) * 8.0 /
201  last_send_rate_);
202 
203  // Adjust the next admission time.
204  if (flow_is_idle_ == true)
205  {
206  LogD(kClassName, __func__, "fid: %" PRIu32 ", flow is idle.\n",
207  encoding_state_.flow_tag());
208 
209  next_admission_time_ = Time::Now() + serialization_time;
210  flow_is_idle_ = false;
211  }
212  else
213  {
214  next_admission_time_ = next_admission_time_.Add(serialization_time);
215  }
216  }
217  }
218  else if (last_send_rate_ < 0)
219  {
220  LogE(kClassName, __func__, "fid: %" PRIu32 ", computed send rate is "
221  "negative %.03f.\n", encoding_state_.flow_tag(), last_send_rate_);
222  }
223 }

226 void AdmissionController::UpdateScheduledAdmissionEvent(Time& now,
227  UtilityFn* utility_fn)
228 {
229  double new_rate = utility_fn->GetSendRate();
230 
231  if ((new_rate == 0) && (last_send_rate_ != 0))
232  {
233  if ((encoding_state_.GetCountFromEncodedPktsQueue() > 0) &&
234  (flow_state() != iron::FLOW_OFF))
235  {
236  LogD(kClassName, __func__, "fid: %" PRIu32 " toggled down.\n",
237  encoding_state_.flow_tag());
238  ++toggle_count_;
239  push_stats_ = true;
240  }
241  }
242  else if ((new_rate != 0) && (last_send_rate_ == 0) &&
243  (start_time_ < (now - kStartupTime)))
244  {
245  LogD(kClassName, __func__, "fid: %" PRIu32 " toggled up.\n",
246  encoding_state_.flow_tag());
247  ++toggle_count_;
248  push_stats_ = true;
249  }
250 
251  // Adjust the next_admission_time_.
252  Time sched_svc_time = encoding_state_.sched_svc_time();
253  Time admission_target_time = next_admission_time_;
254  Time admission_delta = admission_target_time - sched_svc_time;
255 
256  if (admission_delta > Time())
257  {
258  Time
259  next_target_time(static_cast<double>(admission_delta.GetTimeInUsec()) *
260  last_send_rate_ / (new_rate * 1000000.0));
261 
262  next_target_time = next_target_time.Add(sched_svc_time);
263  next_admission_time_ = next_target_time;
264 
265  LogD(kClassName, __func__, "fid: %" PRIu32 ", updated admission time, "
266  "now is %s, original admission time is %s, new admission time is "
267  "%s, last send rate is %f, new send rate is %f.\n",
268  encoding_state_.flow_tag(), now.ToString().c_str(),
269  admission_target_time.ToString().c_str(),
270  next_admission_time_.ToString().c_str(),
271  last_send_rate_, new_rate);
272  }
273  else
274  {
275  LogD(kClassName, __func__, "fid: %" PRIu32 ", next admission time is in "
276  "the past, now is %s, next admission time is %s.\n",
277  encoding_state_.flow_tag(), now.ToString().c_str(),
278  next_admission_time_.ToString().c_str());
279  }
280 
281  // Finally, save the new old rate.
282  last_send_rate_ = new_rate;
283 }
