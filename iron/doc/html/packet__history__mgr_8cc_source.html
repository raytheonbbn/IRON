98 bool PacketHistoryMgr::PacketVisitedBin(Packet* packet, BinId bin_id)
99 {
100  // MCAST TODO: seems like a weird use of bin id. Is this correct?
101  const uint8_t* history = packet->history();
102  uint8_t count = 0;
103 
104  while ((*history != kHistoryEntryUnused) &&
105  (count < kHistoryFieldSizeBytes))
106  {
107  if (*history == bin_id)
108  {
109  return true;
110  }
111 
112  ++history;
113  ++count;
114  }
115 
116  return false;

120 uint32_t PacketHistoryMgr::GetAllVisitedBins(Packet* packet,
121  BinId* visited_bin_array,
122  const uint32_t visited_bin_len)
123 {
124  if (!packet)
125  {
126  return 0;
127  }
128 
129  const uint8_t* history = packet->history();
130  uint32_t count = 0;
131  uint32_t it_count = 0;
132 
133  // MCAST TODO: Check whether this will works with the changes to bin id.
134  // TODO: Note that "(count < bin_map_.GetNumDestBinIds())" cannot be part of
135  // the conditional, as it does not include interior node BinIds, but
136  // interior node BinIds get recorded in the packet history.
137  while ((*history != kHistoryEntryUnused) &&
138  (it_count < kHistoryFieldSizeBytes))
139  {
140  // Figure out if we have already seen this Bin Id.
141  bool already_in = false;
142 
143  for (uint32_t i = 0; i < count; ++i)
144  {
145  if (static_cast<BinId>(*history) == visited_bin_array[i])
146  {
147  already_in = true;
148  break;
149  }
150  }
151 
152  // First time we have seen it, add to visited array.
153  if (!already_in)
154  {
155  if (count >= visited_bin_len)
156  {
157  LogE(kClassName, __func__, "Error, visited bins array size (%" PRIu32
158  ") is too small.\n", visited_bin_len);
159  return count;
160  }
161 
162  visited_bin_array[count] = static_cast<BinId>(*history);
163  ++count;
164  }
165 
166  ++it_count;
167  ++history;
168  }
169 
170  return count;
171 }

174 uint8_t PacketHistoryMgr::GetNumVisits(Packet* packet, BinId bin_id)
175 {
176  const uint8_t* history = packet->history();
177  uint8_t count = 0;
178  uint8_t it_count = 0;
179 
180  while ((*history != kHistoryEntryUnused) &&
181  (it_count < kHistoryFieldSizeBytes))
182  {
183  if (static_cast<BinId>(*history) == bin_id)
184  {
185  ++count;
186  }
187 
188  ++it_count;
189  ++history;
190  }
191 
192  return count;
193 }

196 void PacketHistoryMgr::TrackHistory(Packet* packet, bool local_packet)
197 {
198  uint8_t num_times_visited = GetNumVisits(packet, my_bin_id_);
199 
200  if (local_packet && (num_times_visited > 0))
201  {
202  // Don't track a local packet more than once, since this could be a
203  // retransmission (and can't possibly be a cycle, since local packets
204  // haven't yet entered the network).
205  return;
206  }
207 
208  if (kTrackCycles)
209  {
210  ++total_num_packets_;
211  if (num_times_visited > 0)
212  {
213  ++cycle_count_;
214  }
215  }
216 
217  packet->InsertNodeInHistory(my_bin_id_);
218 }

232 void PacketHistoryMgr::LogCirculationStats()
233 {
234  if (kTrackCycles)
235  {
236  if (total_num_packets_ == 0)
237  {
238  LogI(kClassName, __func__, "No packets observed.\n");
239  }
240  else
241  {
242  LogW(kClassName, __func__, "Observed total of %" PRIu64 " packets, "
243  "including %" PRIu64 " that cycled (%.2f%%).\n",
244  total_num_packets_, cycle_count_,
245  (static_cast<double>(cycle_count_) /
246  static_cast<double>(total_num_packets_) * 100));
247  }
