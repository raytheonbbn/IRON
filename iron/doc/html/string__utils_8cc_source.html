64 void StringUtils::Tokenize(const string& str, const char* delim,
65  List<string>& tokens)
66 {
67  size_t str_len = (str.size() + 1);
68  char* str_to_tokenize;
69 
70  tokens.Clear();
71 
72  str_to_tokenize = new (std::nothrow) char[str_len];
73 
74  if (str_to_tokenize != NULL)
75  {
76  ::memcpy(str_to_tokenize, str.c_str(), str_len);
77 
78  char* p = strtok(str_to_tokenize, delim);
79 
80  while (p)
81  {
82  tokens.Push(p);
83  p = strtok(NULL, delim);
84  }
85 
86  delete [] str_to_tokenize;
87  }
88 }

91 bool StringUtils::Substitute(
92  std::string& str, std::string& start, std::string& end, std::string& val)
93 {
94  // Get the start of the target substring.
95  size_t start_pos = str.find(start);
96 
97  if (start_pos == string::npos)
98  {
99  return false;
100  }
101 
102  // Get the end of the target substring.
103  size_t end_pos = str.find(end, start_pos);
104 
105  if (end_pos == string::npos)
106  {
107  return false;
108  }
109 
110  str = str.substr(0,start_pos + start.size()) + val
111  + str.substr(end_pos, (str.size() - end_pos));
112 
113  return true;
114 }

121  bool replaced = false;
122 
123  while ((pos = input.find(search, pos)) != string::npos)

126  pos += replace.length();
127  replaced = true;
128  }
129 
130  return replaced;

134 bool StringUtils::GetBool(const string& str, const bool default_value)
135 {
136  bool rv = default_value;
137 
138  //
139  // Use strncasecmp() to do a case-insensitive comparisons on "true" and
140  // "false". Also permits the value "0" to be used to represent false or the
141  // value "1" to be used to represent true.
142  //
143 
144  if (::strncasecmp(str.c_str(), "true", 4) == 0)
145  {
146  rv = true;
147  }
148  else if (::strncasecmp(str.c_str(), "false", 5) == 0)
149  {
150  rv = false;
151  }
152  else if (::strncmp(str.c_str(), "0", 1) == 0)
153  {
154  rv = false;
155  }
156  else if (::strncmp(str.c_str(), "1", 1) == 0)
157  {
158  rv = true;
159  }
160 
161  return rv;
162 }

165 int StringUtils::GetInt(const string& str, const int default_value)
166 {
167  char* end_ptr = NULL;
168  const char* str_ptr = str.c_str();
169 
170  // Clear errno before the call, per strtol(3).
171  errno = 0;
172 
173  long val = ::strtol(str_ptr, &end_ptr, 10);
174 
175  // Check for overflow, underflow, and any other conversion error.
176  if (((errno == ERANGE) && ((val == LONG_MAX) || (val == LONG_MIN)))
177  || ((errno != 0) && (val == 0)))
178  {
179  LogE(kClassName, __func__, "Error converting string %s to int: %s\n",
180  str_ptr, strerror(errno));
181  return default_value;
182  }
183 
184  // Check for no conversion.
185  if (end_ptr == str_ptr)
186  {
187  LogE(kClassName, __func__, "Error converting string %s to int.\n",
188  str_ptr);
189  return default_value;
190  }
191 
192  return static_cast<int>(val);
193 }

452 std::string StringUtils::FormatString(int size, const char* format, ...)
453 {
454  char format_str[size];
455  va_list vargs;
456 
457  if ((size < 2) || (format == NULL))
458  {
459  return "";
460  }
461 
462  //
463  // Use vsnprintf(), which is made to take in the variable argument list.
464  //
465 
466  va_start(vargs, format);
467  if (vsnprintf(format_str, size, format, vargs) > size)
468  {
469  LogW(kClassName, __func__, "String was truncated during formatting.\n");
470  }
471  va_end(vargs);
472 
473  return format_str;
474 }
