88 Packet* Framer::GenerateConnHndshk(const ConnHndshkHeader& input)
89 {
90  // Get a packet.
91  Packet* packet = packet_pool_.Get();
92 
93  if (packet == NULL)
94  {
95  LogE(kClassName, __func__, "Error getting packet from pool.\n");
96  return NULL;
97  }
98 
99  // Do not exceed the input array size.
100  uint8_t cnt = ((input.num_cc_algs > SliqApp::kMaxCcAlgPerConn) ?
101  SliqApp::kMaxCcAlgPerConn : input.num_cc_algs);
102 
103  // Build the header.
104  if ((!WriteUint8(CONNECTION_HANDSHAKE_HEADER, packet)) ||
105  (!WriteUint8(cnt, packet)) ||
106  (!WriteUint16(input.message_tag, packet)) ||
107  (!WriteUint32(input.timestamp, packet)) ||
108  (!WriteUint32(input.echo_timestamp, packet)))
109  {
110  LogE(kClassName, __func__, "Error generating connection handshake "
111  "header.\n");
112  TRACK_UNEXPECTED_DROP(kClassName, packet_pool_);
113  packet_pool_.Recycle(packet);
114  return NULL;
115  }
116 
117  // Append all of the congestion control algorithm settings.
118  for (uint8_t i = 0; i < cnt; ++i)
119  {
120  // Generate the flags field.
121  uint8_t flags = ((input.cc_alg[i].deterministic_flag ? 0x02 : 0x00) |
122  (input.cc_alg[i].pacing_flag ? 0x01 : 0x00));
123 
124  if ((!WriteUint8(static_cast<uint8_t>(
125  input.cc_alg[i].congestion_control_alg), packet)) ||
126  (!WriteUint8(flags, packet)) ||
127  (!WriteUint16(0, packet)) ||
128  (!WriteUint32(input.cc_alg[i].congestion_control_params, packet)))
129  {
130  LogE(kClassName, __func__, "Error generating connection handshake "
131  "CC params.\n");
132  TRACK_UNEXPECTED_DROP(kClassName, packet_pool_);
133  packet_pool_.Recycle(packet);
134  return NULL;
135  }
136  }
137 
138  // Append the unique client ID.
139  if (!WriteUint32(input.client_id, packet))
140  {
141  LogE(kClassName, __func__, "Error generating connection handshake client "
142  "ID.\n");
143  TRACK_UNEXPECTED_DROP(kClassName, packet_pool_);
144  packet_pool_.Recycle(packet);
145  return NULL;
146  }
147 
148  return packet;
149 }

300 bool Framer::AppendDataHeader(Packet*& packet, const DataHeader& input,
301  size_t payload_length)
302 {
303  // Verify the number of time-to-go (TTG) values first.
304  if (input.num_ttg > kMaxTtgs)
305  {
306  LogE(kClassName, __func__, "Error, invalid number of TTGs %" PRITtgCount
307  ".\n", input.num_ttg);
308  return false;
309  }
310 
311  if (packet == NULL)
312  {
313  packet = packet_pool_.Get();
314 
315  if (packet == NULL)
316  {
317  LogE(kClassName, __func__, "Error getting packet from pool.\n");
318  return false;
319  }
320  }
321 
322  // Generate the flags field.
323  uint8_t flags = ((input.enc_pkt_len_flag ? 0x40 : 0x00) |
324  (input.fec_flag ? 0x20 : 0x00) |
325  (input.move_fwd_flag ? 0x10 : 0x00) |
326  (input.persist_flag ? 0x02 : 0x00) |
327  (input.fin_flag ? 0x01 : 0x00));
328 
329  // Build the header.
330  if ((!WriteUint8(DATA_HEADER, packet)) ||
331  (!WriteUint8(flags, packet)) ||
332  (!WriteUint8(input.stream_id, packet)) ||
333  (!WriteUint8(input.num_ttg, packet)) ||
334  (!WriteUint8(input.cc_id, packet)) ||
335  (!WriteUint8(input.retransmission_count, packet)) ||
336  (!WriteUint16(payload_length, packet)) ||
337  (!WriteUint32(input.sequence_number, packet)) ||
338  (!WriteUint32(input.timestamp, packet)) ||
339  (!WriteUint32(input.timestamp_delta, packet)))
340  {
341  LogE(kClassName, __func__, "Error generating data header common "
342  "fields.\n");
343  return false;
344  }
345 
346  // Append the move forward packet sequence number field if needed.
347  if (input.move_fwd_flag)
348  {
349  if (!WriteUint32(input.move_fwd_seq_num, packet))
350  {
351  LogE(kClassName, __func__, "Error appending move forward sequence "
352  "number.\n");
353  return false;
354  }
355  }
356 
357  // Append the FEC fields if needed.
358  if (input.fec_flag)
359  {
360  uint16_t tmp =
361  (((static_cast<uint16_t>(input.fec_pkt_type) & 0x01) << 15) |
362  ((static_cast<uint16_t>(input.fec_group_index) & 0x3f) << 8) |
363  ((static_cast<uint16_t>(input.fec_num_src) & 0x0f) << 4) |
364  (static_cast<uint16_t>(input.fec_round) & 0x0f));
365 
366  if ((!WriteUint16(tmp, packet)) ||
367  (!WriteUint16(input.fec_group_id, packet)))
368  {
369  LogE(kClassName, __func__, "Error appending FEC fields.\n");
370  return false;
371  }
372  }
373 
374  // Append the encoded packet length field if needed.
375  if (input.enc_pkt_len_flag)
376  {
377  if (!WriteUint16(input.encoded_pkt_length, packet))
378  {
379  LogE(kClassName, __func__, "Error appending encoded packet length "
380  "field.\n");
381  return false;
382  }
383  }
384 
385  // Append the time-to-go (TTG) fields if needed.
386  if (input.num_ttg > 0)
387  {
388  TtgTime ttg = 0;
389 
390  for (TtgCount i = 0; i < input.num_ttg; ++i)
391  {
392  double ttg_sec = input.ttg[i];
393 
394  if (ttg_sec <= 1.0)
395  {
396  if (ttg_sec < 0.0)
397  {
398  ttg_sec = 0.0;
399  }
400 
401  ttg = (static_cast<TtgTime>((ttg_sec * 32767.0) + 0.5));
402  }
403  else
404  {
405  if (ttg_sec > 33.767)
406  {
407  ttg_sec = 33.767;
408  }
409 
410  ttg = (static_cast<TtgTime>(((ttg_sec - 1.0) * 1000.0) + 0.5) |
411  static_cast<TtgTime>(0x8000));
412  }
413 
414  if (!WriteUint16(ttg, packet))
415  {
416  LogE(kClassName, __func__, "Error appending time-to-go.\n");
417  return false;
418  }
419  }
420  }
421 
422  return true;
423 }

636 HeaderType Framer::GetHeaderType(const Packet* packet, size_t offset)
637 {
638  // Read the header type byte from the packet to determine the header type.
639  size_t local_offset = offset;
640  uint8_t type_byte = 0;
641 
642  if (!ReadUint8(packet, local_offset, type_byte))
643  {
644  LogE(kClassName, __func__, "Unable to read type byte from header.\n");
645  return UNKNOWN_HEADER;
646  }
647 
648  // Test the most likely success case first for efficiency.
649  if (((type_byte >= DATA_HEADER) && (type_byte <= CONN_MEAS_HEADER)) ||
650  ((type_byte >= CONNECTION_HANDSHAKE_HEADER) &&
651  (type_byte <= RESET_STREAM_HEADER)) ||
652  (type_byte == CC_PKT_TRAIN_HEADER))
653  {
654  return static_cast<HeaderType>(type_byte);
655  }
656 
657  LogE(kClassName, __func__, "Invalid header type %" PRIu8 ".\n", type_byte);
658  return UNKNOWN_HEADER;
659 }

662 bool Framer::ParseConnHndshkHeader(const Packet* packet, size_t& offset,
663  ConnHndshkHeader& output)
664 {
665  // Skip the packet type byte.
666  offset += 1;
667 
668  // Parse the header.
669  if ((!ReadUint8(packet, offset, output.num_cc_algs)) ||
670  (!ReadUint16(packet, offset, output.message_tag)) ||
671  (!ReadUint32(packet, offset, output.timestamp)) ||
672  (!ReadUint32(packet, offset, output.echo_timestamp)))
673  {
674  LogE(kClassName, __func__, "Error parsing connection handshake "
675  "header.\n");
676  return false;
677  }
678 
679  // Parse all of the congestion control algorithm settings.
680  uint8_t alg_type = 0;
681  uint8_t flags = 0;
682  uint32_t params = 0;
683 
684  for (uint8_t i = 0; i < output.num_cc_algs; ++i)
685  {
686  if ((!ReadUint8(packet, offset, alg_type)) ||
687  (!ReadUint8(packet, offset, flags)))
688  {
689  LogE(kClassName, __func__, "Error parsing connection handshake "
690  "CC type and flags.\n");
691  return false;
692  }
693 
694  // Skip the unused 2 bytes in the middle.
695  offset += 2;
696 
697  if (!ReadUint32(packet, offset, params))
698  {
699  LogE(kClassName, __func__, "Error parsing connection handshake "
700  "CC param.\n");
