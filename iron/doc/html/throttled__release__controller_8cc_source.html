92 void ThrottledReleaseController::SvcEvents(Time& now)
93 {
94  int64_t now_ms = now.GetTimeInMsec();
95  while (release_pkts_queue_.GetCount())
96  {
97  Packet* next_pkt = release_pkts_queue_.Peek();
98 
99  int64_t origin_ts = next_pkt->origin_ts_ms();
100  if (origin_ts <= last_origin_ts_ms_)
101  {
102  origin_ts += origin_rollover_ms_;
103  }
104 
105  if ((origin_ts + traversal_time_) <= now_ms)
106  {
107  Packet* pkt = release_pkts_queue_.Dequeue();
108  if (!pkt)
109  {
110  LogD(kClassName, __func__, "No packets in queue to be released.\n");
111  break;
112  }
113  size_t bytes_sent = decoding_state_.ReleasePkt(pkt);
114 
115  if (bytes_sent == 0)
116  {
117  packet_pool_.Recycle(pkt);
118  break;
119  }
120  LogD(kClassName, __func__, "Bytes sent: %zu, mtt:%" PRId64 "\n",
121  bytes_sent, traversal_time_);
122  }
123  else
124  {
125  break;
126  }
127  }
128 }

131 bool ThrottledReleaseController::HandlePkt(Packet* pkt)
132 {
133 
134  uint16_t origin_ts_ms = pkt->origin_ts_ms();
135  if (origin_ts_ms != iron::kUnsetOriginTs)
136  {
137  Time now = Time::Now();
138  if ((origin_ts_ms < last_origin_ts_ms_) &&
139  (last_origin_ts_ms_ != iron::kUnsetOriginTs))
140  {
141  LogD(kClassName, __func__, "Origin time rollover\n");
142  origin_rollover_ms_ += kTimestampOverflow;
143  }
144 
145  int64_t traversal_time = now.GetTimeInMsec() -
146  (origin_rollover_ms_ + origin_ts_ms);
147  if (tracking_ == MAX_TT)
148  {
149  if (traversal_time > traversal_time_)
150  {
151  traversal_time_ = traversal_time;
152  LogD(kClassName, __func__, "Increase in max transmission time: %"
153  PRId64"\n", traversal_time );
154  }
155  }
156  else if (tracking_ == AVG_TT)
157  {
158  // Track the windowed exponential average of the traversal time.
159  traversal_time_ = (1 - kDefaultCurTtWeight)*traversal_time_ +
160  kDefaultCurTtWeight*traversal_time;
161  }
162  else if (tracking_ == BURST)
163  {
164  if (traversal_time > traversal_time_)
165  {
166  traversal_time_ = traversal_time;
167  LogD(kClassName, __func__, "Increase in max transmission time: %"
168  PRId64"\n", traversal_time );
169  }
170 
171  Packet* next_pkt = release_pkts_queue_.Peek();
172  if (NULL != next_pkt)
173  {
174  uint16_t next_origin_ts_ms = next_pkt->origin_ts_ms();
175  uint32_t current_exit_latency = 0;
176 
177  // Note: if subtracting two unsigned values would result in a
178  // negative number, the result is the negative number modulo MAX+1,
179  // which is the same as "counting backwards" around the ring of
180  // unsigned values. By casting the result to a signed value of
181  // the same size and comparing to 0, we are examining the most
182  // significant digit in the result, which is essentially checking
183  // whether or not the distance (in the ring) is greater
184  // than or equal to 0x8000 (2^15).
185  if ((int16_t)(origin_ts_ms - next_origin_ts_ms) <= 0)
186  {
187  // If the new timestamp is less that the previous timestamp, and
188  // packets are in order, then the timestamp must have overflowed.
189  // The timestamp is 15 bits long, so we add the 16th bit to
190  // account for the overflow.
191  current_exit_latency = kTimestampOverflow + origin_ts_ms -
192  next_origin_ts_ms;
193  }
194  else
195  {
196  current_exit_latency = origin_ts_ms - next_origin_ts_ms;
197  }
198  if (current_exit_latency > kDefaultMaxExitLatencyMs)
199  {
200  // There is a lot of latency added by packets waiting to exit,
201  // so adjust the traversal time such that a burst will be sent to
202  // reduce the latency.
203  int64_t new_traversal_time = traversal_time_/2;
204  LogD(kClassName, __func__, "Current release latency: %" PRIu16
205  "adjusting traversal time from %" PRId64 " to %" PRId64 ".\n",
206  current_exit_latency, traversal_time_, new_traversal_time);
207  traversal_time_ = new_traversal_time;
208  }
209  }
210  }
211  else
212  {
213  LogF(kClassName, __func__, "Undefined traversal tracking technique.\n");
214  }
215  }
216  last_origin_ts_ms_ = origin_ts_ms;
217 
218  return release_pkts_queue_.Enqueue(pkt);
219 }
