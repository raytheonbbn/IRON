113 void SocketMgr::AddSocket(Socket* sock)
114 {
115  FourTuple ft(sock->his_addr().s_addr,
116  sock->his_port(),
117  sock->my_addr().s_addr,
118  sock->my_port());
119 
120  if (!sockmap_.Insert(ft, sock))
121  {
122  LogE(kClassName, __func__, "Error adding socket for four-tuple %s.\n",
123  ft.ToString().c_str());
124  }
125 
126  // Add the socket to the socket_list_.
127  sock->set_next(socket_list_);
128  sock->set_prev(NULL);
129  if (socket_list_ != NULL)
130  {
131  socket_list_->set_prev(sock);
132  }
133 
134  socket_list_ = sock;
135 }

139 {
140  // Invoke the close() method on the socket.
141  if (!sock->Close())
142  {
143  return;
144  }
145 
146  // Remove the socket from the collection of sockets if the socket's state is
147  // suitable for the removal and destruction of the socket.
148  if ((sock->state() == TCP_LISTEN) || (sock->state() == TCP_NASCENT) ||
149  (sock->state() == TCP_SYN_SENT))
150  {
151  RemoveSocket(sock);
152  }

174 void SocketMgr::RemoveSocket(Socket* s)
175 {
176  LogI(kClassName, __func__, "%s, removing socket.\n", s->flow_id_str());
177 
178  // Remove the socket from the socket map.
179  FourTuple ft(s->his_addr().s_addr,
180  s->his_port(),
181  s->my_addr().s_addr,
182  s->my_port());
183 
184  // Erase the entry for the four-tuple in the hash table.
185  if (sockmap_.Erase(ft) < 1)
186  {
187  LogE(kClassName, __func__, "Error finding socket in hash table.\n");
188  }
189 
190  // Remove the socket from the socket_list_ also.
191  Socket* iter = socket_list_;
192  while (iter != NULL)
193  {
194  if ((s->his_addr().s_addr == iter->his_addr().s_addr) &&
195  (s->his_port() == iter->his_port()) &&
196  (s->my_addr().s_addr == iter->my_addr().s_addr) &&
197  (s->my_port() == iter->my_port()))
198  {
199  if (iter->next() != NULL)
200  {
201  iter->next()->set_prev(iter->prev());
202  }
203 
204  if (iter->prev() != NULL)
205  {
206  iter->prev()->set_next(iter->next());
207  }
208 
209  if (iter == socket_list_)
210  {
211  socket_list_ = iter->next();
212  }
213 
214  break;
215  }
216 
217  iter = iter->next();
218  }
219 
220  // Now, we can destroy the socket.
221  delete s;
222 }

297 Socket* SocketMgr::GetSocket(FourTuple& four_tuple)
298 {
299  // This method finds a socket that matches the provided 4-tuple. This
300  // 4-tuple is normally received from the Admission Planner and should match
301  // a LAN side socket. The comparison depends on whether the current socket
302  // is active or passive. See the class level comment in socket.h for an
303  // example that illustrates why the 4-tuple matches are tested as they are
304  // below.
305 
306  LogD(kClassName, __func__, "Target 4-tuple: (%" PRIu32
307  ", %" PRIu16 ", %" PRIu32 ", %" PRIu16 ").\n",
308  four_tuple.src_addr_nbo(),four_tuple.src_port_nbo(),
309  four_tuple.dst_addr_nbo(),four_tuple.dst_port_nbo());
310 
311  Socket* sock = NULL;
312 
313  if (sockmap_.Find(four_tuple, sock))
314  {
315  if ((sock != NULL) && (sock->is_active()))
316  {
317  LogW(kClassName, __func__, "Socket 4-tuple: (%" PRIu32
318  ", %" PRIu16 ", %" PRIu32 ", %" PRIu16 ").\n", sock->my_addr(),
319  ntohs(sock->my_port()), sock->his_addr(), ntohs(sock->his_port()));
320 
321  // We have a match.
322  LogW(kClassName, __func__, "Found matching socket.\n");
323 
324  return sock;
325  }
326  }
327 
328  // Swap the source and destination info and search again
329  FourTuple flipped_four_tuple (four_tuple.dst_addr_nbo(),
330  four_tuple.dst_port_nbo(),
331  four_tuple.src_addr_nbo(),
332  four_tuple.src_port_nbo());
333 
334  if (sockmap_.Find(flipped_four_tuple, sock))
335  {
336  LogW(kClassName, __func__, "Socket 4-tuple: (%" PRIu32
337  ", %" PRIu16 ", %" PRIu32 ", %" PRIu16 ").\n", sock->my_addr(),
338  ntohs(sock->my_port()), sock->his_addr(), ntohs(sock->his_port()));
339 
340  if (sock->is_active())
341  {
342  // We have a match
343  LogW(kClassName, __func__, "Found matching socket.\n");
344 
345  return sock;
346  }
347  }
348 
349  // No match was found.
350  LogW(kClassName, __func__, "No matching socket found.\n");
351 
352  return NULL;
353 }

356 void SocketMgr::ProcessSvcDefUpdate(const TcpContext* tcp_context)
357 {
358  HashTable<FourTuple, Socket*>::WalkState walk_state;
359  FourTuple four_tuple;
360  Socket* sock = NULL;
361 
362  while (sockmap_.GetNextPair(walk_state, four_tuple, sock))
363  {
364  if ((sock != NULL) && (sock->cfg_if_id() == WAN))
365  {
366  if (sock->is_active())
367  {
368  if (sock->peer() == NULL)
369  {
370  continue;
371  }
372 
373  if ((ntohs(sock->peer()->my_port()) >= tcp_context->lo_port()) &&
374  (ntohs(sock->peer()->my_port()) <= tcp_context->hi_port()))
375  {
376  // The current socket ports fall within the range of the context that
377  // has been modified. Figure out if the utility function definition
378  // needs to be updated. It will be if there is not an existing Flow
379  // level utility function defined. See the class level documentation
380  // in socket.h for an example that illustrates why the 4-tuple is
381  // initialized this way.
382  FourTuple four_tuple(sock->peer()->his_addr().s_addr,
383  sock->peer()->his_port(),
384  sock->peer()->my_addr().s_addr,
385  sock->peer()->my_port());
386 
387  if (tcp_proxy_->HasFlowUtilityFnDef(four_tuple))
388  {
389  // There is an existing Flow definition for the current socket. This
390  // takes precedence over any Service level utility function
391  // definition. So, don't make any changes to the current socket.
392  LogW(kClassName, __func__, "Socket with port %" PRIu16 " has an "
393  "active flow definition. Not modifying.\n",
394  ntohs(sock->peer()->my_port()));
395 
396  continue;
397  }
398 
399  LogW(kClassName, __func__, "Applying new Service Definition update to "
400  "port(%" PRIu16 ").\n", ntohs(sock->peer()->my_port()));
401 
402  // There is no Flow definition for the current socket so update its
403  // utility function definition (as the Service level definition has
404  // changed).
405  sock->ResetUtilityFn(tcp_context->util_fn_defn(),
406  tcp_proxy_->GetQueueDepths());
407  }
408  }
409  else
410  {
411  if ((ntohs(sock->peer()->his_port()) >= tcp_context->lo_port()) &&
412  (ntohs(sock->peer()->his_port()) <= tcp_context->hi_port()))
413  {
414  // The current socket ports fall within the range of the context that
415  // has been modified. Figure out if the utility function definition
416  // needs to be updated. It will be if there is not an existing Flow
417  // level utility function defined. See the class level documentation
418  // in socket.h for an example that illustrates why the 4-tuple is
419  // initialized this way.
420  FourTuple four_tuple(sock->peer()->my_addr().s_addr,
421  sock->peer()->my_port(),
422  sock->peer()->his_addr().s_addr,
423  sock->peer()->his_port());
424 
425  if (tcp_proxy_->HasFlowUtilityFnDef(four_tuple))
426  {
427  // There is an existing Flow definition for the current socket. This
428  // takes precedence over any Service level utility function
429  // definition. So, don't make any changes to the current socket.
430  LogW(kClassName, __func__, "Socket with port %" PRIu16 " has an "
431  "active flow definition. Not modifying.\n",
432  ntohs(sock->peer()->his_port()));
433 
434  continue;
435  }
436 
437  LogW(kClassName, __func__, "Applying new Service Definition update to "
438  "port(%" PRIu16 ").\n", ntohs(sock->peer()->his_port()));
439 
440  // There is no Flow definition for the current socket so update its
441  // utility function definition (as the Service level definition has
442  // changed).
443  sock->peer()->ResetUtilityFn(tcp_context->util_fn_defn(),
444  tcp_proxy_->GetQueueDepths());
445  }
446  }
447  }
448  }

452 void SocketMgr::WriteStats(Writer<StringBuffer>* writer)
453 {
454  // Stats "keyvals" format.
455  // "stats" :
456  // {
457  // "Flows" :
458  // [
459  // {
460  // "flow_id" : "xxx.xxx.xxx.xxx:aaaaa -> yyy.yyy.yyy.yyy:bbbb",
461  // "priority" : xx.xx,
462  // "cumulative_sent_pkt_cnt" : xx,
463  // "cumulative_sent_bytes_cnt" : xx,
464  // "send_rate_bps" : xx.xx,
465  // "send_rate_pps" : xx.xx,
466  // "cumulative_rcvd_pkt_cnt" : xx,
467  // "cumulative_rcvd_bytes_cnt" : xx,
468  // "recv_rate_bps" : xx.xx,
469  // "recv_rate_pps" : xx.xx,
470  // "ave_instantaneous_utility" : xx.xx
471  // },
472  // ],
473  // "NumActiveFlows" : xx,
474  // "CumulativeAveInstantaneousUtility": xx.xx,
475  // "CumulativeAggregateUtility": xx.xx
476  // }
477 
478  bool log_stats = tcp_proxy_->log_stats();
479  if (log_stats)
480  {
481  LogI(kClassName, __func__, "-- TCP Proxy Stats --------------\n");
482  }
483 
484  if (writer)
485  {
486  // Append "stats" : {
487  writer->Key("stats");
488  writer->StartObject();
489  }
490 
491  string log_str;
492 
493  // Append the "Flows" statistics. This will be an array of objects
494  // containing key/value pairs for each active flow.
495  if (log_stats)
496  {
497  log_str.append("FlowStats=");
498  }
499 
500  if (writer)
501  {
502  // Append "Flows" : [
503  writer->Key("Flows");
504  writer->StartArray();
505  }
506 
507  bool first = true;
508  double cumulative_ave_utility = 0.0;
509  double cumulative_aggregate_utility = 0.0;
510  uint8_t active_flow_cnt = 0;
511 
512  HashTable<FourTuple, Socket*>::WalkState walk_state;
513  FourTuple four_tuple;
514  Socket* s = NULL;
515 
516  while (sockmap_.GetNextPair(walk_state, four_tuple, s))
517  {
518  if ((s != NULL) && (s->cfg_if_id() == WAN) &&
519  ((s->peer()) && (s->peer()->peer()) && (s->peer()->peer() == s)))
520  {
521  active_flow_cnt++;
522 
523  if (writer)
524  {
525  // Start the current flow with the '{' character.
526  writer->StartObject();
527  }
528 
529  if (first)
530  {
531  first = false;
532  }
533  else
534  {
535  log_str.append(",");
536  }
537 
538  s->WriteStats(log_str, writer);
539 
540  if (writer)
541  {
542  // End the current flow with the '}' character.
543  writer->EndObject();
544  }
545 
546  cumulative_ave_utility += s->ave_utility();
547  cumulative_aggregate_utility += s->cumulative_utility();
548  }
549  }
550 
551  if (log_stats)
552  {
553  LogI(kClassName, __func__, "%s\n", log_str.c_str());
554  }
555 
556  if (writer)
557  {
558  // End the "flows" array with the ']' character.
559  writer->EndArray();
560  }
561 
562  // Log the NumActiveFlows statistic, if required.
563  if (log_stats)
564  {
565  log_str.clear();
566  log_str.append(
567  StringUtils::FormatString(256, "NumActiveFlows=%" PRIu8,
568  active_flow_cnt));
569 
570  LogI(kClassName, __func__, "%s\n", log_str.c_str());
571  }
572 
573  // Append the "NumActiveFlows" statistic, if required.
574  if (writer)
575  {
576  // Append "NumActiveFlows" : xx
577  writer->Key("NumActiveFlows");
578  writer->Uint(active_flow_cnt);
579  }
580 
581  // Log the CumulativeAveInstantaneousUtility statistic, if required.
582  if (log_stats)
583  {
584  log_str.clear();
585  log_str.append("AggStats=");
586  log_str.append(
587  StringUtils::FormatString(256, "'CumulativeUtility':'%f',",
588  cumulative_ave_utility));
589  }
590 
591  // Append the "CumulativeAveInstantaneousUtility" statistic, if required.
592  if (writer)
593  {
594  // Append "CumulativeAveInstantaneousUtility" : xx.xx
595  writer->Key("CumulativeAveInstantaneousUtility");
596  writer->Double(cumulative_ave_utility);
597  }
598 
599  // Log the CumulativeAggregateUtility statistic, if required.
600  if (log_stats)
601  {
602  log_str.append(
603  StringUtils::FormatString(256, "'HistoricAggregateUtility':'%f'",
604  cumulative_aggregate_utility));
605 
606  LogI(kClassName, __func__, "%s\n", log_str.c_str());
607  }
608 
609  // Append the "CumulativeAggregateUtility" statistic, if required.
610  if (writer)
611  {
612  // Append "CumulativeAggregateUtility" : xx.xx
613  writer->Key("CumulativeAggregateUtility");
614  writer->Double(cumulative_aggregate_utility);
615 
616  // End keyvals object with '}' character.
617  writer->EndObject();
618  }
619 
620  if (log_stats)
621  {
622  LogI(kClassName, __func__, "-------------- TCP Proxy Stats --\n");
623  }
624 }
