75 bool LogAdmissionController::CreateUtilityFn(string& utility_def,
76  uint32_t flow_id,
77  QueueDepths& queue_depths)
78 {
79  // Validate that the provided utility definition string is for the Log
80  // utility.
81  if (GetUtilityFnType(utility_def, flow_id) != "LOG")
82  {
83  LogW(kClassName, __func__, "fid: %" PRIu32 ", utility definition is not "
84  "for a Log utility function.\n", flow_id);
85  return false;
86  }
87 
88  log_utility_ = new (std::nothrow) LogUtility(queue_depths,
89  encoding_state_.bin_idx(),
90  encoding_state_.k_val(),
91  flow_id);
92  if (log_utility_ == NULL)
93  {
94  LogF(kClassName, __func__, "fid: %" PRIu32 "Unable to allocate memory "
95  "for log utility function.\n", flow_id);
96  return false;
97  }
98 
99  ConfigureUtilityFn(log_utility_, utility_def);
100 
101  return true;
102 }

122 void LogAdmissionController::set_flow_state(FlowState flow_state)
123 {
124  if (flow_state == iron::FLOW_TRIAGED)
125  {
126  LogW(kClassName, __func__, "Unable to triage a log utility flow.\n");
127  return;
128  }
129 
130  if (log_utility_ == NULL)
131  {
132  // If we get here and don't have a utility function something is very very
133  // wrong.
134  LogF(kClassName, __func__, "Attempting to set the flow's state and there "
135  "isn't a utility function.\n");
136  return;
137  }
138 
139  if (flow_state == iron::FLOW_OFF)
140  {
141  CancelScheduledEvent(next_admission_time_);
142  }
143 
144  log_utility_->set_flow_state(flow_state);
145 }

170 void LogAdmissionController::UpdateUtilityFn(std::string key_val)
171 {
172  List<string> tokens;
173  StringUtils::Tokenize(key_val, ":", tokens);
174  if (tokens.size() != 2)
175  {
176  LogE(kClassName, __func__, "Parameter %s must be of the form key:value.\n",
177  key_val.c_str());
178  return;
179  }
180 
181  string key;
182  string value;
183  tokens.Peek(key);
184  tokens.PeekBack(value);
185 
186  if (key == "p")
187  {
188  log_utility_->set_priority(StringUtils::GetDouble(value,0.0));
189  }
190  else
191  {
192  LogE(kClassName, __func__, "Update of %s not supported.\n", key.c_str());
193  }
194 }
