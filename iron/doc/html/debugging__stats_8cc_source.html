81 void DebuggingStats::TrackStat(const char* name, uint64_t value)
82 {
83  uint16_t index = GetStatIndex(name, false);
84  if (recent_vals_[index].done_)
85  {
86  return;
87  }
88  Time now = Time::Now();
89  // First check whether it's time to record the next bundle.
90  // If we haven't set the period start time yet, then it's not time to move
91  // on - this allows us to use item 0 (otherwise we'd skip over it, since it
92  // never had it's start time set).
93  if (!recent_vals_[index].period_start_time_.IsZero()
94  && now > recent_vals_[index].period_start_time_ + kStatInterval)
95  {
96  if (stats_[index].num_items_== kMaxItems)
97  {
98  recent_vals_[index].done_ = true;
99  return;
100  }
101  else
102  {
103  uint16_t item = stats_[index].num_items_;
104  stats_[index].num_items_++;
105  stats_[index].items_[item].set_vals(recent_vals_[index]);
106  recent_vals_[index].Clear(now);
107  }
108  }
109  if (recent_vals_[index].period_start_time_.IsZero())
110  {
111  recent_vals_[index].period_start_time_ = now;
112  }
113  recent_vals_[index].total_ += value;
114  recent_vals_[index].count_++;
115  if (value < recent_vals_[index].min_)
116  {
117  recent_vals_[index].min_ = value;
118  }
119  if (value > recent_vals_[index].max_)
120  {
121  recent_vals_[index].max_ = value;
122  }
123 }

126 void DebuggingStats::CountOccurrences(const char* name, Time period)
127 {
128  uint16_t index = GetStatIndex(name, false);
129  if (recent_vals_[index].done_)
130  {
131  return;
132  }
133 
134  recent_vals_[index].count_ = 1;
135  Time now = Time::Now();
136  // First check whether it's time to record the next bundle.
137  // If we haven't set the period start time yet, then it's not time to move
138  // on - this allows us to use item 0 (otherwise we'd skip over it, since it
139  // never had it's start time set).
140  if (!recent_vals_[index].period_start_time_.IsZero()
141  && now > recent_vals_[index].period_start_time_ + period)
142  {
143  if (stats_[index].num_items_== kMaxItems)
144  {
145  recent_vals_[index].done_ = true;
146  return;
147  }
148  else
149  {
150  uint16_t item = stats_[index].num_items_;
151  stats_[index].num_items_++;
152  stats_[index].items_[item].set_vals(recent_vals_[index]);
153  recent_vals_[index].Clear(now);
154  recent_vals_[index].min_ = 0;
155  }
156  }
157  if (recent_vals_[index].period_start_time_.IsZero())
158  {
159  recent_vals_[index].period_start_time_ = now;
160  }
161  recent_vals_[index].total_++;
162  recent_vals_[index].min_++;
163  recent_vals_[index].max_++;
164 }

167 void DebuggingStats::TrackInstant(const char* name, uint64_t value,
168  uint64_t collection_len_usec)
169 {
170  uint16_t index = GetStatIndex(name, true);
171  uint64_t now_usec = Time::GetNowInUsec();
172  if (instant_stats_[index].start_time_usec_ == 0)
173  {
174  instant_stats_[index].start_time_usec_ = now_usec;
175  instant_stats_[index].collection_usec_ = collection_len_usec;
176  }
177 
178  // collection length of 0 implies just fill the values.
179  if (collection_len_usec == 0 &&
180  instant_stats_[index].num_instants_ == kMaxVals)
181  {
182  // Done collecting this stat.
183  return;
184  }
185 
186  if (collection_len_usec > 0 &&
187  (now_usec - instant_stats_[index].start_time_usec_ >
188  instant_stats_[index].collection_usec_))
189  {
190  // Done collecting this stat.
191  return;
192  }
193 
194  uint16_t num_instants = instant_stats_[index].num_instants_;
195 
196  // Now find the current instant to write or replace.
197  num_instants = num_instants % kMaxVals;
198  instant_stats_[index].times_[num_instants] = now_usec;
199  instant_stats_[index].values_[num_instants] = value;
200  instant_stats_[index].num_instants_++;
201 }

213 void DebuggingStats::LogStats()
214 {
215  for (uint16_t index = 0; index < num_instant_stats_; index++)
216  {
217  const char* UNUSED(name) = instant_stats_[index].name_.c_str();
218  for (uint16_t val = 0;
219  val < instant_stats_[index].num_instants_ && val < kMaxVals;
220  val++)
221  {
222  LogA(kClassName, __func__,
223  "%s, %" PRIu64 ", val: %" PRIu64 "\n",
224  name,
225  instant_stats_[index].times_[val],
226  instant_stats_[index].values_[val]);
227  }
228  }
229  for (uint16_t index = 0; index < num_stats_; index++)
230  {
231  uint64_t total = 0.;
232  const char* UNUSED(name) = stats_[index].name_.c_str();
233 
234  stats_[index].items_[stats_[index].num_items_].set_vals(recent_vals_[index]);
235  stats_[index].num_items_++;
236  for (uint16_t item = 0; item < stats_[index].num_items_; item++)
237  {
238  total += stats_[index].items_[item].max_;
239  LogA(kClassName, __func__,
240  "%s, %" PRIu64 ", avg: %f, min: %" PRIu64 ", max: %" PRIu64 "\n",
241  name,
242  stats_[index].items_[item].start_time_usec_,
243  stats_[index].items_[item].average_,
244  stats_[index].items_[item].min_,
245  stats_[index].items_[item].max_);
246  }
247  if (recent_vals_[index].count_ == 1)
248  {
249  LogA(kClassName, __func__,
250  "%s, Num: %" PRIu64 ".\n", name, total);
251  }
252  }
253 }

256 uint16_t DebuggingStats::GetStatIndex(const char* name, bool instant)
257 {
258  string sname = string(name);
259  uint16_t index = kMaxStats - 1;
260  std::map<string, uint16_t>::iterator it =
261  stat_index_map_.find(sname);
262 
263  if (it != stat_index_map_.end())
264  {
265  index = (*it).second;
266  }
267  else if (instant)
268  {
269  if (num_instant_stats_ == kMaxStats)
270  {
271  LogW(kClassName, __func__, "Too many instant statistics (%" PRIu16 ") "
272  "Trying to add %s.\n", num_instant_stats_, name);
273  }
274  else
275  {
276  index = num_instant_stats_;
277  ++num_instant_stats_;
278  stat_index_map_[sname] = index;
279  instant_stats_[index].name_ = sname;
280  LogD(kClassName, __func__, "Instant stats index %" PRIu16
281  " maps to %s.\n", index, name);
282  }
283  }
284  else
285  {
286  if (num_stats_ == kMaxStats)
287  {
288  LogW(kClassName, __func__, "Too many amortized statistics (%" PRIu16 ") "
289  "Trying to add %s.\n", num_stats_, name);
290  }
291  else
292  {
293  index = num_stats_;
294  ++num_stats_;
295  stat_index_map_[sname] = index;
296  stats_[index].name_ = sname;
297  LogD(kClassName, __func__, "Stats index %" PRIu16 " maps to %s.\n",
298  index, name);
299  }
300  }
301  return index;
302 }
