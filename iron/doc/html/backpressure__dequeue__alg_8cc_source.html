542 uint8_t BPDequeueAlg::FindNextTransmission(TxSolution* solutions,
543  uint8_t max_num_solutions)
544 {
545  if (!initialized_)
546  {
547  LogE(kClassName, __func__, "Uber BPF alg is not initialized. Cannot "
548  "compute next transmission opportunity.\n");
549  return 0;
550  }
551 
552  // This optimization isn't explicitly called out in the packet forwarder
553  // description in the BPF design documentation. If there are NO packets in
554  // any queue, skip all ops. Avoid computing latency stats and finding
555  // solutions when there are no queued packets.
556  if (queue_store_->AreQueuesEmpty())
557  {
558  LogD(kClassName, __func__, "All queues empty, no available transmit "
559  "opportunity.\n");
560 
561  BinIndex idx = 0;
562  for (bool idx_valid = bin_map_.GetFirstUcastBinIndex(idx);
563  idx_valid;
564  idx_valid = bin_map_.GetNextUcastBinIndex(idx))
565  {
566  // A packet coming to this queue would experience no queuing delay. Add
567  // 0 to the average.
568  AddDelayToAverage(0, idx);
569  }
570 
571  return 0;
572  }
573 
574  // Steps S1 and S2 from the packet forwarder description in the BPF design
575  // documentation, clean up expired packets and criticalize packets.
576  Time now = Time::Now();
577  int32_t path_ctrl_q_sizes[kMaxPathCtrls];
578  TransmitCandidate critical_candidate;
579  if (!ZombifyAndCriticalizePkts(now, kMaxPathCtrls, path_ctrl_q_sizes,
580  critical_candidate))
581  {
582  return 0;
583  }
584 
585  // Now print all mcast bins, so we have a full list in the log.
586  //
587  // Question: Why is this done here?
588  if (WouldLogD(kClassName))
589  {
590  BinIndex dst_bin_idx = 0;
591  for (bool dst_bin_idx_valid = bin_map_.GetFirstMcastBinIndex(dst_bin_idx);
592  dst_bin_idx_valid;
593  dst_bin_idx_valid = bin_map_.GetNextMcastBinIndex(dst_bin_idx))
594  {
595  queue_store_->GetBinQueueMgr(dst_bin_idx)->Print();
596  }
597  }
598 
599  // Step S3 from the packet forwarder description in the BPF design
600  // documentation, handle critical packets with latency. This is simple
601  // enough that we won't make a separate method call.
602  //
603  // If there is a critical candidate, send it.
604  //
605  // MCAST TODO: If there is a critical candidate, we know it's
606  // unicast. Handle multicast case in the future.
607  uint8_t num_solutions = 0;
608 
609  if (critical_candidate.is_valid && critical_candidate.pkt)
610  {
611  solutions[0].pkt = critical_candidate.q_mgr->DequeueAtIterator(
612  CRITICAL_LATENCY, critical_candidate.dequeue_loc);
613  solutions[0].bin_idx = critical_candidate.bin_idx;
614  solutions[0].path_ctrl_index = critical_candidate.path_ctrl_index;;
615  num_solutions = 1;
616 
617  if (solutions[0].pkt->HasQueuingDelay())
618  {
619  AddDelayToAverage(Time::GetNowInUsec() -
620  solutions[0].pkt->recv_time().GetTimeInUsec(), solutions[0].bin_idx);
621  }
622 
623  LogD(kClassName, __func__, "Selected immediate release of candidate "
624  "%s.\n", critical_candidate.ToString().c_str());
625 
626  return num_solutions;
627  } // END sending critical candidate.
628 
629  // Only packets from the Critical queue can be selected without considering
630  // gradients. Selection from all other queues requires using the
631  // backpressure gradient.
632 
633  // Keep the gradients ordered.
634  OrderedList<Gradient, int64_t> ls_gradients(iron::LIST_DECREASING);
635  OrderedList<Gradient, int64_t> gradients(iron::LIST_DECREASING);
636 
637  // Get the queue depth of the queues for the priority types. If zero, do not
638  // attempt to find a packet for the gradient (during LS processing).
639  has_prio_ttypes_.Clear(false);
640 
641  // Step S4 from the packet forwarder description in the BPF design
642  // documentation, construct forwarding gradients.
643  ComputeGradients(path_ctrl_q_sizes, gradients, ls_gradients);
644 
645  // Provide BinQueueMgr gradient info to help with addressing starvation.
646  queue_store_->ProcessGradientUpdate(ls_gradients, gradients);
647 
648  // Step S5 from the packet forwarder description in the BPF design
649  // documentation, find latency-sensitive gradient-based packets.
650  OrderedList<Gradient, int64_t>* ef_gradients = NULL;
651  if (enable_hierarchical_fwding_)
652  {
653  ef_gradients = &ls_gradients;
654  }
655  else
656  {
657  ef_gradients = &gradients;
658  }
659 
660  FindLatencySensitivePkts(now, ef_gradients, path_ctrl_q_sizes,
661  max_num_solutions, solutions, num_solutions);
662  if (num_solutions > 0)
663  {
664  return num_solutions;
665  }
666 
667  // Step S6 from the packet forwarder description in the BPF design
668  // documentation, find normal (latency-insensitive) gradient-based packets.
669  LogD(kClassName, __func__, "Did not find candidate for priority dequeue "
670  "traffic types.\n");
671 
672  FindLatencyInsensitivePkts(now, gradients, path_ctrl_q_sizes,
673  max_num_solutions, solutions, num_solutions);
674 
675  if (num_solutions == 0)
676  {
677  LogD(kClassName, __func__, "Found no solution, nothing dequeued.\n");
678  }
679 
680  return num_solutions;
681 }

683 //============================================================================
684 void BPDequeueAlg::ComputeOneBinGradient(
685  BinIndex bin, PathController* path_ctrl,
686  QueueDepths* my_qd_for_bin, QueueDepths* nbr_qd_for_bin,
687  QueueDepths* my_v_queue_depth, QueueDepths* nbr_v_queue_depth,
688  bool& is_dst, int64_t& differential, int64_t& ls_differential)
689 {
690  uint32_t my_qd = my_qd_for_bin->GetBinDepthByIdx(bin);
691  uint32_t my_ls = my_qd_for_bin->GetBinDepthByIdx(bin, LOW_LATENCY);
692  uint32_t my_v_queue_len = 0;
693  uint32_t nbr_v_queue_len = 0;
694 
695  if (my_v_queue_depth)
696  {
697  my_v_queue_len = my_v_queue_depth->GetBinDepthByIdx(bin);
698  }
699 
700  // The virtual queue length may be set to UINT32_MAX, which identifies
701  // an unreachable node (i.e., it requires an "infinite" number of hops)
702  // If so, set the differential to zero so that this destination isn't
703  // selected
704 
705  if (my_v_queue_len == UINT32_MAX)
706  {
707  differential = 0;
708  ls_differential = 0;
709  }
710  else
711  {
712  differential = my_qd + my_v_queue_len;
713  ls_differential = my_ls + my_v_queue_len;
714  }
715 
716  LogD(kClassName, __func__,
717  "My queue depth to %s is %" PRIu32 "B (%" PRIu32 "B LS), virtual queue "
718  "depth %" PRIu32 "B.\n",
719  bin_map_.GetIdToLog(bin).c_str(), my_qd, my_ls, my_v_queue_len);
720 
721  if (!nbr_qd_for_bin)
722  {
723  LogF(kClassName, __func__,
724  "No queue depth for bin %s on path to %" PRIBinId ".\n",
725  bin_map_.GetIdToLog(bin).c_str(),
726  path_ctrl->remote_bin_id());
727  return;
728  }
729 
730  // Check if the neighbor happens to be the bin destination: in that
731  // case, this neighbor has an implicit queue size of 0 on that bin.
732  // TODO: use path ctrl index to get path ctrl
733  if (path_ctrl->remote_bin_idx() == bin)
734  {
735  is_dst = true;
736  // This neighbor is the bin's destination! Woohoo it will take it all!
737  LogD(kClassName, __func__,
738  "Nbr %" PRIBinId " is the bin Id %s destination - Q len is 0.\n",
739  path_ctrl->remote_bin_id(), bin_map_.GetIdToLog(bin).c_str());
740  }
741  else
742  {
743  is_dst = false;
744  uint32_t nbr_qd = nbr_qd_for_bin->GetBinDepthByIdx(bin, NORMAL_LATENCY);
745  uint32_t nbr_ls = nbr_qd_for_bin->GetBinDepthByIdx(bin, LOW_LATENCY);
746  if (nbr_v_queue_depth)
747  {
748  nbr_v_queue_len = nbr_v_queue_depth->GetBinDepthByIdx(bin);
749  }
750  LogD(kClassName, __func__,
751  "Nbr has a bin %s depth of %" PRIu32 "B (%" PRIu32 "B LS), virtual queue"
752  " depth %" PRIu32 "B.\n",
753  bin_map_.GetIdToLog(bin).c_str(), nbr_qd, nbr_ls, nbr_v_queue_len);
754 
755  // The virtual queue length may be set to UINT32_MAX, which identifies
756  // a "deadend" that we don't want to use for this destination
757  // If so, set the differential to a large negative number.
758  if (nbr_v_queue_len == UINT32_MAX)
759  {
760  differential = INT32_MIN;
761  ls_differential = INT32_MIN;
762  }
763  else
764  {
765  differential -= nbr_qd + nbr_v_queue_len;
766  ls_differential -= nbr_ls + nbr_v_queue_len;
767  }
768  }
769 
770  LogD(kClassName, __func__, "Gradient differential is %" PRId64
771  "B, LS differential is %" PRId64 "B.\n", differential,
772  ls_differential);
773 }

775 //============================================================================
776 void BPDequeueAlg::ComputeMulticastGradient(
777  PathController* path_ctrl,
778  QueueDepths* my_qd_for_bin, QueueDepths* nbr_qd_for_bin,
779  QueueDepths* my_v_queue_depth, QueueDepths* nbr_v_queue_depth,
780  Gradient& gradient, Gradient& ls_gradient)
781 {
782  DstVec UNUSED(dst_vec) = bin_map_.GetMcastDst(gradient.bin_idx);
783  LogD(kClassName, __func__, "========================================\n");
784  LogD(kClassName, __func__, "Computing multicast gradient for bin %s with "
785  "nbr %" PRIBinId " and dsts 0x%X.\n",
786  bin_map_.GetIdToLog(gradient.bin_idx).c_str(),
787  path_ctrl->remote_bin_id(), static_cast<unsigned int>(dst_vec));
788 
789  // Set value to 0 in case this is called without initializing gradient.
790  gradient.value = 0;
791  ls_gradient.value = 0;
792  Gradient zombie_gradient;
793  Gradient zombie_ls_gradient;
794  zombie_gradient.value = 0;
795  zombie_ls_gradient.value = 0;
796  BinQueueMgr* q_mgr = queue_store_->GetBinQueueMgr(gradient.bin_idx);
797 
798  // Skim through all unicast destination bins and compute gradient for each.
799  // For efficiency, don't even bother looking at bins that aren't
800  // destinations for this multicast group.
801  DstVec send_to = 0;
802  DstVec ls_send_to = 0;
803  DstVec zombie_send_to = 0;
804  DstVec zombie_ls_send_to = 0;
805  BinIndex dst_idx = 0;
806 
807  for (bool valid = bin_map_.GetFirstUcastBinIndex(dst_idx);
808  valid;
809  valid = bin_map_.GetNextUcastBinIndex(dst_idx))
810  {
811  // NOTE: This iteration through the destination vector could include the
812  // node itself, so be mindful of that when computing gradients.
813  if (my_qd_for_bin->GetBinDepthByIdx(dst_idx) > 0)
814  {
815  LogD(kClassName, __func__,
816  "Including bin %" PRIBinId " (index %" PRIBinIndex ") in mcast "
817  "gradient, because it has a non-zero queue depth %" PRIu32 "B.\n",
818  bin_map_.GetPhyBinId(dst_idx), dst_idx,
819  my_qd_for_bin->GetBinDepthByIdx(dst_idx));
820  bool is_dst = false;
821  int64_t differential = 0;
822  int64_t ls_differential = 0;
823  ComputeOneBinGradient(dst_idx, path_ctrl,
824  my_qd_for_bin, nbr_qd_for_bin,
825  my_v_queue_depth, nbr_v_queue_depth,
826  is_dst, differential, ls_differential);
827 
828  // Gradients are the sum of the positive per-destination gradients.
829  // We only want to send to destinations that had a positive
830  // per-destination gradient.
831  if (differential > static_cast<int64_t>(is_dst ? 0 : hysteresis_))
832  {
833  // Keep track of the zombie and non-zombie gradients separately.
834  if (q_mgr->non_zombie_queue_depth_bytes(dst_idx) > 0)
835  {
836  gradient.value += differential;
837  send_to = bin_map_.AddBinToDstVec(send_to,
838  dst_idx);
839  LogD(kClassName, __func__,
840  "With differential %" PRId64 "B (hysteresis %uB), adding bin "
841  "index %" PRIBinIndex " to dst vec, now 0x%X.\n",
842  differential, hysteresis_, dst_idx, send_to);
843  }
844  else
845  {
846  zombie_gradient.value += differential;
847  zombie_send_to = bin_map_.AddBinToDstVec(zombie_send_to,
848  dst_idx);
849  LogD(kClassName, __func__,
850  "Zombie With differential %" PRId64 "B (hysteresis %uB), adding bin "
851  "index %" PRIBinIndex " to dst vec, now 0x%X.\n",
852  differential, hysteresis_, dst_idx, zombie_send_to);
853  }
854  mcast_gradients_[dst_idx] = differential;
855  }
856  else
857  {
858  LogD(kClassName, __func__,
859  "Differential %" PRId64 "B is below hysteresis %uB or is dst, not"
860  " adding.\n",
861  differential, hysteresis_);
862  mcast_gradients_[dst_idx] = 0;
863  }
864  if (ls_differential > static_cast<int64_t>(is_dst ? 0 : hysteresis_))
865  {
866  if (q_mgr->non_zombie_queue_depth_bytes(dst_idx) > 0)
867  {
868  ls_gradient.value += ls_differential;
869  ls_send_to = bin_map_.AddBinToDstVec(ls_send_to, dst_idx);
870  LogD(kClassName, __func__,
871  "With LS differential %" PRId64 "B (hysteresis %uB), adding bin "
872  "index %" PRIBinIndex " to LS dst vec, now 0x%X.\n",
873  ls_differential, hysteresis_, dst_idx, ls_send_to);
874  }
875  else
876  {
877  zombie_ls_gradient.value += ls_differential;
878  zombie_ls_send_to = bin_map_.AddBinToDstVec(zombie_ls_send_to, dst_idx);
879  LogD(kClassName, __func__,
880  "With Zombie LS differential %" PRId64 "B (hysteresis %uB), "
881  "adding bin index %" PRIBinIndex " to LS dst vec, now 0x%X.\n",
882  ls_differential, hysteresis_, dst_idx, ls_send_to);
883  }
884  }
885  }
886  else
887  {
888  LogD(kClassName, __func__,
889  "NOT including bin %" PRIBinId " (index %" PRIBinIndex ") in mcast "
890  "gradient (dsts 0x%x), because it has a 0 queue depth.\n",
891  bin_map_.GetPhyBinId(dst_idx), dst_idx, dst_vec);
892  }
893  }
894 
895  LogD(kClassName, __func__, "Zombie Multicast gradient: %" PRId64
896  ", non zombie: %" PRId64 "\n", zombie_gradient.value, gradient.value);
897 
898  // Set the gradient and destination bit vector.
899 
900  // If the gradient (which at this point only includes contributions
901  // from destinations with physical packets) is greater or equal to the
902  // the zombie_gradient (which only includes contributions from zombies)
903  // then we choose to dequeue physical packets rather than zombies
904 
905  //if (true) //-- use this and uncomment out the gradient assignment below
906  // to run the orig ZLR algorithm
907  if (gradient.value >= zombie_gradient.value)
908  {
909  // gradient.value += zombie_gradient.value;
910  gradient.dst_vec = send_to;
911  gradient.is_zombie = false;
912  }
913 
914  // Else the zombie gradient value is greater than that for just the
915  // physical packets and so we instead choose to dequeue zombies
916 
917  else
918  {
919  gradient.value = zombie_gradient.value;
920  gradient.dst_vec = zombie_send_to;
921  gradient.is_zombie = true;
922  LogD(kClassName, __func__, "Using zombie gradient.\n");
923  }
924 
925  //if (true) //-- use this and uncomment out the gradient assignment below
926  // to run the orig ZLR algorithm
927  if (ls_gradient.value >= zombie_ls_gradient.value)
928  {
929  //ls_gradient.value += zombie_ls_gradient.value;
930  ls_gradient.dst_vec = ls_send_to;
931  ls_gradient.is_zombie = false;
932  }
933 
934  // Else the zombie ls gradient value is greater than that for just the
935  // physical ls packets and we instead choose to dequeue ls zombies
936 
937  else
938  {
939  ls_gradient.value = zombie_ls_gradient.value;
940  ls_gradient.dst_vec = zombie_ls_send_to;
941  ls_gradient.is_zombie = true;
942  LogD(kClassName, __func__, "Using zombie ls gradient.\n");
943  }
944 
945  LogD(kClassName, __func__, "Multicast gradient for bin %s, "
946  "nbr %" PRIBinId " = %" PRId64 ", with dst vec 0x%x\n",
947  bin_map_.GetIdToLog(gradient.bin_idx).c_str(),
948  path_ctrl->remote_bin_id(), gradient.value,
949  static_cast<unsigned int>(gradient.dst_vec));
950 }

952 //============================================================================
953 bool BPDequeueAlg::ZombifyAndCriticalizePkts(
954  const Time& now, size_t max_num_path_ctrls, int32_t* path_ctrl_q_sizes,
955  TransmitCandidate& candidate)
956 {
957  // Steps S1 and S2 from the packet forwarder description in the BPF design
958  // documentation.
959  //
960  // Zombify pkts in Critical & Low-Lat queues. In the process, if we find a
961  // critical candidate that can be sent over an interface that is currently
962  // available send it.
963  Time ttg;
964  ttg.SetInfinite();
965  candidate.ttg.SetInfinite();
966 
967  size_t min_lat_pc_index = 0;
968 
969  for (uint8_t path_ctrl_idx = 0; path_ctrl_idx < max_num_path_ctrls;
970  ++path_ctrl_idx)
971  {
972  path_ctrl_q_sizes[path_ctrl_idx] = -1;
973  }
974 
975  // Iterate through the bins to clean them up, i.e., Criticalize, Zombify, get
976  // some time-to-reach.
977  //
978  // MCAST TODO: This is just cleaning up unicast bins at the
979  // moment. Eventually we probably want to include multicast bins
980  // as well.
981  BinQueueMgr* q_mgr = NULL;
982 
983  BinIndex dst_bin_idx = 0;
984  for (bool dst_bin_idx_valid = bin_map_.GetFirstUcastBinIndex(dst_bin_idx);
985  !base_ && dst_bin_idx_valid;
986  dst_bin_idx_valid = bin_map_.GetNextUcastBinIndex(dst_bin_idx))
987  {
988  q_mgr = queue_store_->GetBinQueueMgr(dst_bin_idx);
989 
990  // Print the BinQueueMgr to see the make up of our queues. Note this is a
991  // little different than the BP values used for gradients (watch out for
992  // NPLB).
993  q_mgr->Print();
994 
995  if (!(q_mgr->ContainsLSNonZombies()))
996  {
997  // A packet coming to this queue would experience no delay. Add to
998  // average.
999  AddDelayToAverage(0, dst_bin_idx);
1000  }
1001 
1002  if (q_mgr->depth_packets() == 0)
1003  {
1004  // There are no packets in the queue (maybe I am the destination),
1005  // therefore nothing to do for this bin.
1006  continue;
1007  }
1008 
1009  uint32_t latency_us[max_num_path_ctrls];
1010  memset(latency_us, 0, sizeof(latency_us));
1011 
1012  Time min_ttr;
1013  min_ttr.SetInfinite();
1014 
1015  if (anti_circ_ == AC_TECH_HEURISTIC_DAG)
1016  {
1017  // Get the per path controller latency, which is same for all packets of
1018  // this bin. Compute best path controller busy-ness.
1019  bpfwder_.GetPerPcLatencyToDst(dst_bin_idx, (uint32_t*)latency_us,
1020  false);
1021 
1022  // Check best path controller queue state: free or busy?
1023  if (GetMinLatencyPath(latency_us, num_path_ctrls_, min_lat_pc_index,
1024  min_ttr) &&
1025  (path_ctrl_q_sizes[min_lat_pc_index] == -1))
1026  {
1027  // Not computed yet.
1028  PathController* path_ctrl = path_ctrls_[min_lat_pc_index].path_ctrl;
1029 
1030  if (!path_ctrl)
1031  {
1032  LogF(kClassName, __func__, "Path controller at index %" PRIu8 " is "
1033  "NULL.\n", min_lat_pc_index);
1034  return false;
1035  }
1036 
1037  size_t current_pc_queue_size = 0;
1038  if (!(path_ctrl->GetXmitQueueSize(current_pc_queue_size)))
1039  {
1040  // This path controller does not have a current transmit queue size.
1041  // Maybe it is still connecting to a peer. Move on.
1042  LogD(kClassName, __func__, "Path to nbr %" PRIBinId " is currently "
1043  "not accepting packets.\n", path_ctrl->remote_bin_id());
1044  current_pc_queue_size = xmit_buf_free_thresh_;
1045  }
1046 
1047  path_ctrl_q_sizes[min_lat_pc_index] = current_pc_queue_size;
1048 
1049  if (WouldLogD(kClassName))
1050  {
1051  if (current_pc_queue_size >= xmit_buf_free_thresh_)
1052  {
1053  // Path Controller full, will not be able to use this neighbor for
1054  // this bin.
1055  LogD(kClassName, __func__, "Path to nbr %" PRIBinId " is full (Q "
1056  "(%zd) > %" PRIu32 ") cannot use.\n",
1057  path_ctrl->remote_bin_id(), current_pc_queue_size,
1058  xmit_buf_free_thresh_);
1059  }
1060  else
1061  {
1062  LogD(kClassName, __func__, "Lowest lat path to nbr %" PRIBinId
1063  " is currently available.\n", path_ctrl->remote_bin_id());
1064  }
1065  }
1066  }
1067  } // End heuristic_dag only.
1068 
1069  // Go through the EF and CRITICAL queues to zombify packets.
1070  for (uint8_t ttype_i = 0; ttype_i < num_zombifiable_ttypes_; ++ttype_i)
1071  {
1072  Packet* prev_pkt = NULL;
1073  uint32_t num_available_bytes = 0;
1074  LatencyClass ttype = zombifiable_ttypes_[ttype_i];
1075 
1076  q_mgr->PrepareIteration(ttype);
1077  PacketQueue::QueueWalkState saved_it;
1078 
1079  // Search inside the queue.
1080  while (num_available_bytes < queue_search_depth_)
1081  {
1082  Packet* pkt = q_mgr->PeekNext(ttype, saved_it);
1083 
1084  if (!pkt || (prev_pkt == pkt))
1085  {
1086  LogD(kClassName, __func__, "No pkt for bin %s for traffic type %s "
1087  "beyond.\n", bin_map_.GetIdToLog(dst_bin_idx).c_str(),
1088  LatencyClass_Name[ttype].c_str());
1089  break;
1090  }
1091 
1092  prev_pkt = pkt;
1093 
1094  if (anti_circ_ == AC_TECH_CONDITIONAL_DAG)
1095  {
1096  // Get the per path controller latency, which is same for all
1097  // packets of this bin. Compute best path controller busy-ness.
1098  bpfwder_.GetPerPcLatencyToDst(dst_bin_idx, (uint32_t*)latency_us,
1099  false, pkt);
1100 
1101  // Check best path controller queue state: free or busy?
1102  GetMinLatencyPath(latency_us, num_path_ctrls_, min_lat_pc_index,
1103  min_ttr);
1104  }
1105 
1106  // Figure out if this packet can still be delivered. Get time to go
1107  // from packet.
1108  if (pkt->time_to_go_valid())
1109  {
1110  ttg = pkt->GetTimeToGo() - (now - pkt->recv_time());
1111  }
1112 
1113  if (ttg < min_ttr)
1114  {
1115  // Packet cannot make it on any interface.
1116  LogD(kClassName, __func__, "Pkt %p with ttg %s cannot be delivered "
1117  "in time on any interface (min_ttr %s). Drop.\n", pkt,
1118  ttg.ToString().c_str(), min_ttr.ToString().c_str());
1119  pkt = q_mgr->DequeueAtCurrentIterator(ttype);
1120 
1121  if (pkt && pkt->HasQueuingDelay())
1122  {
1123  AddDelayToAverage(Time::GetNowInUsec() -
1124  pkt->recv_time().GetTimeInUsec(), dst_bin_idx);
1125  }
1126 
1127  if (drop_expired_ || !q_mgr->ZombifyPacket(pkt))
1128  {
1129  bpfwder_.AddDroppedBytes(dst_bin_idx, pkt->virtual_length());
1130  TRACK_EXPECTED_DROP(kClassName, packet_pool_);
1131  LogD(kClassName, __func__, "Dropped expired packet %p or "
1132  "Zombification failed.\n", pkt);
1133  packet_pool_.Recycle(pkt);
1134  }
1135  continue;
1136  }
1137 
1138  if (anti_circ_ == AC_TECH_HEURISTIC_DAG)
1139  {
1140  // Anti-circulation technique is heuristic_dag, deal with critical.
1141  if ((ttype == CRITICAL_LATENCY) && (ttg < candidate.ttg) &&
1142  (path_ctrl_q_sizes[min_lat_pc_index] <
1143  static_cast<int32_t>(xmit_buf_max_thresh_)))
1144  {
1145  // Critical packet has tighter deadline and can go on non-busy path
1146  // controller.
1147  candidate.is_valid = true;
1148  candidate.pkt = pkt;
1149  candidate.bin_idx = dst_bin_idx;
1150  candidate.id_to_log = bin_map_.GetIdToLog(dst_bin_idx);
1151  candidate.ttg = ttg;
1152  candidate.ttr = min_ttr;
1153  candidate.path_ctrl_index = min_lat_pc_index;
1154  candidate.dequeue_loc = saved_it;
1155  candidate.q_mgr = q_mgr;
1156  LogD(kClassName, __func__, "Critical packet %p with ttg %s on "
1157  "available path controller %" PRIu8 " overtakes "
1158  "candidates.\n", pkt, ttg.ToString().c_str(),
1159  min_lat_pc_index);
1160  }
1161 
1162  if ((ttype == LOW_LATENCY) &&
1163  IsHistoryConstrained(pkt, ttg, latency_us, num_path_ctrls_))
1164  {
1165  // EF packet is history-constrained and not yet in critical. But
1166  // this should not prevent us from assessing it as a candidate.
1167  pkt = q_mgr->DequeueAtCurrentIterator(ttype);
1168 
1169  if (!q_mgr->CriticalizePacket(pkt))
1170  {
1171  if (pkt->HasQueuingDelay())
1172  {
1173  AddDelayToAverage(Time::GetNowInUsec() -
1174  pkt->recv_time().GetTimeInUsec(),
1175  dst_bin_idx);
1176  }
1177 
1178  TRACK_UNEXPECTED_DROP(kClassName, packet_pool_);
1179  LogD(kClassName, __func__, "Dropped packet %p ("
1180  "Criticialization failed).\n", pkt);
1181  packet_pool_.Recycle(pkt);
1182  }
1183  else
1184  {
1185  if ((ttg < candidate.ttg) &&
1186  (path_ctrl_q_sizes[min_lat_pc_index] <
1187  static_cast<int32_t>(xmit_buf_max_thresh_)))
1188  {
1189  // This pkt has a tighter deadline.
1190  candidate.is_valid = true;
1191  candidate.pkt = pkt;
1192  candidate.bin_idx = dst_bin_idx;
1193  candidate.q_mgr = q_mgr;
1194  candidate.id_to_log = bin_map_.GetIdToLog(dst_bin_idx);
1195  candidate.ttg = ttg;
1196  candidate.ttr = min_ttr;
1197  candidate.path_ctrl_index = min_lat_pc_index;
1198  if (!q_mgr->GetIterator(CRITICAL_LATENCY,
1199  pkt, candidate.dequeue_loc))

1202  candidate.is_valid = false;
1203  }

1208  }
1209  }

1211  }
1212  } // End heuristic_dag condition.
1213 
1214  num_available_bytes += pkt->virtual_length();
1215  } // End queue search.
1216  } // End Zombifiable iteration.
1217  } // End bin iteration clean up. MCAST TODO unicast iteration only.
1218  // END Zombification and Criticalization.
1219 
1220  return true;

1260

1262 
1263  if (WouldLogD(kClassName))
1264  {
1265  if (current_pc_queue_size >=
1266  static_cast<size_t>(xmit_buf_free_thresh_))

1278  "available.\n", path_ctrl->remote_bin_id());
1279  }
1280  }
1281  }
1282

1284  static_cast<int32_t>(xmit_buf_free_thresh_))
1285  {
1286  // The path controller is busy.
1287  LogD(kClassName, __func__, "Skip busy path ctrl %" PRIu8 " to nbr %"
1288  PRIBinId ".\n", path_ctrl_idx, path_ctrl->remote_bin_id());

1290  }

1293  // unicast and multicast destination bins.

1295  for (bool dst_bin_idx_valid = bin_map_.GetFirstDstBinIndex(dst_bin_idx);
1296  dst_bin_idx_valid;
1297  dst_bin_idx_valid = bin_map_.GetNextDstBinIndex(dst_bin_idx))
1298  {
1299  LogD(kClassName, __func__, "=========== Nbr %" PRIBinId " (%" PRIBinId
1300  "), Dst Bin %s ================\n",
1301  bin_map_.GetPhyBinId(path_ctrl->remote_bin_idx()),
1302  path_ctrl->remote_bin_id(),
1303  bin_map_.GetIdToLog(dst_bin_idx).c_str());
1304

1306  {

1308  dst_bin_idx)->ContainsPacketsWithTtypes(
1309  priority_dequeue_ttypes_, num_priority_dequeue_ttypes_);
1310  }

1313  {
1314  LogD(kClassName, __func__, "My queue to Bin %s is empty, go on to "
1315  "next bin.\n", bin_map_.GetIdToLog(dst_bin_idx).c_str());
1316 
1317  // There are no packets in the queue (maybe I am the destination),
1318  // therefore nothing to do for this bin.
1319  continue;
1320  }
1321 
1322  // Get neighbor queue depths.
1323  QueueDepths* nbr_queue_depth = queue_store_->GetBinQueueMgr(
1324  dst_bin_idx)->GetNbrQueueDepths(path_ctrl->remote_bin_idx());

1369  }
1370

1379  }
1380

1383  if (gradient.value > 0)
1384  {
1385  gradients.Push(gradient, gradient.value + (gradient.is_dst? 1 : 0));

1387  "%s, pc %" PRIu8 ") %s 0x%X.\n",

1392  }
1393  else
1394  {

1398  gradient.value, hysteresis_);
1399  }
1400 
1401  // Gradient value is given +1 if goes to destination directly to give it
1402  // preference.
1403  if (ls_gradient.value > 0)
1404  {
1405  ls_gradients.Push(ls_gradient,
1406  ls_gradient.value + (ls_gradient.is_dst? 1 : 0));
1407  LogD(kClassName, __func__, "Found LS gradient %" PRId64 "B on (bin "
1408  "%s, pc %" PRIu8 ") %s 0x%X.\n",
1409  ls_gradient.value, bin_map_.GetIdToLog(dst_bin_idx).c_str(),
1410  path_ctrl_idx, ls_gradient.is_dst ? "to dst" : "not to dst",
1411  ls_gradient.dst_vec);
1412  }
1413  else
1414  {
1415  LogD(kClassName, __func__, "LS gradient %dB is negative or below "
1416  "hysteresis %" PRIu32 "B.\n", ls_gradient.value, hysteresis_);
1417  }
1418  } // End bin iterations.
1419  } // END gradient computations.

1465  "%s on path ctrl %" PRIu8 " to nbr %" PRIBinId " / %" PRIBinId

1475  static_cast<int32_t>(xmit_buf_max_thresh_) -
1476  path_ctrl_q_sizes[gradient.path_ctrl_index];

1479  {

1481  // many bytes as the gap between the two largest gradients, and only 1B
1482  // when they were equal. This was observed to be too conservative and led

1534  }

1552

1558  {
1559  LogE(kClassName, __func__, "Invalid candidate in candidates "
1560  "list.\n");

1562  }
1563

1568  selected_candidate.pkt->GetLatencyClass(),
1569  selected_candidate.dequeue_loc, selected_candidate.dst_vec);
1570

1574  solutions[num_solutions].pkt->recv_time().GetTimeInUsec(),

1576  }
1577  }
1578  else
1579  {
1580  // This is a packetless Zombie.
1581  solutions[num_solutions].pkt =
1582  (selected_candidate.q_mgr)->Dequeue(
1583  selected_candidate.latency_class,
1584  selected_candidate.virtual_len, selected_candidate.dst_vec);
1585

1587  {
1588  if (selected_candidate.dst_vec == 0)
1589  {
1590  LogE(kClassName, __func__, "Failed to dequeue %s packet of "
1591  "size %zdB from queue of %zdB.\n",
1592  LatencyClass_Name[selected_candidate.latency_class].c_str(),

1594  (selected_candidate.q_mgr)->GetNextDequeueSize(

1599  LogE(kClassName, __func__, "Failed to dequeue %s packet of "

1604  break;

1624  (path_ctrl_q_sizes[solutions[num_solutions - 1].path_ctrl_index] >

1642  }
1643  else if (cand_bytes_found > 0)

1648  } // END finding a packet for the low-latency traffic.

1760  }

1766  // destinations or path controllers) in the same set of results.
1767  // TODO: consider removing this condition later, but we'll need some
1768  // extra conditions to evaluate max_bytes with consideration for how
1769  // many packets we already picked for a particular path controller.
1770  break;
1771  }
1772 
1773  // Otherwise, look at the case when the next gradient is equal.
1774  } // END treating normal and Zombie packets.

1799  selected_candidate.pkt)
1800  {

1802  }

1804  (selected_candidate.q_mgr)->DequeueAtIterator(
1805  selected_candidate.pkt->GetLatencyClass(),

1807  if (!solutions[num_solutions].pkt)
1808  {

1811  }

1817  // packet than the selected candidate, it means that it was cloned.

1827  selected_candidate.ToString().c_str());

1834  solutions[num_solutions].pkt->recv_time().GetTimeInUsec(),
1835  selected_candidate.bin_idx);
1836  }
1837  }
1838  else
1839  {
1840  // This is a packetless Zombie.
1841  solutions[num_solutions].pkt = (selected_candidate.q_mgr)->Dequeue(
1842  selected_candidate.latency_class,
1843  selected_candidate.virtual_len,
1844  selected_candidate.dst_vec);
1845

1847  {

1849  {

1851  "%zdB from queue of %zdB.\n",

1855  selected_candidate.latency_class));
1856  }
1857  else

1860  "%zdB from multicast zombie queue.\n",
1861  LatencyClass_Name[selected_candidate.latency_class].c_str(),
1862  selected_candidate.virtual_len);
1863  }
1864  }
1865  }
1866  if (solutions[num_solutions].pkt == NULL)
1867  {
1868  LogF(kClassName, __func__, "Error dequeuing a packet.\n");
1869  continue;
1870  }
1871 
1872  solutions[num_solutions].bin_idx = selected_candidate.bin_idx;
1873  solutions[num_solutions].path_ctrl_index =
1874  selected_candidate.path_ctrl_index;
1875 
1876  path_ctrl_q_sizes[selected_candidate.path_ctrl_index] +=
1877  selected_candidate.virtual_len;
1878  max_bytes -= selected_candidate.virtual_len;
1879 
1880  ++num_solutions;
1881 
1882  if (!multi_deq_ ||
1883  (path_ctrl_q_sizes[solutions[num_solutions - 1].path_ctrl_index] >
1884  static_cast<int32_t>(xmit_buf_max_thresh_)) ||

1886  (num_solutions >= max_num_solutions))

1892  : "",
1893  max_bytes <= 0 ? "; num bytes reached" : "",
1894  num_solutions >= max_num_solutions ?
1895  "; max num solutions reached" : "");
1896  break;
1897  }
1898  }
1899  }
