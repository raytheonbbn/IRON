544 void ZLR::DoPerBinEnqueueProcessing(
545  uint16_t bytes, LatencyClass lat, BinIndex bin_index)
546 {
547  // When a packet is enqueued, we need to track it with the appropriate queue
548  // depth dynamics object(s). Whether or not (and how) this class counts
549  // towards each of the two QueueDepthDynamics (normal ZLR or LS-ZLR) is
550  // determined using the ZLR_*_TTYPES constants. The queue depth dynamics
551  // objects are responsible for tracking the minimum queue depth over the
552  // window (of non-zombies only), the general queue depth change direction
553  // (increasing or decreasing, including zombies), and how many zombies we've
554  // already added towards the window. Any of these may change because of an
555  // enqueue'd packet.
556  if (do_ls_zombie_latency_reduction_ &&
557  (is_zlr_ls_decision_ttype_[lat] || is_zlr_ls_zombie_ttype_[lat]))
558  {
559  uint32_t zlr_depth_bytes = q_mgr_.GetTtypeDepthBytes(bin_index,
560  ZLR_LS_DECISION_TTYPES, kNumZlrLsDecisionTtypes);
561  if (kFastRecovery)
562  {
563  UpdateFastRecoveryStateOnEnqueue(bin_index, true, zlr_depth_bytes);
564  }
565 
566  // This latency class counts towards LS-ZLR. Consider it in the LS
567  // dynamics. We want to call this function for both LS zombies and LS
568  // packets, since both are counted towards the change rate. However, we
569  // only want to use the non-zombie packets in the "new depth" passed
570  // in, since that is used to compute the min non-zombie depth over the ZLR
571  // observation window.
572  zlr_ls_queue_depth_dynamics_[bin_index].ProcessBytesAdded(
573  bytes, lat, zlr_depth_bytes);
574  }
575  // NOTE: this is not an else, because some packets count towards both LS ZLR
576  // and normal ZLR.
577  if (is_zlr_decision_ttype_[lat] || is_zlr_zombie_ttype_[lat])
578  {
579  uint32_t zlr_depth_bytes = q_mgr_.GetTtypeDepthBytes(bin_index,
580  ZLR_DECISION_TTYPES, kNumZlrDecisionTtypes);
581  if (kFastRecovery)
582  {
583  UpdateFastRecoveryStateOnEnqueue(bin_index, false, zlr_depth_bytes);
584  }
585 
586  // This latency class counts towards normal ZLR. Consider it in the normal
587  // dynamics. We want to call this function for both zombies and non-zombie
588  // packets, since both are counted towards the change rate. However, we
589  // only want to use the non-zombie packets in the "new depth" passed
590  // in, since that is used to compute the min non-zombie depth over the ZLR
591  // observation window.
592  zlr_queue_depth_dynamics_[bin_index].ProcessBytesAdded(
593  bytes, lat, zlr_depth_bytes);
594  }
595  if (do_ls_zombie_latency_reduction_ && is_zlr_ls_zombie_ttype_[lat])
596  {
597  // This latency class counts against the LS ZLR queue depth floor.
598  zlr_ls_queue_depth_dynamics_[bin_index].ProcessZombieBytesAdded(bytes,
599  lat);
600  }
601  if (is_zlr_zombie_ttype_[lat])
602  {
603  // This latency class counts against the normal ZLR queue depth floor.
604  zlr_queue_depth_dynamics_[bin_index].ProcessZombieBytesAdded(bytes, lat);
605  }
606 }

609 void ZLR::UpdateFastRecoveryStateOnEnqueue(
610  BinIndex bin_idx, bool process_ls, uint32_t zlr_depth_bytes)
611 {
612  FastRecoveryData* data = &(fast_recovery_[bin_idx]);
613  QueueDepthDynamics* dynamics = &(zlr_queue_depth_dynamics_[bin_idx]);
614 
615  if (process_ls)
616  {
617  data = &(ls_fast_recovery_[bin_idx]);
618  dynamics = &(zlr_ls_queue_depth_dynamics_[bin_idx]);
619  }
620 
621  if (dynamics->GetChangeRateBytesPerSec() >
622  zlr_q_change_min_thresh_bytes_per_s_)
623  {
624  data->deq_bytes = 0;
625  }
626  if (data->fast_recovery_state == QUEUE_DEPTH_DIP)
627  {
628  if (zlr_depth_bytes > kFastRecoveryStartThreshBytes)
629  {
630  data->fast_recovery_state = RECOVERY;
631  if (kGraphZLRFastRecovery && qd_xplot_[bin_idx] != NULL)
632  {
633  qd_xplot_[bin_idx]->DrawVerticalLine(
634  Time::Now().GetTimeInUsec() - iron::kStartTime,
635  ORANGE);
636  }
637  LogI(kClassName, __func__, "Bin %s, %s"
638  "Entering FastRecoveryState RECOVERY: zlr_depth_bytes = %"
639  PRIu32 "\n", bin_map_.GetIdToLog(bin_idx).c_str(),
640  (process_ls ? "LS, " : ""), zlr_depth_bytes);
641  }
642  }

669 //============================================================================
670 void ZLR::DoPerBinDequeueProcessing(
671  const DequeuedInfo& dq_info, BinIndex bin_index)
672 {
673  // When a packet is dequeued, we need to track it with the appropriate queue
674  // depth dynamics object(s). Whether or not (and how) this class counts
675  // towards each of the two QueueDepthDynamics (normal ZLR or LS-ZLR) is
676  // determined using the ZLR_*_TTYPES constants. The queue depth dynamics
677  // objects need to know about a dequeue to maintain the minimum non-zombie
678  // queue depth over the window as well as the general queue depth change
679  // direction (increasing or decreasing, including zombies). We will include
680  // zombies in BytesRemoved calls for the sake of updating the change rate,
681  // but not in the ZLR_*_TTYPES constant passed in for updating min depth.
682  if (do_ls_zombie_latency_reduction_ &&
683  (is_zlr_ls_decision_ttype_[dq_info.lat] ||
684  is_zlr_ls_zombie_ttype_[dq_info.lat]))
685  {
686  // This latency class counts towards LS-ZLR. Consider it in the LS
687  // dynamics. We want to call this function for both zombies and non-zombie
688  // packets, since both are counted towards the change rate. However, we
689  // only want to use the non-zombie packets in the "new depth" passed
690  // in, since that is used to compute the min non-zombie depth over the ZLR
691  // observation window.
692  uint32_t zlr_depth_bytes = q_mgr_.GetTtypeDepthBytes(bin_index,
693  ZLR_LS_DECISION_TTYPES, kNumZlrLsDecisionTtypes);
694  if (kFastRecovery)
695  {
696  UpdateFastRecoveryStateOnDequeue(
697  dq_info, bin_index, true, zlr_depth_bytes);
698  }
699 
700  zlr_ls_queue_depth_dynamics_[bin_index].BytesRemoved(
701  dq_info.dequeued_size, dq_info.lat, zlr_depth_bytes);
702  }
703  // NOTE: this is not an else, because some packets count towards both LS ZLR
704  // and normal ZLR.
705  if (is_zlr_decision_ttype_[dq_info.lat] ||
706  is_zlr_zombie_ttype_[dq_info.lat])
707  {
708  // This latency class counts towards normal ZLR. Consider it in the normal
709  // dynamics. We want to call this function for both zombies and non-zombie
710  // packets, since both are counted towards the change rate. However, we
711  // only want to use the non-zombie packets in the "new depth" passed
712  // in, since that is used to compute the min non-zombie depth over the ZLR
713  // observation window.
714  uint32_t zlr_depth_bytes = q_mgr_.GetTtypeDepthBytes(bin_index,
715  ZLR_DECISION_TTYPES, kNumZlrDecisionTtypes);
716  if (kFastRecovery)
717  {
718  UpdateFastRecoveryStateOnDequeue(
719  dq_info, bin_index, false, zlr_depth_bytes);
720  }
721  zlr_queue_depth_dynamics_[bin_index].BytesRemoved(
722  dq_info.dequeued_size, dq_info.lat, zlr_depth_bytes);
723  }
724  if (dq_info.is_ip)
725  {
726  // Perform ZLR zombie addition algorithm.
727  DoZombieLatencyReduction(dq_info, bin_index);
728  }

732 void ZLR::UpdateFastRecoveryStateOnDequeue(const DequeuedInfo& dq_info,
733  BinIndex bin_idx, bool process_ls, uint32_t zlr_depth_bytes)
734 {
735  //
736  // This function has two goals:
737  // 1. Update the Fast Recovery state if necessary (see state machine drawing
738  // in zlr.h).
739  // 2. Update the dynamic observation window size if necessary.
740  //
741  FastRecoveryData* data = &(fast_recovery_[bin_idx]);
742  QueueDepthDynamics* dynamics = &(zlr_queue_depth_dynamics_[bin_idx]);
743 
744  // Number of zombies counted against this ZLR window. This will be used to
745  // determine whether it's appropriate to increase the ZLR window, since we
746  // don't want to increase it unless there's been some action on a queue that
747  // counts. It will also be used to determine whether a fast recovery is
748  // complete.
749  uint32_t zombie_depth_bytes = q_mgr_.GetTtypeDepthBytes(bin_idx,
750  ZLR_ZOMBIE_TTYPES, kNumZlrZombieTtypes);
751 
752  if (process_ls)
753  {
754  // If we are responding to a dequeued LS packet, use the LS-ZLR data
755  // structures and values instead.
756  data = &(ls_fast_recovery_[bin_idx]);
757  dynamics = &(zlr_ls_queue_depth_dynamics_[bin_idx]);
758  zombie_depth_bytes = q_mgr_.GetTtypeDepthBytes(bin_idx,
759  ZLR_LS_ZOMBIE_TTYPES, kNumZlrLsZombieTtypes);
760  }
761 
762  Time now = Time::Now();
763 
764  // First see if it's time to reset the fast recovery state machine to
765  // STEADY_STATE due to the "reset time" amount having passed since our last
766  // state change.
767  if (data->fast_recovery_state != STEADY_STATE &&
768  (now - data->fast_recovery_start_time > kFastRecoveryResetTime))
769  {
770  data->fast_recovery_state = STEADY_STATE;
771  if (!process_ls && kGraphZLRFastRecovery && qd_xplot_[bin_idx] != NULL)
772  {
773  qd_xplot_[bin_idx]->DrawVerticalLine(
774  now.GetTimeInUsec() - iron::kStartTime,
775  WHITE);
776  }
777  LogI(kClassName, __func__, "Bin %s, %s Entering "
778  "FastRecoveryState STEADY_STATE: more than 3 seconds passed.\n",
779  bin_map_.GetIdToLog(bin_idx).c_str(), (process_ls ? "LS, " : ""));
780  }
781 
782  // If we're in steady state and this is the potential beginning of a queue
783  // depth dip (our dequeue counter is 0), record the time that this dip (if
784  // it turns out to be a dip) started and the number of zombies present at
785  // the start of the potential dip. Add to the count of dequeued bytes. The
786  // dequeued byte and dequeue start time values will be used to determine
787  // whether this is a dip. The recovery zombie depth bytes will be used as
788  // part of recovering should fast recovery kick in after this dip.
789  if (data->deq_bytes == 0 && data->fast_recovery_state == STEADY_STATE)
790  {
791  data->deq_start_time.GetNow();
792  data->recovery_zombie_depth_bytes = zombie_depth_bytes;
793  }
794  data->deq_bytes += dq_info.dequeued_size;
795 
796  // If we're dequeueing a zombie, that's a flag for considering whether we're
797  // now in a dip state (fast recovery won't do anything if we haven't
798  // dequeued any zombies, so no need to change the state unless/until we
799  // dequeue a zombie). If we are in a dip (dequeue bytes is big enough over a
800  // small enough dequeue time), then update the state machine accordingly. If
801  // this was the first dip out of steady state, move to "QUEUE_DEPTH_DIP"
802  // (from which we'll use fast recovery). If we're already recovering or
803  // recovered from a dip, then a second (or later) dip before a reset means
804  // we want to consider this oscillatory - i.e., no fast recovery and
805  // increase the dynamic observation window.
806  if (Packet::IsZombie(dq_info.lat))
807  {
808  if ((now - data->deq_start_time < kFastRecoveryDipThreshTime) &&
809  (data->deq_bytes > kFastRecoveryDipThreshBytes))
810  {
811  if (data->fast_recovery_state == STEADY_STATE)
812  {
813  data->fast_recovery_state = QUEUE_DEPTH_DIP;
814  data->fast_recovery_start_time.GetNow();
815  if (!process_ls && kGraphZLRFastRecovery && qd_xplot_[bin_idx] != NULL)
816  {
817  qd_xplot_[bin_idx]->DrawVerticalLine(
818  now.GetTimeInUsec() - iron::kStartTime,
819  RED);
820  }
821  LogI(kClassName, __func__, "Bin %s, %s Entering "
822  "FastRecoveryState QUEUE_DEPTH_DIP: deq_bytes = %" PRIu32
823  ".\n", bin_map_.GetIdToLog(bin_idx).c_str(),
824  (process_ls ? "LS, " : ""),
825  data->deq_bytes);
826  }
827  else if (data->fast_recovery_state >= RECOVERY)
828  {
829  data->fast_recovery_state = OSCILLATORY;
830  // As long as we keep seeing dips, extend the time before we'll
831  // consider entering fast recovery again.
832  data->fast_recovery_start_time.GetNow();
833  LogI(kClassName, __func__, "Bin %s, %s Entering "
834  "FastRecoveryState OSCILLATORY: deq_bytes = %" PRIu32
835  ".\n", bin_map_.GetIdToLog(bin_idx).c_str(),
836  (process_ls ? "LS, " : ""),
837  data->deq_bytes);
838  if (!process_ls && kGraphZLRFastRecovery && qd_xplot_[bin_idx] != NULL)
839  {
840  qd_xplot_[bin_idx]->DrawVerticalLine(
841  now.GetTimeInUsec() - iron::kStartTime,
842  GREEN);
843  }
844  }
845  }
846  }
847  // If we're not in fast recovery mode, adjust the ZLR floor window. i.e.,
848  // over how long into the past we should look for the sake of ignoring
849  // queue depth spikes.
850  //
851  // If we dequeued a zombie packet or have few non-zombies left, then our
852  // observation window is probably too small. If we haven't dequeued a
853  // zombie in a while, then we can probe a smaller window.
854  if ((data->fast_recovery_state == STEADY_STATE ||
855  data->fast_recovery_state == OSCILLATORY) &&
856  (zombie_depth_bytes > 0) &&
857  (Packet::IsZombie(dq_info.lat) ||
858  (zlr_depth_bytes < zlr_low_water_mark_bytes_)))
859  {
860  // Note: rate regulation of window changes is done within
861  // IncrementMinBytesResetPeriod.
862  dynamics->IncrementMinBytesResetPeriod();
863  }
864  // Whenever we dequeue a non-zombie, check whether it's time to shrink the
865  // obvservation window. Logic to determine whether it's time and by how much
866  // to shrink the window is inside DecrementMinBytesResetPeriod.
867  if (!Packet::IsZombie(dq_info.lat))
868  {
869  dynamics->DecrementMinBytesResetPeriod();
870 
871  if (data->fast_recovery_state == RECOVERY &&
872  zombie_depth_bytes > data->recovery_zombie_depth_bytes)
873  {
874  LogI(kClassName, __func__, "Bin %s, %s Entering "
875  "FastRecoveryState RECOVERED: zombie_depth_bytes = %" PRIu32
876  ", recovery_zombie_depth_bytes = %" PRIu32 ".\n",
877  bin_map_.GetIdToLog(bin_idx).c_str(), (process_ls ? "LS, " : ""),
878  zombie_depth_bytes, data->recovery_zombie_depth_bytes);
879  data->fast_recovery_state = RECOVERED;
880  if (!process_ls && kGraphZLRFastRecovery && qd_xplot_[bin_idx] != NULL)
881  {
882  qd_xplot_[bin_idx]->DrawVerticalLine(
883  now.GetTimeInUsec() - iron::kStartTime,
884  YELLOW);
885  }
886  }
887  }

891 void ZLR::DoZombieLatencyReduction(
892  const DequeuedInfo& dq_info, BinIndex bin_idx)
893 {
894  // Function performs the heart of ZLR, where we determine whether or not to
895  // add a zombie based on the queue depth state and dynamics. For more
896  // documentation about the ZLR algorithm, see the \class level doxygen in
897  // zlr.h.
898  bool is_ls = do_ls_zombie_latency_reduction_ &&
899  (Packet::IsLatencySensitive(dq_info.lat));
900  // Which queue depth dynamics we should look at based on this packet to
901  // determine whether to add a zombie (ls or normal). We will only add normal
902  // zombies if we dequeued a normal packet, and we will only add an LS zombie
903  // if we dequeued a LS packet.
904  QueueDepthDynamics* dynamics = &(zlr_queue_depth_dynamics_[bin_idx]);
905  // Fast recovery data: used to determine whether we're using the
906  // instantaneous queue depth or the minimum over the observation window.
907  FastRecoveryData* data = &(fast_recovery_[bin_idx]);
908  // If we're adding a zombie, which type we should add.
909  LatencyClass new_zombie_class = HIGH_LATENCY_ZLR;
910  // The instantaneous queue depth to consider for ZLR, which includes a
911  // different set of latency classes depending on whether this is normal
912  // latency ZLR or LS ZLR.
913  uint32_t zlr_depth_bytes = q_mgr_.GetTtypeDepthBytes(
914  bin_idx, ZLR_DECISION_TTYPES, kNumZlrDecisionTtypes);
915  // Whether or not we have any zombies that should be counted against this
916  // ZLR window. This will be used to determine whether it's appropriate to
917  // increase the ZLR window, since we don't want to increase it unless
918  // there's been some action on a queue that counts.
919 
920  if (is_ls)
921  {
922  dynamics = &(zlr_ls_queue_depth_dynamics_[bin_idx]);
923  data = &(ls_fast_recovery_[bin_idx]);
924  new_zombie_class = HIGH_LATENCY_ZLR_LS;
925  zlr_depth_bytes = q_mgr_.GetTtypeDepthBytes(bin_idx,
926  ZLR_LS_DECISION_TTYPES, kNumZlrLsDecisionTtypes);
927  }
928  int32_t change_rate = dynamics->GetChangeRateBytesPerSec();
929  uint32_t min_depth_bytes = dynamics->GetMinQueueDepthBytes();
930 
931  if (zlr_xplot_[bin_idx])
932  {
933  XPLOT_COLOR color = RED;
934  uint8_t line_num = 0;
935  if (is_ls)
936  {
937  color = GREEN;
938  line_num++;
939  }
940  zlr_xplot_[bin_idx]->ContinueTimeLine(line_num,
941  dynamics->min_bytes_reset_period().GetTimeInMsec(), color);
942  }
943 
944  Time now = Time::Now();
945  if (!is_ls && qd_xplot_[bin_idx] != NULL && kGraphZLRWindows)
946  {
947  uint32_t zombie_depth_bytes = q_mgr_.GetTtypeDepthBytes(
948  bin_idx,
949  ZLR_ZOMBIE_TTYPES, kNumZlrZombieTtypes);
950  qd_xplot_[bin_idx]->ContinueTimeLine(
951  1, zombie_depth_bytes + min_depth_bytes,
952  GREEN);
953  }
954 
955  if (!Packet::IsZombie(dq_info.lat))
956  {
957  LogD(kClassName, __func__,
958  "Reacting to a non-zombie dequeue: min_depth_bytes = %" PRIu32
959  ", zlr_high_water_mark_bytes_ = %" PRIu32 ", change_rate = %" PRId32
960  ", zlr_q_change_min_thresh_bytes_per_s_ = %" PRId32 ".\n",
961  min_depth_bytes, zlr_high_water_mark_bytes_, change_rate,
962  zlr_q_change_min_thresh_bytes_per_s_);
963  // Determine whether to add a zombie packet.
964  // During fast recovery, we use the instantaneous queue depth to make this
965  // decision. Otherwise, we use the minimum depth over the observation
966  // window.
967  if ((kFastRecovery &&
968  data->fast_recovery_state == RECOVERY &&
969  zlr_depth_bytes >= zlr_high_water_mark_bytes_) ||
970  ((min_depth_bytes > zlr_high_water_mark_bytes_)
971  && (change_rate >= zlr_q_change_min_thresh_bytes_per_s_)))
972  {
973  // This is not a zombie packet, and the non-zombie queue is long enough,
974  // and the queue change rate is high enough.
975  // Add a zombie packet (the triggering packet has already been dequeued).
976  if (dq_info.is_ip && dq_info.dst_addr != 0)
977  {
978  size_t zombie_len = dq_info.dequeued_size;
979  if (zombie_len > kMaxZombieLenBytes)
980  {
981  zombie_len = kMaxZombieLenBytes;
982  }
983  // For multicast packets, we want to add zombies just for the single
984  // destination bin we are currently evaluating.
985  DstVec dst_vec = 0;
986  if (is_multicast_)
987  {
988  dst_vec = bin_map_.AddBinToDstVec(dst_vec, bin_idx);
989  }
990  q_mgr_.AddNewZombie(dq_info.dst_addr, zombie_len, new_zombie_class,
991  dst_vec);
992  }
993  }
994  }
