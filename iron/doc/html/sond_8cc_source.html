101 // The state transition diagram for the SOND packet transmission
102 // implementation is shown below. The IDLE state occurs when there is no
103 // packet being transmitted. The XMIT state occurs when there is a packet
104 // being transmitted. The packet transmission timer is set to the packet's
105 // transmission delay.
106 //
107 // +----------+
108 // ------| Set rate |<-------
109 // | +----------+ | R
110 // | __|__
111 // | / \ .
112 // | -------------->| IDLE |<---------
113 // | | \_____/ |
114 // V | Yes | |
115 // /------------\ | P |
116 // | Queue empty? | | |
117 // \------------/ | |
118 // ^ | No V |
119 // | | +----------------+ |
120 // | | | Enqueue packet | |
121 // | | +----------------+ |
122 // | | | |
123 // | | V |
124 // | | /----------\ |
125 // | ------------>| Rate > 0 ? |-------
126 // | \----------/ No
127 // | | Yes
128 // | V
129 // +-------------+ +--------------------------+
130 // | Send packet | | Dequeue packet |
131 // +-------------+ | Set timer for xmit delay |
132 // ^ +--------------------------+
133 // | |
134 // | __V__
135 // | T / \ R +----------+
136 // ---------------------| XMIT |------->| Set rate |
137 // \_____/<--------| |
138 // ^ | +----------+
139 // | | P
140 // | V
141 // +----------------+
142 // | Enqueue packet |
143 // +----------------+
144 //
145 // States:
146 // IDLE = When xmit_pkt_ptr_ is NULL.
147 // XMIT = When xmit_pkt_ptr_ points to a packet.
148 //
149 // Events:
150 // P = Packet arrives from BPF for transmission.
151 // R = Rate change.
152 // T = Timer expiration.

213 bool Sond::Initialize(const ConfigInfo& config_info, uint32_t config_id)
214 {
215  LogI(kClassName, __func__, "Sond %" PRIu32 ": Initializing...\n",
216  config_id);
217 
218  // Store the configuration identifier as this SOND's number.
219  path_controller_number_ = config_id;
220 
221  // Construct the prefix for the configuration names.
222  string config_prefix("PathController.");
223  config_prefix.append(StringUtils::ToString(static_cast<int>(config_id)));
224 
225  // Extract the label, if any.
226  string config_name = config_prefix;
227  config_name.append(".Label");
228  label_ = config_info.Get(config_name);
229 
230  // Extract the endpoint IPv4 addresses and optional UDP port numbers.
231  config_name = config_prefix;
232  config_name.append(".Endpoints");
233  endpoints_str_ = config_info.Get(config_name);
234 
235  if (!ParseEndpointsString(endpoints_str_))
236  {
237  LogE(kClassName, __func__, "Sond %" PRIu32 ": Error, invalid endpoints: "
238  "%s\n", path_controller_number_, endpoints_str_.c_str());
239  return false;
240  }
241 
242  // Extract the maximum line rate, in kilobits per second.
243  config_name = config_prefix;
244  config_name.append(".MaxLineRateKbps");
245  max_line_rate_ = config_info.GetDouble(config_name, kDefLineRate);
246 
247  if (max_line_rate_ < 0.0)
248  {
249  LogE(kClassName, __func__, "Sond %" PRIu32 ": Invalid maximum line rate "
250  "%f kbps specified.\n", path_controller_number_, max_line_rate_);
251  return false;
252  }
253 
254  // Extract the estimated packet delivery delay (PDD) value, in seconds.
255  config_name = config_prefix;
256  config_name.append(".EstPddSec");
257  cb_pdd_ = config_info.GetDouble(config_name, kDefaultPddSec);
258 
259  if (cb_pdd_ < 0.000001)
260  {
261  LogD(kClassName, __func__, "Sond %" PRIu32 ": PDD %f seconds specified, "
262  "disabling PDD reporting.\n", path_controller_number_, cb_pdd_);
263  }
264 
265  // Compute and set the data packet transmit queue size in packets.
266  size_t xmit_thresh = config_info.GetUint("Bpf.XmitQueueThreshBytes",
267  kDefaultBpfXmitQueueThreshBytes);
268  size_t data_queue_size = COMPUTE_XMIT_QUEUE_SIZE(xmit_thresh);
269 
270  ef_data_pkt_queue_.SetQueueLimits(data_queue_size);
271  ef_data_pkt_queue_.set_drop_policy(NO_DROP);
272 
273  data_pkt_queue_.SetQueueLimits(data_queue_size);
274  data_pkt_queue_.set_drop_policy(NO_DROP);
275 
276  // Set the maximum control packet transmit queue size, in packets.
277  control_pkt_queue_.SetQueueLimits(kControlQueueSize);
278  control_pkt_queue_.set_drop_policy(NO_DROP);
279 
280  // Create the UDP socket to communicate with the remote SOND.
281  if ((udp_fd_ = socket(AF_INET, SOCK_DGRAM, IPPROTO_UDP)) < 0)
282  {
283  LogF(kClassName, __func__, "Sond %" PRIu32 ": Error creating socket: "
284  "%s\n", path_controller_number_, strerror(errno));
285  return false;
286  }
287 
288  // Enable port number reuse on the socket.
289  int opt_val = 1;
290 
291  if (setsockopt(udp_fd_, SOL_SOCKET, SO_REUSEPORT, &opt_val,
292  sizeof(opt_val)) < 0)
293  {
294  LogE(kClassName, __func__, "Sond %" PRIu32 ": Error enabling port number "
295  "reuse: %s\n", path_controller_number_, strerror(errno));
296  return false;
297  }
298 
299  // Bind the socket to the local address.
300  struct sockaddr_in addr;
301 
302  local_endpt_.ToSockAddr(reinterpret_cast<struct sockaddr*>(&addr));
303 
304  if (bind(udp_fd_, reinterpret_cast<struct sockaddr*>(&addr),
305  sizeof(addr)) < 0)
306  {
307  LogE(kClassName, __func__, "Sond %" PRIu32 ": Error binding local "
308  "address %s: %s\n", path_controller_number_,
309  local_endpt_.ToString().c_str(), strerror(errno));
310  return false;
311  }
312 
313  // Connect the socket to the remote address.
314  remote_endpt_.ToSockAddr(reinterpret_cast<struct sockaddr*>(&addr));
315 
316  if (connect(udp_fd_, reinterpret_cast<struct sockaddr *>(&addr),
317  sizeof(addr)) < 0)
318  {
319  LogE(kClassName, __func__, "Sond %" PRIu32 ": Error connecting to remote "
320  "address %s: %s\n", path_controller_number_,
321  remote_endpt_.ToString().c_str(), strerror(errno));
322  return false;
323  }
324 
325  // Log the configuration information.
326  LogC(kClassName, __func__, "Sond %" PRIu32 " configuration:\n",
327  path_controller_number_);
328 
329  LogC(kClassName, __func__, "Type : Sond\n");
330  LogC(kClassName, __func__, "Label : %s\n",
331  label_.c_str());
332  LogC(kClassName, __func__, "Endpoints : %s->%s\n",
333  local_endpt_.ToString().c_str(), remote_endpt_.ToString().c_str());
334  LogC(kClassName, __func__, "Max Line Rate : %f Kbps\n",
335  max_line_rate_);
336  LogC(kClassName, __func__, "EF Data Transmit Queue Size : %zu packets\n",
337  data_queue_size);
338  LogC(kClassName, __func__, "Data Transmit Queue Size : %zu packets\n",
339  data_queue_size);
340  LogC(kClassName, __func__, "Control Transmit Queue Size : %zu packets\n",
341  kControlQueueSize);
342  LogC(kClassName, __func__, "PDD Maximum Period : %f seconds\n",
343  cb_max_period_);
344  LogC(kClassName, __func__, "PDD Value : %f seconds\n",
345  cb_pdd_);
346 
347  LogC(kClassName, __func__, "Sond %" PRIu32 " configuration complete.\n",
348  path_controller_number_);
349 
350  bpf_->ProcessCapacityUpdate(this, (max_line_rate_ * 1000.0),
351  (max_line_rate_ * 800.0));
352 
353  return true;
354 }

383 bool Sond::SendPacket(Packet* pkt)
384 {
385  DoCallbacks();
386 
387  if (pkt == NULL)
388  {
389  return false;
390  }
391 
392  // Get the packet's type.
393  int pkt_type = pkt->GetRawType();
394 
395  // Add any necessary Packet object metadata headers to the packet before it
396  // is sent.
397  if (NeedsMetadataHeaders(pkt))
398  {
399  if (!AddMetadataHeaders(pkt))
400  {
401  LogE(kClassName, __func__, "Sond %" PRIu32 ": Error adding necessary "
402  "metadata headers to packet.\n", path_controller_number_);
403  }
404  }
405  else
406  {
407  pkt->SetMetadataHeaderLengthInBytes(0);
408  }
409 
410  // Get the resulting packet's length in bytes.
411  size_t pkt_len = (pkt->GetMetadataHeaderLengthInBytes() +
412  pkt->GetLengthInBytes());
413  size_t drop_len = 0;
414 
415  // Enqueue the packet based on its type.
416  switch (pkt_type)
417  {
418  case IPV4_PACKET:
419  if (pkt->GetLatencyClass() == LOW_LATENCY)
420  {
421  if (!ef_data_pkt_queue_.Enqueue(pkt))
422  {
423  LogF(kClassName, __func__, "Sond %" PRIu32 ": EF data packet "
424  "transmit queue overflow.\n", path_controller_number_);
425  return false;
426  }
427 
428  LogD(kClassName, __func__, "EF DATA: Enqueued in Sond %" PRIu32
429  ", %zu bytes.\n", path_controller_number_, pkt_len);
430  }
431  else
432  {
433  if (!data_pkt_queue_.Enqueue(pkt))
434  {
435  LogF(kClassName, __func__, "Sond %" PRIu32 ": Data packet transmit "
436  "queue overflow.\n", path_controller_number_);
437  return false;
438  }
439 
440  LogD(kClassName, __func__, "DATA: Enqueued in Sond %" PRIu32 ", %zu "
441  "bytes.\n", path_controller_number_, pkt_len);
442  }
443  break;
444 
445  case QLAM_PACKET:
446  // Store this new QLAM packet, replacing any old QLAM packet.
447  if (qlam_pkt_ptr_ != NULL)
448  {
449  drop_len = (qlam_pkt_ptr_->GetMetadataHeaderLengthInBytes() +
450  qlam_pkt_ptr_->GetLengthInBytes());
451  TRACK_UNEXPECTED_DROP(kClassName, packet_pool_);
452  packet_pool_.Recycle(qlam_pkt_ptr_);
453  qlam_pkt_ptr_ = NULL;
454  }
455 
456  qlam_pkt_ptr_ = pkt;
457 
458  LogD(kClassName, __func__, "QLAM: Enqueued in Sond %" PRIu32 ", %zu "
459  "bytes\n", path_controller_number_, pkt_len);
460  break;
461 
462  case LSA_PACKET:
463  if (!control_pkt_queue_.Enqueue(pkt))
464  {
465  LogF(kClassName, __func__, "Sond %" PRIu32 ": Control packet "
466  "transmit queue overflow.\n", path_controller_number_);
467  return false;
468  }
469 
470  LogD(kClassName, __func__, "LSA: Enqueued in Sond %" PRIu32 ", %zu "
471  "bytes\n", path_controller_number_, pkt_len);
472  break;
473 
474  case ZOMBIE_PACKET:
475  if (!data_pkt_queue_.Enqueue(pkt))
476  {
477  LogF(kClassName, __func__, "Sond %" PRIu32 ": Data packet transmit "
478  "queue overflow.\n", path_controller_number_);
479  return false;
480  }
481 
482  LogD(kClassName, __func__, "ZOMBIE: Enqueued in Sond %" PRIu32 ", %zu "
483  "bytes.\n", path_controller_number_, pkt_len);
484  break;
485 
486  default:
487  LogE(kClassName, __func__, "Sond %" PRIu32 ": Unknown packet type "
488  "%d (0x%02x).\n", path_controller_number_, pkt_type,
489  static_cast<unsigned int>(pkt_type));
490  return false;
491  }
492 
493  // Update the total number of bytes queued. Note that this only works
494  // correctly when the three PacketQueue objects have their drop policies set
495  // to NO_DROP.
496  total_bytes_queued_ -= drop_len;
497  total_bytes_queued_ += pkt_len;
498 
499  // If we are currently IDLE and there is a non-zero rate, then reset the
500  // start time and schedule the next transmission.
501  if ((xmit_pkt_ptr_ == NULL) && (max_line_rate_ > 0.0))
502  {
503  if (xmit_start_time_.GetNow())
504  {
505  xmit_delta_time_ = 0.0;
506  ScheduleNextPacket(xmit_start_time_);
507  }
508  else
509  {
510  LogF(kClassName, __func__, "Unable to get current time.\n");
511  }
512  }
513 
514  return true;
515 }

518 void Sond::ServiceFileDescriptor(int fd, FdEvent event)
519 {
520  DoCallbacks();
521 
522  if (event != kFdEventRead)
523  {
524  LogW(kClassName, __func__, "Only file descriptor read events are "
525  "supported.\n");
526  return;
527  }
528 
529  Packet* packet = packet_pool_.Get(PACKET_NOW_TIMESTAMP);
530 
531  if (packet == NULL)
532  {
533  LogF(kClassName, __func__, "Unable to allocate Packet.\n");
534  return;
535  }
536 
537  // Receive the next packet from the UDP socket.
538  ssize_t bytes_read = ::recv(fd, packet->GetBuffer(),
539  packet->GetMaxLengthInBytes(), 0);
540 
541  if (bytes_read > 0)
542  {
543  LogD(kClassName, __func__, "RECV: Sond %" PRIu32
544  ", pkt size: %zd bytes\n", path_controller_number_, bytes_read);
545 
546  packet->SetLengthInBytes(bytes_read);
547 
548  // Process and remove any Packet object metadata headers from the packet.
549  if (!ProcessMetadataHeaders(packet))
550  {
551  LogE(kClassName, __func__, "Sond %" PRIu32 ": Error processing "
552  "metadata headers.\n", path_controller_number_);
553  }
554 
555  if (bpf_ != NULL)
556  {
557  // Pass the packet to the BPF for processing.
558  bpf_->ProcessRcvdPacket(packet, this);
559  }
560  else
561  {
562  LogF(kClassName, __func__, "BPF pointer is NULL.\n");
563  TRACK_UNEXPECTED_DROP(kClassName, packet_pool_);
564  packet_pool_.Recycle(packet);
565  }
566  }
567  else if (bytes_read == 0)
568  {
569  LogE(kClassName, __func__, "Sond %" PRIu32 ": Zero byte recv().\n",
570  path_controller_number_);
571 
572  TRACK_UNEXPECTED_DROP(kClassName, packet_pool_);
573  packet_pool_.Recycle(packet);
574  }
575  else
576  {
577  // Do not log connection refused errors. These are caused by the peer's
578  // socket not being open yet, which can happen at the beginning or end of
579  // a connection.
580  if (errno != ECONNREFUSED)
581  {
582  LogE(kClassName, __func__, "Sond %" PRIu32 ": Error in recv(): %s\n",
583  path_controller_number_, strerror(errno));
584  }
585 
586  TRACK_UNEXPECTED_DROP(kClassName, packet_pool_);
587  packet_pool_.Recycle(packet);
588  }
589 }

649 //============================================================================
650 bool Sond::ParseEndpointsString(const string& ep_str)
651 {
652  // The format to parse is:
653  // LOCAL_IP[:LOCAL_PORT]->REMOTE_IP[:REMOTE_PORT]
654 
655  // Start by tokenizing on the required "->" characters.
656  List<string> tokens;
657  StringUtils::Tokenize(ep_str, "->", tokens);
658 
659  if (tokens.size() != 2)
660  {
661  return false;
662  }
663 
664  string lep_str;
665  string rep_str;
666  tokens.Pop(lep_str);
667  tokens.Peek(rep_str);
668 
669  // IPv4 addresses in dot-decimal notation require at least 7 characters.
670  if ((lep_str.size() < 7) || (rep_str.size() < 7))
671  {
672  return false;
673  }
674 
675  // Add the default port number if port numbers are not specified.
676  if (lep_str.find(":") == string::npos)
677  {
678  lep_str.append(":");
679  lep_str.append(kDefaultPort);
680  }
681 
682  if (rep_str.find(":") == string::npos)
683  {
684  rep_str.append(":");
685  rep_str.append(kDefaultPort);
686  }
687 
688  // Convert the strings to endpoints.
689  if ((!local_endpt_.SetEndpoint(lep_str)) ||
690  (!remote_endpt_.SetEndpoint(rep_str)))
691  {
692  return false;
693  }
694 
695  // The addresses and port numbers must not be zero.
696  if ((local_endpt_.address() == 0) || (local_endpt_.port() == 0) ||
697  (remote_endpt_.address() == 0) || (remote_endpt_.port() == 0))
698  {
699  return false;
700  }
701 
702  return true;
703 }

706 bool Sond::SetMaxLineRate(const char* value)
707 {
708  double rate = StringUtils::GetDouble(value, -1.0);
709 
710  if (rate < 0.0)
711  {
712  LogE(kClassName, __func__, "Sond %" PRIu32 ": Maximum line rate change "
713  "failed, bad rate %f kbps specified, leaving at %f kbps.\n",
714  path_controller_number_, rate, max_line_rate_);
715  return false;
716  }
717 
718  LogI(kClassName, __func__, "Sond %" PRIu32 ": Maximum line rate change "
719  "from %f kbps to %f kbps.\n", path_controller_number_, max_line_rate_,
720  rate);
721 
722  // Update the maximum line rate.
723  max_line_rate_ = rate;
724 
725  // If we are currently IDLE, there is a non-zero rate, and there is a packet
726  // ready to transmit, then reset the start time and schedule the next
727  // transmission.
728  if ((xmit_pkt_ptr_ == NULL) && (max_line_rate_ > 0.0) &&
729  IsPacketReadyToXmit())
730  {
731  if (xmit_start_time_.GetNow())
732  {
733  xmit_delta_time_ = 0.0;
734  ScheduleNextPacket(xmit_start_time_);
735  }
736  else
737  {
738  LogF(kClassName, __func__, "Unable to get current time.\n");
739  }
740  }
741 
742  bpf_->ProcessCapacityUpdate(this, (max_line_rate_ * 1000.0),
743  (max_line_rate_ * 800.0));
744 
745  return true;
746 }

749 void Sond::DoCallbacks()
750 {
751  // Get the current time.
752  Time now = Time::Now();
753 
754  // Report the estimated packet delivery delay (PDD) to the backpressure
755  // forwarder if too much time has passed since the last report.
756  if ((cb_pdd_ >= 0.000001) && (now > cb_prev_time_.Add(cb_max_period_)))
757  {
758  // Update the report time first due to possible re-entrant calls.
759  cb_prev_time_ = now;
760 
761  bpf_->ProcessPktDelDelay(this, cb_pdd_, 0.0);
762  }

766 void Sond::ScheduleNextPacket(const Time& now)
767 {
768  // Send as many packets that are ready for transmitting as possible until
769  // either the queue is empty or a timer needs to be set.
770  while (true)
771  {
772  // Dequeue the next QLAM, control, or data packet to transmit.
773  if (qlam_pkt_ptr_ != NULL)
774  {
775  xmit_pkt_ptr_ = qlam_pkt_ptr_;
776  qlam_pkt_ptr_ = NULL;
777  }
778  else if (ef_data_pkt_queue_.GetCount() > 0)
779  {
780  xmit_pkt_ptr_ = ef_data_pkt_queue_.Dequeue();
781 
782  if (xmit_pkt_ptr_ == NULL)
783  {
784  LogE(kClassName, __func__, "Sond %" PRIu32 ": Dequeued NULL packet "
785  "from EF data transmit queue.\n", path_controller_number_);
786  break;
787  }
788  }
789  else if (control_pkt_queue_.GetCount() > 0)
790  {
791  xmit_pkt_ptr_ = control_pkt_queue_.Dequeue();
792 
793  if (xmit_pkt_ptr_ == NULL)
794  {
795  LogE(kClassName, __func__, "Sond %" PRIu32 ": Dequeued NULL packet "
796  "from control transmit queue.\n", path_controller_number_);
797  break;
798  }
799  }
800  else if (data_pkt_queue_.GetCount() > 0)
801  {
802  xmit_pkt_ptr_ = data_pkt_queue_.Dequeue();
803 
804  if (xmit_pkt_ptr_ == NULL)
805  {
806  LogE(kClassName, __func__, "Sond %" PRIu32 ": Dequeued NULL packet "
807  "from data transmit queue.\n", path_controller_number_);
808  break;
809  }
810  }
811  else
812  {
813  LogD(kClassName, __func__, "Sond %" PRIu32 ": Transmit queues now "
814  "empty.\n", path_controller_number_);
815  break;
816  }
817 
818  // Update the total number of bytes queued. Note that this only works
819  // correctly when the three PacketQueue objects have their drop policies
820  // set to NO_DROP.
821  size_t pkt_len = (xmit_pkt_ptr_->GetMetadataHeaderLengthInBytes() +
822  xmit_pkt_ptr_->GetLengthInBytes());
823 
824  total_bytes_queued_ -= pkt_len;
825 
826  // Compute the packet's send time, taking into account its transmission
827  // delay. It is the transmission delay for each packet that controls the
828  // overall transmission rate.
829  Time xmit_time(xmit_start_time_);
830  double delta_sec = ((static_cast<double>(pkt_len) * 8.0) /
831  (max_line_rate_ * 1000.0));
832  xmit_delta_time_ += delta_sec;
833  xmit_time = xmit_time.Add(xmit_delta_time_);
834 
835  LogD(kClassName, __func__, "TIMER: Sond %" PRIu32 ": Scheduling packet, "
836  "now = %s, xmit_start_time_ = %s, delta_sec = %f sec, "
837  "xmit_delta_time_ = %f sec, xmit_time = %s, %zu bytes.\n",
838  path_controller_number_, now.ToString().c_str(),
839  xmit_start_time_.ToString().c_str(), delta_sec, xmit_delta_time_,
840  xmit_time.ToString().c_str(), pkt_len);
841 
842  // Compare now with the packet's transmission time.
843  if (xmit_time <= now)
844  {
845  // No need for setting a timer. Send the packet now. When this
846  // returns, xmit_pkt_ptr_ is guaranteed to be set to NULL.
847  XmitPacket();
848  }
849  else
850  {
851  // Set a timer for the packet's transmission time.
852  Time delta_time = xmit_time - now;
853  CallbackNoArg<Sond> cb(this, &Sond::TimerCallback);
854 
855  if (!timer_.StartTimer(delta_time, &cb, xmit_timer_handle_))
856  {
857  LogF(kClassName, __func__, "Sond %" PRIu32 ": Error starting timer "
858  "for %s.\n", path_controller_number_, delta_time.ToString().c_str());
859  }
860 
861  break;
862  }
863  }
864 }

866 //============================================================================
867 void Sond::TimerCallback()
868 {
869  // Transmit the packet stored in xmit_pkt_ptr_ now that its transmission
870  // delay is over. When this returns, xmit_pkt_ptr_ is guaranteed to be set
871  // to NULL.
872  XmitPacket();
873 
874  // We are now IDLE (xmit_pkt_ptr_ is NULL). If there is a non-zero rate,
875  // and there is a packet ready to transmit, then schedule the next
876  // transmission. Do not reset the start time.
877  if ((max_line_rate_ > 0.0) && IsPacketReadyToXmit())
878  {
879  Time now;
880 
881  if (now.GetNow())
882  {
883  ScheduleNextPacket(now);
884  }
885  else
886  {
887  LogF(kClassName, __func__, "Unable to get current time.\n");
888  }
889  }
890 }

893 void Sond::XmitPacket()
894 {
895  if (xmit_pkt_ptr_ == NULL)
896  {
897  return;
898  }
899 
900  // Send the packet stored in xmit_pkt_ptr_.
901  size_t pkt_len = static_cast<size_t>(
902  xmit_pkt_ptr_->GetMetadataHeaderLengthInBytes() +
903  xmit_pkt_ptr_->GetLengthInBytes());
904 
905  ssize_t bytes_sent = send(udp_fd_,
906  xmit_pkt_ptr_->GetMetadataHeaderBuffer(),
907  pkt_len, 0);
908 
909  if (bytes_sent >= 0)
910  {
911  if (bytes_sent == static_cast<ssize_t>(pkt_len))
912  {
913  LogD(kClassName, __func__, "SEND: Sond %" PRIu32 ", packet size: %zd "
914  "bytes.\n", path_controller_number_, bytes_sent);
915 
916  total_bytes_sent_ += static_cast<uint32_t>(bytes_sent);
917  }
918  else
919  {
920  LogE(kClassName, __func__, "Error: Sond %" PRIu32 ": sent %zd bytes of "
921  "%zu byte packet.\n", path_controller_number_, bytes_sent, pkt_len);
922  }
923  }
924  else
925  {
926  LogE(kClassName, __func__, "Sond %" PRIu32 ": Error in sendto(): %s.\n",
927  path_controller_number_, strerror(errno));
928  }
929 
930  // Delete the packet.
931  packet_pool_.Recycle(xmit_pkt_ptr_);
932  xmit_pkt_ptr_ = NULL;
933 }
