74 bool EdgeIf::Open()
75 {
76  // Open the transmit socket.
77  if ((xmt_sock_ = socket(AF_INET, SOCK_RAW, IPPROTO_RAW)) < 0)
78  {
79  LogE(kClassName, __func__, "socket error: %s\n", strerror(errno));
80  return false;
81  }
82 
83  // Tell the kernel that we will provide the headers for the transmit
84  // socket.
85  int on = 1;
86  if (setsockopt(xmt_sock_, IPPROTO_IP, IP_HDRINCL, &on, sizeof(on)) < 0)
87  {
88  LogE(kClassName, __func__, "setsockopt IP_HDRINCL error: %s\n",
89  strerror(errno));
90  CloseSockets();
91  return false;
92  }
93 
94  // Set the multicast interface for the transmit socket.
95  struct in_addr if_addr;
96  if (inet_aton(config_.inbound_dev_ip_str().c_str(), &if_addr) == 0)
97  {
98  LogE(kClassName, __func__, "Error getting inbound dev IP Address.\n");
99  CloseSockets();
100  return false;
101  }
102 
103  if (setsockopt(xmt_sock_, IPPROTO_IP, IP_MULTICAST_IF,
104  (const void*)&if_addr, sizeof(if_addr)) < 0)
105  {
106  LogE(kClassName, __func__, "setsockopt IP_MULTICAST_IF error: %s\n",
107  strerror(errno));
108  CloseSockets();
109  return false;
110  }
111 
112  // Open the receive socket, a packet socket for receiving packets from the
113  // kernel. Use SOCK_DGRAM for the type so we don't receive Ethernet
114  // headers. Use ETH_P_IP, instead of ETH_P_ALL, for the protocol so that we
115  // are only notified of incoming packets. Otherwise, we hear echos of the
116  // packets that we transmit.
117  if ((rcv_sock_ = socket(PF_PACKET, SOCK_DGRAM, htons(ETH_P_IP))) < 0)
118  {
119  LogE(kClassName, __func__, "socket error: %s\n", strerror(errno));
120  CloseSockets();
121  return false;
122  }
123 
124  // Bind the receive socket to the identified interface.
125  struct sockaddr_ll bind_addr;
126  memset(&bind_addr, 0, sizeof(bind_addr));
127  bind_addr.sll_family = PF_PACKET;
128  bind_addr.sll_ifindex = if_nametoindex(config_.inbound_dev_name().c_str());
129  bind_addr.sll_protocol = htons(ETH_P_IP);
130 
131  if (bind(rcv_sock_, (struct sockaddr*)&bind_addr, sizeof(bind_addr)) < 0)
132  {
133  LogE(kClassName, __func__, "bind error: %s\n", strerror(errno));
134  CloseSockets();
135  return false;
136  }
137 
138  // Attach the Berkeley Packet Filter to the receive socket.
139  struct sock_fprog* bpf = config_.bpf();
140  if (setsockopt(rcv_sock_, SOL_SOCKET, SO_ATTACH_FILTER, bpf, sizeof(*bpf))
141  < 0)
142  {
143  LogE(kClassName, __func__, "setsockopt SO_ATTACH_FILTER error: %s\n",
144  strerror(errno));
145  CloseSockets();
146  return false;
147  }
148 
149  // Set the receive socket to non-blocking mode.
150  if (fcntl(rcv_sock_, F_SETFL, fcntl(rcv_sock_, F_GETFL, 0) | O_NONBLOCK)
151  < 0)
152  {
153  LogE(kClassName, __func__, "fcntl error: %s\n", strerror(errno));
154  CloseSockets();
155  return false;
156  }
157 
158  if (!config_.external_plumbing())
159  {
160  List<string>& add_rule_list = config_.iptables_add_rule_list();
161  List<string>::WalkState rule_ws;
162  rule_ws.PrepareForWalk();
163 
164  string rule;
165  while (add_rule_list.GetNextItem(rule_ws, rule))
166  {
167  LogD(kClassName, __func__, "iptables rule: %s\n", rule.c_str());
168  ExeSysCmd(rule.c_str());
169  }
170  }
171 
172  return true;
173 }

188 ssize_t EdgeIf::Recv(Packet* pkt, const size_t offset)
189 {
190  if (pkt == NULL)
191  {
192  LogF(kClassName, __func__, "Error: pkt was NULL\n");
193  return -1;
194  }
195 
196  ssize_t num_read = recvfrom(rcv_sock_, pkt->GetBuffer(offset),
197  pkt->GetMaxLengthInBytes() - offset, 0,
198  NULL, NULL);
199 
200  if (num_read < 0)
201  {
202  // Multiple Recv() calls are expected after a single indication from
203  // select() that there is data to read, so don't log EAGAIN.
204  if (errno != EAGAIN)
205  {
206  LogE(kClassName, __func__, "Failed to receive with error: %s\n",
207  strerror(errno));
208  }
209  pkt->SetLengthInBytes(0);
210  return -1;
211  }
212 
213  // Set the tentative packet length
214  pkt->SetLengthInBytes(num_read);
215 
216  // Because of ethernet minimum size rules, the number read
217  // may be bigger that the actual ipv4 packet. Adjust
218  // accordingly
219 
220  size_t real_len;
221  if (pkt->GetIpLen(real_len))
222  {
223  if (num_read != (ssize_t)real_len)
224  {
225  num_read = real_len;
226  pkt->SetLengthInBytes(num_read);
227  }
228  }
229 
230  LogD(kClassName, __func__, "%zd bytes read.\n", num_read);
231  return num_read;
232 }

235 ssize_t EdgeIf::Send(const Packet* pkt)
236 {
237  if (pkt == NULL)
238  {
239  LogF(kClassName, __func__, "Error: pkt was NULL\n");
240  return -1;
241  }
242 
243  uint16_t dport;
244  if (!pkt->GetDstPort(dport))
245  {
246  LogE(kClassName, __func__, "Error getting destination port.\n");
247  return -1;
248  }
249 
250  uint32_t daddr;
251  if (!pkt->GetIpDstAddr(daddr))
252  {
253  LogE(kClassName, __func__, "Error getting packet's destination "
254  "address.\n");
255  return -1;
256  }
257 
258  struct sockaddr_in addr;
259  memset(&addr, 0, sizeof(addr));
260  addr.sin_family = AF_INET;
261  addr.sin_port = dport;
262  addr.sin_addr.s_addr = daddr;
263 
264  ssize_t num_written;
265  if ((num_written = sendto(xmt_sock_, pkt->GetBuffer(),
266  pkt->GetLengthInBytes(), 0,
267  (struct sockaddr*)&addr, sizeof(addr)))
268  != static_cast<ssize_t>(pkt->GetLengthInBytes()))
269  {
270  LogE(kClassName, __func__, "sendto error: %s, wrote=%zd, expected=%zd\n",
271  strerror(errno), num_written, pkt->GetLengthInBytes());
272  return -1;
273  }
274 
275  LogD(kClassName, __func__, "%zd bytes written to edge interface.\n",
276  num_written);
277 
278  return num_written;
279 }

307 void EdgeIf::CloseEdgeIf()
308 {
309  if (IsOpen())
310  {
311  CloseSockets();
312 
313  if (!config_.external_plumbing())
314  {
315  List<string>& del_rule_list = config_.iptables_del_rule_list();
316  List<string>::WalkState rule_ws;
317  rule_ws.PrepareForWalk();
318 
319  string rule;
320  while (del_rule_list.GetNextItem(rule_ws, rule))
321  {
322  LogD(kClassName, __func__, "iptables rule: %s\n", rule.c_str());
323  ExeSysCmd(rule.c_str());
324  }
325  }
326  }

330 void EdgeIf::CloseSockets()
331 {
332  // Close the transmit socket.
333  if (xmt_sock_ != kNoFd)
334  {
335  if (close(xmt_sock_) < 0)
336  {
337  LogE(kClassName, __func__, "Error closing transmit socket: %s.\n",
338  strerror(errno));
339  }
340  xmt_sock_ = kNoFd;
341  }
342 
343  // Close the receive socket.
344  if (rcv_sock_ != kNoFd)
345  {
346  if (close(rcv_sock_) < 0)
347  {
348  LogE(kClassName, __func__, "Error closing receive socket: %s.\n",
349  strerror(errno));
350  }
351  rcv_sock_ = kNoFd;
352  }
353 }
