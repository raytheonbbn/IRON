202  bool OrderedInsert(const K& k, const V& v, const O& order)
203  {
204  typename OrderedList<V, O>::LLElem* lle = linked_list_.GetLle(v,
205  order);
206  if (lle == NULL)
207  {
208  return false;
209  }
210  MTElem* mte = NULL;
211 
212  if (mte_pool_ == NULL)
213  {
214  mte = new (std::nothrow) MTElem(lle, v, order);
215 
216  if (mte == NULL)
217  {
218  linked_list_.Recycle(lle);
219  return false;
220  }
221  }
222  else
223  {
224  mte = mte_pool_;
225  mte_pool_ = mte->next;
226  mte->llelem = lle;
227  mte->val = v;
228  mte->next = NULL;
229  }
230 
231  if(hash_table_.Insert(k, mte))
232  {
233  linked_list_.OrderedInsert(lle);
234  return true;
235  }
236  linked_list_.Recycle(lle);
237  mte->next = mte_pool_;
238  mte_pool_ = mte;
239  return false;
240  }
