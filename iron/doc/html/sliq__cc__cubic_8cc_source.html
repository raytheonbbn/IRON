357 void Cubic::OnRttUpdate(StreamId stream_id, const Time& ack_time,
358  PktTimestamp send_ts, PktTimestamp recv_ts,
359  PktSeqNumber seq_num, PktSeqNumber cc_seq_num,
360  const Time& rtt, uint32_t bytes, float cc_val)
361 {
362 #ifdef SLIQ_CC_DEBUG
363  LogD(kClassName, __func__, "Conn %" PRIEndptId ": Args stream %" PRIStreamId
364  " seq %" PRIPktSeqNumber " cc_seq %" PRIPktSeqNumber " rtt %f bytes %"
365  PRIu32 " cc_val %f.\n", conn_id_, stream_id, seq_num, cc_seq_num,
366  rtt.ToDouble(), bytes, static_cast<double>(cc_val));
367 #endif
368 
369  // Note that the reported RTT is already limited to positive, non-zero
370  // values, so there is no need to test for negative values in this method.
371 
372  // Discard delay samples right after fast recovery (for 1 second). Note
373  // that ack_time is approximately the current time. As per Linux-4.6.3.
374  Time limit(1);
375 
376  if ((!cubic_epoch_start_.IsZero()) &&
377  ((ack_time - cubic_epoch_start_) < limit))
378  {
379 #ifdef SLIQ_CC_DEBUG
380  LogD(kClassName, __func__, "Conn %" PRIEndptId ": Rtt update ignored, "
381  "too soon after fast recovery.\n", conn_id_);
382 #endif
383 
384  return;
385  }
386 
387  // Update the minimum observed delay. From the CUBIC algorithm "On each
388  // ACK" step.
389  // \todo What if a route change increases the base RTT?
390  if (cubic_delay_min_.IsZero() || (cubic_delay_min_ > rtt))
391  {
392  cubic_delay_min_ = rtt;
393 
394 #ifdef SLIQ_CC_DEBUG
395  LogD(kClassName, __func__, "Conn %" PRIEndptId ": Update delay_min %f.\n",
396  conn_id_, cubic_delay_min_.ToDouble());
397 #endif
398  }
399 
400  // Perform HyStart delay increase detection. Note that HyStart triggers
401  // when cwnd is larger than some threshold.
402  if (config_hystart_ && (cwnd_ < ssthresh_) && (cwnd_ >= kHystartLowWindow))
403  {
404  HystartDelayUpdate(rtt);
405  }
406 
407 #ifdef SLIQ_CC_DEBUG
408  PrintState("OnRttUpdate()");
409 #endif
410 }

458 void Cubic::OnAckPktProcessingDone(const Time& ack_time)
459 {
460 #ifdef SLIQ_CC_DEBUG
461  LogD(kClassName, __func__, "Conn %" PRIEndptId ": ACK processing done.\n",
462  conn_id_);
463 #endif
464 
465  // If required, update the lowest unACKed sequence number for all of the
466  // streams in the connection.
467  if (update_snd_una_)
468  {
469  UpdateSndUna();
470  }
471 
472  // Check if fast recovery must be entered or exited.
473  if (in_fast_recovery_)
474  {
475  // Per RFC 6675, section 5, action (A), exit fast recovery when HighACK >
476  // RecoveryPoint. Note that snd_una_ is just beyond HighACK.
477  if (SEQ_GEQ(snd_una_, prr_recovery_point_))
478  {
479  EndPrr();
480  }
481  }
482  else
483  {
484  // Per RFC 6675, section 5.1, paragraph 2, avoid entering fast recovery
485  // after an RTO occurs until HighACK >= RecoveryPoint.
486  if ((enter_fast_recovery_) && SEQ_GEQ(snd_una_, prr_recovery_point_))
487  {
488  BeginPrr();
489  }
490  }
491 
492  // Update the state based on the ACK packets that have just been processed.
493  if (in_fast_recovery_)
494  {
495  // In fast recovery.
496  UpdatePrr();
497  }
498  else
499  {
500  // Perform HyStart ACK-train bandwidth-delay product detection. Note that
501  // HyStart triggers when cwnd is larger than some threshold.
502  if (config_hystart_ && (cwnd_ < ssthresh_) &&
503  (cwnd_ >= kHystartLowWindow))
504  {
505  HystartAckTrainUpdate(ack_time);
506  }
507 
508  // In slow start or congestion avoidance. Update the congestion window.
509  UpdateCwnd(ack_time);
510  }
511 
512 #ifdef SLIQ_CC_DEBUG
513  PrintState("OnAckPktProcessingDone()");
514 #endif
515 }

518 PktSeqNumber Cubic::OnPacketSent(StreamId stream_id, const Time& send_time,
519  PktSeqNumber seq_num, uint32_t pld_bytes,
520  uint32_t /* tot_bytes */, float& cc_val)
521 {
522 #ifdef SLIQ_CC_DEBUG
523  LogD(kClassName, __func__, "Conn %" PRIEndptId ": Args stream %" PRIStreamId
524  " seq %" PRIPktSeqNumber " bytes %" PRIu32 ".\n", conn_id_, stream_id,
525  seq_num, pld_bytes);
526 #endif
527 
528  // Update the maximum bytes in flight per RTT. Don't forget to add the size
529  // of this packet being sent.
530  int64_t bytes_out = (bytes_in_flight_ + pld_bytes);
531 
532  if (SEQ_GEQ(snd_una_, max_bytes_seq_) || (bytes_out > max_bytes_out_))
533  {
534 #ifdef SLIQ_CC_DEBUG
535  LogD(kClassName, __func__, "Conn %" PRIEndptId ": Update max bytes out "
536  "from %" PRId64 " to %" PRId64" seq %" PRIPktSeqNumber ".\n",
537  conn_id_, max_bytes_out_, bytes_out, snd_nxt_);
538 #endif
539 
540  max_bytes_out_ = bytes_out;
541  max_bytes_seq_ = snd_nxt_;
542  }
543 
544  // Assign a congestion control sequence number to the packet.
545  PktSeqNumber cc_seq_num = snd_nxt_;
546  snd_nxt_++;
547 
548  // Update the byte offset for the next packet to be sent.
549  snd_nxt_byte_offset_ += pld_bytes;
550  pkt_byte_offset_[snd_nxt_ % kMaxCongCtrlWindowPkts] = snd_nxt_byte_offset_;
551 
552 #ifdef SLIQ_CC_DEBUG
553  LogD(kClassName, __func__, "Conn %" PRIEndptId ": Assign seq %"
554  PRIPktSeqNumber " to ccseq %" PRIPktSeqNumber ".\n", conn_id_, seq_num,
555  cc_seq_num);
556 #endif
557 
558  // If the application was idle for a while, then shift the
559  // cubic_epoch_start_ to keep the cwnd growth to a cubic curve.
560  if (bytes_in_flight_ == 0)
561  {
562  if (!cubic_epoch_start_.IsZero())
563  {
564  Time delta = (send_time - last_app_send_time_);
565  Time limit = Time::FromUsec(kCubicIdleThreshUsec);
566 
567  if (delta > limit)
568  {
569  cubic_epoch_start_ += delta;
570 
571  if (cubic_epoch_start_ > send_time)
572  {
573  cubic_epoch_start_ = send_time;
574  }
575 
576 #ifdef SLIQ_CC_DEBUG
577  LogD(kClassName, __func__, "Conn %" PRIEndptId ": App was idle, "
578  "shift epoch start to %s.\n", conn_id_,
579  cubic_epoch_start_.ToString().c_str());
580 #endif
581  }
582  }
583  }
584 
585  // Fast recovery must keep track of all packet transmissions.
586  if (in_fast_recovery_)
587  {
588  prr_out_ += pld_bytes;
589  }
590 
591  // Store the last send time.
592  last_app_send_time_ = send_time;
593  last_proto_send_time_ = send_time;
594 
595  // Update the next send time.
596  UpdateNextSendTime(send_time, pld_bytes);
597 
598 #ifdef SLIQ_CC_DEBUG
599  LogD(kClassName, __func__, "Conn %" PRIEndptId ": Sent data on stream %"
600  PRIStreamId " seq %" PRIPktSeqNumber " bytes %" PRIu32 " ccseq %"
601  PRIPktSeqNumber ".\n", conn_id_, stream_id, seq_num, pld_bytes,
602  cc_seq_num);
603 
604  PrintState("OnPacketSent()");
605 #endif
606 
607  return cc_seq_num;
608 }

680 void Cubic::OnRto(bool pkt_rexmit)
681 {
682 #ifdef SLIQ_CC_DEBUG
683  LogD(kClassName, __func__, "Conn %" PRIEndptId ": Args pkt_rexmit %d.\n",
684  conn_id_, static_cast<int>(pkt_rexmit));
685 #endif
686 
687  // Reduce ssthresh if needed. The resend test (after last OR) comes from
688  // RFC 5681, section 3.1, just after equation (4). While this resend test
689  // is not completely accurate due to multiple streams, it should still work.
690  if (((!in_fast_recovery_) && (!in_rto_)) || SEQ_LEQ(high_seq_, snd_una_) ||
691  (in_rto_ && (!pkt_rexmit)))
692  {
693  ssthresh_ = CubicRecalcSsthresh();
694  }
695 
696  // A congestion event has occurred.
697  high_seq_ = snd_nxt_;
698 
699  // Reset cwnd to the loss window, LW, which is one segment, per RFC 5681,
700  // section 3.1, in the next to last paragraph.
701  cwnd_ = kCubicMss;
702  cubic_cwnd_cnt_ = 0;
703 
704  // Reset CUBIC. From the CUBIC algorithm "Timeout" step.
705  CubicReset();
706 
707  // Reset HyStart.
708  if (config_hystart_)
709  {
710  HystartReset();
711  }
712

714  // recovery, exit fast recovery and set RecoveryPoint to HighData (note that
715  // snd_nxt_ is just beyond HighData).
716  if (in_fast_recovery_)
717  {
718  in_fast_recovery_ = false;
719  prr_recovery_point_ = snd_nxt_;
720  }
721 
722  // An RTO event has been started.
723  in_rto_ = true;
724 
725 #ifdef SLIQ_CC_DEBUG
726  LogD(kClassName, __func__, "Conn %" PRIEndptId ": RTO - cwnd %" PRId64
727  " ssthresh %" PRId64 ".\n", conn_id_, cwnd_, ssthresh_);
728

757 bool Cubic::CanSend(const Time& now, uint32_t bytes)
758 {
759  bool rv = false;
760 
761  // Even though cwnd is computed in bytes, make sure that the maximum number
762  // of packets supported in cwnd is never exceeded.
763  if ((snd_nxt_ - snd_una_) >= kMaxCongCtrlWindowPkts)
764  {
765  LogW(kClassName, __func__, "Conn %" PRIEndptId ": CC window size reached "
766  "%zu packets.\n", conn_id_, kMaxCongCtrlWindowPkts);
767  }
768  else
769  {
770  if (in_fast_recovery_)
771  {
772  // In fast recovery. This test just needs to honor prr_out_ and
773  // prr_sndcnt_. Note that RFC 6937, section 3, states that prr_out must
774  // be "strictly less than or equal to sndcnt". Thus, we have to add in
775  // the number of bytes for this request.
776  rv = ((prr_out_ + bytes) <= prr_sndcnt_);
777  }
778  else
779  {
780  // In slow start or congestion avoidance. Perform the normal cwnd test.
781  // Note that bytes_in_flight_ is allowed to go over cwnd_ for the last
782  // packet to "fit" into cwnd and have IsCwndLimited() work correctly.
783  // Thus, we do not add in the number of bytes for this request.
784  rv = (bytes_in_flight_ < cwnd_);
785  }
786  }
787 
788  if (config_idle_restart_ && rv)
789  {
790  // Possibly restart an idle connection.
791  RestartIdleConnection(now);
792  }
793 
794  return rv;
795 }

928 //============================================================================
929 double Cubic::ComputePacingRate()
930 {
931  // The current rate in bps is: rate = ((cwnd * 8) / srtt)
932  Time srtt = rtt_mgr_.smoothed_rtt();
933  double rate_bps = ((static_cast<double>(cwnd_) * 8.0 *
934  kNumMicrosPerSecond) /
935  static_cast<double>(srtt.GetTimeInUsec()));
936 
937  // In slow start, set the pacing rate to 200% of the current rate. In
938  // congestion avoidance, set the pacing rate to 120% of the current rate.
939  //
940  // Note that the normal slow start condition is (cwnd < ssthresh). However,
941  // if (cwnd >= (ssthresh / 2)), then we are approaching the end of slow
942  // start and should start to slow down.
943  if (cwnd_ < (ssthresh_ / 2))
944  {
945  rate_bps *= kPacingSlowStartRatio;
946  }
947  else
948  {
949  rate_bps *= kPacingCongAvoidRatio;
950  }
951 
952  return rate_bps;
953 }

955 //============================================================================
956 void Cubic::UpdateNextSendTime(const Time& now, uint32_t bytes)
957 {
958  // Get the pacing rate.
959  double rate_bps = ComputePacingRate();
960 
961  // Compute the packet inter-send time using the size of the packet just
962  // sent.
963  double pkt_intersend_time = ((static_cast<double>(bytes) * 8.0) /
964  rate_bps);
965 
966  // If the current time is more than kPacingQuiescentThreshold seconds beyond
967  // the stored next send time, then the sender is considered to have been
968  // quiescent for a time, so the next send time must be computed from now.
969  // Otherwise, the send pacing timer must have been used, so add the
970  // inter-send time for this packet to the stored next send time.
971  if (now > next_send_time_.Add(kPacingQuiescentThreshold))
972  {
973  next_send_time_ = now.Add(pkt_intersend_time);
974  }
975  else
976  {
977  next_send_time_ = next_send_time_.Add(pkt_intersend_time);
978  }
979 
980 #ifdef SLIQ_CC_DEBUG
981  LogD(kClassName, __func__, "Conn %" PRIEndptId ": Waiting for packet "
982  "intersend time %f sec for rate %f bps.\n", conn_id_,
983  pkt_intersend_time, rate_bps);
984 #endif
985 }

996 #endif
997 
998  for (int i = 0; i < num_streams; ++i)

1047 void Cubic::RestartIdleConnection(const Time& now)
1048 {
1049  // Possibly restart an idle connection, as per RFC 5681, section 4.1. This
1050  // requires the protocol being idle for the RTO period, at which point, the
1051  // congestion window is set to be:
1052  //
1053  // IW = (3 * MSS);
1054  // RW = min(IW, cwnd);
1055  // cwnd = RW;
1056  //
1057  // Do not restart while in fast recovery or when recovering from an RTO
1058  // timer expiration event.
1059  if ((!in_fast_recovery_) && (!in_rto_) && (cwnd_ > kInitCwnd) &&
1060  ((now - last_proto_send_time_) > rtt_mgr_.GetRtoTime()))
1061  {
1062  // Reset CUBIC.
1063  CubicReset();
1064 
1065  // Reset HyStart.
1066  if (config_hystart_)
1067  {
1068  HystartReset();
1069  }
1070 
1071  // Set the congestion window size to RW. Leave the slow start threshold
1072  // alone.
1073  cwnd_ = kInitCwnd;
1074 
1075 #ifdef SLIQ_CC_DEBUG
1076  LogD(kClassName, __func__, "Conn %" PRIEndptId ": Restart idle "
1077  "connection, cwnd %" PRId64 ".\n", conn_id_, cwnd_);
1078 #endif
1079  }

1107 //============================================================================
1108 void Cubic::UpdateCwnd(const Time& now)
1109 {
1110  if (!IsCwndLimited())
1111  {
1112 #ifdef SLIQ_CC_DEBUG
1113  LogD(kClassName, __func__, "Conn %" PRIEndptId ": Not cwnd limited.\n",
1114  conn_id_);
1115 #endif
1116 
1117  return;
1118  }
1119 
1120  // Compute the number of bytes that were ACKed.
1121  int64_t acked_bytes = (pre_ack_bytes_in_flight_ - bytes_in_flight_);
1122 
1123  // Handle slow start and congestion avoidance separately.
1124  if (cwnd_ < ssthresh_)
1125  {
1126  // Slow start.
1127 
1128  // End the HyStart RTT round if it is time, as per HyStart paper.
1129  if (config_hystart_ && SEQ_GT(snd_una_, hystart_end_seq_))
1130  {
1131  HystartReset();
1132  }
1133 
1134  // When in slow start, increment cwnd by the number of bytes ACKed, as per
1135  // RFC 5681, section 3.1, equation (2).
1136  cwnd_ += MIN(acked_bytes, kCubicMss);
1137  }
1138  else
1139  {
1140  // Congestion avoidance. The following logic comes from the CUBIC
1141  // algorithm "On each ACK" step, with updates from Linux-4.6.3.
1142 
1143  // Update the CUBIC limit cubic_cnt_.
1144  CubicUpdate(now, acked_bytes);
1145 
1146  // Update the congestion window based on the CUBIC parameters. This is
1147  // roughly cwnd += (1 / cwnd) for every packet that was ACKed.
1148  if (cubic_cwnd_cnt_ >= cubic_cnt_)
1149  {
1150 #ifdef SLIQ_CC_DEBUG
1151  LogD(kClassName, __func__, "Conn %" PRIEndptId ": Cubic incr 1, "
1152  "cwnd_cnt %" PRId64 " cnt %" PRId64 " old cwnd %" PRId64 " new "
1153  "cwnd %" PRId64 ".\n", conn_id_, cubic_cwnd_cnt_, cubic_cnt_,
1154  cwnd_, static_cast<int64_t>(cwnd_ + kCubicMss));
1155 #endif
1156 
1157  cubic_cwnd_cnt_ = 0;
1158  cwnd_ += kCubicMss;
1159  }
1160 
1161  cubic_cwnd_cnt_ += acked_bytes;
1162 
1163 #ifdef SLIQ_CC_DEBUG
1164  LogD(kClassName, __func__, "Conn %" PRIEndptId ": Cubic, update cwnd_cnt "
1165  "%" PRId64 ".\n", conn_id_, cubic_cwnd_cnt_);
1166 #endif
1167 
1168  if (cubic_cwnd_cnt_ >= cubic_cnt_)
1169  {
1170  int64_t delta = (cubic_cwnd_cnt_ / cubic_cnt_);
1171 
1172 #ifdef SLIQ_CC_DEBUG
1173  LogD(kClassName, __func__, "Conn %" PRIEndptId ": Cubic incr 2, delta %"
1174  PRId64 " old cwnd_cnt %" PRId64 " new cwnd_cnt %" PRId64 " old "
1175  "cwnd %" PRId64 " new cwnd %" PRId64 ".\n", conn_id_, delta,
1176  cubic_cwnd_cnt_, (cubic_cwnd_cnt_ - (delta * cubic_cnt_)), cwnd_,
1177  (cwnd_ + (delta * kCubicMss)));
1178 #endif
1179 
1180  cubic_cwnd_cnt_ -= (delta * cubic_cnt_);
1181  cwnd_ += (delta * kCubicMss);
1182  }
1183  }
1184 
1185  // Limit the congestion window if needed.
1186  if (cwnd_ > static_cast<int64_t>(kMaxCongCtrlWindowPkts * kCubicMss))
1187  {
1188  cwnd_ = static_cast<int64_t>(kMaxCongCtrlWindowPkts * kCubicMss);
1189  }
1190 
1191 #ifdef SLIQ_CC_DEBUG
1192  LogD(kClassName, __func__, "Conn %" PRIEndptId ": Result cwnd_cnt %" PRId64
1193  " cwnd %" PRId64 ".\n", conn_id_, cubic_cwnd_cnt_, cwnd_);
1194 #endif
1195 }

1197 //============================================================================
1198 void Cubic::CubicUpdate(const Time& now, int64_t acked_bytes)
1199 {
1200 #ifdef SLIQ_CC_DEBUG
1201  LogD(kClassName, __func__, "Conn %" PRIEndptId ": Args acked_bytes %" PRId64
1202  ".\n", conn_id_, acked_bytes);
1203 #endif
1204 
1205  // This logic comes from the CUBIC algorithm "cubic_update()" step.
1206 
1207  // Increment by the number of ACKed packet bytes.
1208  cubic_ack_cnt_ += acked_bytes;
1209 
1210 #ifdef SLIQ_CC_DEBUG
1211  LogD(kClassName, __func__, "Conn %" PRIEndptId ": Update ack_cnt %" PRId64
1212  ".\n", conn_id_, cubic_ack_cnt_);
1213 #endif
1214 
1215  // CUBIC updates with no change to cwnd are limited by time.
1216  Time time_limit = Time::FromUsec(kCubicTimeLimitUsec);
1217 
1218  if ((cubic_last_cwnd_ == cwnd_) && ((now - cubic_last_time_) <= time_limit))
1219  {
1220 #ifdef SLIQ_CC_DEBUG
1221  LogD(kClassName, __func__, "Conn %" PRIEndptId ": No change to cwnd %"
1222  PRId64 " within time limit.\n", conn_id_, cwnd_);
1223 #endif
1224 
1225  return;
1226  }
1227 
1228  // The CUBIC function can update cubic_cnt_ at most once per time interval.
1229  // On all cwnd reduction events, cubic_epoch_start_ is set to zero, which
1230  // will force a recalculation of cubic_cnt_.
1231  time_limit = Time::FromUsec(kCubicUpdateThreshUsec);
1232 
1233  if ((cubic_epoch_start_.IsZero()) ||
1234  ((now - cubic_last_time_) > time_limit))
1235  {
1236  // Record the last cwnd and time.
1237  cubic_last_cwnd_ = cwnd_;
1238  cubic_last_time_ = now;
1239 
1240  // Start a new epoch if required.
1241  if (cubic_epoch_start_.IsZero())
1242  {
1243  cubic_epoch_start_ = now;
1244 
1245  if (cwnd_ < cubic_cwnd_last_max_)
1246  {
1247  cubic_k_ = static_cast<int64_t>(
1248  cbrt(static_cast<double>(kCubeRootFactor) *
1249  static_cast<double>(cubic_cwnd_last_max_ - cwnd_)));
1250  cubic_origin_point_ = cubic_cwnd_last_max_;
1251  }
1252  else
1253  {
1254  cubic_k_ = 0;
1255  cubic_origin_point_ = cwnd_;
1256  }
1257 
1258  cubic_ack_cnt_ = acked_bytes;
1259  cubic_cwnd_tcp_ = cwnd_;
1260 
1261 #ifdef SLIQ_CC_DEBUG
1262  LogD(kClassName, __func__, "Conn %" PRIEndptId ": Epoch start, cwnd %"
1263  PRId64 " cwnd_last_max %" PRId64 " k %" PRId64 " origin_point %"
1264  PRId64 " ack_cnt %" PRId64 " cwnd_tcp %" PRId64 ".\n", conn_id_,
1265  cwnd_, cubic_cwnd_last_max_, cubic_k_, cubic_origin_point_,
1266  cubic_ack_cnt_, cubic_cwnd_tcp_);
1267 #endif
1268  }
1269 
1270  // Compute (now + cubic_delay_min_ - cubic_epoch_start_) as a number of
1271  // 1/1024 second intervals.
1272  Time t_obj = (now + cubic_delay_min_ - cubic_epoch_start_);
1273  int64_t t = ((t_obj.GetTimeInMsec() << 10) / 1000);
1274  int64_t offset = 0;
1275 
1276  // Update cubic_cnt_.
1277  if (t < cubic_k_)
1278  {
1279  offset = (cubic_k_ - t);
1280  }
1281  else
1282  {
1283  offset = (t - cubic_k_);
1284  }
1285 
1286  int64_t delta = (((kCubeCwndScale * offset * offset * offset) >>
1287  kCubeScale) * kCubicMss);
1288  int64_t target = 0;
1289 
1290  if (t < cubic_k_)
1291  {
1292  target = (cubic_origin_point_ - delta);
1293  }
1294  else
1295  {
1296  target = (cubic_origin_point_ + delta);
1297  }
1298 
1299  if (target > cwnd_)
1300  {
1301  cubic_cnt_ = ((cwnd_ / (target - cwnd_)) * kCubicMss);
1302  }
1303  else
1304  {
1305  cubic_cnt_ = (100 * cwnd_);
1306  }
1307 
1308  if ((cubic_cwnd_last_max_ == 0) && (cubic_cnt_ > (20 * kCubicMss)))
1309  {
1310  cubic_cnt_ = (20 * kCubicMss);
1311  }
1312 
1313 #ifdef SLIQ_CC_DEBUG
1314  LogD(kClassName, __func__, "Conn %" PRIEndptId ": Cubic update, t %"
1315  PRId64 " k %" PRId64 " offset %" PRId64 " delta %" PRId64 " target %"
1316  PRId64 " cnt %" PRId64 ".\n", conn_id_, t, cubic_k_, offset, delta,
1317  target, cubic_cnt_);
1318 #endif
1319  }
1320 
1321  // Update TCP-friendly behavior.
1322  if (config_cubic_tcp_friendliness_)

1570 void Cubic::UpdatePrr()
1571 {
1572  // As per RFC 6937, section 3, with additions for CUBIC. Note that pipe is
1573  // always updated, so the RFC 6675 pipe algorithm does not need to be
1574  // executed here.
1575 
1576  int64_t sndcnt = 0;
1577  int64_t delivered_data = (pre_ack_bytes_in_flight_ - bytes_in_flight_);
1578 
1579  prr_delivered_ += delivered_data;
1580 
1581  if (pipe_ > ssthresh_)
1582  {
1583  // Proportional rate reduction.
1584  sndcnt = (static_cast<int64_t>(
1585  ceil(static_cast<double>(prr_delivered_) *
1586  static_cast<double>(ssthresh_) /
1587  static_cast<double>(prr_recover_fs_))) - prr_out_);
1588  }
1589  else
1590  {
1591  int64_t limit = 0;
1592 
1593  // Two versions of the Reduction Bound.
1594  if (config_prr_crb_)
1595  {
1596  // PRR-CRB: Conservative Reduction Bound.
1597  limit = (prr_delivered_ - prr_out_);
1598  }
1599  else
1600  {
1601  // PRR-SSRB: Slow Start Reduction Bound.
1602  limit = (MAX((prr_delivered_ - prr_out_), delivered_data) + kCubicMss);
1603  }
1604 
1605  // Attempt to catch up, as permitted by limit.
1606  sndcnt = MIN((ssthresh_ - pipe_), limit);
1607  }
1608 
1609  // Note that prr_sndcnt_ limits the packets that can be transmitted in
1610  // CanSend() and CanResend(), as per RFC 6937, section 3.
1611  if (sndcnt > 0)
1612  {
1613  prr_sndcnt_ += sndcnt;
1614 
1615 #ifdef SLIQ_CC_DEBUG
1616  LogD(kClassName, __func__, "Conn %" PRIEndptId ": Add %" PRId64 " to "
1617  "sndcnt, total %" PRId64 ".\n", conn_id_, sndcnt, prr_sndcnt_);
1618 #endif
1619  }
1620 
1621  // Note that the NextSeq() logic, as per RFC 6675, page 6, is implemented in
1622  // the connection and stream OnCanWrite(), OnCanSend(), and OnCanResend()
1623  // methods. See the comments in Connection::OnCanWrite() for details.
1624 
1625 #ifdef SLIQ_CC_DEBUG
1626  PrintState("UpdatePrr()");
1627 #endif
1628 }
