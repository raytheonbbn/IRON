194 void Copa::DelayTracker::Update(double delay, const Time& now,
195  double win_sec, double& result)
196 {
197  // Update the recent minimum delay observed.
198  if (delay < recent_min_delay_)
199  {
200  recent_min_delay_ = delay;
201  recent_obs_time_ = now;
202  }
203 
204  // Check if the current bin period is over.
205  Time bin_dur(win_sec / static_cast<double>(kDelayTrackerBins));
206 
207  if (now >= (prev_time_ + bin_dur))
208  {
209  // Record the recent minimum delay in the next bin in the window.
210  bin_[next_index_].min_delay_ = recent_min_delay_;
211  bin_[next_index_].obs_time_ = recent_obs_time_;
212 
213  // Update the number of minimum RTTs in the circular array.
214  if (count_ < kDelayTrackerBins)
215  {
216  ++count_;
217  }
218 
219  // Find the lowest minimum RTT in the specified time window.
220  Time win_start_time = (now - Time(win_sec));
221  double cand_min_delay = recent_min_delay_;
222  uint32_t idx = ((next_index_ == 0) ? (kDelayTrackerBins - 1) :
223  (next_index_ - 1));
224 
225  for (uint32_t i = 1; i < count_; ++i)
226  {
227  if (bin_[idx].obs_time_ < win_start_time)
228  {
229  // Invalidate the bin to prevent increases in the RTT from including
230  // it again later.
231  bin_[idx].min_delay_ = kHugeRtt;
232  }
233  else
234  {
235  if (bin_[idx].min_delay_ < cand_min_delay)
236  {
237  cand_min_delay = bin_[idx].min_delay_;
238  }
239  }
240 
241  idx = ((idx == 0) ? (kDelayTrackerBins - 1) : (idx - 1));
242  }
243 
244  // Pass the lowest minimum delay back to the caller only if a valid
245  // minimum delay was found.
246  if (cand_min_delay < kHugeRtt)
247  {
248  result = cand_min_delay;
249  }
250 
251  // Reset for the next update.
252  recent_min_delay_ = kHugeRtt;
253 
254  next_index_ = ((next_index_ + 1) % kDelayTrackerBins);
255  prev_time_ = now;
256  }
257  else
258  {
259  // Update the result as needed.
260  if (delay < result)
261  {
262  result = delay;
263  }
264  }
265 }

284 void Copa::VelocityState::Update(PktSeqNumber next_cc_seq_num,
285  double current_cwnd, bool cwnd_increasing,
286  uint32_t& result_velocity)
287 {
288  VelDir dir = VEL_DIR_NEITHER;
289 
290  // Determine the current direction.
291  if (current_cwnd > start_cwnd_)
292  {
293  dir = VEL_DIR_UP;
294  }
295  else if (current_cwnd < start_cwnd_)
296  {
297  dir = VEL_DIR_DOWN;
298  }
299 
300  if ((dir != VEL_DIR_NEITHER) && (dir == prev_direction_))
301  {
302  // Direction is the same as in the previous window. Only start doubling
303  // the velocity after the direction has remained the same for 3 RTTs.
304  if ((result_velocity == 1) && (same_direction_cnt_ < 3))
305  {
306  ++same_direction_cnt_;
307  }
308  else
309  {
310  result_velocity *= 2;
311  }
312  }
313  else
314  {
315  // Direction is NEITHER or not the same as in the previous window. Reset
316  // the velocity to 1.
317  same_direction_cnt_ = 0;
318  result_velocity = 1;
319  }
320 
321  // Reset for the next update.
322  prev_direction_ = dir;
323  start_cc_seq_num_ = next_cc_seq_num;
324  start_cwnd_ = current_cwnd;
325  start_cwnd_increasing_ = cwnd_increasing;
326 }

355 bool Copa::Damper::OnRttUpdate(double queueing_delay, double ist,
356  double delta)
357 {
358  bool rv = false;
359 
360  if (state_ == DAMPER_MONITOR_HIGH)
361  {
362  // Watch for instances when there are clearly too many packets queued at
363  // the bottleneck link.
364  if ((queueing_delay / ist) > kDamperThresPkts)
365  {
366  state_ = DAMPER_MONITOR_LOW;
367  }
368  }
369  else if (state_ == DAMPER_MONITOR_LOW)
370  {
371  // Watch for when the queueing delay just drops below 1/delta packets.
372  if ((queueing_delay / ist) < (1.0 / delta))
373  {
374  state_ = DAMPER_HOLD;
375  hold_cnt_ = 0;
376  rv = true;
377  }
378  }
379 
380  return rv;
381 }

489 {
490  if (state_ != NOT_CONNECTED)
491  {
492  LogE(kClassName, __func__, "Conn %" PRIEndptId ": Invalid state %d.\n",
493  conn_id_, static_cast<int>(state_));

495  }
496 
497  // Get the handshake RTT measurement in seconds.
498  double handshake_rtt = rtt.ToDouble();
499 
500  // Initialize the fast startup state.
501  smoothed_rtt_ = handshake_rtt;
502  standing_rtt_ = handshake_rtt;
503  min_rtt_ = handshake_rtt;
504  loc_min_rtt_ = handshake_rtt;
505  fs_.Clear();
506 
507  LogA(kClassName, __func__, "Conn %" PRIEndptId ": Initial min_rtt=%f\n",
508  conn_id_, min_rtt_);
509 
510  if (handshake_rtt > kSlowStartThreshold)
511  {
512  state_ = FAST_STARTUP;
513 
514  // Send the first packet pair immediately.
515  timer_.CancelTimer(fs_.timer_);
516  FsPktPairCallback();
517  }
518  else
519  {
520  state_ = SLOW_START;
521  ist_ = (handshake_rtt / cwnd_);
522  }
523 }

568 //============================================================================
569 void Copa::OnRttUpdate(StreamId stream_id, const Time& ack_time,
570  PktTimestamp send_ts, PktTimestamp recv_ts,
571  PktSeqNumber seq_num, PktSeqNumber cc_seq_num,
572  const Time& rtt, uint32_t bytes, float cc_val)
573 {
574  if (state_ < SLOW_START)
575  {
576  LogE(kClassName, __func__, "Conn %" PRIEndptId ": Invalid state %d.\n",
577  conn_id_, static_cast<int>(state_));
578  return;
579  }
580 
581  // Get the RTT measurement in seconds.
582  double measured_rtt = rtt.ToDouble();
583 
584 #ifdef SLIQ_CC_DEBUG
585  LogD(kClassName, __func__, "** Conn %" PRIEndptId ": On RTT Update: "
586  "stream_id=%" PRIStreamId " seq_num=%" PRIPktSeqNumber " cc_seq_num=%"
587  PRIPktSeqNumber " ack_time=%s measured_rtt=%f bytes=%" PRIu32
588  " cc_val=%f\n", conn_id_, stream_id, seq_num, cc_seq_num,
589  ack_time.ToString().c_str(), measured_rtt, bytes,
590  static_cast<double>(cc_val));
591 #endif
592 
593  // Update the minimum timestamp delta using the tracker with a time window
594  // of 7*RTTmin or 0.2 seconds, whichever is greater.
595  double ts_delta = (static_cast<double>(static_cast<int64_t>(recv_ts) -
596  static_cast<int64_t>(send_ts)) *
597  0.000001);
598  double mtd_win_sec = (7.0 * min_rtt_);
599 
600  if (mtd_win_sec < 0.2)
601  {
602  mtd_win_sec = 0.2;
603  }
604 
605  mtd_.Update(ts_delta, ack_time, mtd_win_sec, min_ts_delta_);
606 
607  // Adjust the RTT measurement in order to eliminate the queueing delay in
608  // the reverse path.
609  double adjusted_rtt = (measured_rtt + min_ts_delta_ - ts_delta);
610 
611  if (adjusted_rtt < kMinRttSec)
612  {
613  adjusted_rtt = kMinRttSec;
614  }
615 
616  // Update the local minimum RTT using the tracker with a time window of
617  // 28*RTTmin or 0.8 seconds, whichever is greater.
618  double old_loc_min_rtt = loc_min_rtt_;
619  double mrt_win_sec = (28.0 * min_rtt_);
620 
621  if (mrt_win_sec < 0.8)
622  {
623  mrt_win_sec = 0.8;
624  }
625 
626  mrt_.Update(adjusted_rtt, ack_time, mrt_win_sec, loc_min_rtt_);
627 
628  // If an anti-jitter value is configured, then reduce the adjusted RTT by
629  // the anti-jitter amount while not going lower than the local minimum RTT.
630  if (anti_jitter_ > 0.0)
631  {
632  adjusted_rtt -= anti_jitter_;
633 
634  if (adjusted_rtt < loc_min_rtt_)
635  {
636  adjusted_rtt = loc_min_rtt_;
637  }
638  }
639 
640  // Update the smoothed RTT.
641  smoothed_rtt_ = ((kSrttAlpha * adjusted_rtt) +
642  ((1.0 - kSrttAlpha) * smoothed_rtt_));
643 
644  // Update the standing RTT using the tracker with a time window of srtt/2.
645  double srt_win_sec = (0.5 * smoothed_rtt_);
646 
647  srt_.Update(adjusted_rtt, ack_time, srt_win_sec, standing_rtt_);
648 
649  // If the local minimum RTT has changed, then recalculate the minimum RTT.
650  if (loc_min_rtt_ != old_loc_min_rtt)
651  {
652  // If the remote minimum RTT came in more than 3 reporting periods ago,
653  // then it is stale and must be eliminated from the calculation.
654  if ((!rmt_min_rtt_time_.IsZero()) &&
655  (ack_time >= (rmt_min_rtt_time_ + Time(3.1 * kReportMinRttPeriod))))
656  {
657  rmt_min_rtt_ = kHugeRtt;
658  rmt_min_rtt_time_.Zero();
659  }
660 
661  // The minimum RTT is the minimum of the local and remote minimum RTTs.
662  if (rmt_min_rtt_ < kHugeRtt)
663  {
664  min_rtt_ = ((loc_min_rtt_ < rmt_min_rtt_) ? loc_min_rtt_ :
665  rmt_min_rtt_);
666  }
667  else
668  {
669  min_rtt_ = loc_min_rtt_;
670  }
671 
672  LogA(kClassName, __func__, "Conn %" PRIEndptId ": Updated min_rtt=%f\n",
673  conn_id_, min_rtt_);
674 
675  // Possibly report the new local minimum RTT to the peer.
676  ReportMinRttOnUpdate();
677  }
678 
679  // Check if it is time to report the local minimum RTT to the peer.
680  if ((!report_min_rtt_) && (ack_time > next_report_min_rtt_time_))
681  {
682  ReportMinRttOnTimeout(ack_time);
683  }
684 
685  // Estimate the queueing delay, which is in seconds.
686  double queueing_delay = (standing_rtt_ - min_rtt_);
687 
688  // Prevent the queueing delay from going negative.
689  if (queueing_delay < 0.0)
690  {
691  queueing_delay = 0.0;
692  }
693 
694  // Compute the current target rate, which is in packets per second.
695  double lambda_target = kMaxRate;
696 
697  if (queueing_delay > 0.0)
698  {
699  lambda_target = (1.0 / (delta_ * queueing_delay));
700  }
701 
702  // Compute the current rate, which is in packets per second.
703  double lambda = (cwnd_ / standing_rtt_);
704 
705  // Update the congestion window differently depending on the current state.
706  if (state_ == SLOW_START)
707  {
708  // Only increase the congestion window if the current number of packets in
709  // flight is close to the current congestion window size. This prevents
710  // the congestion window size from increasing indefinitely when the sender
711  // is not keeping the channel full.
712  if ((cwnd_ < kIncCwndPkts) ||
713  ((cwnd_ - (static_cast<double>(bytes_in_flight_) /
714  static_cast<double>(kNominalPktSizeBytes -
715  kDataHdrBaseSize))) <=
716  (kIncCwndRatio * cwnd_)))
717  {
718  // Double the congestion window each RTT.
719  cwnd_ += (static_cast<double>(bytes) /
720  static_cast<double>(kNominalPktSizeBytes -
721  kDataHdrBaseSize));
722  }
723 
724  // Determine if slow start is over.
725  if (lambda > lambda_target)
726  {
727  state_ = CLOSED_LOOP;
728  }
729  }
730  else // state_ == CLOSED_LOOP
731  {
732  // Update the damper.
733  if (damper_.OnRttUpdate(queueing_delay, ist_, delta_))
734  {
735  // Force the congestion window size to the size when this packet was
736  // sent, and reset the velocity state.
737  cwnd_ = static_cast<double>(cc_val);
738 
739  vel_.Reset(nxt_cc_seq_num_, cwnd_, (lambda <= lambda_target),
740  velocity_);
741 
742  LogA(kClassName, __func__, "Conn %" PRIEndptId ": Damper, hold cwnd at "
743  "%f.\n", conn_id_, cwnd_);
744  }
745 
746  // Only update the velocity and congestion window size if the damper
747  // allows it.
748  if (damper_.CanUpdateVelCwnd())
749  {
750  // Update the velocity.
751  if (SEQ_GEQ(cc_seq_num, vel_.start_cc_seq_num_))
752  {
753  vel_.Update(nxt_cc_seq_num_, cwnd_, (lambda <= lambda_target),
754  velocity_);
755  }
756 
757  if ((lambda <= lambda_target) != vel_.start_cwnd_increasing_)
758  {
759  vel_.Reset(nxt_cc_seq_num_, cwnd_, (lambda <= lambda_target),
760  velocity_);
761  }
762 
763  // Limit the velocity so that the rate can never more than double once
764  // per RTT.
765  uint32_t max_velocity = static_cast<uint32_t>(delta_ * cwnd_);
766 
767  if (velocity_ > max_velocity)
768  {
769  velocity_ = max_velocity;
770  }
771 
772  // Make sure that the velocity is never less than 1.
773  if (velocity_ < 1)
774  {
775  velocity_ = 1;
776  }
777 
778  // Adjust the congestion window size, which is in packets. Scale the
779  // adjustment amount by the size of the packet.
780  double cwnd_adj = ((static_cast<double>(bytes) *
781  static_cast<double>(velocity_)) /
782  (static_cast<double>(kNominalPktSizeBytes -
783  kDataHdrBaseSize) *
784  delta_ * cwnd_));
785 
786  if (lambda <= lambda_target)
787  {
788  // Only increase the congestion window if the current number of
789  // packets in flight is close to the current congestion window size.
790  // This prevents the congestion window size from increasing
791  // indefinitely when the sender is not keeping the channel full.
792  if ((cwnd_ < kIncCwndPkts) ||
793  ((cwnd_ - (static_cast<double>(bytes_in_flight_) /
794  static_cast<double>(kNominalPktSizeBytes -
795  kDataHdrBaseSize))) <=
796  (kIncCwndRatio * cwnd_)))
797  {
798  cwnd_ += cwnd_adj;
799  }
800  }
801  else
802  {
803  cwnd_ -= cwnd_adj;
804  }
805  }
806  }
807 
808  // Limit how small/large the congestion window size can be.
809  if (cwnd_ < kMinCwndPkts)
810  {
811  cwnd_ = kMinCwndPkts;
812  }
813 
814  if (cwnd_ > static_cast<double>(kMaxCongCtrlWindowPkts))
815  {
816  cwnd_ = static_cast<double>(kMaxCongCtrlWindowPkts);
817  }
818 
819  // Update the current inter-send time.
820  ist_ = (standing_rtt_ / cwnd_);
821 
822  // Limit the inter-send time if needed.
823  if (ist_ < kMinIst)
824  {
825  ist_ = kMinIst;
826  }
827 
828 #ifdef SLIQ_CC_DEBUG
829  LogD(kClassName, __func__, "Conn %" PRIEndptId ": Computed rtt=%f "
830  "smoothed_rtt=%f standing_rtt=%f min_rtt=%f queueing_delay=%f "
831  "lambda_target=%f lambda=%f cwnd=%f ist=%f\n", conn_id_, adjusted_rtt,
832  smoothed_rtt_, standing_rtt_, min_rtt_, queueing_delay, lambda_target,
833  lambda, cwnd_, ist_);
834  LogA(kClassName, __func__, "Conn %" PRIEndptId ": PLT_COPA %f %f %f %f %f "
835  "%f %f %f %" PRIu32 " %f %0.9f %f %f %f\n", conn_id_,
836  CurrentTime(ack_time), adjusted_rtt, smoothed_rtt_, standing_rtt_,
837  min_rtt_, queueing_delay, lambda_target, lambda, velocity_, cwnd_,
838  ist_, (static_cast<double>(bytes_in_flight_) /
839  static_cast<double>(kNominalPktSizeBytes)), ts_delta,
840  min_ts_delta_);
841 #endif
842 }

1055 //============================================================================
1056 void Copa::ProcessCcPktTrain(const Time& now, CcPktTrainHeader& hdr)
1057 {
1058  uint32_t pair = (hdr.pt_seq_num / 2);
1059 
1060  // Check if this is an FS_DATA packet.
1061  if (hdr.pt_pkt_type == FS_DATA)
1062  {
1063 #ifdef SLIQ_CC_DEBUG
1064  LogD(kClassName, __func__, "Conn %" PRIEndptId ": Received packet pair "
1065  "FS_DATA with seq=%" PRIu8 ".\n", conn_id_, hdr.pt_seq_num);
1066 #endif
1067 
1068  // Update local state and compute the packet pair inter-receive time to
1069  // send back in the FS_ACK packet.
1070  uint32_t irt_usec = 0;
1071 
1072  if ((hdr.pt_seq_num % 2) == 0)
1073  {
1074  // This is the first FS_DATA packet for a pair. Record its receive
1075  // time.
1076  fs_.pair_recv_time_[pair] = now;
1077  }
1078  else
1079  {
1080  // This is the second FS_DATA packet for a pair. Compute the delay
1081  // between receipt of the two packets.
1082  if (!fs_.pair_recv_time_[pair].IsZero())
1083  {
1084  Time irt = (now - fs_.pair_recv_time_[pair]);
1085 
1086  irt_usec = static_cast<uint32_t>(irt.GetTimeInUsec());
1087 
1088  if (irt_usec == 0)
1089  {
1090  irt_usec = 1;
1091  }
1092 
1093 #ifdef SLIQ_CC_DEBUG
1094  LogD(kClassName, __func__, "Conn %" PRIEndptId ": Second packet of "
1095  "pair received, irt=%f.\n", conn_id_, irt.ToDouble());
1096 #endif
1097  }
1098  }
1099 
1100  // Send an FS_ACK packet immediately.
1101  SendPktPairAck(hdr.pt_seq_num, irt_usec);
1102  return;
1103  }
1104 
1105  // This is an FS_ACK packet.
1106 
1107 #ifdef SLIQ_CC_DEBUG
1108  LogD(kClassName, __func__, "Conn %" PRIEndptId ": Received packet pair "
1109  "FS_ACK with seq=%" PRIu8 ".\n", conn_id_, hdr.pt_seq_num);
1110 #endif
1111 
1112  // Copa must be in the fast startup state to process the FS_ACK packet.
1113  if (state_ != FAST_STARTUP)
1114  {
1115  // It is possible for an FS_ACK packet to arrive late. Thus, if we are in
1116  // the CLOSED_LOOP state, just ignore the packet without logging an error.
1117  if (state_ != CLOSED_LOOP)
1118  {
1119  LogE(kClassName, __func__, "Conn %" PRIEndptId ": Invalid state %d.\n",
1120  conn_id_, static_cast<int>(state_));
1121  }
1122  return;
1123  }
1124 
1125  if ((hdr.pt_seq_num % 2) == 0)
1126  {
1127  // This is an FS_ACK of the first packet in the pair. Use it for an RTT
1128  // estimate. Note that the second packet in the pair is not used for an
1129  // RTT estimate because it was delayed by the first packet.
1130  if (!fs_.pair_send_time_[pair].IsZero())
1131  {
1132  Time diff = (now - fs_.pair_send_time_[pair]);
1133 
1134  fs_.rtt_[pair] = diff.ToDouble();
1135 
1136 #ifdef SLIQ_CC_DEBUG
1137  LogD(kClassName, __func__, "Conn %" PRIEndptId ": First packet of pair "
1138  "FS_ACKed, rtt=%f.\n", conn_id_, fs_.rtt_[pair]);
1139 #endif
1140  }
1141  }
1142  else
1143  {
1144  // This is an FS_ACK of the second packet in the pair. Use it for a
1145  // bottleneck link rate estimate in packets per second.
1146  if (hdr.pt_inter_recv_time != 0)
1147  {
1148  double irt_sec = (static_cast<double>(hdr.pt_inter_recv_time) *
1149  0.000001);
1150 
1151  fs_.rate_[pair] = (1.0 / irt_sec);
1152 
1153 #ifdef SLIQ_CC_DEBUG
1154  LogD(kClassName, __func__, "Conn %" PRIEndptId ": Second packet of "
1155  "pair FS_ACKed, irt=%f rate=%f.\n", conn_id_, irt_sec,
1156  fs_.rate_[pair]);
1157 #endif
1158  }
1159  }
1160 
1161  // If this is the last FS_ACK needed, then end fast startup immediately.
1162  if (hdr.pt_seq_num == ((2 * kNumFsPairs) - 1))
1163  {
1164  timer_.CancelTimer(fs_.timer_);
1165  FsDoneCallback();
1166  }
1167 }

1215 //============================================================================
1216 void Copa::UpdateNextSendTime(const Time& now, size_t bytes)
1217 {
1218  // Update the next send time using the packet size and the stored next send
1219  // time. This maintains inter-send time accuracy.
1220  double pkt_intersend_time =
1221  (ist_ *
1222  (static_cast<double>(bytes + kDataHdrBaseSize + kPktOverheadBytes) /
1223  static_cast<double>(kNominalPktSizeBytes + kPktOverheadBytes)));
1224 
1225  // If the current time is more than kQuiescentThreshold seconds beyond the
1226  // stored next send time, then the sender is considered to have been
1227  // quiescent for a time, so the next send time must be computed from now.
1228  // Otherwise, the send pacing timer must have been used, so add the
1229  // inter-send time for this packet to the stored next send time.
1230  if (now > next_send_time_.Add(kQuiescentThreshold))
1231  {
1232  next_send_time_ = now.Add(pkt_intersend_time);
1233  }
1234  else
1235  {
1236  next_send_time_ = next_send_time_.Add(pkt_intersend_time);
1237  }
1238 
1239 #ifdef SLIQ_CC_DEBUG
1240  LogD(kClassName, __func__, "Conn %" PRIEndptId ": Waiting for "
1241  "intersend_time=%f\n", conn_id_, pkt_intersend_time);
1242 #endif
1243 }

1349 //============================================================================
1350 void Copa::FsPktPairCallback()
1351 {
1352  if (state_ != FAST_STARTUP)
1353  {
1354  LogE(kClassName, __func__, "Conn %" PRIEndptId ": Invalid state %d.\n",
1355  conn_id_, static_cast<int>(state_));
1356  return;
1357  }
1358 
1359  // Record the time that the next packet pairs were sent.
1360  fs_.pair_send_time_[fs_.pairs_sent_] = Time::Now();
1361 
1362  // Send the next packet pair.
1363  SendPktPair(2 * fs_.pairs_sent_);
1364  ++(fs_.pairs_sent_);
1365 
1366  // Start a timer for the next event.
1367  if (fs_.pairs_sent_ < kNumFsPairs)
1368  {
1369  // Start the next timer to send kNumFsPairs packet pairs over 2 RTTs.
1370  Time duration((2.0 * (smoothed_rtt_ + kPktPairRttAdj)) /
1371  static_cast<double>(kNumFsPairs));
1372  CallbackNoArg<Copa> callback(this, &Copa::FsPktPairCallback);
1373 
1374  if (!timer_.StartTimer(duration, &callback, fs_.timer_))
1375  {
1376  LogE(kClassName, __func__, "Conn %" PRIEndptId ": Error starting "
1377  "packet pair timer.\n", conn_id_);
1378  }
1379  }
1380  else
1381  {
1382  // Wait up to twice the kNumFsPairs RTTs, or a maximum of one second, for
1383  // FS_ACKs from the last packet pair sent.
1384  double wait_time = (2.0 * static_cast<double>(kNumFsPairs) *
1385  (smoothed_rtt_ + kPktPairRttAdj));
1386 
1387  if (wait_time > 1.0)
1388  {
1389  wait_time = 1.0;
1390  }
1391 
1392  Time duration(wait_time);
1393  CallbackNoArg<Copa> callback(this, &Copa::FsDoneCallback);
1394 
1395  if (!timer_.StartTimer(duration, &callback, fs_.timer_))
1396  {
1397  LogE(kClassName, __func__, "Conn %" PRIEndptId ": Error starting done "
1398  "timer.\n", conn_id_);
1399  }
1400  }
1401 }

1404 void Copa::FsDoneCallback()
1405 {
1406  if (state_ != FAST_STARTUP)
1407  {
1408  LogE(kClassName, __func__, "Conn %" PRIEndptId ": Invalid state %d.\n",
1409  conn_id_, static_cast<int>(state_));
1410  return;
1411  }
1412 
1413  // Find the minimum, maximum, and mean RTT estimates, as well as the
1414  // bottleneck link rate estimate (mu).
1415  uint32_t rtt_cnt = 0;
1416  uint32_t rate_cnt = 0;
1417  double rtt_min = kHugeRtt;
1418  double rtt_max = -1.0;
1419  double mean_rtt = 0.0;
1420  double mu = 0.0;
1421 
1422  // Start at 1, skipping the first sample. This is because the first sample
1423  // is usually very inaccurate in our testing over high speed networks.
1424  for (size_t pair = 1; pair < kNumFsPairs; ++pair)
1425  {
1426  double rtt_est = fs_.rtt_[pair];
1427 
1428  if (rtt_est > 0.0)
1429  {
1430  mean_rtt += rtt_est;
1431  ++rtt_cnt;
1432 
1433  if (rtt_est < rtt_min)
1434  {
1435  rtt_min = rtt_est;
1436  }
1437 
1438  if (rtt_est > rtt_max)
1439  {
1440  rtt_max = rtt_est;
1441  }
1442  }
1443 
1444  double rate_est = fs_.rate_[pair];
1445 
1446  if (rate_est > 0.0)
1447  {
1448  mu += rate_est;
1449  ++rate_cnt;
1450  }
1451  }
1452 
1453  // If there was not enough data, then restart the fast startup.
1454  if ((rtt_cnt == 0) || (rate_cnt == 0))
1455  {
1456  LogE(kClassName, __func__, "Conn %" PRIEndptId ": Incomplete fast "
1457  "startup, trying again.\n", conn_id_);
1458  fs_.Clear();
1459  timer_.CancelTimer(fs_.timer_);
1460  FsPktPairCallback();
1461  return;
1462  }
1463 
1464  // Complete the mean computations.
1465  mean_rtt = (mean_rtt / static_cast<double>(rtt_cnt));
1466  mu = (mu / static_cast<double>(rate_cnt));
1467 
1468  // Compute the target rate, in packets per second.
1469  double lambda_target = mu;
1470 
1471  if (rtt_max > rtt_min)
1472  {
1473  double tmp = (2.0 / (delta_ * (rtt_max - rtt_min)));
1474 
1475  if (tmp < mu)
1476  {
1477  lambda_target = tmp;
1478  }
1479  }
1480 
1481  // Limit the target rate to the maximum allowable startup rate.
1482  if (lambda_target > kMaxStartupRate)
1483  {
1484  lambda_target = kMaxStartupRate;
1485  }
1486 
1487  // Set the initial parameters based on the fast startup results.
1488  smoothed_rtt_ = mean_rtt;
1489  standing_rtt_ = rtt_min;
1490  min_rtt_ = rtt_min;
1491  loc_min_rtt_ = rtt_min;
1492  cwnd_ = (lambda_target * rtt_min);
1493  ist_ = (1.0 / lambda_target);
1494 
1495  LogA(kClassName, __func__, "Conn %" PRIEndptId ": Fast startup, rtt_min=%f "
1496  "rtt_max=%f mean_rtt=%f mu=%f lambda_target=%f smoothed_rtt=%f "
1497  "standing_rtt=%f min_rtt=%f cwnd=%f ist=%f\n", conn_id_, rtt_min,
1498  rtt_max, mean_rtt, mu, lambda_target, smoothed_rtt_, standing_rtt_,
1499  min_rtt_, cwnd_, ist_);
1500 
1501  // The algorithm is now ready for closed loop operation.
1502  state_ = CLOSED_LOOP;
1503 }
