124 // \todo: Create partner method for sending multiple packet indices aggregated
125 // into a single fifo send (to minimize the number of system calls required).
126 bool PacketFifo::Send(Packet* packet)
127 {
128  if (fifo_->IsOpen() || fifo_->OpenSender())
129  {
130  PktMemIndex msg = packet->mem_index();
131 
132  if (fifo_->Send(reinterpret_cast<uint8_t*>(&msg), sizeof(msg)))
133  {
134 #if defined(PKT_LEAK_DETECT) || defined(PACKET_TRACKING)
135  packet_pool_.TrackPacketRelease(packet, remote_owner_);
136 #endif // PKT_LEAK_DETECT || PACKET_TRACKING
137  return true;
138  }
139  else
140  {
141  LogW(kClassName, __func__, "Unable to send packet index over fifo.\n");
142  return false;
143  }
144  }
145  else
146  {
147  LogW(kClassName, __func__, "Unable to open fifo.\n");
148  return false;
149  }
150  return false;
151 }

154 bool PacketFifo::Recv()
155 {
156  if (num_viewed_pkts_ < num_pkts_rcvd_)
157  {
158  LogW(kClassName, __func__, "%zd packets were received over the fifo from "
159  "owner %d, but only %zd were viewed.\n",
160  num_pkts_rcvd_, num_viewed_pkts_);
161  }
162  num_viewed_pkts_ = 0;
163  num_pkts_rcvd_ = 0;
164 
165  // Read in packet indices from the underlying fifo.
166  size_t bytes = fifo_->Recv(reinterpret_cast<uint8_t*>(
167  &(recv_pkt_index_buf_[0])),
168  max_bytes_to_recv_);
169 
170  // Make sure that whole packet indices were read.
171  if ((bytes % sizeof(PktMemIndex)) != 0)
172  {
173  LogW(kClassName, __func__, "Partial packet index read detected from "
174  "packet owner %d.\n", remote_owner_);
175 
176  bytes += fifo_->Recv((reinterpret_cast<uint8_t*>(
177  &(recv_pkt_index_buf_[0])) + bytes),
178  (sizeof(PktMemIndex) -
179  (bytes % sizeof(PktMemIndex))));
180 
181  if ((bytes % sizeof(PktMemIndex)) != 0)
182  {
183  LogF(kClassName, __func__, "Error correcting for partial packet index "
184  "read from packet owner %d.\n", remote_owner_);
185  return false;
186  }
187  }
188 
189  // Process the packet indices.
190  if (bytes > 0)
191  {
192  last_recv_time_ = Time::Now();
193  num_pkts_rcvd_ = (bytes / sizeof(PktMemIndex));
194 
195  LogD(kClassName, __func__, "Read %zd packets from packet owner %d.\n",
196  num_pkts_rcvd_, remote_owner_);
197  return true;
198  }
199  return false;
200 }

203 bool PacketFifo::GetNextRcvdPacket(Packet** packet)
204 {
205  if (num_viewed_pkts_ >= num_pkts_rcvd_)
206  {
207  return false;
208  }
209 
210  Packet* pkt = packet_pool_.GetPacketFromIndex(
211  recv_pkt_index_buf_[num_viewed_pkts_]);
212  num_viewed_pkts_++;
213  if (pkt != NULL)
214  {
215 #if defined(PKT_LEAK_DETECT) || defined(PACKET_TRACKING)
216  packet_pool_.TrackPacketClaim(pkt, remote_owner_);
217 #endif // PKT_LEAK_DETECT || PACKET_TRACKING
218  pkt->set_recv_time(last_recv_time_);
219  }
220  else
221  {
222  LogW(kClassName, __func__, "Invalid packet index received over fifo.\n");
223  }
224  *packet = pkt;
225  return true;
226 }
