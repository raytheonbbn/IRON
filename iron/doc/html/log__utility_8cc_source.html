89 bool LogUtility::Initialize(const ConfigInfo& ci)
90 {
91  m_val_ = ci.GetDouble("m", 0, false);
92  if (m_val_ == 0)
93  {
94  LogF(kClassName, __func__, "fid: %" PRIu32 ", m value not provided.\n",
95  flow_id_);
96  return false;
97  }
98 
99  a_val_ = ci.GetDouble("a", 0, false);
100  if (a_val_ == 0)
101  {
102  LogF(kClassName, __func__, "fid: %" PRIu32 ", a value not provided.\n",
103  flow_id_);
104  return false;
105  }
106 
107  p_val_ = ci.GetDouble("p", kDefaultPriority, false);
108 
109  LogC(kClassName, __func__, "LOG configuration :\n");
110  LogC(kClassName, __func__, "k : %.2e\n",
111  static_cast<double>(k_val_.GetValue()));
112  LogC(kClassName, __func__, "m : %.03e\n", m_val_);
113  LogC(kClassName, __func__, "a : %.03f\n", a_val_);
114  LogC(kClassName, __func__, "p : %.03f\n", p_val_);
115  LogC(kClassName, __func__, "LOG configuration complete\n");
116 
117  LogI(kClassName, __func__, "Log initialized.\n");
118 
119  return true;
120 }

123 double LogUtility::GetSendRate()
124 {
125  if (flow_state_ == FLOW_OFF)
126  {
127  return 0.0;
128  }
129 
130  double send_rate = 0.0;
131  double queue_depth_bits = queue_depths_.GetBinDepthByIdx(bin_idx_)*8;
132 
133  if (queue_depth_bits >= k_val_.GetValue() * p_val_ * a_val_)
134  {
135  LogD(kClassName, __func__, "fid: %" PRIu32 ", queue is too large, not "
136  "sending.\n", flow_id_);
137  send_rate = 0.0;
138  }
139  else if (queue_depth_bits == 0.0)
140  {
141  send_rate = m_val_;
142  }
143  else
144  {
145  send_rate = min((a_val_ * k_val_.GetValue() * p_val_ - queue_depth_bits) /
146  (a_val_ * queue_depth_bits), m_val_);
147  }
148 
149  LogA(kClassName, __func__, "f_id: %" PRIu32 ", queue: %.03fb, rate: "
150  "%.03fbps.\n", flow_id_, queue_depth_bits, send_rate);
151 
152  return send_rate;
153 }

156 double LogUtility::ComputeUtility(double send_rate)
157 {
158  if (a_val_ * send_rate <= -1)
159  {
160  LogF(kClassName, __func__, "fid: %" PRIu32 ", Error: Cannot take log "
161  "of negative value a*r + 1 = %.3f.\n", flow_id_,
162  a_val_ * send_rate + 1);
163  return 0;
164  }
165 
166  // p * log(ar + 1)
167  return p_val_ * log(a_val_ * send_rate + 1);
168 }
