352 void CopaBeta2::OnRttUpdate(StreamId stream_id, const Time& ack_time,
353  PktTimestamp /* send_ts */,
354  PktTimestamp /* recv_ts */, PktSeqNumber seq_num,
355  PktSeqNumber cc_seq_num, const Time& rtt,
356  uint32_t bytes, float cc_val)
357 {
358  if (state_ != CLOSED_LOOP)
359  {
360  LogE(kClassName, __func__, "Conn %" PRIEndptId ": Invalid state %d.\n",
361  conn_id_, static_cast<int>(state_));
362  return;
363  }
364 
365  // Get the RTT measurement in seconds.
366  double measured_rtt = rtt.ToDouble();
367 
368 #ifdef SLIQ_CC_DEBUG
369  LogD(kClassName, __func__, "** Conn %" PRIEndptId ": On RTT Update: "
370  "stream_id=%" PRIStreamId " seq_num=%" PRIPktSeqNumber " cc_seq_num=%"
371  PRIPktSeqNumber " ack_time=%s measured_rtt=%f size=%" PRIu32
372  " cc_val=%f\n", conn_id_, stream_id, seq_num, cc_seq_num,
373  ack_time.ToString().c_str(), measured_rtt, bytes,
374  static_cast<double>(cc_val));
375 #endif
376 
377  // Update the minimum RTT value observed.
378  if (measured_rtt < min_rtt_)
379  {
380  min_rtt_ = measured_rtt;
381 
382  LogA(kClassName, __func__, "Conn %" PRIEndptId ": Updated min_rtt=%f\n",
383  conn_id_, min_rtt_);
384  }
385 
386  if (measured_rtt < mrt_.recent_min_rtt_)
387  {
388  mrt_.recent_min_rtt_ = measured_rtt;
389 
390 #ifdef SLIQ_CC_DEBUG
391  LogD(kClassName, __func__, "Conn %" PRIEndptId ": Updated "
392  "recent_min_rtt=%f\n", conn_id_, mrt_.recent_min_rtt_);
393 #endif
394  }
395 
396  // Save the last rtt measurement, which is in seconds.
397  last_rtt_ = measured_rtt;
398 
399  // Update the velocity.
400  if (SEQ_GEQ(cc_seq_num, vel_cc_seq_num_))
401  {
402  // Adjust the velocity.
403  VelDir direction = VEL_DIR_NEITHER;
404 
405 #ifdef SLIQ_CC_DEBUG
406  LogD(kClassName, __func__, "Conn %" PRIEndptId ": Cwnd adjustments up=%"
407  PRIu32 " down=%" PRIu32 "\n", conn_id_, cwnd_adj_up_,
408  cwnd_adj_down_);
409 #endif
410 
411  if ((cwnd_adj_up_ + cwnd_adj_down_) > 0)
412  {
413  double threshold = ((2.0 * (cwnd_adj_up_ + cwnd_adj_down_)) / 3.0);
414 
415  if (cwnd_adj_up_ >= threshold)
416  {
417  direction = VEL_DIR_UP;
418  }
419  else if (cwnd_adj_down_ >= threshold)
420  {
421  direction = VEL_DIR_DOWN;
422  }
423 
424  if ((prev_direction_ != VEL_DIR_NEITHER) &&
425  (direction == prev_direction_))
426  {
427  if ((velocity_ == 1) && (vel_same_direction_cnt_ < 3))
428  {
429  ++vel_same_direction_cnt_;
430  }
431  else
432  {
433  velocity_ *= 2;
434  }
435  }
436  else
437  {
438  vel_same_direction_cnt_ = 0;
439  velocity_ = 1;
440  }
441  }
442  else
443  {
444  vel_same_direction_cnt_ = 0;
445  velocity_ = 1;
446  }
447 
448  // Prepare for the next velocity update.
449  cwnd_adj_up_ = 0;
450  cwnd_adj_down_ = 0;
451  prev_direction_ = direction;
452  vel_cc_seq_num_ = nxt_cc_seq_num_;
453 
454 #ifdef SLIQ_CC_DEBUG
455  LogD(kClassName, __func__, "Conn %" PRIEndptId ": Update velocity=%"
456  PRIu32 "\n", conn_id_, velocity_);
457 #endif
458  }
459 
460  // Check if an RTT period has ended.
461  if (ack_time > rtt_period_end_)
462  {
463  // Update the minimum RTT tracking.
464  double duration_sec = ((ack_time - mrt_.prev_time_).ToDouble());
465 
466  if (duration_sec >= kMinRttMinPeriod)
467  {
468  mrt_.ist_[mrt_.next_ist_index_] = ist_;
469  mrt_.next_ist_index_ =
470  ((mrt_.next_ist_index_ + 1) % kMinRttIstPeriods);
471 
472  if (mrt_.recent_min_rtt_ < kHugeRtt)
473  {
474  mrt_.min_rtt_[mrt_.next_rtt_index_] = mrt_.recent_min_rtt_;
475  mrt_.recent_min_rtt_ = kHugeRtt;
476  mrt_.next_rtt_index_ =
477  ((mrt_.next_rtt_index_ + 1) % kMinRttPeriods);
478 
479  if (mrt_.count_ < kMinRttPeriods)
480  {
481  ++(mrt_.count_);
482  }
483 
484  // If not in TCP mode and enough RTT samples are available, then check
485  // if the minimum RTT should be adjusted upward.
486  if ((!tc_.in_tcp_mode_) && (mrt_.count_ >= kMinRttPeriods))
487  {
488  // Find the oldest minimum RTT sample in the array, and the smallest
489  // minimum RTT of the other samples in the array.
490  double oldest_cand_min_rtt = mrt_.min_rtt_[mrt_.next_rtt_index_];
491  double cand_min_rtt = kHugeRtt;
492 
493  for (uint32_t i = 0; i < kMinRttPeriods; ++i)
494  {
495  if ((i != mrt_.next_rtt_index_) &&
496  (mrt_.min_rtt_[i] < cand_min_rtt))
497  {
498  cand_min_rtt = mrt_.min_rtt_[i];
499  }
500  }
501 
502  // Only include the oldest minimum RTT sample if it is not a
503  // half-step sample.
504  if ((oldest_cand_min_rtt >
505  (min_rtt_ + (0.6 * (cand_min_rtt - min_rtt_)))) ||
506  (oldest_cand_min_rtt <
507  (min_rtt_ + (0.4 * (cand_min_rtt - min_rtt_)))))
508  {
509  if (oldest_cand_min_rtt < cand_min_rtt)
510  {
511  cand_min_rtt = oldest_cand_min_rtt;
512  }
513  }
514 
515  // If the candidate minimum RTT exceeds the current minimum RTT by
516  // the specified factor, then adjust the minimum RTT upward.
517  if ((cand_min_rtt > (kMinRttTrackFactor * min_rtt_)) &&
518  (cand_min_rtt > (min_rtt_ + kMinRttTrackAmount)))
519  {
520  min_rtt_ = cand_min_rtt;
521 
522  LogA(kClassName, __func__, "Conn %" PRIEndptId ": Increased "
523  "min_rtt=%f\n", conn_id_, min_rtt_);
524 
525  // Use the updated minimum RTT value to decide what to do with the
526  // congestion window size and the inter-send time.
527  if (min_rtt_ <= kMinRttResetThreshold)
528  {
529  // Reset both.
530  cwnd_ = kMinCwndPkts;
531  ist_ = (min_rtt_ / cwnd_);
532  }
533  else
534  {
535  // Adjust the current congestion window for the change in
536  // minimum RTT. This assumes that the channel capacity has not
537  // changed, and is the fastest way to adapt on high-latency
538  // channels.
539  cwnd_ = (min_rtt_ / mrt_.ist_[mrt_.next_ist_index_]);
540  ist_ = mrt_.ist_[mrt_.next_ist_index_];
541  }
542 
543  // Reset the rest of the Copa Beta 2 parameters.
544  delta_ = kDefaultDelta;
545  velocity_ = 1;
546  cwnd_adj_up_ = 0;
547  cwnd_adj_down_ = 0;
548  prev_direction_ = VEL_DIR_NEITHER;
549 
550  // Reset the damper state.
551  damper_.state_ = DAMPER_MONITOR_HIGH;
552  damper_.hold_cnt_ = 0;
553  }
554  }
555  }
556 
557  // Record the current time.
558  mrt_.prev_time_ = ack_time;
559  }
560 
561  // Update the TCP compatibility state.
562  ++tc_.rtt_period_cnt_;
563 
564  // Check if the TCP compatibility period is over or not.
565  if (tc_.rtt_period_cnt_ >= tc_.rtt_periods_)
566  {
567  // If in default mode and there were no nearly empty queue events during
568  // the period, then test if the maximum and minimum queueing delays
569  // observed during this period fit the pattern of a very stable queueing
570  // condition. When this condition occurs, the observed queueing delays
571  // never drop down to the nearly empty queueing delay threshold value.
572  if ((!tc_.in_tcp_mode_) && (tc_.recent_neq_ == 0))
573  {
574  if ((tc_.recent_min_qd_ >= (kStableQueueLoFactor * ist_)) &&
575  (tc_.recent_max_qd_ <= (kStableQueueHiFactor * ist_)))
576  {
577 #ifdef SLIQ_CC_DEBUG
578  LogD(kClassName, __func__, "Stable queue detected\n");
579 #endif
580  tc_.recent_neq_ = 1;
581  }
582  }
583 
584  tc_.recent_min_qd_ = kHugeRtt;
585 
586  // Record the maximum queueing delay witnessed during the period.
587  tc_.max_qd_[tc_.next_index_] = tc_.recent_max_qd_;
588  tc_.recent_max_qd_ = 0.0;
589 
590  // Record the number of nearly empty queue events witnessed during the
591  // period.
592  tc_.neq_[tc_.next_index_] = tc_.recent_neq_;
593  tc_.recent_neq_ = 0;
594 
595  // Update the nearly empty queueing delay threshold value.
596  double overall_max_qd = tc_.max_qd_[0];
597 
598  for (uint32_t j = 1; j < kTcpCompStateSize; ++j)
599  {
600  if (tc_.max_qd_[j] > overall_max_qd)
601  {
602  overall_max_qd = tc_.max_qd_[j];
603  }
604  }
605 
606  tc_.nearly_empty_threshold_ = (kNeqFactor * overall_max_qd);
607 
608 #ifdef SLIQ_CC_DEBUG
609  LogD(kClassName, __func__, "Conn %" PRIEndptId ": Update nearly empty "
610  "queue threshold %f (%f %f %f %f)\n", conn_id_,
611  tc_.nearly_empty_threshold_, tc_.max_qd_[0], tc_.max_qd_[1],
612  tc_.max_qd_[2], tc_.max_qd_[3]);
613 #endif
614 
615  // Update the TCP compatibility mode. If the sender detects a nearly
616  // empty queue in all of the most recent TCP compatibility periods that
617  // are to be checked, then use default mode. Otherwise, use TCP mode.
618 
619  // \todo Complete the Copa Beta 2 TCP compatibility algorithm. Until
620  // completed, the algorithm is disabled. To re-enable it, delete the
621  // following #ifdef.
622 #ifdef DISABLE_TCP_COMPAT
623  bool new_in_tcp_mode = false;
624  size_t lim = (tc_.in_tcp_mode_ ? kTcpModePeriods :
625  kDfltModePeriods);
626  size_t idx = ((tc_.next_index_ + kTcpCompStateSize + 1 -
627  lim) % kTcpCompStateSize);
628 
629  for (size_t k = 0; k < lim; ++k)
630  {
631  if (tc_.neq_[idx] == 0)
632  {
633  new_in_tcp_mode = true;
634 
635  // Reset the damper state.
636  damper_.state_ = DAMPER_MONITOR_HIGH;
637  damper_.hold_cnt_ = 0;
638 
639  break;
640  }
641 
642  idx = ((idx + 1) % kTcpCompStateSize);
643  }
644 
645  tc_.in_tcp_mode_ = new_in_tcp_mode;
646  tc_.rtt_periods_ = (new_in_tcp_mode ? kTcpModeRttPeriods :
647  kDfltModeRttPeriods);
648 #endif
649 
650  if (!tc_.in_tcp_mode_)
651  {
652  delta_ = kDefaultDelta;
653 
654 #ifdef SLIQ_CC_DEBUG
655  LogD(kClassName, __func__, "Conn %" PRIEndptId ": Updated delta=%f\n",
656  conn_id_, delta_);
657 #endif
658  }
659 
660 #ifdef SLIQ_CC_DEBUG
661  LogD(kClassName, __func__, "Conn %" PRIEndptId ": Updated TCP "
662  "compatibility mode %d (%" PRIu32 ": %" PRIu32 " %" PRIu32 " %"
663  PRIu32 " %" PRIu32 " )\n", conn_id_,
664  static_cast<int>(tc_.in_tcp_mode_), tc_.next_index_, tc_.neq_[0],
665  tc_.neq_[1], tc_.neq_[2], tc_.neq_[3]);
666 #endif
667 
668  // Start a new TCP compatibility RTT period.
669  tc_.next_index_ = ((tc_.next_index_ + 1) % kTcpCompStateSize);
670  tc_.rtt_period_cnt_ = 0;
671  }
672 
673  // Start a new RTT period.
674  rtt_period_end_ = (ack_time + rtt);
675  }
676 
677  // Estimate the queueing delay, which is in seconds.
678  double queueing_delay = (measured_rtt - min_rtt_);
679 
680  // Update the TCP compatibility state based on the queueing delay.
681  if (queueing_delay > tc_.recent_max_qd_)
682  {
683  tc_.recent_max_qd_ = queueing_delay;
684  }
685 
686  if (queueing_delay < tc_.recent_min_qd_)
687  {
688  tc_.recent_min_qd_ = queueing_delay;
689  }
690 
691  if (queueing_delay < tc_.nearly_empty_threshold_)
692  {
693  ++(tc_.recent_neq_);
694  }
695 
696  // Update the damper state.
697  if (damper_.state_ == DAMPER_MONITOR_HIGH)
698  {
699  // Watch for instances when there are clearly too many packets queued at
700  // the bottleneck link. This cannot be done in TCP mode.
701  if ((!tc_.in_tcp_mode_) && ((queueing_delay / ist_) > kDamperThresPkts))
702  {
703  damper_.state_ = DAMPER_MONITOR_LOW;
704 
705 #ifdef SLIQ_CC_DEBUG
706  LogD(kClassName, __func__, "Conn %" PRIEndptId ": Damper MONITOR_HIGH "
707  "-> MONITOR_LOW.\n", conn_id_);
708 #endif
709  }
710  }
711  else if (damper_.state_ == DAMPER_MONITOR_LOW)
712  {
713  // Watch for when the queueing delay just drops below 1/delta packets.
714  if ((queueing_delay / ist_) < (1.0 / kDefaultDelta))
715  {
716  damper_.state_ = DAMPER_HOLD;
717  damper_.hold_cnt_ = 0;
718 
719  cwnd_ = static_cast<double>(cc_val);
720  velocity_ = 1;
721  cwnd_adj_up_ = 0;
722  cwnd_adj_down_ = 0;
723  prev_direction_ = VEL_DIR_NEITHER;
724 
725 #ifdef SLIQ_CC_DEBUG
726  LogD(kClassName, __func__, "Conn %" PRIEndptId ": Damper MONITOR_LOW "
727  "-> HOLD at cwnd=%f.\n", conn_id_, cwnd_);
728 #endif
729  }
730  }
731 
732  // Compute the current target rate, which is in packets per second.
733  double lambda_target = kMaxRate;
734 
735  if (queueing_delay > 0.0)
736  {
737  lambda_target = (1.0 / (delta_ * queueing_delay));
738  }
739 
740  // Compute the current rate, which is in packets per second.
741  double lambda = (cwnd_ / measured_rtt);
742 
743  // Limit the velocity so that the rate can never more than double once per
744  // RTT.
745  uint32_t max_velocity = static_cast<uint32_t>(delta_ * cwnd_);
746 
747  if (velocity_ > max_velocity)
748  {
749  velocity_ = max_velocity;
750 
751 #ifdef SLIQ_CC_DEBUG
752  LogD(kClassName, __func__, "Conn %" PRIEndptId ": Reduce velocity=%"
753  PRIu32 "\n", conn_id_, velocity_);
754 #endif
755  }
756 
757  // Make sure that the velocity is never less than 1.
758  if (velocity_ < 1)
759  {
760  velocity_ = 1;
761  }
762 
763  // Adjust the congestion window size, which is in packets. Scale the
764  // adjustment amount by the size of the packet.
765  double cwnd_adj = ((static_cast<double>(bytes) *
766  static_cast<double>(velocity_)) /
767  (static_cast<double>(kNominalPktSizeBytes -
768  kDataHdrBaseSize) *
769  delta_ * cwnd_));
770 
771  if (damper_.state_ != DAMPER_HOLD)
772  {
773  if (lambda <= lambda_target)
774  {
775  // Only increase the congestion window if the current number of packets
776  // in flight is at least one-half of the current congestion window size.
777  // This prevents the congestion window size from increasing indefinitely
778  // when the sender is not keeping the channel full.
779  double pif = (static_cast<double>(bytes_in_flight_) /
780  static_cast<double>(kNominalPktSizeBytes -
781  kDataHdrBaseSize));
782 
783  if (((cwnd_ <= 8.0) && (pif >= (cwnd_ - 4.0))) ||
784  ((cwnd_ > 8.0) && (pif >= (0.5 * cwnd_))))
785  {
786  cwnd_ += cwnd_adj;
787  ++cwnd_adj_up_;
788 
789 #ifdef SLIQ_CC_DEBUG
790  LogD(kClassName, __func__, "Conn %" PRIEndptId ": Increased "
791  "cwnd=%f\n", conn_id_, cwnd_);
792 #endif
793  }
794 #ifdef SLIQ_CC_DEBUG
795  else
796  {
797  LogD(kClassName, __func__, "Conn %" PRIEndptId ": Froze cwnd=%f "
798  "pif=%f\n", conn_id_, cwnd_, pif);
799  }
800 #endif
801  }
802  else
803  {
804  cwnd_ -= cwnd_adj;
805  ++cwnd_adj_down_;
806 
807 #ifdef SLIQ_CC_DEBUG
808  LogD(kClassName, __func__, "Conn %" PRIEndptId ": Decreased cwnd=%f\n",
809  conn_id_, cwnd_);
810 #endif
811  }
812  }
813 
814  // Limit how small/large the congestion window size can be.
815  if (cwnd_ < kMinCwndPkts)
816  {
817  cwnd_ = kMinCwndPkts;
818  }
819 
820  if (cwnd_ > static_cast<double>(kMaxCongCtrlWindowPkts))
821  {
822  cwnd_ = static_cast<double>(kMaxCongCtrlWindowPkts);
823  }
824 
825  // Update the current inter-send time.
826  ist_ = (last_rtt_ / cwnd_);
827 
828  // Limit the inter-send time if needed.
829  if (ist_ < kMinIst)
830  {
831  ist_ = kMinIst;
832  }
833 
834 #ifdef SLIQ_CC_DEBUG
835  LogD(kClassName, __func__, "Conn %" PRIEndptId ": Computed rtt=%f "
836  "min_rtt=%f queueing_delay=%f lambda_target=%f lambda=%f cwnd=%f "
837  "ist=%f\n", conn_id_, measured_rtt, min_rtt_, queueing_delay,
838  lambda_target, lambda, cwnd_, ist_);
839  LogA(kClassName, __func__, "Conn %" PRIEndptId ": PLT_COPA2 %f %f %f %f %f "
840  "%f %" PRIu32 " %f %0.9f %f\n", conn_id_, CurrentTime(ack_time),
841  last_rtt_, min_rtt_, queueing_delay, lambda_target, lambda, velocity_,
842  cwnd_, ist_,
843  (static_cast<double>(bytes_in_flight_) /
844  static_cast<double>(kNominalPktSizeBytes - kDataHdrBaseSize)));
845 #endif
846 }

848 //============================================================================
849 bool CopaBeta2::OnPacketLost(StreamId /* stream_id */,
850  const Time& /* ack_time */,
851  PktSeqNumber /* seq_num */,
852  PktSeqNumber /* cc_seq_num */,
853  uint32_t /* bytes */)
854 {
855  // If a packet has been lost, we are in TCP mode, and an RTT period has
856  // passed, then increase the delta value.
857  //
858  // delta' = 2 * delta
859  //
860  // This has the effect of decreasing the target send rate, which will force
861  // the congestion window size to shrink. The net result is a TCP-like
862  // multiplicative decrease in the window size.
863  //
864  // Given that the number of buffered packets is equal to (1 / delta), this
865  // has the effect of halving the buffer size.
866  //
867  // new_buf_size = (1 / delta')
868  // = (1 / 2) * (1 / delta)
869  //
870  if (tc_.in_tcp_mode_)
871  {
872  Time now = Time::Now();
873 
874  if (now > tc_.next_delta_update_time_loss_)
875  {
876  delta_ = (2.0 * delta_);
877 
878  if (delta_ < kMaxDelta)
879  {
880  delta_ = kMaxDelta;
881  }
882 
883 #ifdef SLIQ_CC_DEBUG
884  LogD(kClassName, __func__, "Conn %" PRIEndptId ": Increased delta=%f\n",
885  conn_id_, delta_);
886 #endif
887 
888  tc_.next_delta_update_time_loss_ = (now + Time(last_rtt_));
889  }
890  }
891 
892  return true;
893 }

895 //============================================================================
896 void CopaBeta2::OnPacketAcked(StreamId /* stream_id */, const Time& ack_time,
897  PktSeqNumber /* seq_num */,
898  PktSeqNumber /* cc_seq_num */,
899  PktSeqNumber /* ne_seq_num */,
900  uint32_t /* bytes */)
901 {
902  // If in TCP mode and an RTT period has passed, then decrease the delta
903  // value.
904  //
905  // delta' = 1 / (1 + (1 / delta))
906  //
907  // This has the effect of increasing the target send rate, which
908  // will allow the congestion window size to grow. The net result is a
909  // TCP-like additive increase in the window size.
910  //
911  // Given that the number of buffered packets is equal to (1 / delta), this
912  // has the effect of adding one packet to the buffer size.
913  //
914  // new_buf_size = (1 / delta')
915  // = (1 / delta) + 1
916  //
917  if (tc_.in_tcp_mode_ && (ack_time > tc_.next_delta_update_time_ack_))
918  {
919  delta_ = (1.0 / (1.0 + (1.0 / delta_)));
920 
921  if (delta_ < kMinDelta)
922  {
923  delta_ = kMinDelta;
924  }
925 
926 #ifdef SLIQ_CC_DEBUG
927  LogD(kClassName, __func__, "Conn %" PRIEndptId ": Decreased delta=%f\n",
928  conn_id_, delta_);
929 #endif
930 
931  tc_.next_delta_update_time_ack_ = (ack_time + Time(last_rtt_));
932  }
933 }

1140 //============================================================================
1141 void CopaBeta2::ProcessCcPktTrain(const Time& now, CcPktTrainHeader& hdr)
1142 {
1143  uint32_t pair = (hdr.pt_seq_num / 2);
1144 
1145  // Check if this is an FS_DATA packet.
1146  if (hdr.pt_pkt_type == FS_DATA)
1147  {
1148 #ifdef SLIQ_CC_DEBUG
1149  LogD(kClassName, __func__, "Conn %" PRIEndptId ": Received packet pair "
1150  "FS_DATA with seq=%" PRIu8 ".\n", conn_id_, hdr.pt_seq_num);
1151 #endif
1152 
1153  // Update local state and compute the packet pair inter-receive time to
1154  // send back in the FS_ACK packet.
1155  uint32_t irt_usec = 0;
1156 
1157  if ((hdr.pt_seq_num % 2) == 0)
1158  {
1159  // This is the first FS_DATA packet for a pair. Record its receive
1160  // time.
1161  fs_.pair_recv_time_[pair] = now;
1162  }
1163  else
1164  {
1165  // This is the second FS_DATA packet for a pair. Compute the delay
1166  // between receipt of the two packets.
1167  if (!fs_.pair_recv_time_[pair].IsZero())
1168  {
1169  Time irt = (now - fs_.pair_recv_time_[pair]);
1170 
1171  irt_usec = static_cast<uint32_t>(irt.GetTimeInUsec());
1172 
1173  if (irt_usec == 0)
1174  {
1175  irt_usec = 1;
1176  }
1177 
1178 #ifdef SLIQ_CC_DEBUG
1179  LogD(kClassName, __func__, "Conn %" PRIEndptId ": Second packet of "
1180  "pair received, irt=%f.\n", conn_id_, irt.ToDouble());
1181 #endif
1182  }
1183  }
1184 
1185  // Send an FS_ACK packet immediately.
1186  SendPktPairAck(hdr.pt_seq_num, irt_usec);
1187  return;
1188  }
1189 
1190  // This is an FS_ACK packet.
1191 
1192 #ifdef SLIQ_CC_DEBUG
1193  LogD(kClassName, __func__, "Conn %" PRIEndptId ": Received packet pair "
1194  "FS_ACK with seq=%" PRIu8 ".\n", conn_id_, hdr.pt_seq_num);
1195 #endif
1196 
1197  // Copa Beta 2 must be in the fast startup state to process the FS_ACK
1198  // packet.
1199  if (state_ != FAST_STARTUP)
1200  {
1201  // It is possible for an FS_ACK packet to arrive late. Thus, if we are in
1202  // the CLOSED_LOOP state, just ignore the packet without logging an error.
1203  if (state_ != CLOSED_LOOP)
1204  {
1205  LogE(kClassName, __func__, "Conn %" PRIEndptId ": Invalid state %d.\n",
1206  conn_id_, static_cast<int>(state_));
1207  }
1208  return;
1209  }
1210 
1211  if ((hdr.pt_seq_num % 2) == 0)
1212  {
1213  // This is an FS_ACK of the first packet in the pair. Use it for an RTT
1214  // estimate.
1215  if (!fs_.pair_send_time_[pair].IsZero())
1216  {
1217  Time diff = (now - fs_.pair_send_time_[pair]);
1218 
1219  fs_.rtt_[pair] = diff.ToDouble();
1220 
1221 #ifdef SLIQ_CC_DEBUG
1222  LogD(kClassName, __func__, "Conn %" PRIEndptId ": First packet of pair "
1223  "FS_ACKed, rtt=%f.\n", conn_id_, fs_.rtt_[pair]);
1224 #endif
1225  }
1226  }
1227  else
1228  {
1229  // This is an FS_ACK of the second packet in the pair. Use it for a
1230  // bottleneck link rate estimate in packets per second.
1231  if (hdr.pt_inter_recv_time != 0)
1232  {
1233  double irt_sec = (static_cast<double>(hdr.pt_inter_recv_time) *
1234  0.000001);
1235 
1236  fs_.rate_[pair] = (1.0 / irt_sec);
1237 
1238 #ifdef SLIQ_CC_DEBUG
1239  LogD(kClassName, __func__, "Conn %" PRIEndptId ": Second packet of "
1240  "pair FS_ACKed, irt=%f rate=%f.\n", conn_id_, irt_sec,
1241  fs_.rate_[pair]);
1242 #endif
1243  }
1244  }
1245 
1246  // If this is the last FS_ACK needed, then end fast startup immediately.
1247  if (hdr.pt_seq_num == ((2 * kNumFsPairs) - 1))
1248  {
1249  timer_.CancelTimer(fs_.timer_);
1250  FsDoneCallback();
1251  }
1252 }

1301 void CopaBeta2::UpdateNextSendTime(const Time& now, size_t bytes)
1302 {
1303  // Update the next send time using the packet size and the stored next send
1304  // time. This maintains inter-send time accuracy.
1305  double pkt_intersend_time =
1306  (ist_ *
1307  (static_cast<double>(bytes + kDataHdrBaseSize + kPktOverheadBytes) /
1308  static_cast<double>(kNominalPktSizeBytes + kPktOverheadBytes)));
1309

1311  // stored next send time, then the sender is considered to have been
1312  // quiescent for a time, so the next send time must be computed from now.
1313  // Otherwise, the send pacing timer must have been used, so add the
1314  // inter-send time for this packet to the stored next send time.
1315  if (now > next_send_time_.Add(kQuiescentThreshold))
1316  {
1317  next_send_time_ = now.Add(pkt_intersend_time);
1318  }
1319  else
1320  {
1321  next_send_time_ = next_send_time_.Add(pkt_intersend_time);
1322  }
1323 
1324 #ifdef SLIQ_CC_DEBUG
1325  LogD(kClassName, __func__, "Conn %" PRIEndptId ": Waiting for "
1326  "intersend_time=%f\n", conn_id_, pkt_intersend_time);
1327 #endif
1328 }

1426 void CopaBeta2::FsDoneCallback()
1427 {
1428  if (state_ != FAST_STARTUP)
1429  {
1430  LogE(kClassName, __func__, "Conn %" PRIEndptId ": Invalid state %d.\n",
1431  conn_id_, static_cast<int>(state_));
1432  return;
1433  }
1434 
1435  // Find the minimum, maximum, and mean RTT estimates, as well as the
1436  // bottleneck link rate estimate (mu).
1437  uint32_t rtt_cnt = 0;
1438  uint32_t rate_cnt = 0;
1439  double rtt_min = kHugeRtt;
1440  double rtt_max = -1.0;
1441  double mean_rtt = 0.0;
1442  double mu = 0.0;
1443 
1444  // Start at 1, skipping the first sample. This is because the first sample
1445  // is usually very inaccurate.
1446  for (size_t pair = 1; pair < kNumFsPairs; ++pair)
1447  {
1448  double rtt_est = fs_.rtt_[pair];
1449 
1450  if (rtt_est > 0.0)
1451  {
1452  mean_rtt += rtt_est;
1453  ++rtt_cnt;
1454 
1455  if (rtt_est < rtt_min)
1456  {
1457  rtt_min = rtt_est;
1458  }
1459 
1460  if (rtt_est > rtt_max)
1461  {
1462  rtt_max = rtt_est;
1463  }
1464  }
1465 
1466  double rate_est = fs_.rate_[pair];
1467 
1468  if (rate_est > 0.0)
1469  {
1470  mu += rate_est;
1471  ++rate_cnt;
1472  }
1473  }
1474 
1475  // If there was not enough data, then restart the fast startup.
1476  if ((rtt_cnt == 0) || (rate_cnt == 0))
1477  {
1478  LogE(kClassName, __func__, "Conn %" PRIEndptId ": Incomplete fast "
1479  "startup, trying again.\n", conn_id_);
1480  fs_.Clear();
1481  timer_.CancelTimer(fs_.timer_);
1482  FsPktPairCallback();
1483  return;
1484  }
1485 
1486  // Complete the mean computations.
1487  mean_rtt = (mean_rtt / static_cast<double>(rtt_cnt));
1488  mu = (mu / static_cast<double>(rate_cnt));
1489 
1490  // Compute the target rate, in packets per second.
1491  double lambda_target = mu;
1492 
1493  if (rtt_max > rtt_min)
1494  {
1495  double tmp = (2.0 / (delta_ * (rtt_max - rtt_min)));
1496 
1497  if (tmp < mu)
1498  {
1499  lambda_target = tmp;
1500  }
1501  }
1502 
1503  // Limit the target rate to the maximum allowable startup rate.
1504  if (lambda_target > kMaxStartupRate)
1505  {
1506  lambda_target = kMaxStartupRate;
1507  }
1508 
1509  // Set the initial parameters based on the fast startup results.
1510  last_rtt_ = mean_rtt;
1511  min_rtt_ = rtt_min;
1512  cwnd_ = (lambda_target * rtt_min);
1513  ist_ = (1.0 / lambda_target);
1514 
1515  LogA(kClassName, __func__, "Conn %" PRIEndptId ": Fast startup, rtt_min=%f "
1516  "rtt_max=%f mean_rtt=%f mu=%f lambda_target=%f last_rtt=%f min_rtt=%f "
1517  "cwnd=%f ist=%f\n", conn_id_, rtt_min, rtt_max, mean_rtt, mu,
1518  lambda_target, last_rtt_, min_rtt_, cwnd_, ist_);
1519 
1520  // The algorithm is now ready for closed loop operation.
1521  state_ = CLOSED_LOOP;
1522 }
