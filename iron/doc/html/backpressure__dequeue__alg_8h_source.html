164  // ToString method.
165  std::string ToString() const
166  {
167  std::string str;
168 
169  if (!is_valid)
170  {
171  str.append("INVALID CANDIDATE! ");
172  }
173  if (pkt)
174  {
175  char pkt_addr[24];
176  snprintf (pkt_addr, sizeof(pkt_addr), "%p", pkt);
177  str.append("Pkt ").append(pkt_addr).append("(").append(
178  pkt->GetPacketMetadataString()).append("):");
179  }
180  else
181  {
182  str.append("Pkt UNKNOWN:");
183  }
184  str.append(" gradient: ").append(StringUtils::ToString(gradient));
185  str.append("B, len: ").append(StringUtils::ToString(virtual_len));
186  str.append(" to (bin ").append(id_to_log);
187  str.append(", pc ");
188  str.append(StringUtils::ToString(
189  static_cast<int>(path_ctrl_index)));
190  if (dst_vec != 0)
191  {
192  str.append(", dst_vec ").append(StringUtils::ToString(dst_vec));
193  }
194  str.append(") with expected ttr ").append(ttr.ToString());
195  str.append(" and ").append(ttg.ToString()).append(" to make it");
196 
197  return str;
198  }
