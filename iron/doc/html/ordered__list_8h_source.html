563  inline void InsertBefore(LLElem* ref_elem, LLElem* elem)
564  {
565  // Insert before the reference element.
566  if (ref_elem->prev)
567  {
568  ref_elem->prev->next = elem;
569  }
570  else
571  {
572  // It goes before the existing head, so it becomes the new head_.
573  head_ = elem;
574  }
575  elem->prev = ref_elem->prev;
576  elem->next = ref_elem;
577  ref_elem->prev = elem;
578  size_++;
579  }

587  inline void Remove(LLElem* e)
588  {
589  if (e == head_)
590  {
591  head_ = e->next;
592  }
593 
594  if (e == tail_)
595  {
596  tail_ = e->prev;
597  }
598 
599  if (e->next)
600  {
601  e->next->prev = e->prev;
602  }
603 
604  if (e->prev)
605  {
606  e->prev->next = e->next;
607  }
608  size_--;
609  e->prev = NULL;
610  e->next = NULL;
611  }
