170 bool Packet::RemoveBytesFromBeginning(size_t length)
171 {
172  // As an optimization, the requested number of bytes are NOT actually
173  // removed from the internal buffer. Rather, the start_ and length_ class
174  // variables are adjusted to reflect the requested smaller packet. This
175  // enables us to efficiently "strip off" outer encapsulating headers
176  // (without having to copy the buffer data).
177 
178  // If there are any metdata headers, then bytes cannot be removed from the
179  // beginning of the internal buffer.
180  if (metadata_length_ > 0)
181  {
182  LogW(kClassName, __func__, "Request to remove %zu bytes from a packet "
183  "with metadata headers (%zu bytes).\n", length, metadata_length_);
184  return false;
185  }
186 
187  // Ensure that there are enough bytes in the internal buffer to support the
188  // requested reduction.
189  if (length > length_)
190  {
191  LogW(kClassName, __func__, "Request to remove %zu bytes from a packet "
192  "with a length of %zu bytes.\n", length, length_);
193  return false;
194  }
195 
196  // Adjust the "start of" the packet and its length and set its type to
197  // unknown as we don't know the type of the packet after removing bytes.
198  start_ += length;
199  length_ -= length;
200  type_ = UNKNOWN_PACKET;
201  latency_ = UNSET_LATENCY;
202 
203  return true;
204 }

243 {
244  // Must have enough room.
245  if ((start_ + length_ + len) > kMaxPacketSizeBytes)
246  {
247  LogW(kClassName, __func__, "Unable to append %zu bytes to packet with "
248  "current size of %zu bytes, a start at offset %zu, and a maximum "
249  "size of %zu bytes.\n", len, length_, start_, kMaxPacketSizeBytes);
250  return false;
251  }
252 
253  memcpy(buffer_+ start_ + length_, data, len);
254  length_ += len;
255 
256  if (type_ == UNKNOWN_PACKET)
257  {
258  ParseType();
259  }
260 
261  // Make sure to adjust the various embedded lengths for IPv4 packets.
262  if (type_ == IPV4_PACKET)
263  {
264  struct iphdr* ip = GetIpHdr();
265  ip->tot_len = htons(ntohs(ip->tot_len) + len);
266 
267  uint8_t protocol;
268  if (!GetIpProtocol(protocol))
269  {
270  return false;
271  }
272 
273  if (protocol == IPPROTO_UDP)
274  {
275  // Make sure the length of the packet is long enough to have an UDP
276  // header.
277  if (length_ >= (size_t)((ip->ihl * 4) + sizeof(struct udphdr)))
278  {
279  struct udphdr* udp = reinterpret_cast<struct udphdr*>(
280  buffer_ + start_ + (ip->ihl * 4));
281  udp->len = htons(ntohs(udp->len) + len);
282  }
283  else
284  {
285  return false;
286  }
287  }
288  }

568 bool Packet::SetIpDscp(uint8_t dscp)
569 {
570  struct iphdr* ip_hdr = GetIpHdr();
571 
572  if (!ip_hdr)
573  {
574  return false;
575  }
576 
577  if (dscp >= (1 << 6))
578  {
579  LogI(kClassName, __func__, "Cannot set DSCP value %" PRIu8 " (exceeds "
580  "6-bit field).\n", dscp);
581  return false;
582  }
583 
584  // Clear only the DSCP field (leave ECN intact).
585  //
586  // TOS field:
587  // ---------------------------------
588  // | 7 | 6 | 5 | 4 | 3 | 2 | 1 | 0 |
589  // ---------------------------------
590  // | DSCP | ECN |
591  // ---------------------------------
592  ip_hdr->tos &= 0x03;
593 
594  ip_hdr->tos |= (dscp << 2);
595 
596  if (ip_hdr->check != 0)
597  {
598  if (UpdateIpChecksum() == 0)
599  {
600  return false;
601  }
602  }
603  latency_ = UNSET_LATENCY;
604  return true;
605 }

609 {

611

630 LatencyClass Packet::GetLatencyClass() const
631 {
632  if (latency_ != UNSET_LATENCY)
633  {
634  return latency_;
635  }
636 
637  if (type_ == UNKNOWN_PACKET)
638  {
639  ParseType();
640  }
641 
642  uint8_t dscp = 0;
643 
644  // Use the packet's type to help determine the latency class.
645  switch (type_)
646  {
647  case IPV4_PACKET:
648  // For IPv4 packets, use the DSCP value to determine the latency class.
649  if (!GetIpDscp(dscp))
650  {
651  LogE(kClassName, __func__, "Error: Could not get DSCP value out of "
652  "packet, treat as normal latency.\n");
653  latency_ = NORMAL_LATENCY;
654  return latency_;
655  }
656 
657  if (dscp == DSCP_EF)
658  {
659  latency_ = LOW_LATENCY;
660  }
661  else if (dscp == DSCP_TOLERANT)
662  {
663  latency_ = HIGH_LATENCY_RCVD;
664  }
665  else
666  {
667  latency_ = NORMAL_LATENCY;
668  }
669  break;
670 
671  case LSA_PACKET:
672  // These are all system-level control packets with their own latency
673  // class.
674  latency_ = CONTROL_TRAFFIC_LATENCY;
675  break;
676 
677  case ZOMBIE_PACKET:
678  latency_ = HIGH_LATENCY_RCVD;
679  break;
680 
681  case QLAM_PACKET:
682  // Qlam packets should not be assigned a latency class. They are sent
683  // and processed outside of backpressure forwarding. This falls through
684  // to the default case.
685  default:
686  LogE(kClassName, __func__, "Error: Unexpected packet type 0x%02x, "
687  "treating as normal latency.\n", type_);
688  latency_ = NORMAL_LATENCY;
689  }
690 
691  return latency_;
692 }

764 size_t Packet::GetIpPayloadOffset() const
765 {
766  // Need at least an IP header to go further.
767  if (length_ < sizeof(struct iphdr))
768  {
769  return length_;
770  }
771 
772  const struct iphdr* ip_hdr =
773  reinterpret_cast<const struct iphdr*>(buffer_ + start_);
774 
775  uint8_t protocol = 0;
776  if (!GetIpProtocol(protocol))
777  {
778  return length_;
779  }
780 
781  if (protocol == IPPROTO_TCP)
782  {
783  // Need at least a TCP header to go further.
784  if (length_ < (size_t)((ip_hdr->ihl * 4) + sizeof(struct tcphdr)))
785  {
786  return length_;
787  }
788 
789  const struct tcphdr* tcp_hdr =
790  reinterpret_cast<const struct tcphdr*>(buffer_ + start_ +
791  (ip_hdr->ihl * 4));
792  return ((ip_hdr->ihl * 4) + (tcp_hdr->doff * 4));
793  }
794  else if (protocol == IPPROTO_UDP)
795  {
796  // Need at least a UDP header to go further.
797  if (length_ < (size_t)((ip_hdr->ihl * 4) + sizeof(struct udphdr)))
798  {
799  return length_;
800  }
801 
802  return ((ip_hdr->ihl * 4) + sizeof(struct udphdr));
803  }
804  else if (protocol == IPPROTO_ESP)
805  {
806  // For ESP packets, everything beyond the IP header is considered data.
807  return (ip_hdr->ihl * 4);
808  }
809  else
810  {
811  // We have an IP packet with no transport layer header. Payload starts
812  // after the header.
813  return (ip_hdr->ihl * 4);
814  }
815 }

1134 //============================================================================
1135 bool Packet::UpdateIpChecksum()
1136 {
1137  struct iphdr* ip_hdr = GetIpHdr();
1138 
1139  int sum = 0;
1140  int nleft = sizeof(struct iphdr);
1141 
1142  unsigned short* w = (unsigned short *)ip_hdr;
1143  unsigned short odd_byte = 0;
1144 
1145  // Need at least an IP header.
1146  if (!ip_hdr)
1147  {
1148  return false;
1149  }
1150 
1151  // We absolutely must clear the checksum.
1152  ip_hdr->check = 0;
1153 
1154  // Our algorithm is simple. Using a 32 bit accumulator (sum), we add
1155  // sequential 16 bit words to it, and at the end, fold back all the carry
1156  // bits from the top 16 bits into the lower 16 bits.
1157  while (nleft > 1)
1158  {
1159  sum += *w++;
1160  nleft -= 2;
1161  }
1162 
1163  // Mop up an odd byte, if necessary.
1164  if (nleft == 1)
1165  {
1166  *(unsigned char *)(&odd_byte) = *(unsigned char *)w;
1167  sum += odd_byte;
1168  }
1169 
1170  // Add back carry outs from top 16 bits to low 16 bits.
1171  sum = (sum >> 16) + (sum & 0xffff); // add hi 16 to low 16
1172  sum += (sum >> 16); // add carry
1173 
1174  ip_hdr->check = (unsigned short)~sum;
1175 
1176  return true;
1177 }

1223 //============================================================================
1224 bool Packet::ComputeTransportChecksum(size_t len, uint16_t& csum)
1225 {
1226  struct iphdr* ip_hdr = GetIpHdr();
1227 
1228  if (!ip_hdr)
1229  {
1230  return false;
1231  }
1232 
1233  unsigned char* hdr;
1234  struct udphdr* udp_hdr;
1235  struct tcphdr* tcp_hdr;
1236 
1237  uint32_t src_addr;
1238  uint32_t dst_addr;
1239 
1240  if (!GetIpSrcAddr(src_addr))
1241  {
1242  return false;
1243  }
1244 
1245  if (!GetIpDstAddr(dst_addr))
1246  {
1247  return false;
1248  }
1249 
1250  uint8_t protocol;
1251  if (!GetIpProtocol(protocol))
1252  {
1253  return false;
1254  }
1255 
1256  hdr = (unsigned char *)(buffer_ + start_ + (ip_hdr->ihl * 4));
1257 
1258  if (protocol == IPPROTO_TCP)
1259  {
1260  tcp_hdr = reinterpret_cast<struct tcphdr*>(hdr);
1261  tcp_hdr->check = 0;
1262  }
1263  else if (protocol == IPPROTO_UDP)
1264  {
1265  udp_hdr = reinterpret_cast<struct udphdr*>(hdr);
1266  udp_hdr->check = 0;
1267  }
1268  else if (protocol == IPPROTO_ESP)
1269  {
1270  // There are no transport checksums for ESP packets.
1271  return true;
1272  }
1273  else
1274  {
1275  return false;
1276  }
1277 
1278  // Compute the transport psuedo header checksum. Do not actually assemble
1279  // the psuedo header in memory, just add in the fields required. These are:
1280  //
1281  // 0 7 8 15 16 23 24 31
1282  // +--------+--------+--------+--------+
1283  // | source address |
1284  // +--------+--------+--------+--------+
1285  // | destination address |
1286  // +--------+--------+--------+--------+
1287  // | zero |protocol| length |
1288  // +--------+--------+--------+--------+
1289 
1290  uint64_t sum = 0;
1291 
1292  // The following four quantities will not generate an overflow so no
1293  // roll-over handling is required.
1294  sum += src_addr;
1295  sum += dst_addr;
1296  sum += htons((unsigned short)protocol);
1297  sum += htons((unsigned short)len);
1298 
1299  uint64_t* b = reinterpret_cast<uint64_t *>(hdr);
1300  uint32_t size = len;
1301 
1302  // Main loop - 8 bytes at a time
1303  while (size >= sizeof(uint64_t))
1304  {
1305  uint64_t s = *b++;
1306  sum += s;
1307  if (sum < s)
1308  {
1309  sum++;
1310  }
1311  size -= sizeof(uint64_t);
1312  }
1313 
1314  // Handle tails less than 8-bytes long
1315  uint8_t* bb = (uint8_t*) b;
1316 
1317  if (size & 4)
1318  {
1319  uint32_t s = *reinterpret_cast<uint32_t *>(bb);
1320  sum += s;
1321  if (sum < s)
1322  {
1323  sum++;
1324  }
1325  bb += 4;
1326  }
1327 
1328  if (size & 2)
1329  {
1330  uint16_t s = *reinterpret_cast<int16_t *>(bb);
1331  sum += s;
1332  if (sum < s)
1333  {
1334  sum++;
1335  }
1336  bb += 2;
1337  }
1338 
1339  if (size & 1)
1340  {
1341  uint8_t s = *(uint8_t *)bb;
1342  sum += s;
1343  if (sum < s)
1344  {
1345  sum++;
1346  }
1347  }
1348 
1349  // Fold down to 16 bits.
1350  uint32_t t1;
1351  uint32_t t2;
1352 
1353  t1 = sum;
1354  t2 = sum >> 32;
1355  t1 += t2;
1356 
1357  if (t1 < t2)
1358  {
1359  t1++;
1360  }
1361 
1362  uint16_t t3;
1363 
1364  t3 = t1;
1365  csum = t1 >> 16;
1366  csum += t3;
1367 
1368  if (csum < t3)
1369  {
1370  csum++;
1371  }
1372 
1373  csum = ~csum;
1374 
1375  return true;
1376 }
