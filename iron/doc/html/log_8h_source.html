56 #ifdef __COVERITY__
57 // LogF is a macro that calls InternalLog(LOG_FATAL, ....), which
58 // calls abort. However, Coverity does not infer that this happens,
59 // and thus has false positives for code errors after LogF returns.
60 // Work around this by adding an abort call directly into the LogF
61 // definition, when compiling for Coverity.
62 
63 // Declare abort, cribbed from stdlib.h. Obviously we should just
64 // include stdlib.h, but then files that include it will get a defect
65 // for a redundant include. Since this is a hack to convince Coverity
66 // that LogF will not return, accept the ick.
67 extern "C" {
68 extern void abort (void) __THROW __attribute__ ((__noreturn__));
69 }
70 #define COVERITY_ABORT ; abort()
71 #else // __COVERITY__
72 #define COVERITY_ABORT
73 #endif // __COVERITY__
