100 void PacketSet::Initialize(size_t num_packets)
101 {
102  // A packet set should manage a minimum number of packets.
103  if (num_packets < kMinPktSetSize)
104  {
105  LogW(kClassName, __func__, "Number of packets specified was %zu, using "
106  "%zu.\n", num_packets, kMinPktSetSize);
107  num_packets = kMinPktSetSize;
108  }
109 
110  if (num_packets > max_size_)
111  {
112  if (pkt_info_ != NULL)
113  {
114  delete [] pkt_info_;
115  }
116 
117  if (msg_hdr_ != NULL)
118  {
119  delete [] msg_hdr_;
120  }
121 
122  // Allocate the arrays.
123  pkt_info_ = new (std::nothrow) PktInfo[num_packets];
124  msg_hdr_ = new (std::nothrow) mmsghdr[num_packets];
125 
126  if ((pkt_info_ == NULL) || (msg_hdr_ == NULL))
127  {
128  LogF(kClassName, __func__, "Memory allocation error.\n");
129  }
130 
131  // Initialize the state.
132  max_size_ = num_packets;
133  cur_size_ = 0;
134  ret_idx_ = 0;
135  walk_idx_ = 0;
136 
137  PktInfo::SetPacketPool(&pkt_pool_);
138 
139  // Set up all of the mmsghdr, msghdr, and iovec structures.
140  for (size_t i = 0; i < max_size_; ++i)
141  {
142  pkt_info_[i].packet_ = pkt_pool_.Get();
143 
144  if (pkt_info_[i].packet_ == NULL)
145  {
146  LogF(kClassName, __func__, "Unable to get packet from pool.\n");
147  }
148 #ifdef PACKET_TRACKING
149  NEW_HELD_PKT_LOC(pkt_pool_, pkt_info_[i].packet_);
150 #endif // PACKET_TRACKING
151 
152  msg_hdr_[i].msg_len = 0;
153 
154  msg_hdr_[i].msg_hdr.msg_name =
155  static_cast<void*>(&(pkt_info_[i].src_addr_));
156  msg_hdr_[i].msg_hdr.msg_namelen = sizeof(pkt_info_[i].src_addr_);
157 
158  msg_hdr_[i].msg_hdr.msg_iov = &(pkt_info_[i].io_vec_);
159  msg_hdr_[i].msg_hdr.msg_iovlen = 1;
160 
161  msg_hdr_[i].msg_hdr.msg_control =
162  static_cast<void*>(pkt_info_[i].cmsg_buf_);
163  msg_hdr_[i].msg_hdr.msg_controllen = kCmsgSize;
164 
165  msg_hdr_[i].msg_hdr.msg_flags = 0;
166 
167  pkt_info_[i].io_vec_.iov_base = pkt_info_[i].packet_->GetBuffer();
168  pkt_info_[i].io_vec_.iov_len =
169  pkt_info_[i].packet_->GetMaxLengthInBytes();
170  }
171  }
172 
173  // If the monotonic clock to real time clock offset has already been
174  // computed, then return.
175  if (clock_init_)
176  {
177  return;
178  }
179 
180  // Compute the amount to adjust the real time clock values from the kernel
181  // to monotonic clock values. Retry until the delta value range is within a
182  // threshold.
183  int64_t range_nsec = (kTimeRangeThresholdNsec + 1);
184 
185  while (range_nsec > kTimeRangeThresholdNsec)
186  {
187  bool init = false;
188  int64_t pedestal_nsec = 0;
189  int64_t max_nsec = 0;
190  int64_t min_nsec = 0;
191  timespec cgt_mono = {0, 0};
192  timespec cgt_real = {0, 0};
193  double sum_delta = 0.0;
194 
195  for (size_t i = 0; i < kNumTimeSamples; i++)
196  {
197  timespec sleep_time = {0, 1000000};
198 
199  nanosleep(&sleep_time, NULL);
200 
201  clock_gettime(CLOCK_MONOTONIC, &cgt_mono);
202  clock_gettime(CLOCK_REALTIME, &cgt_real);
203 
204  int64_t cgt_mono_nsec = ((static_cast<int64_t>(cgt_mono.tv_sec) *
205  1000000000LL) +
206  static_cast<int64_t>(cgt_mono.tv_nsec));
207  int64_t cgt_real_nsec = ((static_cast<int64_t>(cgt_real.tv_sec) *
208  1000000000LL) +
209  static_cast<int64_t>(cgt_real.tv_nsec));
210 
211  int64_t delta_nsec = (cgt_real_nsec - cgt_mono_nsec);
212 
213  if (!init)
214  {
215  init = true;
216  pedestal_nsec = (delta_nsec - (delta_nsec % 1000000LL));
217  max_nsec = (delta_nsec - pedestal_nsec);
218  min_nsec = (delta_nsec - pedestal_nsec);
219  }
220 
221  delta_nsec -= pedestal_nsec;
222 
223  if (delta_nsec > max_nsec)
224  {
225  max_nsec = delta_nsec;
226  }
227 
228  if (delta_nsec < min_nsec)
229  {
230  min_nsec = delta_nsec;
231  }
232 
233  sum_delta += static_cast<double>(delta_nsec);
234  }
235 
236  double avg_delta = (sum_delta / static_cast<double>(kNumTimeSamples));
237  int64_t delta_nsec = (pedestal_nsec +
238  static_cast<int64_t>(avg_delta + 0.5));
239 
240  mono_to_real_.tv_sec = static_cast<time_t>(delta_nsec / 1000000000LL);
241  mono_to_real_.tv_nsec = static_cast<long>(delta_nsec % 1000000000LL);
242 
243  range_nsec = (max_nsec - min_nsec);
244  }
245 
246  clock_init_ = true;
247 }

250 bool PacketSet::PrepareForRecvMmsg()
251 {
252  if (max_size_ == 0)
253  {
254  LogE(kClassName, __func__, "Set not initialized.\n");
255  return false;
256  }
257 
258  // Only prepare the elements that are missing.
259  for (size_t i = 0; i < ret_idx_; ++i)
260  {
261  if (pkt_info_[i].packet_ == NULL)
262  {
263  pkt_info_[i].packet_ = pkt_pool_.Get();
264 
265  if (pkt_info_[i].packet_ == NULL)
266  {
267  LogF(kClassName, __func__, "Unable to get packet from pool.\n");
268  }
269 #ifdef PACKET_TRACKING
270  NEW_HELD_PKT_LOC(pkt_pool_, pkt_info_[i].packet_);
271 #endif // PACKET_TRACKING
272 
273  pkt_info_[i].io_vec_.iov_base = pkt_info_[i].packet_->GetBuffer();
274  pkt_info_[i].io_vec_.iov_len =
275  pkt_info_[i].packet_->GetMaxLengthInBytes();
276  }
277 
278  msg_hdr_[i].msg_hdr.msg_controllen = kCmsgSize;
279  }
280 
281  cur_size_ = 0;
282  ret_idx_ = 0;
283  walk_idx_ = 0;
284 
285  return true;
286 }

289 void PacketSet::FinalizeRecvMmsg(int packets_read, bool record_rcv_time)
290 {
291  if (packets_read <= 0)
292  {
293  cur_size_ = 0;
294  return;
295  }
296 
297  // Get the current time.
298  Time now = Time::Now();
299 
300  // Loop over the received messages.
301  for (int i = 0; i < packets_read; ++i)
302  {
303  // Get any receive timestamp that the kernel has set for the packet. See
304  // the man page cmsg(3) for details on the CMSG_*() macros that are used.
305  bool has_kernel_ts = false;
306  msghdr* msg = &(msg_hdr_[i].msg_hdr);
307  cmsghdr* cmsg = NULL;
308  timespec ts_rcv_real;
309 
310  for (cmsg = CMSG_FIRSTHDR(msg); cmsg != NULL;
311  cmsg = CMSG_NXTHDR(msg, cmsg))
312  {
313  if ((cmsg->cmsg_level == SOL_SOCKET) &&
314  (cmsg->cmsg_type == SO_TIMESTAMPNS))
315  {
316  has_kernel_ts = true;
317  memcpy(&ts_rcv_real, CMSG_DATA(cmsg), sizeof(ts_rcv_real));
318  break;
319  }
320  }
321 
322  // Set the received message length in the packet.
323  pkt_info_[i].packet_->SetLengthInBytes(msg_hdr_[i].msg_len);
324 
325  // Compute the packet's receive time.
326  if (has_kernel_ts)
327  {
328  // Adjust the kernel's real time clock value to a monotonic clock value
329  // using the offset computed in Initialize().
330  timespec ts_rcv_mono;
331 
332  if (ts_rcv_real.tv_nsec < mono_to_real_.tv_nsec)
333  {
334  ts_rcv_mono.tv_sec = (ts_rcv_real.tv_sec - mono_to_real_.tv_sec - 1);
335  ts_rcv_mono.tv_nsec = (ts_rcv_real.tv_nsec - mono_to_real_.tv_nsec +
336  1000000000L);
337  }
338  else
339  {
340  ts_rcv_mono.tv_sec = (ts_rcv_real.tv_sec - mono_to_real_.tv_sec);
341  ts_rcv_mono.tv_nsec = (ts_rcv_real.tv_nsec - mono_to_real_.tv_nsec);
342  }
343 
344  pkt_info_[i].rcv_time_ = Time(ts_rcv_mono);
345  }
346  else
347  {
348  // Simply use the current time from the monotonic clock.
349  pkt_info_[i].rcv_time_ = now;
350  }
351 
352  // Set the optional receive time in the packet.
353  if (record_rcv_time)
354  {
355  pkt_info_[i].packet_->set_recv_time(pkt_info_[i].rcv_time_);
356  }
357 
358  // Set the source address and port number in the IPv4 endpoint.
359  pkt_info_[i].src_endpt_.set_address(
360  pkt_info_[i].src_addr_.sin_addr.s_addr);
361  pkt_info_[i].src_endpt_.set_port(pkt_info_[i].src_addr_.sin_port);
362  }
363 
364  // Store the number of packets with data.
365  cur_size_ = packets_read;
366 }

369 bool PacketSet::GetNextPacket(Packet*& packet, Ipv4Endpoint& src_endpoint,
370  Time& rcv_time)
371 {
372  if (walk_idx_ > 0)
373  {
374  LogE(kClassName, __func__, "Cannot return packets while walking the "
375  "packet set.\n");
376  return false;
377  }
378 
379  if (ret_idx_ >= cur_size_)
380  {
381  return false;
382  }
383 
384  // Return the next packet and source address to the caller. The packet
385  // ownership is transferred to the caller.
386  packet = pkt_info_[ret_idx_].packet_;
387  src_endpoint = pkt_info_[ret_idx_].src_endpt_;
388  rcv_time = pkt_info_[ret_idx_].rcv_time_;
389  pkt_info_[ret_idx_].packet_ = NULL;
390  ++ret_idx_;
391 #ifdef PACKET_TRACKING
392  NEW_PKT_LOC(pkt_pool_, packet);
393 #endif // PACKET_TRACKING
394  return true;
395 }
