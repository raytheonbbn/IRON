221 int
222 main(int argc, char** argv)
223 {
224  TcpProxyOpts tcp_proxy_opts;
225  tcp_proxy_opts.ParseArgs(argc, argv);
226 
227  LogI(kClassName, __func__, "Starting TCP Proxy.\n");
228 
229  // Set the default logging level.
230  Log::SetDefaultLevel(tcp_proxy_opts.config_info().Get("Log.DefaultLevel",
231  "All", false));
232 
233  // Set the class level logging. The format for this configurable item is as
234  // follows:
235  //
236  // ClassName1=LogLevel1;ClassName2=LogLevel2;...;ClassNameN=LogLevelN
237  string class_levels = tcp_proxy_opts.config_info().Get("Log.ClassLevels",
238  "", false);
239 
240  List<string> tokens;
241  StringUtils::Tokenize(class_levels, ";", tokens);
242 
243  List<string>::WalkState tokens_ws;
244  tokens_ws.PrepareForWalk();
245 
246  string token;
247  while (tokens.GetNextItem(tokens_ws, token))
248  {
249  if (token.find("=") == string::npos)
250  {
251  // This entry is syntactically incorrect, so move on to the next entry.
252  LogW(kClassName, __func__, "Syntactically incorrect class log level: "
253  "%s\n", token.c_str());
254 
255  continue;
256  }
257 
258  List<string> token_values;
259  StringUtils::Tokenize(token, "=", token_values);
260 
261  string token_name;
262  string token_value;
263 
264  token_values.Pop(token_name);
265  token_values.Peek(token_value);
266 
267  LogI(kClassName, __func__, "Setting class %s logging level to %s.\n",
268  token_name.c_str(), token_value.c_str());
269 
270  Log::SetClassLevel(token_name, token_value);
271  }
272 
273  // Set the signal handlers for this process.
274  SetSigHandler();
275 
276  proxy_config = new TcpProxyConfig();
277 
278  // Load the gateway configuration information.
279  proxy_config->Initialize(tcp_proxy_opts.config_info());
280 
281  edge_if_config = new (std::nothrow) TcpEdgeIfConfig();
282  if (!edge_if_config)
283  {
284  LogF(kClassName, __func__, "Error creating new TcpEdgeIfConfig.\n");
285  return -1;
286  }
287 
288  if (!edge_if_config->Initialize(tcp_proxy_opts.config_info()))
289  {
290  LogE(kClassName, __func__, "Edge interface initialization failed. "
291  "Shutting down.\n");
292  CleanUp();
293  return -1;
294  }
295 
296  edge_if = new (std::nothrow) EdgeIf(*edge_if_config);
297  if (!edge_if)
298  {
299  LogF(kClassName, __func__, "Error creating new EdgeIf.\n");
300  return -1;
301  }
302 
303  packet_pool = new PacketPoolShm(PACKET_OWNER_TCP_PROXY);
304  if (!packet_pool->Attach(iron::kPacketPoolSemKey,
305  kPacketPoolShmName))
306  {
307  LogE(kClassName, __func__, "Cannot continue: error attaching to "
308  "shared memory for packet pool.\n");
309  CleanUp();
310  return -1;
311  }
312  LogI(kClassName, __func__, "Connected to shared memory for packet pool.\n");
313 
314  weight_qd_shared_memory = new SharedMemory();
315  if (weight_qd_shared_memory == NULL)
316  {
317  LogF(kClassName, __func__, "Error allocating new SharedMemory.\n");
318  CleanUp();
319  return -1;
320  }
321 
322  bin_map_shared_memory = new (std::nothrow) SharedMemory();
323  if (bin_map_shared_memory == NULL)
324  {
325  LogF(kClassName, __func__, "Error allocating new SharedMemory.\n");
326  CleanUp();
327  return -1;
328  }
329 
330  key_t w_key = tcp_proxy_opts.config_info().GetUint("Tcp.BinMap.SemKey",
331  iron::kDefaultBinMapSemKey);
332  string w_name = tcp_proxy_opts.config_info().Get("Tcp.BinMap.ShmName",
333  kDefaultBinMapShmName);
334 
335  LogI(kClassName, __func__, "Attaching bin map shared memory...\n");
336 
337  uint32_t wait_count = 0;
338 
339  while (true)
340  {
341  if (bin_map_shared_memory->Attach(w_key, w_name.c_str(), sizeof(BinMap)))
342  {
343  break;
344  }
345 
346  sleep(1);
347 
348  ++wait_count;
349  if (wait_count % 120 == 0)
350  {
351  LogW(kClassName, __func__, "... Waiting to attach to bin map shared "
352  "memory.\n");
353  }
354  else
355  {
356  LogD(kClassName, __func__, "... Waiting to attach.\n");
357  }
358  }
359 
360  bin_map = reinterpret_cast<BinMap*>(bin_map_shared_memory->GetShmPtr());
361 
362  bpf_to_tcp_pkt_fifo = new Fifo(kDefaultBpfToTcpPktFifoPath);
363  tcp_to_bpf_pkt_fifo = new Fifo(kDefaultTcpToBpfPktFifoPath);
364 
365  remote_control_server_ = new RemoteControlServer();
366 
367  // Start the TCP Proxy.
368  tcp_proxy = new TcpProxy(*proxy_config, *packet_pool, *edge_if,
369  *bin_map,
370  *weight_qd_shared_memory,
371  bpf_to_tcp_pkt_fifo,
372  tcp_to_bpf_pkt_fifo,
373  *remote_control_server_);
374 
375  tcp_proxy->Initialize(tcp_proxy_opts.config_info());
376  tcp_proxy->Start();
377 
378  CleanUp();
379 }
