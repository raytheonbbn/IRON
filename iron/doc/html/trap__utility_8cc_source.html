134 bool TrapUtility::Initialize(const ConfigInfo& ci)
135 {
136  m_val_ = ci.GetDouble("m", 0);
137  if (m_val_ == 0)
138  {
139  LogF(kClassName, __func__,
140  "m value not provided.\n");
141  return false;
142  }
143  b_val_ = ci.GetDouble("b", 0);
144  if (b_val_ == 0)
145  {
146  LogF(kClassName, __func__,
147  "b value not provided.\n");
148  return false;
149  }
150  p_val_ = ci.GetDouble("p", kDefaultPriority, false);
151  delta_ = ci.GetDouble("delta", kDefaultDelta, false);
152  // TODO: Explain magic numbers.
153  restart_interval_us_ =
154  ci.GetUint("resint", kDefaultResIntS, false) - p_val_ * 100000/2
155  + rng_.GetInt(500000);
156  n_steps_ = ci.GetInt("nsteps", kDefaultNSteps, false);
157  step_interval_us_ = ci.GetInt("stepint", kDefaultStepIntMs, false);
158 
159  avg_interval_usec_ = ci.GetInt("avgint", kDefaultAvgIntUsec, false);
160  avg_interval_usec_ += p_val_*30000 + rng_.GetInt(30000);
161  time_interval_end_ = Time::GetNowInUsec() + avg_interval_usec_;
162 
163  LogC(kClassName, __func__, "TRAP configuration :\n");
164  LogC(kClassName, __func__, "k : %.2e\n",
165  static_cast<double>(k_val_.GetValue()));
166  LogC(kClassName, __func__, "m : %.03e\n", m_val_);
167  LogC(kClassName, __func__, "b : %.03e\n", b_val_);
168  LogC(kClassName, __func__, "p : %.03f\n", p_val_);
169  LogC(kClassName, __func__, "delta : %.03f\n", delta_);
170  LogC(kClassName, __func__, "Interval length : %" PRIu64 "\n",
171  avg_interval_usec_);
172  LogC(kClassName, __func__, "Step duration : %" PRIu64 "\n",
173  step_interval_us_);
174  LogC(kClassName, __func__, "Number steps : %" PRIu8 "\n", n_steps_);
175  LogC(kClassName, __func__, "Restart interval : %" PRIu64 "\n",
176  restart_interval_us_);
177  LogC(kClassName, __func__, "TRAP configuration complete\n");
178 
179  LogI(kClassName, __func__, "TRAP initialized. Now %" PRId64 " , interval end: "
180  "%" PRId64 "\n", Time::GetNowInUsec(), time_interval_end_);
181  return true;
182 }

185 double TrapUtility::GetSendRate()
186 {
187  if (flow_state_ != FLOW_ON)
188  {
189  LogD(kClassName, __func__, "flow %" PRIu32 " is off\n", flow_id_);
190  return 0.0;
191  }
192 
193  // keep track of the queue length over the
194  Time now = Time::Now();
195  double send_rate = (static_cast<float>(curr_step_)/
196  static_cast<float>(n_steps_))*b_val_;
197 
198  if (!time_of_last_update_)
199  {
200  time_of_last_update_ = now.GetTimeInUsec();
201  }
202  else
203  {
204  if (last_step_size_ == 0)
205  {
206  // if the last step was down, increase the penalty proportional
207  // to the deviation from the top step.
208  penalty_ += (b_val_ - last_send_rate_)*
209  (now.GetTimeInUsec() - time_of_last_update_);
210  LogD(kClassName, __func__,
211  "Flow %" PRIu32 ", penalty now: %" PRIu64 ".\n", flow_id_, penalty_);
212  }
213  interval_length_ += now.GetTimeInUsec() - time_of_last_update_;
214  time_of_last_update_ = now.GetTimeInUsec();
215  }
216 
217  last_send_rate_ = send_rate;
218  LogD(kClassName, __func__, "Send rate: %f.\n", send_rate);
219 
220  return send_rate;
221 }

224 bool TrapUtility::ConsiderTriage()
225 {
226  // Check for early triage.
227  uint64_t max_penalty = b_val_ * (1 - delta_) *
228  (avg_interval_usec_ + inertia_usec_);
229  LogD(kClassName, __func__,
230  "Flow %" PRIu32 ": Curr penalty: %" PRIu64 ", max: %" PRIu64
231  "\n, delta: %f", flow_id_, penalty_, max_penalty, delta_);
232 
233  if (penalty_ > max_penalty)
234  {
235  // The flow is not being properly serviced, we get no utility.
236  LogA(kClassName, __func__, "Triage of Flow %" PRIu32 ", with inertia %u.\n",
237  flow_id_, inertia_usec_);
238 
239  flow_state_ = FLOW_TRIAGED;
240  current_utility_ = 0;
241  // Set last_step_size_ positive so when flow is turned on, penalty is not
242  // added.
243  last_step_size_ = 1;
244 
245  // timer_admit_control will turn the timers off and set the restart
246  // timer if we return true.
247  return true;
248  }
249  return false;
250 }

253 void TrapUtility::Step()
254 {
255  if (flow_state_ != FLOW_ON)
256  {
257  return;
258  }
259 
260  // Check if we should step up or step down if cost is greater than utility
261  // then step down.
262  uint32_t cur_qd = queue_depths_.GetBinDepthByIdx(bin_idx_);
263 
264  LogD(kClassName, __func__, "Checking Step for flow % " PRIu32
265  " cur step: % " PRIu8 " \n", flow_id_, curr_step_);
266 
267  // If the queues are large, then we should not be sending. We do this
268  // gradually by stepping down rather than stopping. If the queues are
269  // less tham the k*p/m threshold, then we should step up. The size of
270  // the step depends on the previous stepping action. If the previous
271  // step was down, we ramp up more slowly than if the previous step
272  // was up.
273  if (cur_qd * 8 > k_val_.GetValue() * p_val_ / m_val_)
274  {
275  curr_step_ = std::max(0,curr_step_-1);
276  last_step_size_ = 0;
277  LogD(kClassName, __func__, "Stepping down to %" PRIu8 "\n", curr_step_);
278  }
279  else if (last_step_size_ == 0)
280  {
281  last_step_size_ = 1;
282  curr_step_ =
283  std::min(static_cast<int>(n_steps_), curr_step_ + last_step_size_);
284  LogD(kClassName, __func__, "Stepping up to %" PRIu8 ".\n" ,curr_step_);
285  }
286  else
287  {
288  last_step_size_ =
289  std::min(last_step_size_*2, static_cast<int>(n_steps_));
290  curr_step_ =
291  std::min(static_cast<int>(n_steps_), curr_step_ + last_step_size_);
292  LogD(kClassName, __func__, "Stepping up to %" PRIu8 ".\n", curr_step_);
293  }
294  double UNUSED(send_rate) = (static_cast<float>(curr_step_)/
295  static_cast<float>(n_steps_))*b_val_;
296 
297  LogD(kClassName, __func__,
298  "At step %" PRIu8 ", for tag %" PRIu32 ", with queue %" PRIu8
299  "B, rate %f.\n", curr_step_, flow_id_, cur_qd, send_rate);
300 
301  // This output is for the netanim trace parser.
302  LogA(kClassName, __func__,
303  "f_id: %" PRIu32 ", queue: %" PRIu32 "b, rate: %.03fbps, "
304  "step:%" PRIu8 "\n", flow_id_, cur_qd*8, send_rate, curr_step_);
305 }

308 bool TrapUtility::CheckUtility()
309 {
310  if (flow_state_ != FLOW_ON)
311  {
312  LogD(kClassName, __func__, "flow %" PRIu32 " is off\n", flow_id_);
313  return false;
314  }
315 
316  Time now;
317  if (!now.GetNow())
318  {
319  LogF(kClassName, __func__, "Failed to get current time\n");
320  return false;
321  }
322 
323  if ((time_of_last_update_ != 0) && (last_step_size_ == 0))
324  {
325  // if the last step was down, increase the penalty proportional
326  // to the deviation from the top step.
327  penalty_ += (b_val_ - last_send_rate_)*
328  (now.GetTimeInUsec() - time_of_last_update_);
329  LogD(kClassName, __func__, "Flow %" PRIu32 ", penalty now: %" PRIu64 "\n",
330  flow_id_, penalty_);
331  }
332 
333  interval_length_ += now.GetTimeInUsec() - time_of_last_update_;
334 
335  // Check if the time interval has ended and if so, update the state.
336  uint64_t max_penalty = b_val_ * (1 - delta_) *
337  (avg_interval_usec_ + inertia_usec_);
338  if ((now.GetTimeInUsec() > time_interval_end_ && interval_length_) ||
339  (penalty_ > max_penalty))
340  {
341  LogD(kClassName, __func__, "Flow %" PRIu32 ", penalty now: %" PRIu64 "\n",
342  flow_id_, penalty_);
343 
344  if (ConsiderTriage())
345  {
346  return true;
347  }
348  else
349  {
350  // The flow is being properly serviced, we get full utility
351  current_utility_ = p_val_;
352  if (inertia_usec_ < kDefaultMaxInertia)
353  {
354  inertia_usec_ += kDefaultInertiaInc;
355  }
356  }
357 
358  // reset the counters
359  penalty_ = 0;
360  interval_length_ = 0;
361  time_interval_end_ = now.GetTimeInUsec() + avg_interval_usec_ +
362  inertia_usec_;
363  }
364 
365  time_of_last_update_ = now.GetTimeInUsec();
366 
367  return false;
368 }

371 void TrapUtility::SetFlowOn()
372 {
373  Time now = Time::Now();
374 
375  // If the flow is already on, we don't need to do anything.
376  if (flow_state_ == FLOW_ON)
377  {
378  LogW(kClassName, __func__, "Attempt to turn on flow %" PRIu32
379  " but it is already on.\n", flow_id_);
380  return;
381  }
382 
383  // Check that the flow has been off for at last the duration of the
384  // restart interval. The following if statement should never be true.
385  if ((now.GetTimeInUsec() - time_of_last_update_) < restart_interval_us_)
386  {
387  LogE(kClassName, __func__, "Attempt to turn flow on before duration"
388  " of restart interval. Flow has been off for %" PRId64 ", and "
389  " restart interval is %" PRId64 ".\n",
390  now.GetTimeInUsec() - time_of_last_update_, restart_interval_us_);
391  return;
392  }
393  LogD(kClassName, __func__, "Turning flow %" PRIu32 " on.\n", flow_id_);
394  flow_state_ = FLOW_ON;
395 
396  curr_step_ = 1;
397 
398  // Set the callback to resume checking utility at intervals.
399  interval_length_ = 0;
400  penalty_ = 0;
401  inertia_usec_ = 0;
402  time_of_last_update_ = now.GetTimeInUsec();
403  avg_interval_usec_ = std::min(static_cast<double>(avg_interval_usec_),
404  p_val_*100000/2 + rng_.GetInt(50000));
405  time_interval_end_ = Time::GetNowInUsec() + avg_interval_usec_;
406 }
