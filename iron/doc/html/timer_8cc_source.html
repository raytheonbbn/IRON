67 Timer::Timer()
68  : next_handle_(1), events_head_(NULL), events_tail_(NULL),
69  next_event_(NULL), pool_(NULL)
70 {
71  Time exp_time;
72 
73  // Add an initial number of timer elements to the pool.
74  for (size_t i = 0; i < kInitPoolSize; ++i)
75  {
76  TimerElem* te = new (std::nothrow) TimerElem(0, exp_time);
77 
78  if (te != NULL)
79  {
80  te->next = pool_;
81  pool_ = te;
82  }
83  }
84 }

102 bool Timer::StartTimer(const Time& delta_time, CallbackInterface* cb,
103  Handle& handle)
104 {
105  // Compute the absolute expiration time.
106  Time timeout;
107 
108  if (!timeout.GetNow())
109  {
110  LogF(kClassName, __func__, "Error getting current time.\n");
111  return false;
112  }
113 
114  timeout += delta_time;
115 
116  // Fetch a new element for the timer event.
117  TimerElem* new_te = NULL;
118 
119  if (pool_ == NULL)
120  {
121  new_te = new (std::nothrow) TimerElem(next_handle_, timeout);
122 
123  if (new_te == NULL)
124  {
125  LogF(kClassName, __func__, "Cannot allocate timer element.\n");
126  return false;
127  }
128  }
129  else
130  {
131  new_te = pool_;
132  pool_ = new_te->next;
133 
134  new_te->handle_id = next_handle_;
135  new_te->event_time = timeout;
136  }
137 
138  new_te->cb = cb->Clone();
139 
140  // Return the assigned handle for the timer.
141  handle.id_ = next_handle_;
142  handle.elem_ = new_te;
143 
144  // Increment the next handle to assign, avoiding zero.
145  ++next_handle_;
146 
147  if (next_handle_ == 0)
148  {
149  next_handle_ = 1;
150  }
151 
152  // Add the new timer element to the tail of the event list.
153  if (events_tail_ == NULL)
154  {
155  new_te->next = NULL;
156  new_te->prev = NULL;
157  events_head_ = new_te;
158  events_tail_ = new_te;
159  next_event_ = new_te;
160  }
161  else
162  {
163  events_tail_->next = new_te;
164  new_te->next = NULL;
165  new_te->prev = events_tail_;
166  events_tail_ = new_te;
167 
168  // If this timer event has an earlier expiration time than the current
169  // next timer event, then update the next timer event to this timer event.
170  if ((next_event_ != NULL) && (timeout < next_event_->event_time))
171  {
172  next_event_ = new_te;
173  }
174  }
175 
176  return true;
177 }

180 bool Timer::ModifyTimer(const Time& delta_time, Handle& handle)
181 {
182  bool rv = false;
183 
184  // Check if the timer event is still set.
185  if ((handle.id_ != 0) && (handle.elem_ != NULL) &&
186  (handle.id_ == handle.elem_->handle_id))
187  {
188  // Compute the new absolute expiration time.
189  Time timeout;
190 
191  if (!timeout.GetNow())
192  {
193  LogF(kClassName, __func__, "Error getting current time.\n");
194  return false;
195  }
196 
197  timeout += delta_time;
198 
199  // If this timer element is the current next timer event, and the
200  // expiration time is being pushed out, then invalidate the next timer
201  // event.
202  if ((handle.elem_ == next_event_) && (timeout > handle.elem_->event_time))
203  {
204  next_event_ = NULL;
205  }
206 
207  // Update the expiration time.
208  handle.elem_->event_time = timeout;
209 
210  rv = true;
211  }
212 
213  return rv;
214 }

217 bool Timer::CancelTimer(Handle& handle)
218 {
219  bool rv = false;
220 
221  // Check if the timer event is still set.
222  if ((handle.id_ != 0) && (handle.elem_ != NULL) &&
223  (handle.id_ == handle.elem_->handle_id))
224  {
225  TimerElem* te = handle.elem_;
226 
227  // If this timer element is the current next timer event, then invalidate
228  // the next timer event.
229  if (te == next_event_)
230  {
231  next_event_ = NULL;
232  }
233 
234  // Remove it from the list and add it to the pool.
235  if (te->next != NULL)
236  {
237  te->next->prev = te->prev;
238  }
239  if (te->prev != NULL)
240  {
241  te->prev->next = te->next;
242  }
243  if (te == events_head_)
244  {
245  events_head_ = te->next;
246  }
247  if (te == events_tail_)
248  {
249  events_tail_ = te->prev;
250  }
251 
252  te->handle_id = 0;
253  if (te->cb != NULL)
254  {
255  te->cb->ReleaseClone();
256  te->cb = NULL;
257  }
258  te->next = pool_;
259  te->prev = NULL;
260  pool_ = te;
261 
262  rv = true;
263  }
264 
265  // Invalidate the handle.
266  handle.id_ = 0;
267  handle.elem_ = NULL;
268 
269  return rv;
270 }

273 void Timer::CancelAllTimers()
274 {
275  // Navigate the list of events, removing each element and adding it to the
276  // pool.
277  while (events_head_ != NULL)
278  {
279  TimerElem* te = events_head_;
280  events_head_ = te->next;
281 
282  te->handle_id = 0;
283  if (te->cb != NULL)
284  {
285  te->cb->ReleaseClone();
286  te->cb = NULL;
287  }
288  te->next = pool_;
289  te->prev = NULL;
290  pool_ = te;
291  }
292 
293  events_tail_ = NULL;
294  next_event_ = NULL;
295 }

298 Time Timer::GetNextExpirationTime(const Time& max_wait)
299 {
300  // If the event list is empty, then return the maximum wait.
301  if (events_head_ == NULL)
302  {
303  return max_wait;
304  }
305 
306  // Get the current time.
307  Time now;
308 
309  if (!now.GetNow())
310  {
311  LogF(kClassName, __func__, "Error getting current time.\n");
312  return max_wait;
313  }
314 
315  // Get the next timer event wait time.
316  Time wait_time;
317 
318  if (next_event_ == NULL)
319  {
320  if (!FindNextEvent())
321  {
322  return max_wait;
323  }
324  }
325 
326  if (next_event_->event_time > now)
327  {
328  // The next expiration time has not been reached yet, so return the time
329  // difference from now until the event, limited to max_wait.
330  wait_time = Time::Min((next_event_->event_time - now), max_wait);
331  }
332  else
333  {
334  // The next expiration time has passed, so return a zero time difference.
335  Time time_difference = (now - events_head_->event_time);
336  Time limit(0, 1000);
337 
338  if (time_difference > limit)
339  {
340  LogW(kClassName, __func__,
341  "Timer handle %" PRIu32 " late by more than 1 ms! (diff %s)\n",
342  events_head_->handle_id, time_difference.ToString().c_str());
343  }
344  }
345 
346  return wait_time;
347 }

350 void Timer::DoCallbacks()
351 {
352  while (events_head_ != NULL)
353  {
354  // Get the current time.
355  Time now;
356 
357  if (!now.GetNow())
358  {
359  LogF(kClassName, __func__, "Error getting current time.\n");
360  break;
361  }
362 
363  // Get the next timer event.
364  if (next_event_ == NULL)
365  {
366  if (!FindNextEvent())
367  {
368  break;
369  }
370  }
371 
372  // Check if the next timer event has expired.
373  if (next_event_->event_time <= now)
374  {
375  // It has expired, so remove it from the event list.
376  TimerElem* te = next_event_;
377 
378  if (te->next != NULL)
379  {
380  te->next->prev = te->prev;
381  }
382  if (te->prev != NULL)
383  {
384  te->prev->next = te->next;
385  }
386  if (te == events_head_)
387  {
388  events_head_ = te->next;
389  }
390  if (te == events_tail_)
391  {
392  events_tail_ = te->prev;
393  }
394  next_event_ = NULL;
395 
396  // Invalidate any handles to this timer.
397  te->handle_id = 0;
398 
399  // Do the callback.
400  if (te->cb != NULL)
401  {
402  te->cb->PerformCallback();
403  te->cb->ReleaseClone();
404  te->cb = NULL;
405  }
406  else
407  {
408  LogF(kClassName, __func__, "Callback pointer is NULL.\n");
409  }
410 
411  // Add it to the pool, then allow another loop.
412  te->next = pool_;
413  te->prev = NULL;
414  pool_ = te;
415  }
416  else
417  {
418  // If the next timer event hasn't expired yet, then the other events are
419  // guaranteed to not be expired either.
420  break;
421  }
422  }
423 }

426 bool Timer::FindNextEvent()
427 {
428  // Search for the timer event that will expire next.
429  TimerElem* ne = events_head_;
430 
431  if (events_head_ != NULL)
432  {
433  for (TimerElem* te = events_head_->next; te != NULL; te = te->next)
434  {
435  if (te->event_time < ne->event_time)
436  {
437  ne = te;
438  }
439  }
440  }
441 
442  next_event_ = ne;
443 
444  return (ne != NULL);
445 }
