78 bool BinMap::Initialize(const ConfigInfo& config_info)
79 {
80  if (initialized_)
81  {
82  LogF(kClassName, __func__, "BinMap already initialized or BinMap memory "
83  "not zeroed before call.\n");
84  return false;
85  }
86 
87  // Assert that all assignable BinIndex values are less than the invalid
88  // BinIndex value. Note that 0 is always a valid BinIndex value.
89  if (((kDefaultDstBinIdxOffset + kMaxNumDsts - 1) >= kInvalidBinIndex) ||
90  ((kDefaultIntBinIdxOffset + kMaxNumIntNodes - 1) >= kInvalidBinIndex) ||
91  ((kDefaultMcastBinIdxOffset + kMaxNumMcastGroups - 1) >=
92  kInvalidBinIndex))
93  {
94  LogF(kClassName, __func__, "Error: The maximum BinIndex values (%" PRIu32
95  ",%" PRIu32 ",%" PRIu32 ") cannot interfere with kInvalidBinIndex "
96  "(%" PRIBinIndex ").\n",
97  static_cast<uint32_t>(kDefaultDstBinIdxOffset + kMaxNumDsts - 1),
98  static_cast<uint32_t>(kDefaultIntBinIdxOffset + kMaxNumIntNodes - 1),
99  static_cast<uint32_t>(kDefaultMcastBinIdxOffset + kMaxNumMcastGroups
100  - 1), kInvalidBinIndex);
101  }
102 
103  // Assert that kMaxBinId is small enough that we can represent that many
104  // values of BinId, starting at 0 and going to kMaxBinId. Also assert that
105  // kMaxBinId does not collide with kInvalidBinId.
106  if ((kMaxBinId > std::numeric_limits<BinId>::max()) ||
107  (kMaxBinId >= kInvalidBinId))
108  {
109  LogF(kClassName, __func__, "Error: kMaxBinId (%" PRIu32 ") must be "
110  "representable by BinId and must be less than kInvalidBinId (%"
111  PRIBinId ".\n", kMaxBinId, kInvalidBinId);
112  }
113 
114  // Assert that all unicast destination BinId values, from 0 to
115  // kMaxUcastBinId, are within the valid BinId range of 0 to kMaxBinId.
116  if (kMaxUcastBinId >= kMaxBinId)
117  {
118  LogF(kClassName, __func__, "Error: kMaxUcastBinId (%" PRIu32 ") must be "
119  "less than kMaxBinId (%" PRIu32 ").", kMaxUcastBinId, kMaxBinId);
120  }
121 
122  // Assert that all unicast destination BinId values, from 0 to
123  // kMaxUcastBinId, can be represented in the DstVec bit vector.
124  if (kMaxUcastBinId >= kDstVecBitsUsed)
125  {
126  LogF(kClassName, __func__, "Error: kMaxUcastBinId (%" PRIu32 ") must map "
127  "to a valid bit index in DstVec, in which we only use %" PRIu8
128  " bits.", kMaxUcastBinId, kDstVecBitsUsed);
129  }
130 
131  // Assert that kMaxMcastId is small enough that we can represent that many
132  // values of it, starting at 1 (since kInvalidMcastId is 0) and going all
133  // the way up to kMaxMcastId.
134  if (kMaxMcastId > std::numeric_limits<McastId>::max())
135  {
136  LogF(kClassName, __func__, "Error: kMaxMcastId (%" PRIu32 ") must be "
137  "representable by McastID.", kMaxMcastId);
138  }
139 
140  // Set the Bin Index starting offsets.
141  dst_info_.Initialize(kDefaultDstBinIdxOffset);
142  int_info_.Initialize(kDefaultIntBinIdxOffset);
143  mcast_info_.Initialize(kDefaultMcastBinIdxOffset);
144 
145  // Initialize the Bin ID to Bin Index mapping entries to the invalid Bin
146  // Index value. This marks all of the Bin IDs as "unused".
147  for (BinId i = 0; i <= kMaxBinId; ++i)
148  {
149  bin_id_to_idx_[i] = kInvalidBinIndex;
150  }
151 
152  // Determine if configured to SendGrams.
153  bool grams_enabled = config_info.GetBool("Bpf.SendGrams",
154  kDefaultSendGrams);
155 
156  // Extract the Unicast Destination (Edge Node) Bin ID information.
157  string dst_bin_ids_str = config_info.Get("BinMap.BinIds", "");
158 
159  if (dst_bin_ids_str.empty())
160  {
161  LogF(kClassName, __func__, "Error: No BinMap.BinIds value in BinMap "
162  "configuration file.\n");
163  return false;
164  }
165 
166  List<string> dst_bin_ids;
167 
168  StringUtils::Tokenize(dst_bin_ids_str, ",", dst_bin_ids);
169 
170  while (dst_bin_ids.size() > 0)
171  {
172  string dst_bin_id_str;
173 
174  dst_bin_ids.Pop(dst_bin_id_str);
175 
176  BinId dst_bin_id = static_cast<BinId>(
177  StringUtils::GetUint(dst_bin_id_str, kInvalidBinId));
178  BinIndex dst_bin_idx = kInvalidBinIndex;
179 
180  // Make sure that the Bin ID value is within the valid range.
181  if (dst_bin_id > kMaxUcastBinId)
182  {
183  LogF(kClassName, __func__, "Error: Configured unicast destination Bin "
184  "ID %" PRIBinId " exceeds the max Bin ID value (%" PRIu32 ").\n",
185  dst_bin_id, kMaxUcastBinId);
186  return false;
187  }
188 
189  // Make sure that the Bin ID value is not already being used.
190  if (bin_id_to_idx_[dst_bin_id] != kInvalidBinIndex)
191  {
192  LogF(kClassName, __func__, "Error: Configured unicast destination Bin "
193  "ID %" PRIBinId " already in use.\n", dst_bin_id);
194  return false;
195  }
196 
197  // Add the Bin ID to the Unicast Destination information.
198  if (!dst_info_.AddBinId(config_info, dst_bin_id_str, dst_bin_id,
199  dst_bin_idx))
200  {
201  LogF(kClassName, __func__, "Error: Unable to add the unicast "
202  "destination Bin ID %" PRIBinId ".\n", dst_bin_id);
203  return false;
204  }
205 
206  // Add the Bin ID to the Bin ID to Bin Index mapping.
207  bin_id_to_idx_[dst_bin_id] = dst_bin_idx;
208 
209  // If GRAMs are enabled, then add this destination to the static GRAM
210  // multicast group.
211  if (grams_enabled)
212  {
213  PrivAddDstToMcastGroup(kDefaultGramGrpAddr, dst_bin_idx, true, true);
214  }
215  }
216 
217  // Extract the Interior Node Bin ID information.
218  string int_node_bin_ids_str = config_info.Get("BinMap.IntBinIds",
219  "");
220  List<string> int_node_bin_ids;
221 
222  StringUtils::Tokenize(int_node_bin_ids_str, ",", int_node_bin_ids);
223 
224  while (int_node_bin_ids.size() > 0)
225  {
226  string int_node_bin_id_str;
227 
228  int_node_bin_ids.Pop(int_node_bin_id_str);
229 
230  BinId int_node_bin_id = static_cast<BinId>(
231  StringUtils::GetUint(int_node_bin_id_str, kInvalidBinId));
232  BinIndex int_node_bin_idx = kInvalidBinIndex;
233 
234  // Make sure that the Bin ID value is within the valid range.
235  if (int_node_bin_id > kMaxBinId)
236  {
237  LogF(kClassName, __func__, "Error: Configured interior node Bin ID %"
238  PRIBinId " exceeds the max Bin ID value (%" PRIu32 ").\n",
239  int_node_bin_id, kMaxBinId);
240  return false;
241  }
242 
243  // Make sure that the Bin ID value is not already being used.
244  if (bin_id_to_idx_[int_node_bin_id] != kInvalidBinIndex)
245  {
246  LogF(kClassName, __func__, "Error: Configured interior node Bin ID %"
247  PRIBinId " already in use.\n", int_node_bin_id);
248  return false;
249  }
250 
251  // Add the Bin ID to the Interior Node information.
252  if (!int_info_.AddBinId(int_node_bin_id, int_node_bin_idx))
253  {
254  LogF(kClassName, __func__, "Error: Unable to add the interior node Bin "
255  "ID %" PRIBinId ".\n", int_node_bin_id);
256  return false;
257  }
258 
259  // Add the Bin ID to the Bin ID to Bin Index mapping.
260  bin_id_to_idx_[int_node_bin_id] = int_node_bin_idx;
261  }
262 
263  // Extract the Multicast Group information. These are all static multicast
264  // groups.
265  uint32_t num_mcast_grps = config_info.GetUint("BinMap.NumMcastGroups", 0,
266  false);
267 
268  for (uint32_t j = 0; j < num_mcast_grps; ++j)
269  {
270  string config_prefix("BinMap.McastGroup.");
271  config_prefix.append(StringUtils::ToString(static_cast<int>(j)));
272 
273  // Add the static Multicast Group to the Multicast Group information.
274  if (!mcast_info_.CfgAddMcastGrp(*this, bin_id_to_idx_, config_info,
275  config_prefix, j))
276  {
277  LogF(kClassName, __func__, "Error: Unable to add the multicast group "
278  "at index %" PRIu32 ".\n", j);
279  return false;
280  }
281  }
282 
283  // Dump the configuration information to the log file.
284  LogC(kClassName, __func__, "Bin Map configuration:\n");
285 
286  // Log Unicast Destination Bin ID information.
287  BinId bid = 0;
288  BinIndex bidx = 0;
289 
290  for (bid = 0; bid <= kMaxUcastBinId; ++bid)
291  {
292  bidx = bin_id_to_idx_[bid];
293 
294  if ((bidx != kInvalidBinIndex) && (bidx >= dst_info_.offset_) &&
295  (bidx < (dst_info_.offset_ + dst_info_.num_)))
296  {
297  LogC(kClassName, __func__, "Bin ID %" PRIBinId " (Idx %" PRIBinIndex
298  "): %s\n", bid, bidx, dst_info_.ToString(bidx).c_str());
299  }
300  }
301 
302  // Log the Interior Node Bin ID information.
303  for (bid = 0; bid <= kMaxBinId; ++bid)
304  {
305  bidx = bin_id_to_idx_[bid];
306 
307  if ((bidx != kInvalidBinIndex) && (bidx >= int_info_.offset_) &&
308  (bidx < (int_info_.offset_ + int_info_.num_)))
309  {
310  LogC(kClassName, __func__, "Interior Node BinId %" PRIBinId " (Idx %"
311  PRIBinIndex ")\n", bid, bidx);
312  }
313  }
314 
315  // Log Multicast ID information.
316  for (size_t k = 0; k < mcast_info_.num_; ++k)
317  {
318  bidx = (mcast_info_.offset_ + static_cast<BinIndex>(k));
319 
320  LogC(kClassName, __func__, "Mcast Id %" PRIMcastId " (Idx %" PRIBinIndex
321  "): %s\n", mcast_info_.idx_to_mcast_id_[k], bidx,
322  mcast_info_.ToString(bidx).c_str());
323  }
324 
325  LogC(kClassName, __func__, "Bin Map configuration complete\n");
326 
327  initialized_ = true;
328 
329  return true;
330 }

633 BinIndex BinMap::GetDstBinIndexFromAddress(const Ipv4Address& ip_addr) const
634 {
635  if (ip_addr.IsMulticast())
636  {
637  // Look for an exact Multicast ID match.
638  McastId mcast_id = GetMcastIdFromAddress(ip_addr);
639 
640  for (size_t i = 0; i < mcast_info_.num_; ++i)
641  {
642  if (mcast_info_.idx_to_mcast_id_[i] == mcast_id)
643  {
644  return (mcast_info_.offset_ + i);
645  }
646  }
647  }
648  else
649  {
650  // Look for the IP address being within one of the unicast destination
651  // subnets.
652  for (size_t j = 0; j < dst_info_.num_; ++j)
653  {
654  for (size_t k = 0; k < dst_info_.ucast_dst_[j].num_subnets_; ++k)
655  {
656  if (dst_info_.ucast_dst_[j].subnet_[k].IsInSubnet(ip_addr))
657  {
658  return (dst_info_.offset_ + j);
659  }
660  }
661  }
662  }
663 
664  return kInvalidBinIndex;
665 }

680 BinIndex BinMap::GetMcastBinIndex(McastId mcast_id) const

916 //============================================================================
917 DstVec BinMap::DstVecSubtract(DstVec original, DstVec subtract)
918 {
919  // First, check if "subtract" is a subset of "original". xor (^) sets 1's
920  // where the two are different. If any of the xor result 1 values are also
921  // 1 in the "subtract" DstVec (i.e., xor_result & subtract), then "subtract"
922  // was not a subset of "original".
923  if (((original ^ subtract) & subtract) != 0)
924  {
925  LogF(kClassName, __func__, "Subtracting a DstVec 0x%X that is not a "
926  "subset of the original DstVec 0x%X\n", subtract, original);
927  }
928 
929  // Subtract by removing the 1's in "subtract" from "original".
930  return (original & (~subtract));
931 }
