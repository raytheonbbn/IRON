56 Packet* Rrm::CreateNewRrm(PacketPool& pkt_pool, iron::FourTuple& four_tuple)
57 {
58  // RRM packets have the following headers and format:
59  // 20B: IP header (no IP option).
60  // 8B: UDP header with the destination port set to special RRM port.
61  // 2B: Flow dest port.
62  // 2B: Padding.
63  // --- Report buffer starts here ---
64  // 8B: bytes sourced.
65  // 8B: bytes released.
66  // 4B: packets sourced.
67  // 4B: packets released.
68  // 4B: average loss rate.
69  Packet* rrm = pkt_pool.Get(PACKET_NOW_TIMESTAMP);
70  rrm->InitIpPacket();
71  struct iphdr* iphdr = rrm->GetIpHdr();
72 
73  if (!iphdr)
74  {
75  LogF(kClassName, __func__,
76  "Failed to get IP header in new packet.\n");
77  return NULL;
78  }
79 
80  iphdr->id = htons(pkt_pool.GetNextIpId());
81  iphdr->protocol = IPPROTO_UDP;
82  // In RRM, saddr is address of destination in flow four tuple, and vice versa.
83  iphdr->saddr = four_tuple.dst_addr_nbo();
84  iphdr->daddr = four_tuple.src_addr_nbo();
85 
86  rrm->SetIpDscp(DSCP_DEFAULT);
87  rrm->SetTimeToGo(
88  Time::FromUsec(static_cast<int64_t>(iron::kUnsetTimeToGo)));
89  // Length is 20B.
90  size_t length = rrm->GetLengthInBytes();
91 
92  length += sizeof(struct udphdr);
93  // Length is 28B.
94  rrm->SetLengthInBytes(length);
95 
96  rrm->SetSrcPort(four_tuple.src_port_nbo());
97  rrm->SetDstPort(htons(kDefaultRrmPort));
98 
99  uint8_t* buf = rrm->GetBuffer(rrm->GetIpPayloadOffset());
100  uint16_t dst_port_nbo = four_tuple.dst_port_nbo();
101 
102  // Length is expected to be 32B.
103  length += sizeof(uint16_t) + sizeof(uint16_t);
104 
105  if (length > kMaxPacketSizeBytes)
106  {
107  LogF(kClassName, __func__,
108  "RRM length %zd is larger than max packet size.\n",
109  length);
110  return NULL;
111  }
112 
113  memcpy(buf, &dst_port_nbo, sizeof(dst_port_nbo));
114 
115  buf += sizeof(dst_port_nbo);
116  memset(buf, 0, sizeof(uint16_t));
117 
118  rrm->SetLengthInBytes(length);
119  rrm->UpdateIpLen();
120 
121  // Don't need to do this because RRMs are internal to GNAT.
122  // rrm->UpdateChecksums();
123 
124  LogD(kClassName, __func__,
125  "Created RRM with length %zuB.\n", length);
126  rrm->DumpIpHdr();
127  return rrm;
128 }

131 void Rrm::FillReport(Packet* rrm,
132  uint64_t tot_bytes, uint32_t tot_pkts,
133  uint64_t rel_bytes, uint32_t rel_pkts,
134  uint32_t loss_rate)
135 {
136  size_t length = rrm->GetLengthInBytes();
137  length += sizeof(tot_bytes);
138  length += sizeof(rel_bytes);
139  length += sizeof(tot_pkts);
140  length += sizeof(rel_pkts);
141  length += sizeof(loss_rate);
142 
143  if (length > kMaxPacketSizeBytes)
144  {
145  LogF(kClassName, __func__,
146  "RRM Length exceeds max packet size.\n");
147  return;
148  }
149 
150  // 2B: dest port.
151  // 2B: Padding.
152  // --- Report buffer starts here ---
153  // 8B: bytes sourced.
154  // 8B: bytes released.
155  // 4B: packets sourced.
156  // 4B: packets released.
157  // 4B: average loss rate.
158  uint8_t* buf = iron::Rrm::GetReportBuffer(rrm);
159 
160  tot_bytes = htonl(tot_bytes);
161  memcpy(buf, &tot_bytes, sizeof(tot_bytes));
162  buf += sizeof(tot_bytes);
163 
164  rel_bytes = htonl(rel_bytes);
165  memcpy(buf, &rel_bytes, sizeof(rel_bytes));
166  buf += sizeof(rel_bytes);
167 
168  tot_pkts = htonl(tot_pkts);
169  memcpy(buf, &tot_pkts, sizeof(tot_pkts));
170  buf += sizeof(tot_pkts);
171 
172  rel_pkts = htonl(rel_pkts);
173  memcpy(buf, &rel_pkts, sizeof(rel_pkts));
174  buf += sizeof(rel_pkts);
175 
176  loss_rate = htonl(loss_rate);
177  memcpy(buf, &loss_rate, sizeof(loss_rate));
178  buf += sizeof(loss_rate);
179 
180  rrm->SetLengthInBytes(length);
181  rrm->UpdateIpLen();
182 
183  // Don't need to do this because RRMs are internal to GNAT.
184  // rrm->UpdateChecksums();
185 }

188 void Rrm::GetFlowFourTuple(Packet* rrm, iron::FourTuple& four_tuple)
189 {
190  uint32_t saddr_nbo;
191  uint32_t daddr_nbo;
192  uint16_t sport_nbo;
193  uint16_t dport_nbo;
194  uint32_t proto_nbo;
195 
196  rrm->GetFiveTuple(saddr_nbo, daddr_nbo, sport_nbo, dport_nbo, proto_nbo);
197 
198  if (ntohs(dport_nbo) != Rrm::kDefaultRrmPort)
199  {
200  LogF(kClassName, __func__,
201  "RRM packet does not have port destination %" PRIu16 ".\n",
202  kDefaultRrmPort);
203 
204  four_tuple.Set(0, 0, 0, 0);
205  return;
206  }
207 
208  dport_nbo = GetFlowDstPort(rrm);
209  four_tuple.Set(daddr_nbo, sport_nbo, saddr_nbo, dport_nbo);
210 }

213 void Rrm::GetReport(Packet* rrm,
214  uint64_t& tot_bytes, uint32_t& tot_pkts,
215  uint64_t& rel_bytes, uint32_t& rel_pkts,
216  uint32_t& loss_rate)
217 {
218  // 2B: dest port.
219  // 2B: Padding.
220  // --- Report buffer starts here ---
221  // 8B: bytes sourced.
222  // 8B: bytes released.
223  // 4B: packets sourced.
224  // 4B: packets released.
225  // 4B: average loss rate.
226 
227  size_t pkt_length = rrm->GetLengthInBytes();
228 
229  if (pkt_length < sizeof(struct iphdr) + sizeof(struct udphdr) +
230  sizeof(uint16_t) + sizeof(uint16_t) + sizeof(uint64_t) + sizeof(uint32_t) +
231  sizeof(uint64_t) + sizeof(uint32_t) + sizeof(uint32_t))
232 
233  {
234  LogF(kClassName, __func__,
235  "Packet of size %zd is too short.\n",
236  pkt_length);
237  return;
238  }
239 
240  uint8_t* buf = iron::Rrm::GetReportBuffer(rrm);
241 
242  memcpy(&tot_bytes, buf, sizeof(tot_bytes));
243  tot_bytes = ntohl(tot_bytes);
244  buf += sizeof(tot_bytes);
245 
246  memcpy(&rel_bytes, buf, sizeof(rel_bytes));
247  rel_bytes = ntohl(rel_bytes);
248  buf += sizeof(rel_bytes);
249 
250  memcpy(&tot_pkts, buf, sizeof(tot_pkts));
251  tot_pkts = ntohl(tot_pkts);
252  buf += sizeof(tot_pkts);
253 
254  memcpy(&rel_pkts, buf, sizeof(rel_pkts));
255  rel_pkts = ntohl(rel_pkts);
256  buf += sizeof(rel_pkts);
257 
258  memcpy(&loss_rate, buf, sizeof(loss_rate));
259  loss_rate = ntohl(loss_rate);
260  buf += sizeof(loss_rate);
261 }
