57  // The following Berkeley Packet Filter is meant to limit packets sent to the raw
58  // socket used by the UDP proxy so that the packets are:
59  // 1) any udp packets that are not being sent to the local interface and not marked
60  // with a TOS value of 4 (which we use to bypass IRON/GNAT for demos)
61  // and are not VXLAN packets carrying TCP traffic (these will be handled by the
63  // ((udp and ip[1] != 0x4 and not dst %s%s) and not
64  // (udp dst port 8472 and udp[39]==6))
65  // OR
66  // 2) any esp packets that are not being sent to the local interface address
67  // (esp and not dst %s)
68  // OR
69  // 3) any pim registration packets that are not being sent to the local interface
70  // address. pim registration packets encapsulate multicast traffic in order
71  // to deliver it (via unicast) to the pim rendezvous point. IRON/GNAT
72  // decapsulates these packets and handles the multicast traffic directly
73  // (pim and ip[20]&0xF==1 and not dst %s)";
74 
76  //
77  // %s : String representation of the LAN-facing interface identified by
78  // the configuration item InboundDevName.
79  // %s : Bypass string portion of BPF string.
80  // %s : String representation of the LAN-facing interface identified by
81  // the configuration item InboundDevName.
82  // %s : String representation of the LAN-facing interface identified by
83  // the configuration item InboundDevName.
84  const char* kBpfStr =
85  "((udp and ip[1] != 0x4 and not dst %s%s) and not (udp dst port 8472 and udp[39]==6)) or "
86  "(esp and not dst %s) or "
87  "(pim and ip[20]&0xF==1 and not dst %s)";
88

143 bool UdpEdgeIfConfig::Initialize(ConfigInfo& ci)
144 {
145  if (!EdgeIfConfig::Initialize(ci))
146  {
147  return false;
148  }
149 
150  // Finalize the BPF string.
151  bpf_str_ = StringUtils::FormatString(256, kBpfStr,
152  inbound_dev_ip_str_.c_str(),
153  bpf_bypass_str_.c_str(),
154  inbound_dev_ip_str_.c_str(),
155  inbound_dev_ip_str_.c_str());
156 
157  LogC(kClassName, __func__, "BPF string: %s\n", bpf_str_.c_str());
158 
159  // "Compile" the BPF string into the required micro-code program for the
160  // edge interface implementation.
161  InitializeBpf();
162 
163  // Populate the iptables add and delete rules lists.
164  iptables_add_rule_list_.Push(
165  StringUtils::FormatString(256, iptables_rule_spec1, iptables_cmd_.c_str(),
166  "A", inbound_dev_name_.c_str(),
167  inbound_dev_ip_str_.c_str()));
168  iptables_add_rule_list_.Push(
169  StringUtils::FormatString(256, iptables_rule_spec2, iptables_cmd_.c_str(),
170  "A", inbound_dev_name_.c_str(),
171  inbound_dev_ip_str_.c_str()));
172  iptables_add_rule_list_.Push(
173  StringUtils::FormatString(256, iptables_rule_spec3, iptables_cmd_.c_str(),
174  "A", inbound_dev_name_.c_str(),
175  inbound_dev_ip_str_.c_str()));
176 
177  iptables_del_rule_list_.Push(
178  StringUtils::FormatString(256, iptables_rule_spec1, iptables_cmd_.c_str(),
179  "D", inbound_dev_name_.c_str(),
180  inbound_dev_ip_str_.c_str()));
181  iptables_del_rule_list_.Push(
182  StringUtils::FormatString(256, iptables_rule_spec2, iptables_cmd_.c_str(),
183  "D", inbound_dev_name_.c_str(),
184  inbound_dev_ip_str_.c_str()));
185  iptables_del_rule_list_.Push(
186  StringUtils::FormatString(256, iptables_rule_spec3, iptables_cmd_.c_str(),
187  "D", inbound_dev_name_.c_str(),
188  inbound_dev_ip_str_.c_str()));
189 
190  return true;
191 }
