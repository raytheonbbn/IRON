213  bool operator==(const CongCtrl& cc)
214  {
215  return ((algorithm == cc.algorithm) &&
216  (((algorithm != TCP_CUBIC_BYTES_CC) &&
217  (algorithm != TCP_RENO_BYTES_CC)) ||
218  (cubic_reno_pacing == cc.cubic_reno_pacing)) &&
219  (((algorithm != COPA1_CONST_DELTA_CC) &&
220  (algorithm != COPA1_M_CC)) ||
221  (deterministic_copa == cc.deterministic_copa)) &&
222  ((algorithm != COPA1_CONST_DELTA_CC) ||
223  (static_cast<int>(copa_delta * 1000.0) ==
224  static_cast<int>(cc.copa_delta * 1000.0))) &&
225  ((algorithm != COPA_CC) ||
226  (static_cast<int>((copa_anti_jitter * 1000000.0) + 0.5) ==
227  static_cast<int>((cc.copa_anti_jitter * 1000000.0) +
228  0.5))) &&
229  ((algorithm != FIXED_RATE_TEST_CC) ||
230  (fixed_send_rate == cc.fixed_send_rate)));
231  }
