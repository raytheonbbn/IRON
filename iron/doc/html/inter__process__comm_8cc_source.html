81 bool InterProcessComm::Open(const string local_path)
82 {
83  static const char* UNUSED(mn) = "Open";
84 
85  // Validate the local pathname specified.
86  if ((local_path.size() < 1) ||
87  (local_path.size() >= sizeof(local_addr_.sun_path)))
88  {
89  LogE(CLASS_NAME, mn, "Error, invalid path: \"%s\"\n", local_path.c_str());
90  return false;
91  }
92 
93  // Close any existing endpoint.
94  if (socket_fd_ >= 0)
95  {
96  Close();
97  }
98 
99  // Create the UNIX domain socket using datagram packets.
100  if ((socket_fd_ = socket(AF_UNIX, SOCK_DGRAM, 0)) < 0)
101  {
102  LogE(CLASS_NAME, mn, "Error opening socket: %s\n", strerror(errno));
103  Close();
104  return false;
105  }
106 
107  // Set the socket's send buffer size. Note that the kernel doubles these
108  // values to allow for bookkeeping overhead.
109  int buf_size_bytes = (2 * 1024 * 1024); // 2 MB
110  socklen_t opt_size = sizeof(buf_size_bytes);
111 
112  if (setsockopt(socket_fd_, SOL_SOCKET, SO_SNDBUF, &buf_size_bytes,
113  opt_size) != 0)
114  {
115  LogE(CLASS_NAME, mn, "Error setting send buffer size to %d bytes: %s\n",
116  buf_size_bytes, strerror(errno));
117  Close();
118  return false;
119  }
120 
121  // Store the local pathname.
122  local_path_ = local_path;
123 
124  // Create the local address from the local pathname.
125  memset(&local_addr_, 0, sizeof(local_addr_));
126  local_addr_.sun_family = AF_UNIX;
127  // Note: the use of local_path instead of local_path_ in the following so
128  // that coverity does not complain that the length of local_path_ is not
129  // checked.
130  strcpy(local_addr_.sun_path, local_path.c_str());
131 
132  // Delete any old pathname file.
133  unlink(local_path_.c_str());
134 
135  // Bind the local address to the socket.
136  if (bind(socket_fd_,
137  reinterpret_cast<const struct sockaddr *>(&local_addr_),
138  sizeof(local_addr_)) < 0)
139  {
140  LogE(CLASS_NAME, mn, "Error binding socket: %s\n", strerror(errno));
141  Close();
142  return false;
143  }
144 
145  LogD(CLASS_NAME, mn, "Opened IPC socket %s.\n", local_path_.c_str());
146 
147  return true;
148 }

151 bool InterProcessComm::Connect(const string remote_path)
152 {
153  static const char* UNUSED(mn) = "Connect";
154 
155  // Validate the remote pathname specified.
156  if ((remote_path.size() < 1) ||
157  (remote_path.size() >= sizeof(remote_addr_.sun_path)))
158  {
159  LogE(CLASS_NAME, mn, "Error, invalid path: \"%s\"\n",
160  remote_path.c_str());
161  return false;
162  }
163 
164  // The socket must already be open.
165  if (socket_fd_ < 0)
166  {
167  LogE(CLASS_NAME, mn, "Error, socket not open for connect call.\n");
168  return false;
169  }
170 
171  // Store the remote pathname.
172  remote_path_ = remote_path;
173 
174  // Create the remote address from the remote pathname.
175  memset(&remote_addr_, 0, sizeof(remote_addr_));
176  remote_addr_.sun_family = AF_UNIX;
177  // Note: the use of remote_path instead of remote_path_ in the following so
178  // that coverity does not complain that the length of remote_path_ is not
179  // checked.
180  strcpy(remote_addr_.sun_path, remote_path.c_str());
181 
182  // The endpoint is now connnected.
183  is_connected_ = true;
184 
185  LogD(CLASS_NAME, mn, "Connected IPC socket %s to %s.\n", local_path_.c_str(),
186  remote_path_.c_str());
187 
188  return true;
189 }

206 bool InterProcessComm::SendMessage(uint8_t* buf, size_t len, bool blocking)
207 {
208  static const char* UNUSED(mn) = "SendMessage";
209 
210  // Validate the arguments.
211  if ((buf == NULL) || (len < 1))
212  {
213  LogE(CLASS_NAME, mn, "Error, invalid argument: buf=%p len=%zu.\n", buf,
214  len);
215  return false;
216  }
217 
218  // The socket must already be open and connected.
219  if ((socket_fd_ < 0) || (!is_connected_))
220  {
221  LogE(CLASS_NAME, mn, "Error, socket not open or connected for send "
222  "call.\n");
223  return false;
224  }
225 
226  // Perform the send.
227  ssize_t bytes_sent =
228  sendto(socket_fd_, static_cast<const void *>(buf), len,
229  (blocking ? 0 : MSG_DONTWAIT),
230  reinterpret_cast<const struct sockaddr *>(&remote_addr_),
231  static_cast<socklen_t>(sizeof(remote_addr_)));
232 
233  if (bytes_sent != static_cast<ssize_t>(len))
234  {
235  if (bytes_sent < 0)
236  {
237  if (!blocking && ((errno == EAGAIN) || (errno == EWOULDBLOCK)))
238  {
239  LogD(CLASS_NAME, mn, "Send would block on socket to %s.\n",
240  remote_path_.c_str());
241  }
242  else
243  {
244  LogE(CLASS_NAME, mn, "Error sending on socket: %s\n", strerror(errno));
245  }
246  }
247  else
248  {
249  LogE(CLASS_NAME, mn,
250  "Error, only sent %zd bytes of %zu byte message.\n", bytes_sent,
251  len);
252  }
253  return false;
254  }
255 
256  LogD(CLASS_NAME, mn, "SEND: IPC to %s, size: %zu bytes.\n", remote_path_.c_str(), len);
257 
258  return true;
259 }

262 size_t InterProcessComm::ReceiveMessage(uint8_t* buf, size_t max_len,
263  bool blocking)
264 {
265  static const char* UNUSED(mn) = "ReceiveMessage";
266 
267  // Validate the arguments.
268  if ((buf == NULL) || (max_len < 1))
269  {
270  LogE(CLASS_NAME, mn, "Error, invalid argument: buf=%p max_len=%zu.\n",
271  buf, max_len);
272  return static_cast<size_t>(0);
273  }
274 
275  // The socket must already be open.
276  if (socket_fd_ < 0)
277  {
278  LogE(CLASS_NAME, mn, "Error, socket not open for receive call.\n");
279  return static_cast<size_t>(0);
280  }
281 
282  // Perform the receive.
283  sockaddr_un src_addr;
284  socklen_t src_addr_len = sizeof(src_addr);
285  ssize_t bytes_received =
286  recvfrom(socket_fd_, static_cast<void *>(buf), max_len,
287  (blocking ? 0 : MSG_DONTWAIT),
288  reinterpret_cast<struct sockaddr *>(&src_addr), &src_addr_len);
289 
290  if (bytes_received < 0)
291  {
292  if (!blocking && ((errno == EAGAIN) || (errno == EWOULDBLOCK)))
293  {
294  LogD(CLASS_NAME, mn, "No messages to receive on socket from %s.\n",
295  remote_path_.c_str());
296  }
297  else
298  {
299  LogE(CLASS_NAME, mn, "Error receiving on socket: %s\n",
300  strerror(errno));
301  }
302  return static_cast<size_t>(0);
303  }
304 
305  LogD(CLASS_NAME, mn, "RECV: IPC from %s, size: %zd bytes.\n",
306  src_addr.sun_path, bytes_received);
307 
308  return static_cast<size_t>(bytes_received);
309 }

312 size_t InterProcessComm::ReceiveMessages(PacketSet& packet_set,
313  bool blocking,
314  int num_pkts)
315 {
316  // Prepare for the recvmmsg call. To do so, we prep the PacketSet to be the
317  // destination of the data that gets read from the socket.
318  if (!packet_set.PrepareForRecvMmsg())
319  {
320  LogE(CLASS_NAME, __func__, "Error preparing PacketSet for reading "
321  "packets.\n");
322  return static_cast<size_t>(0);
323  }
324 
325  int packets_read = recvmmsg(socket_fd_, packet_set.GetVecPtr(),
326  packet_set.GetVecLen(),
327  (blocking ? 0 : MSG_DONTWAIT), NULL);
328 
329  if (packets_read <= 0)
330  {
331  LogE(CLASS_NAME, __func__, "recvmmsg error: %s\n", strerror(errno));
332  return static_cast<size_t>(0);
333  }
334 
335  packet_set.FinalizeRecvMmsg(packets_read, true);
336 
337  return static_cast<size_t>(packets_read);
338 }
