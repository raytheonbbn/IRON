125 bool FlogUtility::Initialize(const ConfigInfo& ci)
126 {
127  a_val_ = ci.GetDouble("a", 0, false);
128  if (a_val_ == 0)
129  {
130  LogF(kClassName, __func__, "fid: %" PRIu32 ", a value not provided.\n",
131  flow_id_);
132  return false;
133  }
134 
135  Time now = Time::Now();
136  p_val_ = ci.GetDouble("p", kDefaultPriority, false);
137  scale_factor_ = ((2*p_val_) + 10)/(p_val_ + 10);
138  int_length_sec_ = (ci.GetDouble("avgint", kDefaultIntSec, false))*scale_factor_;
139  min_rate_bps_ = (ci.GetDouble("f", kDefaultMinRateBps, false));
140  time_interval_end_usec_ = now.GetTimeInUsec() + int_length_sec_*1000000;
141 
142  LogC(kClassName, __func__, "FLOG configuration :\n");
143  LogC(kClassName, __func__, "a : %.03f\n", a_val_);
144  LogC(kClassName, __func__, "k : %.2e\n",
145  static_cast<double>(k_val_.GetValue()));
146  LogC(kClassName, __func__, "p : %.03f\n", p_val_);
147  LogC(kClassName, __func__, "min acceptable rate : %.03f\n", min_rate_bps_);
148  LogC(kClassName, __func__, "scale factor : %.03f\n", scale_factor_);
149  LogC(kClassName, __func__, "Interval length : %.03f\n",
150  int_length_sec_);
151  LogC(kClassName, __func__, "Min admission rate : %.03f\n",
152  min_rate_bps_);
153  LogC(kClassName, __func__, "FLOG configuration complete\n");
154 
155  LogI(kClassName, __func__, "FLOG initialized. Now %" PRId64 " , interval end: "
156  "%" PRId64 "\n", Time::GetNowInUsec(), time_interval_end_usec_);
157  return true;
158 }

161 double FlogUtility::GetSendRate()
162 {
163  if (flow_state_ != FLOW_ON)
164  {
165  LogD(kClassName, __func__, "fid: %" PRIu32 ", is off\n", flow_id_);
166  return 0.0;
167  }
168 
169  double send_rate = 0.0;
170  double queue_depth_bits = queue_depths_.GetBinDepthByIdx(bin_idx_)*8;
171 
172  if (queue_depth_bits >= k_val_.GetValue() * p_val_ * a_val_)
173  {
174  LogD(kClassName, __func__, "fid: %" PRIu32 ", queue is too large, not "
175  "sending.\n", flow_id_);
176  send_rate = 0.0;
177  }
178  else if (queue_depth_bits == 0.0)
179  {
180  send_rate = m_val_;
181  }
182  else
183  {
184  send_rate = min((a_val_ * k_val_.GetValue() * p_val_ - queue_depth_bits) /
185  (a_val_ * queue_depth_bits), m_val_);
186  }
187 
188  LogA(kClassName, __func__, "f_id: %" PRIu32 ", queue: %.03fb, rate: "
189  "%.03fbps.\n", flow_id_, queue_depth_bits, send_rate);
190 
191  if (avg_adm_rate_bps_ == 0.0)
192  {
193  avg_adm_rate_bps_ = send_rate;
194  }
195  else
196  {
197  avg_adm_rate_bps_ = send_rate*0.2 + avg_adm_rate_bps_*0.8;
198  LogD(kClassName, __func__, "Avg adm rate is %f\n", avg_adm_rate_bps_);
199  }
200 
201  return send_rate;
202 }

205 bool FlogUtility::ConsiderTriage()
206 {
207  double current_backlog = src_info_.cur_backlog_bytes() * 8;
208  double current_queue_lat = 0;
209 
210  // Incur penalty if the backlog is above a threshold and the admission
211  // rate is not growing.
212  if (avg_adm_rate_bps_ > 0)
213  {
214  current_queue_lat = current_backlog/avg_adm_rate_bps_;
215  LogD(kClassName, __func__,
216  "f_id: %" PRIu32 "Backlog: %f, current admission rate: %f\n",
217  flow_id_, current_backlog, avg_adm_rate_bps_);
218 
219  if ((current_queue_lat > kDefaultMaxLatencySec) &&
220  (avg_adm_rate_bps_ < prev_adm_rate_*1.1))
221  {
222  ++size_penalty_;
223  LogE(kClassName, __func__, "f_id: %" PRIu32 "Backlog is too large: %f, "
224  "given current admission rate: %f .\n",
225  flow_id_, current_backlog, avg_adm_rate_bps_);
226  }
227  else
228  {
229  size_penalty_ = 0;
230  }
231 
232  // Incur penalty if the backlog is above a threshold and is increasing.
233  if ((current_queue_lat > (kDefaultMaxLatencySec)) &&
234  (current_backlog > prev_backlog_))
235  {
236  ++growth_penalty_;
237  LogD(kClassName, __func__, "f_id: %" PRIu32 "Backlog is growing: %f, "
238  "given current admission rate: %f\n",
239  flow_id_, current_backlog, avg_adm_rate_bps_);
240  }
241  else
242  {
243  growth_penalty_ = 0;
244  }
245 
246  // Incur penalty if the admission rate is less that the minimum acceptable rate.
247  if (avg_adm_rate_bps_ < min_rate_bps_)
248  {
249  ++rate_penalty_;
250  LogD(kClassName, __func__, "f_id: %" PRIu32 "Low admission rate: %f, "
251  "given current admission rate: %f.",
252  flow_id_, avg_adm_rate_bps_, min_rate_bps_);
253  }
254  else
255  {
256  rate_penalty_ = 0;
257  }
258  }
259 
260  prev_backlog_ = current_backlog;
261  prev_adm_rate_ = avg_adm_rate_bps_;
262  if ((size_penalty_ > kMaxPenalty) || (growth_penalty_ > kMaxPenalty ) ||
263  (rate_penalty_ > kMaxPenalty ))
264  {
265  flow_state_ = FLOW_TRIAGED;
266  LogD(kClassName, __func__, "Triage of flow: %" PRIu32 "\n", flow_id_);
267  return true;
268  }
269 
270  return false;
271 }

274 void FlogUtility::SetFlowOn()
275 {
276  Time now = Time::Now();
277 
278  // If the flow is already on, we don't need to do anything.
279  if (flow_state_ == FLOW_ON)
280  {
281  LogW(kClassName, __func__, "Attempt to turn on flow %" PRIu32
282  " but it is already on.\n", flow_id_);
283  return;
284  }
285 
286  LogD(kClassName, __func__, "Turning flow %" PRIu32 " ON.\n", flow_id_);
287  flow_state_ = FLOW_ON;
288 
289  time_interval_end_usec_ = now.GetTimeInUsec() + int_length_sec_*1000000;
290 }

293 double FlogUtility::ComputeUtility(double send_rate)
294 {
295  if (a_val_ * send_rate <= -1)
296  {
297  LogF(kClassName, __func__, "fid: %" PRIu32 ", Error: Cannot take log "
298  "of negative value a*r + 1 = %.3f.\n", flow_id_,
299  a_val_ * send_rate + 1);
300  return 0;
301  }
302 
303  // utility = p * log(ar + 1)
304  return p_val_ * log(a_val_ * send_rate + 1);
305 }
