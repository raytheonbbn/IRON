93 GenXplot::~GenXplot()
94 {
95  if (!output_fd_)
96  {
97  return;
98  }
99  uint64_t next_y = max_y_
100  + (20 * line_key_entries_.size())
101  + (20 * x_key_entries_.size());
102  std::map<std::string,XPLOT_COLOR>::iterator it = line_key_entries_.begin();
103  double new_min_x = 0;
104  // If we have no entries at all, just start at x=0.
105  if (min_x_ == std::numeric_limits<int64_t>::max())
106  {
107  min_x_ = 0;
108  }
109  else
110  {
111  min_x_ = static_cast<double>(min_x_) / 1e6;
112  }
113 
114  while (it != line_key_entries_.end())
115  {
116  if (convert_x_to_sec_)
117  {
118  fprintf(output_fd_, "rtext %f %" PRId64 " %" PRIu8 "\n",
119  new_min_x, next_y, it->second);
120  }
121  else
122  {
123  fprintf(output_fd_, "rtext %" PRId64 " %" PRId64 " %" PRIu8 "\n",
124  min_x_, next_y, it->second);
125  }
126  fprintf(output_fd_, "--- %s\n", it->first.c_str());
127  ++it;
128  next_y += 20;
129  }
130 
131  std::map<std::string,XPLOT_COLOR>::iterator it2 = x_key_entries_.begin();
132  while (it2 != x_key_entries_.end())
133  {
134  if (convert_x_to_sec_)
135  {
136  fprintf(output_fd_, "rtext %f %" PRId64 " %" PRIu8 "\n",
137  new_min_x, next_y, it2->second);
138  }
139  else
140  {
141  fprintf(output_fd_, "rtext %" PRId64 " %" PRId64 " %" PRIu8 "\n",
142  min_x_, next_y, it2->second);
143  }
144  fprintf(output_fd_, "x %s\n", it2->first.c_str());
145  ++it2;
146  next_y += 20;
147  }
148 
149  //
150  // If the current output file descriptor is not equal to stdout or stderr,
151  // then we must close it without disrupting users of output_fd_.
152  //
153 
154  if ((output_fd_ != stdout) && (output_fd_ != stderr))
155  {
156  FILE* old_fd = output_fd_;
157  output_fd_ = stdout;
158 
159  fflush(old_fd);
160  fclose(old_fd);
161  }
162 
163  fflush(output_fd_);
164 }

167 bool GenXplot::Initialize(const std::string& file_name,
168  const std::string& graph_name,
169  bool convert_x_to_sec)
170 {
171 #ifdef XPLOT
172  convert_x_to_sec_ = convert_x_to_sec;
173 
174  // Write the file to the same directory where we're logging.
175  std::string logfile = Log::GetOutputFileName();
176  size_t pos = logfile.find_last_of('/');
177  std::string newfile =
178  logfile.substr(0, pos + 1).append(file_name);
179  FILE* new_fd = fopen(newfile.c_str(), "w");
180 
181  if (new_fd == NULL)
182  {
183  LogE(kClassName, __func__, "Error opening xplot file %s.\n",
184  newfile.c_str());
185  return false;
186  }
187 
188  // The following lines allow re-initialization to change where the graph
189  // prints.
190  FILE* old_fd = output_fd_;
191  output_fd_ = new_fd;
192 
193  if (old_fd && (old_fd != stdout) && (old_fd != stderr))
194  {
195  fflush(old_fd);
196  fclose(old_fd);
197  }
198 
199  if (convert_x_to_sec_)
200  {
201  fprintf(output_fd_, "double double\n");
202  }
203  else
204  {
205  fprintf(output_fd_, "signed signed\n");
206  }
207  fprintf(output_fd_, "title\n%s\n", graph_name.c_str());
208 #endif // XPLOT
209  return true;
210 }

214  int64_t y1,

217  XPLOT_COLOR color)

223  }
224  if (x2 < min_x_)
225  {
226  min_x_ = x2;
227  }
228  if (y1 > max_y_)
229  {
230  max_y_ = y1;
231  }
232  if (y2 > max_y_)
233  {
234  max_y_ = y2;
235  }
236  if (convert_x_to_sec_)
237  {
238  double new_x1 = static_cast<double>(x1) / 1e6;
239  double new_x2 = static_cast<double>(x2) / 1e6;
240  fprintf(output_fd_,
241  "line %f %" PRId64 " %f %" PRId64 " %" PRIu8 "\n",
242  new_x1, y1, new_x2, y2,
243  static_cast<XPLOT_COLOR>(color % NUM_COLORS));
244  }
245  else
246  {
247  fprintf(output_fd_,
248  "line %" PRId64 " %" PRId64 " %" PRId64 " %" PRId64 " %" PRIu8
249  "\n", x1, y1, x2, y2,
250  static_cast<XPLOT_COLOR>(color % NUM_COLORS));
251 
252  }
253 #endif // XPLOT
254 }

318 void GenXplot::ContinueLine(uint8_t line_index,
319  int64_t new_x,
320  int64_t new_y,
321  XPLOT_COLOR color)
322 {
323 #ifdef XPLOT
324  if (line_index > 15)
325  {
326  return;
327  }
328  if (new_x < min_x_)
329  {
330  min_x_ = new_x;
331  }
332  if (new_y > max_y_)
333  {
334  max_y_ = new_y;
335  }
336  if (line_started_[line_index])
337  {
338  DrawLine(line_previous_x_[line_index],
339  line_previous_y_[line_index],
340  new_x,
341  new_y,
342  static_cast<XPLOT_COLOR>(color % NUM_COLORS));
343  }
344  line_started_[line_index] = true;
345  line_previous_x_[line_index] = new_x;
346  line_previous_y_[line_index] = new_y;
347 #endif // XPLOT
348 }
