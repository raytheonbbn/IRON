132 size_t CubicBytesHelper::CongestionWindowAfterPacketLoss(size_t cur_cwnd)
133 {
134 #ifdef SLIQ_CC_DEBUG
135  LogD(kClassName, __func__, "Conn %" PRIEndptId ": Args cwnd %zu #conn %d\n",
136  conn_id_, cur_cwnd, num_streams_);
137 #endif
138 
139  if (cur_cwnd < last_max_cwnd_)
140  {
141  // We never reached the old max, so assume we are competing with another
142  // flow. Use our extra back off factor to allow the other flow to go up.
143  last_max_cwnd_ = static_cast<size_t>(kBetaLastMax * cur_cwnd);
144 
145 #ifdef SLIQ_CC_DEBUG
146  LogD(kClassName, __func__, "Conn %" PRIEndptId ": Never reached old cwnd "
147  "max, set last_max_cwnd_ %zu\n", conn_id_, last_max_cwnd_);
148 #endif
149  }
150  else
151  {
152  last_max_cwnd_ = cur_cwnd;
153 
154 #ifdef SLIQ_CC_DEBUG
155  LogD(kClassName, __func__, "Conn %" PRIEndptId ": Reached old cwnd max, "
156  "set last_max_cwnd_ %zu\n", conn_id_, last_max_cwnd_);
157 #endif
158  }
159 
160  // Reset the cycle start time.
161  epoch_.Zero();
162 
163  size_t target_cwnd = static_cast<size_t>(cur_cwnd * Beta());
164 
165 #ifdef SLIQ_CC_DEBUG
166  LogD(kClassName, __func__, "Conn %" PRIEndptId ": New cwnd %zu.\n",
167  conn_id_, target_cwnd);
168 #endif
169 
170  return target_cwnd;
171 }

174 size_t CubicBytesHelper::CongestionWindowAfterAck(
175  size_t acked_bytes, size_t cur_cwnd, const Time& delay_min, const Time& now)
176 {
177 #ifdef SLIQ_CC_DEBUG
178  LogD(kClassName, __func__, "Conn %" PRIEndptId ": Args acked_bytes %zu "
179  "cwnd %zu min_delay %s #conn %d\n", conn_id_, acked_bytes, cur_cwnd,
180  delay_min.ToString().c_str(), num_streams_);
181 #endif
182 
183  acked_bytes_count_ += acked_bytes;
184 
185 #ifdef SLIQ_CC_DEBUG
186  LogD(kClassName, __func__, "Conn %" PRIEndptId ": Update "
187  "acked_bytes_count_ %zu\n", conn_id_, acked_bytes_count_);
188 #endif
189 
190  // Cubic is "independent" of RTT, the update is limited by the time elapsed.
191  if ((last_cwnd_ == cur_cwnd) &&
192  (now.Subtract(last_update_time_) <= MaxCubicTimeInterval()))
193  {
194  size_t rv = ((last_target_cwnd_ > estimated_tcp_cwnd_) ?
195  last_target_cwnd_ : estimated_tcp_cwnd_);
196 
197 #ifdef SLIQ_CC_DEBUG
198  LogD(kClassName, __func__, "Conn %" PRIEndptId ": No cwnd change and not "
199  "enough elapsed time, new cwnd %zu\n", conn_id_, rv);
200 #endif
201 
202  return rv;
203  }
204 
205  last_cwnd_ = cur_cwnd;
206  last_update_time_ = now;
207 
208  if (epoch_.IsZero())
209  {
210  // First ACK after a loss event.
211 #ifdef SLIQ_CC_DEBUG
212  LogD(kClassName, __func__, "Conn %" PRIEndptId ": Start of cubic "
213  "epoch.\n", conn_id_);
214 #endif
215 
216  epoch_ = now; // Start of epoch.
217  acked_bytes_count_ = acked_bytes; // Reset count.
218 
219  // Reset estimated_tcp_cwnd_ to be in sync with cubic.
220  estimated_tcp_cwnd_ = cur_cwnd;
221 
222  if (last_max_cwnd_ <= cur_cwnd)
223  {
224  time_to_origin_point_ = 0;
225  origin_point_cwnd_ = cur_cwnd;
226 
227 #ifdef SLIQ_CC_DEBUG
228  LogD(kClassName, __func__, "Conn %" PRIEndptId ": Last cwnd <= "
229  "current cwnd.\n", conn_id_);
230 #endif
231  }
232  else
233  {
234  time_to_origin_point_ = static_cast<uint32_t>(
235  cbrt(kCubeFactor * (last_max_cwnd_ - cur_cwnd)));
236  origin_point_cwnd_ = last_max_cwnd_;
237 
238 #ifdef SLIQ_CC_DEBUG
239  LogD(kClassName, __func__, "Conn %" PRIEndptId ": Last cwnd > "
240  "current cwnd.\n", conn_id_);
241 #endif
242  }
243  }
244 
245  // Change the time unit from microseconds to 2^10 fractions per second.
246  // Take the round trip time in account. This is done to allow us to use
247  // shift as a divide operator.
248  int64_t elapsed_time =
249  ((now.Add(delay_min).Subtract(epoch_).GetTimeInUsec() << 10) /
250  kNumMicrosPerSecond);
251 
252  int64_t offset = (time_to_origin_point_ - elapsed_time);
253 
254  size_t delta_cwnd =
255  (((kCubeCongestionWindowScale * offset * offset * offset) >> kCubeScale) *
256  kDefaultTcpMss);
257 
258  size_t target_cwnd = (origin_point_cwnd_ - delta_cwnd);
259 
260 #ifdef SLIQ_CC_DEBUG
261  LogD(kClassName, __func__, "Conn %" PRIEndptId ": Computation, "
262  "elapsed_time %" PRId64 " offset %" PRId64 " delta_cwnd %zu "
263  "target_cwnd %zu\n", conn_id_, elapsed_time, offset, delta_cwnd,
264  target_cwnd);
265 #endif
266 
267  // Increase the window by Alpha * 1 MSS of bytes every time we ACK an
268  // estimated TCP window of bytes.
269  if (estimated_tcp_cwnd_ > 0)
270  {
271  estimated_tcp_cwnd_ += ((acked_bytes_count_ *
272  (Alpha() * kDefaultTcpMss)) /
273  estimated_tcp_cwnd_);
274 
275 #ifdef SLIQ_CC_DEBUG
276  LogD(kClassName, __func__, "Conn %" PRIEndptId ": Computed "
277  "estimated_tcp_cwnd_ %zu\n", conn_id_, estimated_tcp_cwnd_);
278 #endif
279  }
280  else
281  {
282  LogW(kClassName, __func__, "Conn %" PRIEndptId ": Warning, "
283  "estimated_tcp_cwnd_ = 0.\n", conn_id_);
284  }
285 
286  acked_bytes_count_ = 0;
287 
288 #ifdef SLIQ_CC_DEBUG
289  LogD(kClassName, __func__, "Conn %" PRIEndptId ": Resulting "
290  "acked_bytes_count_ %zu estimated_tcp_cwnd_ %zu.\n", conn_id_,
291  acked_bytes_count_, estimated_tcp_cwnd_);
292 #endif
293 
294  // We have a new cubic congestion window.
295  last_target_cwnd_ = target_cwnd;
296 
297  // Compute target cwnd based on cubic target and estimated TCP
298  // cwnd, use highest (fastest).
299  if (target_cwnd < estimated_tcp_cwnd_)
300  {
301  target_cwnd = estimated_tcp_cwnd_;
302  }
303 
304 #ifdef SLIQ_CC_DEBUG
305  LogD(kClassName, __func__, "Conn %" PRIEndptId ": New cubic cwnd (target) "
306  "%zu.\n", conn_id_, target_cwnd);
307 #endif
308 
309  return target_cwnd;
310 }

313 double CubicBytesHelper::Alpha() const
314 {
315  // TCPFriendly alpha is described in Section 3.3 of the CUBIC paper. Note
316  // that beta here is a cwnd multiplier, and is equal to 1-beta from the
317  // paper. We derive the equivalent alpha for an N-connection emulation as:
318  const double beta = Beta();
319 
320  double alpha = ((3.0 * num_streams_ * num_streams_ * (1.0 - beta)) /
321  (1.0 + beta));
322 
323 #ifdef SLIQ_CC_DEBUG
324  LogD(kClassName, __func__, "Conn %" PRIEndptId ": Cubic alpha %f "
325  "(#conn %d).\n", conn_id_, alpha, num_streams_);
326 #endif
327 
328  return alpha;
329 }

332 double CubicBytesHelper::Beta() const
333 {
334  // kNConnectionBeta is the backoff factor after loss for our N-connection
335  // emulation, which emulates the effective backoff of an ensemble of N
336  // TCP-Reno connections on a single loss event. The effective multiplier is
337  // computed as:
338  if (num_streams_ == 0)
339  {
340  LogW(kClassName, __func__, "Conn %" PRIEndptId ": Warning, num_streams_ "
341  "= 0.\n", conn_id_);
342  return kBeta;
343  }
344 
345  double beta = (((num_streams_ - 1) + kBeta) / num_streams_);
346 
347 #ifdef SLIQ_CC_DEBUG
348  LogD(kClassName, __func__, "Conn %" PRIEndptId ": Cubic beta %f (#conn "
349  "%d).\n", conn_id_, beta, num_streams_);
350 #endif
351 
352  return beta;
353 }
