225  bool Insert(const K& k, const V& v)
226  {
227  if (buckets_ == NULL)
228  {
229  return false;
230  }
231 
232  HTElem* e = NULL;
233 
234  if (pool_ == NULL)
235  {
236  e = new (std::nothrow) HTElem(k, v);
237 
238  if (e == NULL)
239  {
240  return false;
241  }
242  }
243  else
244  {
245  e = pool_;
246  pool_ = e->next;
247  e->key = k;
248  e->val = v;
249  }
250 
251  size_t i = (k.Hash() % num_buckets_);
252  HTElem* b = buckets_[i];
253 
254  if (b != NULL)
255  {
256  b->prev = e;
257  }
258 
259  e->next = b;
260  e->prev = NULL;
261  buckets_[i] = e;
262  ++size_;
263 
264  return true;
