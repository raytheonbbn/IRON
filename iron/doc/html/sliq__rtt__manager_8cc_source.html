121 void RttManager::UpdateRtt(const Time& now, EndptId conn_id,
122  const Time& rtt_sample)
123 {
124  // Get the RTT sample value as a double in seconds, and make sure that it is
125  // a valid RTT sample.
126  double rtt_val = rtt_sample.ToDouble();
127 
128  if (rtt_val <= 0.0)
129  {
130  return;
131  }
132 
133  // Store the latest RTT sample.
134  latest_rtt_ = rtt_sample;
135 
136  // Next, update the smoothed RTT and the RTT's smoothed mean deviation.
137  if (!initialized_)
138  {
139  // This is the first RTT sample received. Update following RFC 6298.
140  srtt_ = rtt_val;
141  srtt_obj_ = rtt_sample;
142  mdev_ = (0.5 * rtt_val);
143  mdev_obj_ = Time(mdev_);
144 
145  // Initialize the max/min filter smoothed RTT value.
146  mmf_interval_srtt_ = rtt_val;
147 
148  initialized_ = true;
149  }
150  else
151  {
152  // This is a subsequent RTT sample. Update following RFC 6298.
153  mdev_ = ((kOneMinusBeta * mdev_) + (kBeta * fabs(srtt_ - rtt_val)));
154  mdev_obj_ = Time(mdev_);
155  srtt_ = ((kOneMinusAlpha * srtt_) + (kAlpha * rtt_val));
156  srtt_obj_ = Time(srtt_);
157 
158  // Update the max/min filter smoothed RTT value.
159  mmf_interval_srtt_ = (((1.0 - kMmfAlpha) * mmf_interval_srtt_) +
160  (kMmfAlpha * rtt_val));
161  }
162 
163  // Pass the updated max/min filter smoothed RTT value into the filters.
164  mmf_rtt_.interval_srtt_ = mmf_interval_srtt_;
165  mmf_owd_.interval_srtt_ = mmf_interval_srtt_;
166 
167  // Update the max/min filter for RTTs.
168  mmf_rtt_.Update(now, rtt_val);
169 
170 #ifdef SLIQ_DEBUG
171  LogD(kClassName, __func__, "Conn %" PRIEndptId ": RTT sample %f srtt %f "
172  "mdev %f min %f max %f.\n", conn_id, rtt_val, srtt_, mdev_,
173  mmf_rtt_.min_est_, mmf_rtt_.max_est_);
174 #endif
175 }

207 Time RttManager::GetRtoTime() const
208 {
209  // Until a RTT measurement has been made, the RTO should be set to 1 second.
210  // See RFC 6298.
211  if (!initialized_)
212  {
213  return Time::FromMsec(kInitialRtoMsec);
214  }
215 
216  // The RTO is the smoothed RTT plus 4 times the RTT mean deviation. Since
217  // ACKs can be delayed at the receiver, include that time too.
218  double ack_del = (0.000001 * static_cast<double>(kAckTimerUsec +
219  kRtoToleranceUsec));
220  double rto_val = (srtt_ + (4.0 * mdev_) + ack_del);
221 
222  // Round up to a mimimum allowable RTO. See RFC 6298.
223  double min_rto = (0.001 * static_cast<double>(kMinRtoMsec));
224 
225  if (rto_val < min_rto)
226  {
227  rto_val = min_rto;
228  }
229 
230  return Time(rto_val);
231 }

234 Time RttManager::GetRexmitTime(int multiplier) const
235 {
236  // Until a RTT measurement has been made, the retransmission time should be
237  // set to 1 second. See RFC 6298.
238  if (!initialized_)
239  {
240  return Time::FromMsec(kInitialRtoMsec);
241  }
242 
243  // The retransmission time is the smoothed RTT plus the specified multiplier
244  // times the RTT mean deviation. Since ACKs can be delayed at the receiver,
245  // include that time too.
246  double k = static_cast<double>(multiplier);
247  double ack_del = (0.000001 * static_cast<double>(kAckTimerUsec +
248  kRtoToleranceUsec));
249  double rxt_val = (srtt_ + (k * mdev_) + ack_del);
250 
251  return Time(rxt_val);
252 }

272 void RttManager::MaxMinFilter::Update(const Time& now, double sample)
273 {
274  if (!init_)
275  {
276  // This is the first sample received.
277  init_ = true;
278  curr_min_ = sample;
279  curr_max_ = sample;
280  curr_end_time_ = (now + Time(kMmfIntvMult * interval_srtt_));
281  min_est_ = sample;
282  max_est_ = sample;
283 
284  return;
285  }
286 
287  // This is a subsequent sample. Test if the current interval's end time has
288  // been reached.
289  if (now < curr_end_time_)
290  {
291  // Add the sample to the current interval.
292  if (sample < curr_min_)
293  {
294  curr_min_ = sample;
295  }
296 
297  if (sample > curr_max_)
298  {
299  curr_max_ = sample;
300  }
301  }
302  else
303  {
304  // Complete the current interval.
305  size_t next_idx = ((prev_end_idx_ + 1) % kNumMmfIntv);
306 
307  prev_min_[next_idx] = curr_min_;
308  prev_max_[next_idx] = curr_max_;
309 
310  if (prev_cnt_ < kNumMmfIntv)
311  {
312  ++prev_cnt_;
313  }
314 
315  prev_end_idx_ = next_idx;
316 
317  // Update the minimum and maximum value estimates.
318  min_est_ = MinValue();
319  max_est_ = (outlier_rejection_ ? MedianFilterMaxValue() : MaxValue());
320 
321  // Start the next interval.
322  curr_min_ = sample;
323  curr_max_ = sample;
324  curr_end_time_ = (now + Time(kMmfIntvMult * interval_srtt_));
325  }
326 
327  // Update the minimum value estimate immediately.
328  if (sample < min_est_)
329  {
330  min_est_ = sample;
331  }
332 
333  // Possibly update the maximum value estimate immediately.
334  if ((!outlier_rejection_) && (sample > max_est_))
335  {
336  max_est_ = sample;
337  }
338 }

341 double RttManager::MaxMinFilter::MinValue()
342 {
343  // Return the minimum of the previous two intervals.
344  if (prev_cnt_ < 2)
345  {
346  return prev_min_[prev_end_idx_];
347  }
348 
349  double val_1 = prev_min_[((prev_end_idx_ + kNumMmfIntv - 1) %
350  kNumMmfIntv)];
351  double val_2 = prev_min_[prev_end_idx_];
352  double min_val = ((val_1 < val_2) ? val_1 : val_2);
353 
354  return min_val;
355 }

375 double RttManager::MaxMinFilter::MedianFilterMaxValue()
376 {
377  // Return the median of the previous five intervals. Handle the easy cases
378  // first.
379  if (prev_cnt_ == 1)
380  {
381  // The median of a single value is the value.
382  return prev_max_[prev_end_idx_];
383  }
384 
385  if (prev_cnt_ == 2)
386  {
387  // The median of two values is the average of the two values.
388  return ((prev_max_[((prev_end_idx_ + kNumMmfIntv - 1) % kNumMmfIntv)] +
389  prev_max_[prev_end_idx_]) * 0.5);
390  }
391 
392  // There are 3, 4, or 5 previous interval maximum values. Copy the previous
393  // maximum values into another array that can be sorted.
394  size_t i = 0;
395  size_t idx = 0;
396  size_t start_idx = ((prev_end_idx_ + kNumMmfIntv + 1 - prev_cnt_) %
397  kNumMmfIntv);
398  size_t num = prev_cnt_;
399  double buf[kNumMmfIntv];
400 
401  for (i = 0, idx = start_idx; i < num; ++i, idx = ((idx + 1) % kNumMmfIntv))
402  {
403  buf[i] = prev_max_[idx];
404  }
405 
406  // Sort the copied values. Since this is such a small array, just use
407  // bubblesort. The worst case (num=5 with reversed values) requires 5
408  // loops, 10 value swaps, and 20 value comparisons.
409  bool sorted = false;
410  double tmp = 0.0;
411 
412  while (!sorted)
413  {
414  sorted = true;
415 
416  for (i = 1; i < num; ++i)
417  {
418  if (buf[i] < buf[i - 1])
419  {
420  tmp = buf[i];
421  buf[i] = buf[i - 1];
422  buf[i - 1] = tmp;
423  sorted = false;
424  }
425  }
426  }
427 
428  // Return the median of the sorted array.
429  if ((num % 2) == 0)
430  {
431  // There is an even number of values. Return an average of the middle
432  // two elements in the array.
433  tmp = ((buf[(num / 2) - 1] + buf[num / 2]) * 0.5);
434  }
435  else
436  {
437  // There is an odd number of values. Return the middle element in the
438  // array.
439  tmp = buf[num / 2];
440  }
441 
442  return tmp;
443 }
