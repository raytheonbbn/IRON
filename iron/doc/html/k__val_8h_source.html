113  inline void Update(uint64_t new_k_val,
114  uint64_t start_transition_time_ms,
115  uint64_t transition_time_ms)
116  {
117  // Start the new transition where ever we left off on the previous
118  // transition. This avoids big jumps if K is updated is rapid
119  // succession.
120  k_current_ = GetValue();
121  // Compute the new difference.
122  k_diff_ = new_k_val - k_current_;
123  // When to start the next transition.
124  start_time_ms_ = start_transition_time_ms;
125  // How long the next transition should take.
126  transition_time_ms_ = transition_time_ms;
127  }
128 
139  inline uint64_t GetValue()
140  {
141  uint64_t value = k_current_;
142  // If k_diff_ is 0, we're not currently adapting K.
143  if (k_diff_ != 0)
144  {
145  uint64_t now = Time::Now().GetTimeInMsec();
146  if (start_time_ms_ + transition_time_ms_ < now)
147  {
148  // We're done adapting. Jump to the new value.
149  k_current_ += k_diff_;
150  k_diff_ = 0;
151  value = k_current_;
152  }
153  else
154  {
155  double fraction =
156  static_cast<double>(now - start_time_ms_) / transition_time_ms_;
157  value = k_current_ + (k_diff_ * fraction);
158  }
159  }
160  return value;
161  }
