179 PktSeqNumber PacingSender::OnPacketSent(StreamId stream_id,
180  const Time& send_time,
181  PktSeqNumber seq_num,
182  uint32_t pld_bytes,
183  uint32_t tot_bytes, float& cc_val)
184 {
185  // Call into the congestion control algorithm first.
186  PktSeqNumber cc_seq_num = cc_alg_->OnPacketSent(stream_id, send_time,
187  seq_num, pld_bytes,
188  tot_bytes, cc_val);
189 
190  // If there are no bytes in flight, then update the number of burst tokens
191  // allowed. Note that bytes in flight does not reflect the packet just sent
192  // yet.
193  if (bytes_in_flight_ == 0)
194  {
195  // Add more burst tokens anytime the connection is leaving quiescence, but
196  // limit it to the equivalent of a single bulk write, not exceeding the
197  // current cwnd in packets.
198  if (kLimitPacingBurst)
199  {
200  burst_tokens_ = (cc_alg_->GetCongestionWindow() / kDefaultTcpMss);
201 
202  if (burst_tokens_ > initial_packet_burst_)
203  {
204  burst_tokens_ = initial_packet_burst_;
205  }
206  }
207  else
208  {
209  burst_tokens_ = initial_packet_burst_;
210  }
211  }
212 
213  // If there are burst tokens left, consume one for the packet just sent and
214  // return.
215  if (burst_tokens_ > 0)
216  {
217  --burst_tokens_;
218  was_last_send_delayed_ = false;
219  last_delayed_packet_sent_time_.Zero();
220  ideal_next_packet_send_time_.Zero();
221 
222  return cc_seq_num;
223  }
224 
225  // The next packet should be sent as soon as the current packets have been
226  // transferred.
227  Capacity bits_per_second = cc_alg_->SendPacingRate();
228  Time delay;
229 
230  if (bits_per_second > 0)
231  {
232  delay = Time::FromUsec(pld_bytes * 8 * kNumMicrosPerSecond /
233  bits_per_second);
234  }
235 
236  // If the last send was delayed, and the timer took a long time to get
237  // invoked, allow the connection to make up for lost time.
238  if (was_last_send_delayed_)
239  {
240  ideal_next_packet_send_time_ += delay;
241 
242  // The send was application limited if it takes longer than the pacing
243  // delay between sent packets.
244  bool application_limited =
245  ((!last_delayed_packet_sent_time_.IsZero()) &&
246  (send_time > last_delayed_packet_sent_time_.Add(delay)));
247 
248  bool making_up_for_lost_time =
249  (ideal_next_packet_send_time_ <= send_time);
250 
251  // As long as we're making up time and not application limited, continue
252  // to consider the packets delayed, allowing the packets to be sent
253  // immediately.
254  if (making_up_for_lost_time && !application_limited)
255  {
256  last_delayed_packet_sent_time_ = send_time;
257  }
258  else
259  {
260  was_last_send_delayed_ = false;
261  last_delayed_packet_sent_time_.Zero();
262  }
263  }
264  else
265  {
266  ideal_next_packet_send_time_ = Time::Max(
267  ideal_next_packet_send_time_.Add(delay), send_time.Add(delay));
268  }
269 
270  return cc_seq_num;
271 }

336 Time PacingSender::TimeUntilSend(const Time& now)
337 {
338  // Call into the congestion control algorithm first.
339  Time time_until_send = cc_alg_->TimeUntilSend(now);
340 
341  // Don't pace if there are burst tokens available or we are leaving
342  // quiescence.
343  if ((burst_tokens_ > 0) || (bytes_in_flight_ == 0))
344  {
345  return time_until_send;
346  }
347 
348  // If the underlying sender prevents sending right now, then pass on the
349  // time value.
350  if (!time_until_send.IsZero())
351  {
352  return time_until_send;
353  }
354 
355  // If the next send time is larger than the timer granularity, then wait to
356  // send.
357  if (ideal_next_packet_send_time_ > now.Add(timer_granularity_))
358  {
359  was_last_send_delayed_ = true;
360  return ideal_next_packet_send_time_.Subtract(now);
361  }
362 
363  // Send the packet immediately.
364  return Time();
365 }
