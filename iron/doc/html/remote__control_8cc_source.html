109 int EndpointInfo::ReceiveMessage()
110 {
111  if (msg_size_ == 0)
112  {
113  // Receive the message delimiter, which is a 4-byte integer (in network
114  // byte order) containing the following JSON message length, in bytes.
115  if (!Receive(static_cast<int>(sizeof(uint32_t))))
116  {
117  return -1;
118  }
119 
120  // Check if all four bytes have been received.
121  if (rcv_offset_ == static_cast<int>(sizeof(uint32_t)))
122  {
123  // Get the message length, and prepare for receiving the message.
124  uint32_t* len_nbo = (uint32_t*)(rcv_buf_);
125  msg_size_ = static_cast<int>(ntohl(*len_nbo));
126  rcv_offset_ = 0;
127 
128  LogD(kSClassName, __func__, "Message length is %d bytes for remote "
129  "control endpoint: %s:%d\n", msg_size_, addr_.ToString().c_str(),
130  static_cast<int>(port_));
131 
132  // Avoid overflowing the receive buffer. Leave room for a
133  // null-terminated string (JSON is a string format).
134  if (msg_size_ > (MAX_RC_MSG_SIZE - 1))
135  {
136  LogE(kSClassName, __func__, "Error, message length %d is too large "
137  "for receive buffer length %d.\n", msg_size_, MAX_RC_MSG_SIZE);
138  return -1;
139  }
140  }
141  }
142  else
143  {
144  // Receive more of the JSON message.
145  if (!Receive(msg_size_))
146  {
147  return -1;
148  }
149 
150  // Check if all of the JSON message has been received.
151  if (rcv_offset_ == msg_size_)
152  {
153  // The entire JSON message has been received. Null-terminate the
154  // string so it can be parsed.
155  rcv_buf_[msg_size_] = '\0';
156 
157  return 1;
158  }
159  }
160  return 0;
161 }

164 bool EndpointInfo::Receive(int total_size)
165 {
166  // Perform the receive.
167  ssize_t bytes = ::recv(sock_, (void*)&(rcv_buf_[rcv_offset_]),
168  (size_t)(total_size - rcv_offset_), 0);
169 
170  if (bytes > 0)
171  {
172  LogD(kEClassName, __func__, "Received %d bytes from the remote control "
173  "endpoint: %s:%d\n", static_cast<int>(bytes), addr_.ToString().c_str(),
174  static_cast<int>(port_));
175 
176  rcv_offset_ += static_cast<int>(bytes);
177 
178  return true;
179  }
180 
181  if (bytes == 0)
182  {
183  // The endpoint has closed its end of the connection.
184  LogD(kEClassName, __func__, "Remote control endpoint %s:%d has closed its "
185  "end of the connection.\n", addr_.ToString().c_str(),
186  static_cast<int>(port_));
187  }
188  else
189  {
190  // An error has occurred.
191  LogE(kEClassName, __func__, "Error receiving from endpoint %s:%d: %s.\n",
192  addr_.ToString().c_str(), static_cast<int>(port_), strerror(errno));
193  }
194 
195  return false;
196 }

199 bool EndpointInfo::SendMessage(uint8_t* msg_buf, int msg_len)
200 {
201  // Validate the arguments.
202  if ((msg_buf == NULL) || (msg_len < 0))
203  {
204  return false;
205  }
206 
207  // Send the message.
208  ssize_t bytes = ::send(sock_, static_cast<void*>(msg_buf), msg_len, 0);
209 
210  if (bytes >= 0)
211  {
212  if (static_cast<int>(bytes) == msg_len)
213  {
214  LogD(kEClassName, __func__, "Sent %d bytes to the remote control "
215  "endpoint: %s:%d\n", static_cast<int>(bytes),
216  addr_.ToString().c_str(), static_cast<int>(port_));
217  }
218  else
219  {
220  LogD(kEClassName, __func__, "Only sent %d bytes of %d bytes to the "
221  "remote control endpoint: %s:%d\n", static_cast<int>(bytes), msg_len,
222  addr_.ToString().c_str(), static_cast<int>(port_));
223  return false;
224  }
225  }
226  else
227  {
228  // An error has occurred.
229  LogE(kEClassName, __func__, "Error sending to endpoint %s:%d: %s.\n",
230  addr_.ToString().c_str(), static_cast<int>(port_), strerror(errno));
231  return false;
232  }
233 
234  return true;

332 bool iron::RemoteControl::SendMessage(EndpointInfo* ep,
333  rapidjson::StringBuffer& str_buf)
334 {
335  LogD(kClassName, __func__, "Sending message: %s\n",
336  str_buf.GetString());
337 
338  if (!ep)
339  {
340  LogE(kClassName, __func__, "Unable to send message.\n");
341  return false;
342  }
343 
344  // Copy the passed message into the send buffer. Prepend with a 4-byte
345  // integer containing the following JSON message length in network byte
346  // order.
347  int json_len = static_cast<int>(str_buf.GetSize());
348  uint32_t json_len_nbo = static_cast<uint32_t>(htonl(json_len));
349  int msg_len = (json_len + sizeof(json_len_nbo));
350 
351  if (msg_len > (MAX_RC_MSG_SIZE - 1))
352  {
353  LogE(kClassName, __func__, "Error, message length %d is too large for "
354  "send buffer length %d.\n", msg_len, MAX_RC_MSG_SIZE);
355  // Close the connection.
356  endpoints_.erase(ep->id_);
357  delete ep;
358  return false;
359  }
360 
361  LogD(kClassName, __func__, "buffer: %s\n", str_buf.GetString());
362  ::memcpy(&(snd_buf_[0]), &json_len_nbo, sizeof(json_len_nbo));
363  ::memcpy(&(snd_buf_[sizeof(json_len_nbo)]), str_buf.GetString(), json_len);
364 
365  // Send the JSON reply message.
366  if (!ep->SendMessage(snd_buf_, msg_len))
367  {
368  // Close the connection.
369  endpoints_.erase(ep->id_);
370  delete ep;
371  return false;
372  }
373  return true;
374 }

415  // Make sure that a client has a message ready.
416  if (endpoint_ready_ == NULL)

420

422  endpoint_ready_->PrepareForNextMessage();

424  // Get access to the "keys" array.

426 
427  // Check that "keys" is an array of strings.
428  if (!keys.IsArray())

434  (char*)(endpoint_ready_->rcv_buf_));
435  return false;
436  }
437 
438  for (SizeType index = 0; index < keys.Size(); ++index)
439  {
440  if (!keys[index].IsString())

442  LogE(kSClassName, __func__, "Received JSON get message from %s:%d has "

445  static_cast<int>(endpoint_ready_->port_),
446  (char*)(endpoint_ready_->rcv_buf_));
447  return false;

453  key_array = &(keys);

455  return true;

500 bool RemoteControl::ServiceEndpoints(fd_set& read_fds)
501 {
502 
503  EndpointInfo* ep = NULL;
504  // Only service sockets until one client has a message fully received and
505  // ready for processing. This keeps the API and implementation simple.
506  // Check that this is the case before we start.
507  if (endpoint_ready_ != NULL)
508  {
509  LogW(kClassName, __func__, "A remote control client (%s:%d) appears to "
510  "have a message ready for servicing when it should not.\n",
511  endpoint_ready_->addr_.ToString().c_str(),
512  static_cast<int>(endpoint_ready_->port_));
513 
514  }
515 
516  // Check the client sockets last.
517  if (!endpoints_.empty())
518  {
519  // If there is a received message waiting for processing, then we cannot
520  // service another client right now. Notify the caller that a message is
521  // ready for processing again.
522  if (endpoint_ready_ != NULL)
523  {
524  return true;
525  }
526 
527  map<uint32_t, EndpointInfo*>::iterator it = endpoints_.begin();
528 
529  while (it != endpoints_.end())
530  {
531  ep = it->second;
532  //LogD(kClassName, __func__, "Looking at remote control endpoint: "
533  // "%s:%d\n", ci->addr_.ToString().c_str(),
534  // static_cast<int>(ci->port_));
535 
536  if ((ep != NULL) && InSet(ep->sock_, read_fds))
537  {
538  LogD(kClassName, __func__, "Receiving from remote control endpoint: "
539  "%s:%d\n", ep->addr_.ToString().c_str(),
540  static_cast<int>(ep->port_));
541 
542  // Receive message data from the endpoint.
543  int rv = ep->ReceiveMessage();
544 
545  if (rv > 0)
546  {
547  LogD(kClassName, __func__, "Received JSON request message from "
548  "%s:%d: %s\n", ep->addr_.ToString().c_str(),
549  static_cast<int>(ep->port_), (char*)(ep->rcv_buf_));
550 
551  // Attempt to parse the received JSON message.
552  if (ParseJsonMessage(ep))
553  {
554  // The endpoint now has a message ready. Record the endpoint pointer.
555  endpoint_ready_ = ep;
556  LogD(kClassName, __func__, "Endpoint ready %u\n", ep->id_);
557  break;
558  }
559  else
560  {
561  // The message cannot be parsed. Close the connection.
562  LogW(kClassName, __func__, "Cannot parse JSON message\n");
563  endpoints_.erase(it);
564  delete ep;
565  break;
566  }
567  }
568  else if (rv < 0)
569  {
570  // The endpoint has closed the connection.
571  if (endpoint_ready_ == ep)
572  {
573  endpoint_ready_ = NULL;
574  }
575  endpoints_.erase(it);
576  delete ep;
577  LogD(kClassName, __func__, "Client has closed the connection.\n");
578  break;
579  }
580  else
581  {
582  LogD(kClassName, __func__, "rv is zero\n");
583  }
584  }
585 
586  ++it;
587  }
588  }
589 
590  return(endpoint_ready_ != NULL);
591 }

667 bool RemoteControlClient::ParseJsonMessage(EndpointInfo* si)
668 {
669  // Initialize.
670  msg_type_ = RC_INVALID;
671  msg_id_ = 0;
672  err_msg_ = "";
673  msg_target_ = "";
674 
675  LogD(kCClassName, __func__, "Parsing message: %s\n", si->rcv_buf_);
676 
677  // Parse the JSON message into a document object (to use DOM-style APIs).
678  if (document_.ParseInsitu((char*)(si->rcv_buf_)).HasParseError())
679  {
680  LogE(kCClassName, __func__, "Error parsing received JSON reply "
681  "message from %s:%d: %s\n", si->addr_.ToString().c_str(),
682  static_cast<int>(si->port_), (char*)(si->rcv_buf_));
683  return false;
684  }
685 
686  // Make sure that the message is an object (begins with '{' and ends with
687  // '}').
688  if (!document_.IsObject())
689  {
690  LogE(kSClassName, __func__, "Request message from %s:%d is not an "
691  "object: %s\n", si->addr_.ToString().c_str(),
692  static_cast<int>(si->port_), (char*)(si->rcv_buf_));
693  return false;
694  }
695 
696  // Get the message type.
697  string msg_type_str;
698 
699  if ((document_.HasMember("msg")) && (document_["msg"].IsString()))
700  {
701  msg_type_str = document_["msg"].GetString();
702 
703  if (msg_type_str == "setreply")
704  {
705  msg_type_ = RC_SETREPLY;
706  }
707  else if (msg_type_str == "getreply")
708  {
709  msg_type_ = RC_GETREPLY;
710  }
711  else if (msg_type_str == "pusherror")
712  {
713  msg_type_ = RC_PUSHERR;
714  }
715  else if (msg_type_str == "push")
716  {
717  msg_type_ = RC_PUSH;
718  }
719  else
720  {
721  LogE(kCClassName, __func__, "Unknown message type from %s:%d: "
722  "%s\n", si->addr_.ToString().c_str(), static_cast<int>(si->port_),
723  msg_type_str.c_str());
724  return false;
725  }
726  }
727  else
728  {
729  LogE(kSClassName, __func__, "Message from %s:%d does not have a "
730  "message type: %s\n", si->addr_.ToString().c_str(),
731  static_cast<int>(si->port_), (char*)(si->rcv_buf_));
732  return false;
733  }
734 
735  // Get the messge identifier.
736  if ((document_.HasMember("msgid")) && (document_["msgid"].IsUint()))
737  {
738  msg_id_ = document_["msgid"].GetUint();
739  }
