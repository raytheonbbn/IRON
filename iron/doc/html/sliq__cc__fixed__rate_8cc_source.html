198 PktSeqNumber FixedRate::OnPacketSent(StreamId stream_id,
199  const Time& send_time,
200  PktSeqNumber seq_num, uint32_t pld_bytes,
201  uint32_t tot_bytes, float& cc_val)
202 {
203  // Assign a CC sequence number to the packet.
204  PktSeqNumber cc_seq_num = nxt_cc_seq_num_;
205  ++nxt_cc_seq_num_;
206 
207 #ifdef SLIQ_CC_DEBUG
208  LogD(kClassName, __func__, "Conn %" PRIEndptId ": On Send: stream=%"
209  PRIStreamId " seq_num=%" PRIPktSeqNumber " cc_seq_num=%"
210  PRIPktSeqNumber " send_time=%s size=%" PRIu32 "/%" PRIu32
211  " cc_val=%f\n", conn_id_, stream_id, seq_num, cc_seq_num,
212  send_time.ToString().c_str(), pld_bytes, tot_bytes,
213  static_cast<double>(cc_val));
214 #endif
215 
216  // Update the next send time.
217  UpdateNextSendTime(send_time, tot_bytes);
218 
219  return cc_seq_num;
220 }

273 Time FixedRate::TimeUntilSend(const Time& now)
274 {
275  // Check if the send can happen immediately.
276  if (now.Add(timer_tolerance_) >= next_send_time_)
277  {
278  return Time();
279  }
280 
281  // Wait to send.
282  return (next_send_time_ - now);
283 }

356 void FixedRate::UpdateNextSendTime(const Time& now, size_t bytes)
357 {
358  // Update the next send time using the packet size, the send rate, and the
359  // stored next send time. This maintains inter-send time accuracy.
360  double ist = ((8.0 * static_cast<double>(bytes + kPktOverheadBytes)) /
361  static_cast<double>(send_rate_bps_));
362 
363  // If the current time is more than kQuiescentThreshold seconds beyond the
364  // stored next send time, then the sender is considered to have been
365  // quiescent for a time, so the next send time must be computed from now.
366  // Otherwise, the send pacing timer must have been used, so add the
367  // inter-send time for this packet to the stored next send time.
368  if (now > next_send_time_.Add(kQuiescentThreshold))
369  {
370  next_send_time_ = now.Add(ist);
371  }
372  else
373  {
374  next_send_time_ = next_send_time_.Add(ist);
375  }
376 
377 #ifdef SLIQ_CC_DEBUG
378  LogD(kClassName, __func__, "Conn %" PRIEndptId ": Waiting for "
379  "intersend_time=%f\n", conn_id_, ist);
380 #endif
381 }
