181 bool QueueDepthOsc::Initialize(const QueueDepthOscConfig& config)
182 {
183  fft_sample_size_ = config.fft_sample_size;
184  fft_sample_time_interval_ = Time(config.fft_sample_time_interval_secs);
185  fft_compute_time_interval_ = Time(config.fft_compute_time_interval_secs);
186  max_considered_period_usec_ =
187  static_cast<uint64_t>(config.max_considered_period_secs * 1000000);
188  min_time_between_resets_ = Time(config.min_time_between_resets_secs);
189  reset_trigger_fraction_ = config.reset_trigger_fraction;
190  reset_trigger_sample_count_ = config.reset_trigger_sample_count;
191  use_soft_reset_ = config.use_soft_reset;
192 
193  fft_samples_ = new (std::nothrow) double[fft_sample_size_];
194  fft_input_ = new (std::nothrow) double[fft_sample_size_];
195  fft_input_times_ = new (std::nothrow) Time[fft_sample_size_];
196  fft_output_ = static_cast<fftw_complex*>(
197  fftw_malloc(sizeof(fftw_complex) * fft_sample_size_));
198  if ((fft_samples_ == NULL) ||
199  (fft_input_ == NULL) ||
200  (fft_input_times_ == NULL) ||
201  (fft_output_ == NULL))
202  {
203  LogF(kClassName, __func__,
204  "Error allocating memory for oscillation manager.\n");
205  return false;
206  }
207  // This function sets up the FFTW computation, telling it where to get the
208  // input and where to put the output. FFTW_MEASURE will give a more exact
209  // frequency that FFTW_ESTIMATE (the alternative), at the expense of
210  // efficiency in the case where the input sizes keep changing. In our case,
211  // we always have the same size input, so FFTW_MEASURE only takes an
212  // efficiency hit the first time we call it (in the constructor).
213  fft_plan_ = fftw_plan_dft_r2c_1d(
214  fft_sample_size_, fft_input_, fft_output_, FFTW_MEASURE);
215  if (fft_plan_ == NULL)
216  {
217  LogF(kClassName, __func__, "Error setting up FFT plan.\n");
218  return false;
219  }
220  return true;
221 }

224 void QueueDepthOsc::Reset(Time now)
225 {
226  // \TODO Hard resets are really a workaround for not being able to identify
227  // which period of oscillation to use.
228  //
229  // We want to be able to smooth based on the period of oscillation caused by
230  // transmission and QLAM delays. Smoothing basically makes these
231  // oscillations disappear. Therefore, if we're continuing to smooth after a
232  // network event (i.e., a soft reset), then we're more likely to pick up and
233  // use a period that's really convergence after the network event because we
234  // don't have any better period. Hard resets make the "right" period of
235  // oscillation show up again.
236  //
237  // Performance would be better if we could continue to smooth on the old
238  // value and not have these false periods coming from post-network-event
239  // convergence, but it's not obvious how to do this.
240  have_sufficient_data_ = false;
241  next_index_ = 0;
242  num_low_for_reset_ = 0;
243  num_high_for_reset_ = 0;
244  if (!use_soft_reset_)
245  {
246  have_usable_period_ = false;
247  }
248  last_reset_ = now;
249 }

252 void QueueDepthOsc::QueueDepthOscCheckPoint(
253  uint32_t new_depth, uint32_t smoothed)
254 {
255  Time now = Time::Now();
256  if (now - last_checkpoint_ < fft_sample_time_interval_)
257  {
258  return;
259  }
260 
261  if ((new_depth == 0) && (next_index_ == 0))
262  {
263  // Don't start collecting samples until we have data.
264  return;
265  }
266 
267  double threshold = smoothed * reset_trigger_fraction_;
268  if (new_depth < smoothed - threshold)
269  {
270  num_low_for_reset_++;
271  num_high_for_reset_ = 0;
272  }
273  else if (new_depth > smoothed + threshold)
274  {
275  num_high_for_reset_++;
276  num_low_for_reset_ = 0;
277  }
278  else
279  {
280  num_low_for_reset_ = 0;
281  num_high_for_reset_ = 0;
282  }
283 
284  if (WouldLogD(kClassName))
285  {
286  LogD(kClassName, __func__,
287  "%s Bin %s: "
288  " new_depth = %" PRIu32 ", smoothed = %" PRIu32
289  ", num_low_for_reset_ = %" PRIu16 ", num_high_for_reset_ = %" PRIu16
290  ", time since last reset = %" PRId64 "\n",
291  (ls_queue_ ? "LS" : "ALL"), log_id_,
292  new_depth, smoothed, num_low_for_reset_, num_high_for_reset_,
293  (now - last_reset_).GetTimeInUsec());
294  }
295 
296  if ((num_low_for_reset_ > reset_trigger_sample_count_ ||
297  num_high_for_reset_ > reset_trigger_sample_count_)
298  && now - last_reset_ > min_time_between_resets_)
299  {
300  // \TODO Resets will toss out all samples and start again for the sake of
301  // the FFT computation. This is correct because we don't want the FFT to
302  // compute a period that includes the convergence after whatever network
303  // event triggered the reset. Ideally, we'd wait until convergence was
304  // complete before we start collecting samples again, and then compute the
305  // FFT as soon as we have enough data (maybe less than a full sample size)
306  // to start smoothing again. However, it's not obvious how to do this.
307  Reset(now);
308  if (qd_xplot_ && kGraphPeriods)
309  {
310  qd_xplot_->DrawLine(
311  now.GetTimeInUsec() - iron::kStartTime, 0,
312  now.GetTimeInUsec() - iron::kStartTime, qd_xplot_->max_y(), WHITE);
313  }
314  }
315 
316  if (qd_xplot_ && kGraphPeriods)
317  {
318  qd_xplot_->DrawPoint(
319  now.GetTimeInUsec() - iron::kStartTime, new_depth, MAGENTA);
320  }
321  double new_sample = static_cast<double>(new_depth);
322  // Taking the log of the sample gives a low pass filter over the data to
323  // help reduce the impact of long-timescape fluctuations (which we can't
324  // address via smoothing).
325  if (new_sample != 0)
326  {
327  new_sample = log(new_sample);
328  }
329  fft_samples_[next_index_] = new_sample;
330  fft_input_times_[next_index_] = now;
331  if (WouldLogD(kClassName))
332  {
333  LogD(kClassName, __func__,
334  "%s Bin %s: "
335  "Index %" PRIu16 " now has depth %f, time %f\n",
336  (ls_queue_ ? "LS" : "ALL"), log_id_,
337  next_index_, fft_samples_[next_index_],
338  static_cast<double>(now.GetTimeInUsec())/1e6);
339  }
340 
341  next_index_++;
342  if (next_index_ >= fft_sample_size_)
343  {
344  next_index_ = 0;
345  // Now we have enough data to start computing FFTs.
346  have_sufficient_data_ = true;
347  }
348  last_checkpoint_ = now;
349 
350  if (have_sufficient_data_ && ((now - last_fft_) >
351  fft_compute_time_interval_))
352  {
353  ComputeFft();
354  last_fft_ = now;
355  }
356 }

359 void QueueDepthOsc::ComputeFft()
360 {
361  uint64_t new_computed_period_usec = 0;
362  // FFTW needs the input to be in adjacent memory, and in order
363  // to avoid re-planning each time (which overwrites input), the
364  // input must always be in the same memory location. Therefore,
365  // before executing the FFT, we copy the samples into a single
366  // location (fft_input_). This takes two memcpys because of the
367  // circular buffer of samples.
368  memcpy(&(fft_input_[0]), &(fft_samples_[next_index_]),
369  sizeof(fft_input_[0]) * (fft_sample_size_ - next_index_));
370  memcpy(&(fft_input_[fft_sample_size_ - next_index_]),
371  &(fft_samples_[0]),
372  sizeof(fft_input_[0]) * next_index_);
373  if (WouldLogD(kClassName))
374  {
375  LogD(kClassName, __func__,
376  "%s Bin %s: "
377  "Computing period.\n",
378  (ls_queue_ ? "LS" : "ALL"), log_id_);
379  uint16_t sample_index = next_index_;
380  for (uint16_t count = 0; count < fft_sample_size_; ++count)
381  {
382  LogD(kClassName, __func__,
383  "*** sample %" PRIu16 " = %f\n", sample_index, fft_input_[count]);
384  sample_index = (sample_index + 1) % fft_sample_size_;
385  }
386  }
387  // fftw_execute computes the FFT with the parameters set up in the
388  // Initialize method.
389  fftw_execute(fft_plan_);
390 
391  double norm = 0;
392  double max_norm = 0;
393  double max_norm_low_freq = 0;
394  uint16_t max_norm_index = 0;
395  uint16_t max_norm_low_freq_index = 0;
396  uint16_t last_index = 0;
397  if (next_index_ != 0)
398  {
399  last_index = next_index_ - 1;
400  }
401  int64_t time_span_usec =
402  (fft_input_times_[last_index]
403  - fft_input_times_[next_index_]).GetTimeInUsec();
404  LogI(kClassName, __func__,
405  "%s Bin %s: "
406  "FFT results are shown below:\n", (ls_queue_ ? "LS" : "ALL"), log_id_);
407 
408  // We only want to look at periods less than the nominal time span.
409  uint16_t min_i = time_span_usec / max_considered_period_usec_;
410  bool update_period = false;
411  // This look finds the norm for each FFT output entry, and then evaluates it
412  // to find out whether this is (so far) the right frequency for us to
413  // use. We want the highest-energy frequency that is for a period of less
414  // than max_considered_period_usec_. Anything longer than that would be too
415  // long for effective smoothing of queue depths. However, if the
416  // highest-energy useable frequency is neglible (and is negligible compared
417  // to the overall highest-energy frequency), we will just ignore it and
418  // continue smoothing based on the last period we computed.
419  for (uint16_t i = 2; i < (fft_sample_size_ + 2) / 2; i++)
420  {
421  norm = (fft_output_[i][0] * fft_output_[i][0])
422  + (fft_output_[i][1] * fft_output_[i][1]);
423  if (i < min_i)
424  {
425  // We can't smooth based on long periods. So for the lowest frequencies
426  // (those that would result in a period longer than
427  // max_considered_period_usec_), just keep track of the norm so we'll
428  // know whether the best period in a usable range is oris not neglible
429  // compared to this.
430  if (norm > max_norm_low_freq)
431  {
432  max_norm_low_freq = norm;
433  max_norm_low_freq_index = i;
434  }
435  }
436  else
437  {
438  // Otherwise, this would be a period usable for smoothing, so consider
439  // it as a candidate if it's the max energy.
440  if (norm > max_norm)
441  {
442  max_norm = norm;
443  max_norm_index = i;
444  }
445  }
446  if (WouldLogI(kClassName) && i < 100)
447  {
448  // This log message generates a table in the log file. The explanation
449  // of what's in the table can be seen in the log message before the for
450  // loop. Don't bother printing the lowest frequencies - it just clutters
451  // up the log file and (based on preivous experience) is never the right
452  // one data look at anyway.
453  LogI(kClassName, __func__,
454  "*** index %" PRIu16 " | %f + %f i | norm = %f "
455  "| index/T = %f s | period = %f usec \n",
456  i,
457  fft_output_[i][0], fft_output_[i][1],
458  norm/1e6,
459  (static_cast<double>(i)*1e6)/time_span_usec,
460  static_cast<double>(time_span_usec) / i);
461  }
462  }
463  if (max_norm_index == 0)
464  {
465  LogD(kClassName, __func__,
466  "Max norm index is 0. That's weird except during start-up.\n");
467  have_usable_period_ = false;
468  return;
469  }
470  // Period in seconds is 1/frequency * sample time in seconds. (We want
471  // usec.)
472  new_computed_period_usec = time_span_usec / max_norm_index;
473  if (WouldLogI(kClassName))
474  {
475  LogI(kClassName, __func__,
476  "New FFT results (time span %" PRId64 " usec): Max considered "
477  "norm = %f, at index = %" PRIu16 ", period = %" PRIu64 " usec.\n",
478  time_span_usec, max_norm, max_norm_index, new_computed_period_usec);
479  if (max_norm_low_freq_index != 0)
480  {
481  LogI(kClassName, __func__, "Max norm with frequency too low to use = %f, "
482  "at index = %" PRIu16 ", period = %" PRIu64 " usec.\n",
483  max_norm_low_freq, max_norm_low_freq_index,
484  time_span_usec / max_norm_low_freq_index);
485  }
486  }
487 
488  // We want to use the newly-computed period if the max norm usuable period
489  // has enough energy OR if the max norm usable period is close enough to the
490  // overall max norm. This effectively ignores tiny fluctuations that are
491  // totally overshadowed by large fluctuations.
492  update_period = ((max_norm > kEnergyThreshold) ||
493  (max_norm > max_norm_low_freq * kEnergyFraction));
494  have_usable_period_ = have_usable_period_ || update_period;
495 
496  if (update_period)
497  {
498  // Interpolation helps if we didn't have enough samples to exactly capture
499  // the right frequency.
500  if (kDoInterpolation)
501  {
502  Interpolate(new_computed_period_usec, max_norm_index, time_span_usec);
503  }
504  oscillation_period_usec_ = new_computed_period_usec;
505  }
506 
507  LogI(kClassName, __func__, "Period is usable? %c\n",
508  (update_period ? 'Y' : 'N'));
509 
510 #ifdef XPLOT
511  if (qd_xplot_ && kGraphPeriods)
512  {
513  // Usable periods are graphed in colors GREEN, RED, BLUE, and YELLOW.
514  // Non-usable periods are graphed in colors PURPLE, ORANGE, MAGENTA, and
515  // PINK.
516  uint8_t color = next_color_ + 1;
517  if (!update_period)
518  {
519  color += 4;
520  }
521  uint64_t t =
522  fft_input_times_[last_index].GetTimeInUsec() - iron::kStartTime;
523  int64_t y = qd_xplot_->max_y();
524  // This log message can help trace back to how the FFT results looked for
525  // a particular period displayed on the graph.
526  // The graph includes 4 lines for each computed period. One at the start
527  // of the sample set, and then three evenly spaced showing the computed
528  // period.
529  LogI(kClassName, __func__,
530  "Printing periods in color %d, compute time %" PRId64 "\n",
531  color, t);
532  qd_xplot_->DrawLine(t, 0, t, y, static_cast<XPLOT_COLOR>(color));
533  t -= new_computed_period_usec;
534  qd_xplot_->DrawLine(t, 0, t, y, static_cast<XPLOT_COLOR>(color));
535  t -= new_computed_period_usec;
536  qd_xplot_->DrawLine(t, 0, t, y, static_cast<XPLOT_COLOR>(color));
537  t = fft_input_times_[next_index_].GetTimeInUsec() - iron::kStartTime;
538  qd_xplot_->DrawLine(t, 0, t, y, static_cast<XPLOT_COLOR>(color));
539  next_color_ = ((next_color_ + 1) % 4);
540  }
541 #endif // XPLOT
542 }

545 void QueueDepthOsc::Interpolate(
546  uint64_t& computed_period_usec, uint16_t used_index, int64_t time_span_usec)
547 {
548  if (used_index + 1 < (fft_sample_size_ + 2) / 2)
549  {
550  // Second order lagrange interpolator to get a finer grained estimate,
551  // using one sample on either side of the peak index.
552  //
553  // f1 = peak index - 1, f2 = peak index, f3 = peak index + 1
554  // q1 = val(f1), q2 = val(f2), q3 = val(f3)
555 
556  double norm =
557  (fft_output_[used_index][0] * fft_output_[used_index][0])
558  + (fft_output_[used_index][1] * fft_output_[used_index][1]);
559 
560  double norm_above =
561  (fft_output_[used_index + 1][0] * fft_output_[used_index + 1][0])
562  + (fft_output_[used_index + 1][1] * fft_output_[used_index + 1][1]);
563 
564  double norm_below =
565  (fft_output_[used_index - 1][0] * fft_output_[used_index - 1][0])
566  + (fft_output_[used_index - 1][1] * fft_output_[used_index - 1][1]);
567 
568 
569  // denominator = (f1 - f2) * (f1 - f3) * (f2 - f3) = -2
570  // a = ((f3 * (q2 - q1)) + (f2 * (q1 - q3)) + (f1 * (q3 - q2)))
571  // / denominator
572  // b = ((f3^2 * (q1 - q2)) + (f2^2 * (q3 - q1)) + (f1^2 * (q2 - q3)))
573  // / denominator
574  // c = ((f2 * f3 * (f2 - f3) * q1) + (f3 * f1 * (f3 - f1) * q3)
575  // + (f1 * f2 * (f1 - f2) * q3))
576 
577  // The interpolator fits a quadratic of the form q = af^2 + bf + c
578  // Hence the estimated peak location should be when dq/df = 2af + b = 0,
579  // which implies the estimated peak location (i.e., estimated frequency)
580  // should be at f = -b/2a
581  //
582  // The estimated peak value is a(-b/2a)^2 + b(-b/2a) + c = -b^2/4a + c,
583  // but this is unused.
584  double f2 = static_cast<double>(used_index);
585  // Note that denominator of a = -2, so multiplying the numerator by -1
586  // effectively multiplies a by 2.
587  double a_times_2 =
588  (((f2 + 1) * (norm - norm_below))
589  + (f2 * (norm_below - norm_above))
590  + ((f2 - 1) * (norm_above - norm)))
591  * -1;
592  // Similarly, dividing the numerator of b by 2 is the same as multiplying
593  // all of b by -1, since the denominator of b is -2.
594  double neg_b =
595  (((f2 + 1) * (f2 + 1) * (norm_below - norm))
596  + (f2 * f2 * (norm_above - norm_below))
597  + ((f2 - 1) * (f2 - 1) * (norm - norm_above))
598  ) / 2;
599 
600  double frequency = neg_b / a_times_2;
601  computed_period_usec =
602  static_cast<uint64_t>(static_cast<double>(time_span_usec) / frequency);
603  }
604 }
