135 bool StrapUtility::Initialize(const ConfigInfo& ci)
136 {
137  Time now = Time::Now();
138  p_val_ = ci.GetDouble("p", kDefaultPriority, false);
139  scale_factor_ = ((2*p_val_) + 10)/(p_val_ + 10);
140  delta_ = ci.GetDouble("delta", kDefaultMaxLossThreshold,
141  false);
142  restart_interval_sec_ = ci.GetDouble("resint",
143  kDefaultResIntSec,
144  false)/scale_factor_;
145  n_steps_ = ci.GetInt("nsteps", kDefaultNSteps, false);
146  step_interval_sec_ = ci.GetDouble("stepint", kDefaultStepIntSec, false);
147  avg_interval_sec_ = (ci.GetDouble("avgint", kDefaultAvgIntSec, false) +
148  rng_.GetDouble(0.02))*scale_factor_;
149  int ttg_usec = ci.GetInt("ttg", 0, false);
150  time_to_go_ = Time::FromUsec(ttg_usec);
151  inertia_ = ci.GetDouble("inertia", kInertiaInc, false);
152 
153  if (ttg_usec > 0)
154  {
155  avg_interval_sec_ = 6 * static_cast<double>(ttg_usec)/1000000.0;
156  }
157  time_interval_end_ = now.GetTimeInUsec() + avg_interval_sec_*1000000;
158  last_step_time_ = now;
159  last_acked_time_ = now;
160 
161  // Make sure that the is more than 0 steps.
162  if (n_steps_ == 0)
163  {
164  n_steps_ = kDefaultNSteps;
165  LogW(kClassName, __func__, "STRAP utility configured with 0 steps, "
166  "using default value instead\n");
167  }
168 
169  LogC(kClassName, __func__, "STRAP configuration :\n");
170  LogC(kClassName, __func__, "flow id : %" PRIu32 "\n", flow_id_);
171  LogC(kClassName, __func__, "k : %.2e\n",
172  static_cast<double>(k_val_.GetValue()));
173  LogC(kClassName, __func__, "p : %.03f\n", p_val_);
174  LogC(kClassName, __func__, "scale factor : %.03f\n", scale_factor_);
175  LogC(kClassName, __func__, "delta : %.03f\n", delta_);
176  LogC(kClassName, __func__, "Interval length : %.03f\n",
177  avg_interval_sec_);
178  LogC(kClassName, __func__, "Step duration : %.03f\n",
179  step_interval_sec_);
180  LogC(kClassName, __func__, "Number steps : %" PRIu8 "\n", n_steps_);
181  LogC(kClassName, __func__, "Restart interval : %.03f\n",
182  restart_interval_sec_);
183  LogC(kClassName, __func__, "Time to go : %s\n",
184  time_to_go_.ToString().c_str());
185  LogC(kClassName, __func__, "Inertia : %.03f\n", inertia_);
186  LogC(kClassName, __func__, "STRAP configuration complete\n");
187 
188  LogI(kClassName, __func__, "STRAP initialized. Now %" PRId64 " , interval end: "
189  "%" PRId64 "\n", Time::GetNowInUsec(), time_interval_end_);
190 
191  return true;
192 }

195 double StrapUtility::GetSendRate()
196 {
197  if (flow_state_ != FLOW_ON)
198  {
199  LogD(kClassName, __func__, "fid: %" PRIu32 ", is off\n", flow_id_);
200  return 0.0;
201  }
202 
203  double current_backlog = src_info_.cur_backlog_bytes() * 8;
204  return (static_cast<double>(curr_step_)/n_steps_)*
205  (current_backlog*1000000);

209 double StrapUtility::GetAdmFrac()
210 {
211  if (flow_state_ != FLOW_ON)
212  {
213  LogD(kClassName, __func__, "fid: %" PRIu32 ", is off\n", flow_id_);
214  return 0.0;
215  }
216 
217  uint32_t cur_qd = queue_depths_.GetBinDepthByIdx(bin_idx_);
218  if (cur_qd > max_queue_depths_)
219  {
220  max_queue_depths_ = cur_qd;
221  max_queue_time_ms_ = Time::GetNowInUsec();
222  }
223 
224  return static_cast<double>(curr_step_)/n_steps_;
225 }

228 void StrapUtility::Step()
229 {
230  if (flow_state_ != FLOW_ON)
231  {
232  return;
233  }
234 
235  LogD(kClassName, __func__, "Last acked pkt: %" PRIu32 ", curr loss rate: %"
236  PRIu32 "\n", acked_seq_num_, curr_loss_rate_pct_);
237 
238  // If we don't yet have an average of the nominal rate, use a small rate
239  // to avoid incurring penalty until we have a better handle on the rate.
240  double nom_rate = std::max(1.0, src_rate_estimator_.avg_src_rate());
241  uint32_t cur_qd = queue_depths_.GetBinDepthByIdx(bin_idx_);
242  Time now = Time::Now();
243 
244  LogD(kClassName, __func__, "Checking Step for flow % " PRIu32
245  " cur step: % " PRIu8 " \n", flow_id_, curr_step_);
246 
247  // If the queues are large, then we should not be sending. We do this
248  // gradually by stepping down rather than stopping. If the queues are
249  // less tham the k*p/m threshold, then we should step up.
250  if (cur_qd * 8 > k_val_.GetValue() *
251  p_val_ / (nom_rate * (1 - (delta_*inertia_))))
252  {
253  curr_step_ = std::max(0,curr_step_-1);
254  LogD(kClassName, __func__, "Flow %" PRIu32 " Stepping down to %" PRIu8
255  ", using m=%f, inertia=%f.\n", flow_id_, curr_step_, nom_rate, inertia_);
256 
257  // Incur penalty if the backlog is growning.
258  penalty_ += std::max(0.0, (1.0 - GetAdmFrac()) * step_interval_sec_);
259  }
260  else if (time_to_go_.IsZero() || !enable_loss_triage())
261  {
262  curr_step_ = std::min(static_cast<int>(n_steps_), curr_step_ + 1);
263  LogD(kClassName, __func__, "Stepping up to %" PRIu8 ", using m=%f.\n",
264  curr_step_, nom_rate);
265  }
266  else if ((now - last_step_time_) > time_to_go_)
267  {
268  if (curr_loss_rate_pct_ > (delta_*inertia_*100))
269  {
270  curr_step_ = std::max(0,curr_step_-1);
271  LogD(kClassName, __func__, "Flow %" PRIu32 " Stepping down to %" PRIu8
272  ", using m=%f, inertia=%f due to loss %u.\n",
273  flow_id_, curr_step_, nom_rate, inertia_, curr_loss_rate_pct_);
274 
275  // Incur penalty if the backlog is growning.
276  penalty_ += std::max(0.0, (1.0 - GetAdmFrac()) * 2 *
277  (static_cast<double>(time_to_go_.GetTimeInUsec())/1000000));
278  }
279  else
280  {
281  curr_step_ = std::min(static_cast<int>(n_steps_), curr_step_ + 1);
282  }
283  if (last_acked_seq_num_ < acked_seq_num_)
284  {
285  last_acked_time_ = now;
286  }
287 
288  last_step_time_ = now;
289  last_acked_seq_num_ = acked_seq_num_;
290  last_admitted_seq_num_ = admitted_seq_num_;
291 
292  }
293 
294  double UNUSED(send_rate) = (static_cast<float>(curr_step_)/
295  static_cast<float>(n_steps_))*nom_rate;
296 
297  LogD(kClassName, __func__,
298  "At step %" PRIu8 ", for tag %" PRIu32 ", with queue %" PRIu8
299  "B, rate %f.\n",
300  curr_step_, flow_id_, cur_qd, send_rate);
301 
302  // This output is for the netanim trace parser.
303  LogD(kClassName, __func__,
304  "f_id: %" PRIu32 ", queue: %" PRIu32 "b, rate: %.03fbps, "
305  "step:%" PRIu8 "\n", flow_id_, cur_qd*8, send_rate,
306  curr_step_);
307 }

310 bool StrapUtility::CheckUtility()
311 {
312  if (flow_state_ != FLOW_ON)
313  {
314  LogD(kClassName, __func__, "flow %" PRIu32 " is off\n", flow_id_);
315  return false;
316  }
317 
318  Time now;
319  if (!now.GetNow())
320  {
321  LogF(kClassName, __func__, "Failed to get current time\n");
322  return false;
323  }
324 
325  double max_penalty = delta_ * inertia_ * avg_interval_sec_;
326 
327  LogD(kClassName, __func__, "Flow %" PRIu32 ", penalty now: %f, inertia: %f\n",
328  flow_id_, penalty_, inertia_);
329 
330  if (penalty_ > max_penalty)
331  {
332  if (curr_loss_rate_pct_ > delta_*inertia_*100)
333  {
334  flow_state_ = LOSS_TRIAGED;
335  }
336  else
337  {
338  flow_state_ = FLOW_TRIAGED;
339  }
340  current_utility_ = 0;
341  return true;
342  }
343  else
344  {
345  // The flow is being properly serviced, we get full utility and
346  // increased inertia.
347  current_utility_ = p_val_;
348  }
349 
350  // Reset the counters if the averaging interval has ended.
351  if (now.GetTimeInUsec() > time_interval_end_)
352  {
353  penalty_ = 0;
354  time_interval_end_ = now.GetTimeInUsec() + (avg_interval_sec_ * 1000000);
355  if (max_queue_time_ms_ <
356  (now.GetTimeInUsec() - (avg_interval_sec_ * 1000000)))
357  {
358  inertia_ = std::min(1.0, inertia_ + kInertiaInc);
359  }
360  }
361  return false;
362 }

366 {
367  Time now = Time::Now();
368 
369  // If the flow is already on, we don't need to do anything.

376

378

382  curr_loss_rate_pct_ = 0;
