73  // TODO: These initializations should be inside of an Initialize() method,
74  // not the constructor, as they can fail.
75  if (!local_ls_queue_depths_.Initialize(bin_map_))
76  {
77  LogF(kClassName, __func__, "Unable to initialize local latency-sensitive "
78  "queue depths array.\n");
79  }
80  local_ls_queue_depths_.Clear(0);
81 
82  // TODO: The initialization of local_queue_depths_ is not necessary if using
83  // shared memory. However, the APIs do not allow avoiding this call (and
84  // the associated memory allocations) when using shared memory, as there is
85  // no "normal" initialization method. There really should be two different
86  // flavors of initialization methods, one for local memory direct access and
87  // one for shared memory direct access. Note than a complication for all of
88  // this is the need to have GetShmSize() working properly before a call to
89  // InitializeShmDirectAccess() can be made.
90  if (!local_queue_depths_.Initialize(bin_map_))
91  {
92  LogF(kClassName, __func__, "Unable to initialize local queue depths "
93  "array.\n");
94  }
95  local_queue_depths_.Clear(0);
96 
97  // Note that this initialization has to occur here in order for the
98  // GetShmSize() to work properly before calls to InitializeShmDirectAccess()
99  // can be made.
100  if (!shm_queue_depths_.Initialize(bin_map_))
101  {
102  LogF(kClassName, __func__, "Unable to initialize shared memory queue "
103  "depths array.\n");
104  }

118 bool QueueDepths::InitializeShmDirectAccess(SharedMemoryIF* shm)
119 {
120  uint32_t wait_count = 0;
121 
122  if (shm == NULL)
123  {
124  return false;
125  }
126 
127  // Wait until the shared memory is initialized, as required by
128  // SetShmDirectAccess().
129  while (!shm->IsInitialized())
130  {
131  sleep(1);
132  ++wait_count;
133 
134  if (wait_count % 10 == 0)
135  {
136  if (wait_count % 120 == 0)
137  {
138  LogW(kClassName, __func__, "... Still trying to attach to shared "
139  "memory queue depths (%" PRIu32 " s).\n", wait_count);
140  }
141  else
142  {
143  LogD(kClassName, __func__, "... Waiting to attach to shared memory "
144  "queue depths.\n");
145  }
146  }
147  }
148 
149  // Set up the shared memory queue depths array to point to the shared memory
150  // segment.
151  if (!shm_queue_depths_.SetShmDirectAccess(*shm))
152  {
153  LogE(kClassName, __func__, "Unable to set shared memory direct access on "
154  "queue depths array.\n");
155  return false;
156  }
157 
158  // Clear the shared memory queue depths.
159  shm_queue_depths_.Clear(0);
160 
161  // Update to use shared memory directly.
162  access_shm_directly_ = true;
163  shm_if_ = shm;
164 
165  return true;
166 }

256 void QueueDepths::Increment(BinIndex bin_idx, uint32_t incr_amt_bytes,
257  uint32_t ls_incr_amt_bytes)
258 {
259  if (ls_incr_amt_bytes > incr_amt_bytes)
260  {
261  LogF(kClassName, __func__, "Latency-sensitive adjustment amount %" PRIu32
262  "B cannot be more than normal latency adjustment amount %" PRIu32
263  "B.\n", ls_incr_amt_bytes, incr_amt_bytes);
264  return;
265  }
266 
267  // Note that there is no need to check the latency-sensitive (LS) queue
268  // depths for overflow, because we maintain the invariant that LS queue
269  // depth < NORMAL queue depth, and we already checked that we aren't
270  // incrementing LS by more than we're incrementing NORMAL.
271 
272  IntLock();
273 
274 #ifdef SHM_STATS
275  bool updated = false;
276 #endif /* SHM_STATS */
277 
278  uint32_t curr_depth = IntGet(bin_idx);
279 
280  if ((UINT32_MAX - incr_amt_bytes) < curr_depth)
281  {
282  LogD(kClassName, __func__, "Unable to increment Queue depth for bin %s "
283  "by %" PRIu32 " because current depth (%" PRIu32 ") is too large.\n",
284  bin_map_.GetIdToLog(bin_idx).c_str(), incr_amt_bytes, curr_depth);
285  IntSet(bin_idx, UINT32_MAX);
286  }
287  else
288  {
289  IntSet(bin_idx, (curr_depth + incr_amt_bytes));
290  local_ls_queue_depths_[bin_idx] += ls_incr_amt_bytes;
291 
292 #ifdef SHM_STATS
293  updated = true;
294 #endif /* SHM_STATS */
295  }
296 
297  IntUnlock();
298 
299 #ifdef SHM_STATS
300  if ((shm_stats_ != NULL) && updated)
301  {
302  shm_stats_->DepthChanged(bin_idx, (curr_depth + incr_amt_bytes));
303  }
304 #endif /* SHM_STATS */
305 }

308 void QueueDepths::Decrement(BinIndex bin_idx, uint32_t decr_amt_bytes,
309  uint32_t ls_decr_amt_bytes)
310 {
311  if (ls_decr_amt_bytes > decr_amt_bytes)
312  {
313  LogF(kClassName, __func__, "Latency-sensitive adjustment amount %" PRIu32
314  "B cannot be more than normal latency adjustment amount %" PRIu32
315  "B.\n", ls_decr_amt_bytes, decr_amt_bytes);
316  return;
317  }
318 
319  IntLock();
320 
321  uint32_t curr_depth = IntGet(bin_idx);
322 
323  // Since queue depths are unsigned, make sure we don't wrap.
324  if (decr_amt_bytes > curr_depth)
325  {
326  LogW(kClassName, __func__, "Attempting to decrement queue depth for Bin "
327  "%s below 0 from %" PRIu32 "B.\n",
328  bin_map_.GetIdToLog(bin_idx).c_str(), curr_depth);
329  IntSet(bin_idx, 0);
330  curr_depth = 0;
331  }
332  else
333  {
334  IntSet(bin_idx, (curr_depth - decr_amt_bytes));
335  curr_depth -= decr_amt_bytes;
336  }
337 
338  IntUnlock();
339 
340  // Need to check LS overflow separately, because LS depth is less than full
341  // depth.
342  uint32_t ls_curr_depth = local_ls_queue_depths_[bin_idx];
343 
344  if (ls_decr_amt_bytes > ls_curr_depth)
345  {
346  LogW(kClassName, __func__, "Attempting to decrement LS queue depth for "
347  "Bin %s below 0 from %" PRIu32 "B.\n",
348  bin_map_.GetIdToLog(bin_idx).c_str(), ls_curr_depth);
349  local_ls_queue_depths_[bin_idx] = 0;
350  }
351  else
352  {
353  local_ls_queue_depths_[bin_idx] = (ls_curr_depth - ls_decr_amt_bytes);
354  }
355 
356 #ifdef SHM_STATS
357  if (shm_stats_ != NULL)
358  {
359  shm_stats_->DepthChanged(bin_idx, curr_depth);
360  }
361 #endif /* SHM_STATS */
362 }

410 size_t QueueDepths::Serialize(uint8_t* buf, size_t max_len,
411  uint8_t& num_pairs)
412 {
413  size_t length = 0;
414  uint32_t bin_depth_nbo = 0;
415 
416  num_pairs = 0;
417 
418  if (access_shm_directly_)
419  {
420  LogF(kClassName, __func__, "Cannot call Serialize on a shared memory "
421  "direct access queue depths object.\n");
422  return 0;
423  }
424 
425  // Check that a buffer has been specified.
426  if (buf == NULL)
427  {
428  LogE(kClassName, __func__, "Missing buffer.\n");
429  return 0;
430  }
431 
432  // Add all of the (bin,depth,ls_depth) tuples.
433  BinIndex bin_idx = 0;
434 
435  for (bool more_bin_idx = bin_map_.GetFirstUcastBinIndex(bin_idx);
436  more_bin_idx;
437  more_bin_idx = bin_map_.GetNextUcastBinIndex(bin_idx))
438  {
439  BinId bin_id = bin_map_.GetPhyBinId(bin_idx);
440  uint32_t depth = local_queue_depths_[bin_idx];
441  uint32_t ls_depth = local_ls_queue_depths_[bin_idx];
442 
443  if (depth < ls_depth)
444  {
445  LogD(kClassName, __func__, "LS queue depths %" PRIu32 "B is larger "
446  "than all queue depths %" PRIu32 "B, wrong if not HvyBall or "
447  "EWMA.\n", ls_depth, depth);
448  }
449 
450  if ((depth + ls_depth) == 0)
451  {
452  continue;
453  }
454 
455  // Verify that this addition will not go off of the end of the buffer.
456  // Given the checks above, this should never happen.
457  // Dest bin id: 1B, Queue Depth: 4B, LS Queue Depth: 4B.
458  if ((length + sizeof(uint8_t) + (2 * sizeof(uint32_t))) > max_len)
459  {
460  LogW(kClassName, __func__, "Serialization of %" PRIu8 "th group would "
461  "overshoot max length %zuB. Fail.\n", (num_pairs + 1), max_len);
462  return 0;
463  }
464 
465  // Copy the destination bin id.
466  *buf = static_cast<uint8_t>(bin_id);
467  buf += sizeof(uint8_t);
468  length += sizeof(uint8_t);
469 
470  // Copy in the bin depth in bytes.
471  bin_depth_nbo = htonl(depth);
472  std::memcpy(buf, &bin_depth_nbo, sizeof(bin_depth_nbo));
473  buf += sizeof(bin_depth_nbo);
474  length += sizeof(bin_depth_nbo);
475 
476  // Copy in the latency-sensitive-only bin depth in bytes.
477  bin_depth_nbo = htonl(ls_depth);
478  std::memcpy(buf, &bin_depth_nbo, sizeof(bin_depth_nbo));
479  buf += sizeof(bin_depth_nbo);
480  length += sizeof(bin_depth_nbo);
481 
482  ++num_pairs;
483 
484  LogD(kClassName, __func__, "Bin ID %" PRIBinId " (Index %" PRIBinIndex
485  ", translates to %s) depth: %" PRIu32 "B ls-depth: %" PRIu32
486  "B added to QLAM.\n", bin_id, bin_idx,
487  bin_map_.GetIdToLog(bin_idx).c_str(), depth, ls_depth);
488  }
489 
490  return length;
491 }

494 size_t QueueDepths::Deserialize(const uint8_t *buf, size_t len,
495  uint8_t num_pairs)
496 {
497  uint8_t dst_bin_id = 0;
498  BinIndex dst_bin_idx = 0;
499  uint32_t q_depth = 0;
500  uint32_t ls_q_depth = 0;
501  size_t length = 0;
502 
503  if (access_shm_directly_)
504  {
505  LogF(kClassName, __func__, "Cannot call Deserialize on a shared memory "
506  "direct access queue depths object.\n");
507  return 0;
508  }
509 
510  // Check that a buffer has been specified.
511  if (buf == NULL)
512  {
513  LogE(kClassName, __func__, "Missing buffer.\n");
514  return 0;
515  }
516 
517  if (num_pairs == 0)
518  {
519  LogD(kClassName, __func__, "Deserializing queue depth with 0 pairs.\n");
520  }
521 
522  // Check that the number of pairs matches the buffer length given.
523  // There are 1B dest bin id, 4B queue depth, 4B LS queue depth per pair.
524  if (len < (length + (num_pairs * (sizeof(uint8_t) + sizeof(uint32_t) +
525  sizeof(uint32_t)))))
526  {
527  LogE(kClassName, __func__, "%" PRIu8 " (dst bin id, depth) pairs would "
528  "exceed remaining buffer length %zuB.\n", num_pairs, len);
529  return 0;
530  }
531 
532  // It is a new QLAM, so clear all (bin,depth,ls_depth) tuple in the object.
533  local_queue_depths_.Clear(0);
534 
535  // Parse the remainder of the buffer to get the (bin,depth) pairs.
536  for (uint16_t num_bins_created = 0; num_bins_created < num_pairs;
537  ++num_bins_created)
538  {
539  // Verify that this parsing will not go off of the end of the buffer.
540  // Given the checks above, this should never happen.
541  if (len < (length + sizeof(uint8_t) + sizeof(uint32_t) +
542  sizeof(uint32_t)))
543  {
544  LogF(kClassName, __func__, "Max length %zuB exceeded after %" PRIu16
545  " pairs with %zuB.\n", len, num_bins_created,
546  (length + sizeof(uint8_t) + sizeof(uint32_t) + sizeof(uint32_t)));
547  return 0;
548  }
549 
550  // Read the destination bin. Note that these are all destination bins,
551  // not multicast group bins, since Deserialize is called once we're inside
552  // the portion of the QLAM for a single multicast group.
553  dst_bin_id = *buf;
554  buf += sizeof(dst_bin_id);
555  length += sizeof(dst_bin_id);
556 
557  // Read the queue depth for this destination.
558  std::memcpy(&q_depth, buf, sizeof(q_depth));
559  q_depth = ntohl(q_depth);
560  buf += sizeof(q_depth);
561  length += sizeof(q_depth);
562 
563  // Read the LS queue depth for this destination.
564  std::memcpy(&ls_q_depth, buf, sizeof(ls_q_depth));
565  ls_q_depth = ntohl(ls_q_depth);
566  buf += sizeof(ls_q_depth);
567  length += sizeof(ls_q_depth);
568 
569  // Make sure the bin exists.
570  dst_bin_idx = bin_map_.GetPhyBinIndex(dst_bin_id);
571 
572  if ((dst_bin_idx == kInvalidBinIndex) ||
573  (!bin_map_.IsUcastBinIndex(dst_bin_idx)))
574  {
575  LogW(kClassName, __func__, "Invalid unicast bin_id %" PRIu32 ".\n",
576  dst_bin_id);
577  }
578  else
579  {
580  if (ls_q_depth > q_depth)
581  {
582  LogD(kClassName, __func__, "Latency-sensitive-only depth %" PRIu32
583  "B is larger than overall depth %" PRIu32 "B in QLAM for bin id "
584  "%" PRIu8 ", wrong if not HvyBall or EWMA.\n", ls_q_depth,
585  q_depth, dst_bin_id);
586  }
587 
588  local_queue_depths_[dst_bin_idx] = q_depth;
589  local_ls_queue_depths_[dst_bin_idx] = ls_q_depth;
590 
591  LogD(kClassName, __func__, "Dest bin id %" PRIBinId ": Q Depth: %"
592  PRIu32 "B | LS Q Depth: %" PRIu32 "B.\n", dst_bin_id, q_depth,
593  ls_q_depth);
594  }
595  }
596 
597  return length;
598 }

655 {
656  std::stringstream ret_ss;
657 
658  IntLock();
659 
660  BinIndex bin_idx = 0;
661 
662  // Append the bin:depth tuples in a long string. All multicast bins should
