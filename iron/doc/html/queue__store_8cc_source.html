142 bool QueueStore::Initialize(const ConfigInfo& config_info,
143  BinIndex node_bin_idx)
144 {
145  // The bin queue mgrs are allocated immediately because the constructor is
146  // followed by a call to Initialize with the config info, which is not
147  // retained. The config info contains items used to initialized each queue
148  // set. In the future, we may decide to retain the config info object in
149  // order to allocate bin queue mgrs on demand.
150  if (!q_mgrs_.Initialize(bin_map_))
151  {
152  LogF(kClassName, __func__, "Unable to initialize queue managers "
153  "array.\n");
154  return false;
155  }
156  q_mgrs_.Clear(NULL);
157 
158  string q_mgr_alg = config_info.Get("Bpf.Alg.QDMgr", kDefaultBpfQMgr);
159 
160  use_anti_starvation_zombies_ =
161  config_info.GetBool("Bpf.UseAntiStarvationZombies",
162  kDefaultUseAntiStarvationZombies);
163 
164  // Test for NPLB combined with ASAP (illegal) outside the for loop.
165  if (q_mgr_alg == "NPLB")
166  {
167  if (use_anti_starvation_zombies_)
168  {
169  LogF(kClassName, __func__,
170  "Cannot use NPLB with Anti Starvation Zombies.\n");
171  return false;
172  }
173  }
174 
175  // We use the hysteresis as the minimum value that needs to be overcome for
176  // anti-starvation. (ProcessGradientUpdate uses this.)
177  hysteresis_ =
178  config_info.GetInt("Bpf.Alg.HysteresisBytes", kBpfAlgHysteresisBytes);
179 
180  // Initialize the neighbor virtual queue depths array.
181  if (!nbr_virtual_queue_depths_.Initialize(bin_map_))
182  {
183  LogF(kClassName, __func__, "Unable to initialize neighbor virtual queue "
184  "depths array.\n");
185  return false;
186  }
187  nbr_virtual_queue_depths_.Clear(NULL);
188 
189  q_mgrs_.Clear(NULL);
190 
191  // Create the bin queue managers for each unicast or multicast destination
192  // bin index.
193  BinIndex idx = 0;
194 
195  for (bool valid = bin_map_.GetFirstDstBinIndex(idx);
196  valid;
197  valid = bin_map_.GetNextDstBinIndex(idx))
198  {
199  if (q_mgr_alg == "HvyBall")
200  {
201  q_mgrs_[idx] = new (std::nothrow) HvyballBinQueueMgr(
202  idx, packet_pool_, bin_map_);
203  }
204  else if (q_mgr_alg == "EWMA")
205  {
206  q_mgrs_[idx] = new (std::nothrow) EWMABinQueueMgr(
207  idx, packet_pool_, bin_map_);
208  }
209  else if (q_mgr_alg == "NPLB")
210  {
211  q_mgrs_[idx] = new (std::nothrow) NPLBBinQueueMgr(
212  idx, packet_pool_, bin_map_);
213  }
214  else
215  {
216  q_mgrs_[idx] = new (std::nothrow) BinQueueMgr(
217  idx, packet_pool_, bin_map_);
218  }
219 
220  if (!q_mgrs_[idx])
221  {
222  LogF(kClassName, __func__,
223  "Failed to allocate bin queue mgr object of size at least %zdB.\n",
224  sizeof(BinQueueMgr));
225  return false;
226  }
227  }
228 
229  // Create the shared memory segments, starting with queue depths for weights.
230  key_t sem_key = config_info.GetUint("Bpf.Weight.SemKey",
231  kDefaultWeightSemKey);
232  string name = config_info.Get("Bpf.Weight.ShmName",
233  kDefaultWeightShmName);
234 
235  // Find a place to store the queue depths to share, allocate
236  // enough shared memory for each group, etc.
237  if (!weight_qd_shared_memory_.Create(
238  sem_key, name.c_str(),
239  proxy_depths_.GetShmSize()))
240  {
241  LogF(kClassName, __func__,
242  "Failed to create the shared memory segment for weights.\n");
243  return false;
244  }
245 
246  // Initialize the bin queue managers for each unicast or multicast
247  // destination bin index.
248  for (bool valid = bin_map_.GetFirstDstBinIndex(idx);
249  valid;
250  valid = bin_map_.GetNextDstBinIndex(idx))
251  {
252  if (!q_mgrs_[idx] || !q_mgrs_[idx]->Initialize(config_info, node_bin_idx))
253  {
254  LogF(kClassName, __func__,
255  "Failed to initialize bin queue mgr for %s bin id %" PRIBinIndex
256  " at %p.\n",
257  (bin_map_.IsMcastBinIndex(idx) ? "mcast" : "ucast"), idx,
258  q_mgrs_[idx]);
259  return false;
260  }
261  }
262  proxy_depths_.InitializeShmStats();
263 
264  if (!max_gradient_set_.Initialize(bin_map_))
265  {
266  LogF(kClassName, __func__, "Unable to initialize maximum gradient set "
267  "array.\n");
268  return false;
269  }
270 
271  if (!max_gradient_val_.Initialize(bin_map_))
272  {
273  LogF(kClassName, __func__, "Unable to initialize maximum gradient value "
274  "array.\n");
275  return false;
276  }
277 
278  LogC(kClassName, __func__,
279  "Bpf.Weight.SemKey : %" PRId32 "\n",
280  static_cast<int32_t>(sem_key));
281  LogC(kClassName, __func__,
282  "Bpf.Weight.ShmName : %s\n", name.c_str());
283  LogC(kClassName, __func__,
284  "Bpf.Alg.QDMgr : %s\n",
285  q_mgr_alg.c_str());
286  LogC(kClassName, __func__,
287  "Bpf.UseAntiStarvationZombies : %s\n",
288  (use_anti_starvation_zombies_ ? "Enabled" : "Disabled"));
289 
290  return true;
291 }

451 void QueueStore::SetASAPCap(
452  OrderedList<Gradient, int64_t>& gradients,
453  bool is_ls)
454 {
455  if (!use_anti_starvation_zombies_)
456  {
457  return;
458  }
459 
460  // Compute overall max
461  int64_t max_gradient = 0;
462  Gradient grad;
463  if (gradients.Peek(grad))
464  {
465  max_gradient = grad.value;
466  }
467  // \todo This is here to mimic the original ASAP implementation, but it's
468  // not clear why we're doing this. (And why <0, not <=0). Needs thought and
469  // at least an explanatory comment.
470  if (max_gradient < 0)
471  {
472  max_gradient = 1;
473  }
474 
475  // This will skim through all the gradients and find the max value for each
476  // bin (which could be the gradient to any neighbor).
477  max_gradient_set_.Clear(false);
478  max_gradient_val_.Clear(0);
479 
480  // Find max gradient for this bin
481  OrderedList<Gradient, int64_t>::WalkState grad_ws;
482  grad_ws.PrepareForWalk();
483  Gradient gradient;
484  while (gradients.GetNextItem(grad_ws, gradient))
485  {
486  if ((!max_gradient_set_[gradient.bin_idx]) ||
487  (gradient.value > max_gradient_val_[gradient.bin_idx]))
488  {
489  max_gradient_val_[gradient.bin_idx] = gradient.value;
490  max_gradient_set_[gradient.bin_idx] = true;
491  }
492  }
493 
494  // Note that the max gradient will remain 0 if there was no gradient from
495  // the forwarding algorithm. This includes bins for which all possible
496  // egress links have busy CATs as well as bins for which the gradients to
497  // all neighbors are 0 or negative.
498  //
499  // \todo It's questionable whether this is the right thing to do in that
500  // case. If the gradient is 0 or negative, perhaps we do want to add ASAP
501  // zombies to build up the gradient to make it positive - because packets
502  // may be starved because of a large (potentially lower priority) flow
503  // entering at the neighbor node to the same bin. However, if the gradient
504  // is missing because all egress links are full, adding ASAP zombies is
505  // pretty clearly not the right thing to do.
506 
507  // If there's a hysteresis set, then even if the gradient is 0, we will
508  // still need queue depths of at least the hysteresis to avoid starvation.
509  // This must be done for all unicast and multicast bin indexes.
510  BinIndex idx = 0;
511 
512  for (bool valid = bin_map_.GetFirstDstBinIndex(idx);
513  valid;
514  valid = bin_map_.GetNextDstBinIndex(idx))
515  {
516  if (max_gradient_val_[idx] <= hysteresis_)
517  {
518  max_gradient_val_[idx] = hysteresis_ + 1;
519  }
520  // Now use max_gradient_val_ to hold the initial cap on ASAP additions.
521  // Just use the hysteresis + 1 if the max is smaller than this. That will
522  // let any bin experiencing starvation add enough zombies to overcome the
523  // hysteresis, even if that means jumping past the max gradient.
524  if (max_gradient >= hysteresis_)
525  {
526  max_gradient_val_[idx] = max_gradient - max_gradient_val_[idx];
527  }
528  if (max_gradient_val_[idx] < 0)
529  {
530  LogW(kClassName, __func__, "Inconsistency - zombie cap negative\n");
531  max_gradient_val_[idx] = 0;
532  }
533  else if (max_gradient_val_[idx] * 1.05 > UINT32_MAX)
534  {
535  LogW(kClassName, __func__,
536  "Difference in gradients would overflow max uint32\n.");
537  max_gradient_val_[idx] = UINT32_MAX;
538  q_mgrs_[idx]->SetASAPCap(UINT32_MAX, is_ls);
539  return;
540  }
541  // \todo Why multiply by 1.05 instead of, for instance, adding 1?
542  q_mgrs_[idx]->SetASAPCap(
543  static_cast<uint32_t>(max_gradient_val_[idx] * 1.05), is_ls);
544  LogD(kClassName, __func__, "%sZombie cap set to %" PRIu32 " for bin %s"
545  ", max grad is %" PRIi64 ".\n",
546  (is_ls ? "LS " : ""),
547  static_cast<uint32_t>(max_gradient_val_[idx] * 1.05),
548  bin_map_.GetIdToLog(idx).c_str(),
549  max_gradient);
550  }
551 }
