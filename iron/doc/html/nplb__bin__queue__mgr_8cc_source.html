214 void NPLBBinQueueMgr::IncrementDelayStickiness(BinIndex bin_idx,
215  Time dequeued_pkt_recv_time)
216 {
217  // MCAST TODO: do we need the bin id?
218  // This is a private function. We can assume we've already checked the
219  // validity of the bin id when we were dequeuing the packet.
220 
221  Time oldest_pkt_recv_time = Time::Now();
222  Packet* pkt = NULL;
223  uint32_t oldest_pkt_dst_addr = 0;
224 
225  // Find the oldest enqueued packet (which could be the next packet in any
226  // latency queue). We ignore the high latency (aka zombie) queue, since
227  // these are not real data packets.
228  for (uint8_t it = 0; it < NUM_LATENCY_DEF; ++it)
229  {
230  if (Packet::IsZombie(static_cast<LatencyClass>(it)) ||
231  IS_PKTLESS_Z_QUEUE[it])
232  {
233  // Ignore zombie packets.
234  continue;
235  }
236  PacketQueue* queue =
237  static_cast<PacketQueue*>(phy_queue_.lat_queues[it]);
238  if (!queue)
239  {
240  // No queue, nothing to consider.
241  continue;
242  }
243  pkt = queue->Peek();
244  if (!pkt)
245  {
246  // Nothing in the queue, nothing to consider
247  continue;
248  }
249  if (pkt->recv_time() < oldest_pkt_recv_time)
250  {
251  oldest_pkt_recv_time = pkt->recv_time();
252  pkt->GetIpDstAddr(oldest_pkt_dst_addr);
253  }
254  }
255  Time diff = oldest_pkt_recv_time - dequeued_pkt_recv_time;
256  if (diff > delay_stickiness_threshold_)
257  {
258  uint32_t stickiness_incr = static_cast<uint32_t>(
259  (diff - delay_stickiness_threshold_).GetTimeInUsec());
260 
261  if (oldest_pkt_dst_addr != 0)
262  {
263  AddNewZombie(oldest_pkt_dst_addr, stickiness_incr, HIGH_LATENCY_NPLB);
264  }
265  }
266 }

269 QueueDepths* NPLBBinQueueMgr::ComputeNPLB()
270 {
271  if (!initialized_ || delay_weight_ == 0)
272  {
273  // Don't bother updating the delay portion of the values if we're not
274  // going to use them.
275  return &queue_depths_;
276  }
277 
278  uint32_t depth = 0;
279  uint32_t ls_depth = 0;
280  Time now = Time::Now();
281  Packet* pkt = NULL;
282  uint32_t nplb_zombies = 0;
283 
284  BinIndex bin_idx = 0;
285 
286  for (bool valid = bin_map_.GetFirstUcastBinIndex(bin_idx);
287  valid;
288  valid = bin_map_.GetNextUcastBinIndex(bin_idx))
289  {
290  depth = queue_depths_.GetBinDepthByIdx(bin_idx, NORMAL_LATENCY);
291  ls_depth = queue_depths_.GetBinDepthByIdx(bin_idx, LOW_LATENCY);
292 
293  Time oldest_pkt_recv_time = now;
294  Time oldest_ls_pkt_recv_time = now;
295 
296  // Find the oldest enqueued packet (which could be the next packet in any
297  // latency queue). We ignore the zombie queue, since these are not real
298  // data packets.
299  for (uint8_t it = 0; it < NUM_LATENCY_DEF; ++it)
300  {
301  if (Packet::IsZombie(static_cast<LatencyClass>(it)) ||
302  IS_PKTLESS_Z_QUEUE[it])
303  {
304  if (it == HIGH_LATENCY_NPLB)
305  {
306  nplb_zombies = phy_queue_.lat_queues[it]->GetSize();
307  }
308  // Ignore zombie packets.
309  continue;
310  }
311  PacketQueue* queue =
312  static_cast<PacketQueue*>(phy_queue_.lat_queues[it]);
313  if (!queue)
314  {
315  // No queue, nothing to consider.
316  continue;
317  }
318  pkt = queue->Peek();
319  if (!pkt)
320  {
321  // Nothing in the queue, nothing to consider
322  continue;
323  }
324 
325  if (it <= LOW_LATENCY)
326  {
327  if (pkt->recv_time() < oldest_ls_pkt_recv_time)
328  {
329  oldest_ls_pkt_recv_time = pkt->recv_time();
330  }
331  }
332 
333  if (pkt->recv_time() < oldest_pkt_recv_time)
334  {
335  oldest_pkt_recv_time = pkt->recv_time();
336  }
337  }
338  // Note: if all queues were empty, then depth = 0 and oldest_pkt_recv_time
339  // is still Now, so we'll set the depth to
340  // 0 + (weight * bp_delay_stickiness_.stickiness_).
341  uint32_t delay = static_cast<uint32_t>(
342  (now - oldest_pkt_recv_time).GetTimeInUsec());
343  uint32_t adjusted_depth = depth + (delay_weight_ * delay);
344  nplb_values_.SetBinDepthByIdx(bin_idx, adjusted_depth);
345  delay = static_cast<uint32_t>(
346  (now - oldest_ls_pkt_recv_time).GetTimeInUsec());
347  uint32_t adjusted_ls_depth = ls_depth + (delay_weight_ * delay);
348  nplb_values_.SetBinDepthByIdx(bin_idx, adjusted_ls_depth, LOW_LATENCY);
349 
350  if (adjusted_ls_depth > adjusted_depth)
351  {
352  LogW(kClassName, __func__,
353  "LS adjusted depth is larger than for all %" PRIu32 "B %" PRIu32 "B.\n",
354  adjusted_ls_depth, adjusted_depth);
355  }
356  if (nplb_xplot_[bin_idx])
357  {
358  nplb_xplot_[bin_idx]->ContinueTimeLine(
359  0,
360  depth - nplb_zombies,
361  YELLOW);
362  nplb_xplot_[bin_idx]->ContinueTimeLine(
363  1,
364  depth + (delay_weight_ * delay),
365  RED);
366  }
367  }
368  return &nplb_values_;
369 }
