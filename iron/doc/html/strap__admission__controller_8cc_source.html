80 bool StrapAdmissionController::CreateUtilityFn(string& utility_def,
81  uint32_t flow_id,
82  QueueDepths& queue_depths)
83 {
84  // Validate that the provided utility definition string is for the Strap
85  // utility.
86  if (GetUtilityFnType(utility_def, flow_id) != "STRAP")
87  {
88  LogW(kClassName, __func__, "fid: %" PRIu32 ", utility definition is not "
89  "for a Strap utility function.\n", flow_id);
90  return false;
91  }
92 
93  strap_utility_ = new (std::nothrow) StrapUtility(src_rate_estimator_,
94  src_info_,
95  queue_depths,
96  encoding_state_.bin_idx(),
97  encoding_state_.k_val(),
98  flow_id);
99  if (strap_utility_ == NULL)
100  {
101  LogF(kClassName, __func__, "fid: %" PRIu32 "Unable to allocate memory "
102  "for Strap utility function.\n", flow_id);
103  return false;
104  }
105 
106  ConfigureUtilityFn(strap_utility_, utility_def);
107 
108  strap_utility_->set_enable_loss_triage(
109  encoding_state_.udp_proxy()->enable_loss_triage());
110 
111  ScheduleStepTime();
112 
113  return true;
114 }

117 void StrapAdmissionController::SvcEvents(Time& now)
118 {
119  // First, service the admission event.
120  SvcAdmissionEvent(now, strap_utility_);
121 
122  // Now, service the Strap admission controller specific events.
123  if (restart_time_ <= now)
124  {
125  RestartTimeout(now);
126  }
127 
128  if (step_time_ <= now)
129  {
130  StepTimeout(now);
131  }

146 void StrapAdmissionController::set_flow_state(iron::FlowState flow_state)
147 {
148  if (strap_utility_ == NULL)
149  {
150  // If we get here and don't have a utility function something is very very
151  // wrong.
152  LogF(kClassName, __func__, "Attempting to set the flow's state and there "
153  "isn't a utility function.\n");
154  return;
155  }
156 
157  Time now = Time::Now();
158 
159  if ((flow_state == iron::FLOW_OFF) || (flow_state == iron::FLOW_TRIAGED))
160  {
161  CancelScheduledEvent(next_admission_time_);
162  CancelScheduledEvent(step_time_);
163 
164  if (flow_state == iron::FLOW_OFF)
165  {
166  CancelScheduledEvent(restart_time_);
167  }
168 
169  if ((flow_state == iron::FLOW_TRIAGED) &&
170  (strap_utility_->flow_state() == iron::FLOW_ON))
171  {
172  ScheduleRestartTime();
173  }
174  }
175  else if (flow_state == iron::FLOW_ON)
176  {
177  start_time_ = now;
178  next_admission_time_ = now;
179  strap_utility_->SetAvgIntervalEnd();
180  ScheduleStepTime();
181  CancelScheduledEvent(restart_time_);
182  }
183 
184  strap_utility_->set_flow_state(flow_state);
185 }

218 void StrapAdmissionController::AdmitPkts(iron::Time& now)
219 {
220  LogD(kClassName, __func__,
221  "fid: %" PRIu32 ", adm frac: %f, backlog: %" PRIu32 ".\n",
222  encoding_state_.flow_tag(), strap_utility_->GetAdmFrac(),
223  src_info_.cur_backlog_bytes());
224 
225  double bytes_to_send = strap_utility_->GetAdmFrac() *
226  src_info_.cur_backlog_bytes();
227  size_t bytes_sent = 0;
228  if (strap_utility_->curr_step() > 0)
229  {
230  while (bytes_to_send > bytes_sent)
231  {
232  bytes_sent += AdmitPkt();
233  if (bytes_sent == 0)
234  {
235  break;
236  }
237  }
238  }
239  next_admission_time_ = now.Add(bpf_min_burst_);
240 }

242 void StrapAdmissionController::UpdateUtilityFn(std::string key_val)
243 {
244  List<string> tokens;
245  StringUtils::Tokenize(key_val, ":", tokens);
246  if (tokens.size() != 2)
247  {
248  LogE(kClassName, __func__, "Parameter %s must be of the form key:value.\n",
249  key_val.c_str());
250  return;
251  }
252 
253  string key;
254  string value;
255  tokens.Peek(key);
256  tokens.PeekBack(value);
257 
258  if (key == "delta")
259  {
260  // This is currently used by AMP for making probes less aggressive.
261  // STRAP utility has its own delta management that makes probing less
262  // agressive without AMP's aid. Should this change, uncomment the
263  // following line.
264  //strap_utility_->set_delta(StringUtils::GetDouble(value));
265  return;
266  }
267  else if (key == "p")
268  {
269  strap_utility_->set_priority(StringUtils::GetDouble(value,0.0));
270  }
271  else
272  {
273  LogE(kClassName, __func__, "Update of %s not supported.\n", key.c_str());
274  }
275 }

278 void StrapAdmissionController::RestartTimeout(Time& now)
279 {
280  // Turn the flow on. All packets in the backlog are flushed because they are
281  // old.
282  strap_utility_->SetFlowOn();
283  strap_utility_->ResetInertia();
284  encoding_state_.FlushBacklog();
285 
286  // Adjust next admission time to be now. We are turning the flow back on so
287  // we can send any packets that are in the proxy queue.
288  next_admission_time_ = now;
289  SvcAdmissionEvent(now, strap_utility_);
290 
291  // Set the end time for the current interval.
292  strap_utility_->SetAvgIntervalEnd();
293 
294  // Adjust the event timers.
295  ScheduleStepTime();
296  CancelScheduledEvent(restart_time_);
297 }

300 void StrapAdmissionController::StepTimeout(Time& now)
301 {
302  // Update the receiver stats.
303  strap_utility_->set_acked_seq_num(encoding_state_.acked_seq_num());
304  strap_utility_->set_curr_loss_rate_pct(encoding_state_.loss_rate_pct());
305 
306  uint8_t prev_step = strap_utility_->curr_step();
307  strap_utility_->Step();
308  ScheduleStepTime();
309  uint8_t curr_step = strap_utility_->curr_step();
310 
311  // Check if the flow toggled on/off.
312  if ((curr_step == 0) && (prev_step != 0) && (flow_state() != iron::FLOW_OFF))
313  {
314  LogD(kClassName, __func__, "fid: %" PRIu32 " toggled down.\n",
315  encoding_state_.flow_tag());
316  ++toggle_count_;
317  push_stats_ = true;
318  }
319  else if ((curr_step != 0) && (prev_step == 0.0) &&
320  (start_time_ < (now - kStartupTime)))
321  {
322  LogD(kClassName, __func__, "fid: %" PRIu32 " toggled up.\n",
323  encoding_state_.flow_tag());
324  ++toggle_count_;
325  push_stats_ = true;
326  }
327 
328  if (strap_utility_->CheckUtility())
329  {
330  // The flow is not being properly serviced and should be triaged.
331  //
332  // - Cancel the step event
333  // - Schedule the restart event
334  LogI(kClassName, __func__, "fid: %" PRIu32 " was triaged.\n",
335  encoding_state_.flow_tag());
336  ++toggle_count_;
337  push_stats_ = true;
338  CancelScheduledEvent(step_time_);
339  ScheduleRestartTime();
340  }
341 
342  // Update admitted_seq_num
343  strap_utility_->set_admitted_seq_num(encoding_state_.admitted_seq_num());
344 
345 }
