82 ConnectionManager::~ConnectionManager()
83 {
84  // Destroy all of the connections.
85  for (size_t i = 0; i < kNumBlocks; ++i)
86  {
87  if (connections_[i] != NULL)
88  {
89  for (size_t j = 0; j < kNumConnsPerBlock; ++j)
90  {
91  Connection* conn = connections_[i][j];
92 
93  if (conn != NULL)
94  {
95  conn->DisableCallbacks();
96  delete conn;
97  connections_[i][j] = NULL;
98  }
99  }
100 
101  delete [] connections_[i];
102  connections_[i] = NULL;
103  }
104  }
105 
106  // Cancel any timers.
107  timer_.CancelTimer(reaper_timer_);
108 
109  // Clean up the timer callback object pools.
110  CallbackNoArg<ConnectionManager>::EmptyPool();
111 }

114 bool ConnectionManager::AddConnection(EndptId endpt_id, Connection* conn)
115 {
116  if ((endpt_id < 0) || (conn == NULL))
117  {
118  return false;
119  }
120 
121  // Make sure that the new connection object will fit in the 2D array.
122  if ((endpt_id < 0) ||
123  (static_cast<size_t>(endpt_id) >= (kNumBlocks * kNumConnsPerBlock)))
124  {
125  LogE(kClassName, __func__, "Endpoint ID %" PRIEndptId " cannot be "
126  "stored.\n", endpt_id);
127  return false;
128  }
129 
130  // Compute the location for this connection pointer.
131  size_t block_index = (static_cast<size_t>(endpt_id) / kNumConnsPerBlock);
132  size_t conn_index = (static_cast<size_t>(endpt_id) % kNumConnsPerBlock);
133 
134  // Add a new block of connection pointers if needed.
135  if (connections_[block_index] == NULL)
136  {
137  connections_[block_index] = new (std::nothrow)
138  Connection*[kNumConnsPerBlock];
139 
140  if (connections_[block_index] == NULL)
141  {
142  LogE(kClassName, __func__, "Error allocating block of connection "
143  "pointers.\n");
144  return false;
145  }
146 
147  for (size_t i = 0; i < kNumConnsPerBlock; ++i)
148  {
149  connections_[block_index][i] = NULL;
150  }
151  }
152 
153  // Destroy any existing connection object at the index.
154  Connection* old_conn = connections_[block_index][conn_index];
155 
156  if (old_conn != NULL)
157  {
158  LogF(kClassName, __func__, "Existing connection object found for "
159  "endpoint ID %" PRIEndptId ".\n", endpt_id);
160  delete old_conn;
161  }
162 
163  // Store the connection pointer.
164  connections_[block_index][conn_index] = conn;
165 
166  return true;
167 }

214 bool ConnectionManager::DeleteConnection(EndptId endpt_id)
215 {
216  if ((endpt_id >= 0) &&
217  (static_cast<size_t>(endpt_id) < (kNumBlocks * kNumConnsPerBlock)))
218  {
219  size_t block_index = (static_cast<size_t>(endpt_id) / kNumConnsPerBlock);
220 
221  if (connections_[block_index] != NULL)
222  {
223  size_t conn_index = (static_cast<size_t>(endpt_id) %
224  kNumConnsPerBlock);
225 
226  if (connections_[block_index][conn_index] != NULL)
227  {
228  // The connection object was found. Add the connection to the reaper
229  // stack to be destroyed later.
230  if (reaper_size_ >= kMaxReaperSize)
231  {
232  LogF(kClassName, __func__, "Reaper list size exceeded.\n");
233  return false;
234  }
235 
236  reaper_list_[reaper_size_] = endpt_id;
237  ++reaper_size_;
238 
239  // If the reaper timer has not been started already, then start it
240  // now.
241  if (!timer_.IsTimerSet(reaper_timer_))
242  {
243  Time duration(kReaperTimerSec);
244  CallbackNoArg<ConnectionManager> callback(
245  this, &ConnectionManager::ReaperTimeout);
246 
247  if (!timer_.StartTimer(duration, &callback, reaper_timer_))
248  {
249  LogE(kClassName, __func__, "Error starting reaper timer.\n");
250  return false;
251  }
252  }
253 
254  return true;
255  }
256  }
257  }
258 
259  LogE(kClassName, __func__, "Error, connection for endpoint ID %" PRIEndptId
260  " not found.\n", endpt_id);
261 
262  return false;
263 }

266 void ConnectionManager::ReaperTimeout()
267 {
268  // Destroy all of the connections on the reaper stack.
269  for (size_t i = 0; i < reaper_size_; ++i)
270  {
271  EndptId endpt_id = reaper_list_[i];
272 
273  if ((endpt_id >= 0) &&
274  (static_cast<size_t>(endpt_id) < (kNumBlocks * kNumConnsPerBlock)))
275  {
276  size_t block_index = (static_cast<size_t>(endpt_id) /
277  kNumConnsPerBlock);
278 
279  if (connections_[block_index] != NULL)
280  {
281  size_t conn_index = (static_cast<size_t>(endpt_id) %
282  kNumConnsPerBlock);
283  Connection* conn = connections_[block_index][conn_index];
284 
285  if (conn != NULL)
286  {
287  delete conn;
288  connections_[block_index][conn_index] = NULL;
289  }
290  else
291  {
292  LogE(kClassName, __func__, "Error, connection to be reaped is "
293  "missing.\n");
294  }
295  }
296  else
297  {
298  LogE(kClassName, __func__, "Error, block for connection to be reaped "
299  "is missing.\n");
300  }
301  }
302  else
303  {
304  LogE(kClassName, __func__, "Error, connection to be reaped has "
305  "invalid endpoint ID %" PRIEndptId ".\n", endpt_id);
306  }
307 
308  reaper_list_[i] = -1;
309  }
310 
311  reaper_size_ = 0;
312 }
