91  DropRule drop_rule)
92 {

94  {
95  return false;
96  }
97 
98  // Reallocate the circular buffer.
99  if (queue_ != NULL)

101  delete [] queue_;

103 
104  queue_ = new (std::nothrow) QueueElement[max_size_pkts];
105 
106  if (queue_ == NULL)

108  return false;

112  max_cnt_ = max_size_pkts;

123 bool PacketQueue::Enqueue(Packet* pkt, const Time& now)
124 {
125  if ((pkt == NULL) || (queue_ == NULL))
126  {
127  return false;
128  }
129 
130  // Drop a packet if needed and allowed.
131  if (cnt_ == max_cnt_)
132  {
133  switch (drop_rule_)
134  {
135  case NO_DROP:
136  // Enqueue must fail.
137  return false;
138 
139  case HEAD_DROP:
140  {
141  Packet* old_pkt = queue_[head_].pkt;
142  cnt_ -= 1;
143  size_ -= (old_pkt->GetMetadataHeaderLengthInBytes() +
144  old_pkt->GetLengthInBytes());
145  queue_[head_].pkt = NULL;
146  head_ = ((head_ + 1) % max_cnt_);
147  pkt_pool_.Recycle(old_pkt);
148  // Head drops are QLAM packets from the SLIQ CAT.
149  TRACK_EXPECTED_DROP(kClassName, pkt_pool_);
150  break;
151  }
152 
153  case TAIL_DROP:
154  {
155  size_t tail = ((head_ + cnt_ - 1) % max_cnt_);
156  Packet* old_pkt = queue_[tail].pkt;
157  cnt_ -= 1;
158  size_ -= (old_pkt->GetMetadataHeaderLengthInBytes() +
159  old_pkt->GetLengthInBytes());
160  queue_[tail].pkt = NULL;
161  pkt_pool_.Recycle(old_pkt);
162  TRACK_UNEXPECTED_DROP(kClassName, pkt_pool_);
163  break;
164  }
165 
166  default:
167  LogF(kClassName, __func__, "Invalid drop rule.\n");
168  return false;
169  }
170  }
171 
172  // Store the new packet at the tail.
173  size_t tail = ((head_ + cnt_) % max_cnt_);
174  queue_[tail].enqueue_time = now;
175  queue_[tail].pkt = pkt;
176  cnt_ += 1;
177  size_ += (pkt->GetMetadataHeaderLengthInBytes() +
178  pkt->GetLengthInBytes());
179 
180  return true;
181 }

184 Packet* PacketQueue::Dequeue()
185 {
186  Packet* pkt = NULL;
187 
188  // Dequeue the packet at the head.
189  if ((queue_ != NULL) && (cnt_ > 0))
190  {
191  switch (dequeue_rule_)
192  {
193  case FIFO_QUEUE:
194  pkt = queue_[head_].pkt;
195  cnt_ -= 1;
196  size_ -= (pkt->GetMetadataHeaderLengthInBytes() +
197  pkt->GetLengthInBytes());
198  queue_[head_].pkt = NULL;
199  head_ = ((head_ + 1) % max_cnt_);
200  break;
201 
202  case LIFO_QUEUE:
203  {
204  size_t tail = ((head_ + cnt_ - 1) % max_cnt_);
205  pkt = queue_[tail].pkt;
206  cnt_ -= 1;
207  size_ -= (pkt->GetMetadataHeaderLengthInBytes() +
208  pkt->GetLengthInBytes());
209  queue_[tail].pkt = NULL;
210  break;
211  }
212 
213  default:
214  LogF(kClassName, __func__, "Invalid dequeueing rule.\n");
215  return NULL;
216  }
217  }

264 void PacketQueue::Purge()
265 {
266  if ((queue_ != NULL) && (cnt_ > 0))
267  {
268  Packet* pkt = NULL;
269 
270  while (cnt_ > 0)
271  {
272  pkt = queue_[head_].pkt;
273  queue_[head_].pkt = NULL;
274  head_ = ((head_ + 1) % max_cnt_);
275  --cnt_;
276 
277  pkt_pool_.Recycle(pkt);
278  // Purges are only when leaving an outage.
279  TRACK_EXPECTED_DROP(kClassName, pkt_pool_);
280  }
281 
282  cnt_ = 0;
283  size_ = 0;
284  }
285 }

291

293  if ((queue_ != NULL) && (cnt_ > 0))
294  {

296 
297  switch (dequeue_rule_)
298  {

301  break;
302 
303  case LIFO_QUEUE:
304  pkt = queue_[(head_ + cnt_ - 1) % max_cnt_].pkt;
305  break;
306 
307  default:
308  LogF(kClassName, __func__, "Invalid dequeueing rule.\n");
309  return 0;
310  }
311 
312  size = (pkt->GetMetadataHeaderLengthInBytes() + pkt->GetLengthInBytes());
313  }
314
