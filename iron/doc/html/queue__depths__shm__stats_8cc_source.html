105 void QueueDepthsShmStats::DepthChanged(BinIndex idx, uint32_t new_depth)
106 {
107  Time now = Time::Now();
108  if (new_depth > max_val_since_write_[idx])
109  {
110  max_val_since_write_[idx] = new_depth;
111  }
112  if (new_depth < min_val_since_write_[idx])
113  {
114  min_val_since_write_[idx] = new_depth;
115  }
116  if (last_write_time_ > first_change_time_)
117  {
118  // This was our first change since the last write was performed.
119  first_change_time_ = now;
120  }
121 }

124 void QueueDepthsShmStats::ValuesShared(const QueueDepths* shared_vals)
125 {
126  Time now = Time::Now();
127 
128  if (first_change_time_ > last_write_time_)
129  {
130  // There were changes since the last write. Process them.
131 
132  // Delta is used to track the biggest change since the last write.
133  uint32_t delta = 0;
134  // Outside_range, range_max and range_min are used to track biggest delta
135  // from the previous-to-current range.
136  uint32_t outside_range = 0;
137  // We want 0s for all multicast bins, but we still want a complete array.
138  BinIndex idx = kInvalidBinIndex;
139  for (bool idx_valid = bin_map_.GetFirstBinIndex(idx);
140  idx_valid;
141  idx_valid = bin_map_.GetNextBinIndex(idx))
142  {
143  uint32_t new_val = 0;
144  if (!bin_map_.IsMcastBinIndex(idx))
145  {
146  new_val = shared_vals->GetBinDepthByIdx(idx);
147  }
148  uint32_t range_max = new_val;
149  uint32_t range_min = new_val;
150  if (last_shared_values_[idx] > range_max)
151  {
152  range_max = last_shared_values_[idx];
153  }
154  else
155  {
156  range_min = last_shared_values_[idx];
157  }
158 
159  if (max_val_since_write_[idx] > last_shared_values_[idx])
160  {
161  uint32_t diff = max_val_since_write_[idx] -
162  last_shared_values_[idx];
163  if (diff > delta)
164  {
165  delta = diff;
166  }
167  }
168  if (min_val_since_write_[idx] < last_shared_values_[idx])
169  {
170  uint32_t diff = last_shared_values_[idx] -
171  min_val_since_write_[idx];
172  if (diff > delta)
173  {
174  delta = diff;
175  }
176  }
177  if (max_val_since_write_[idx] > range_max)
178  {
179  uint32_t diff = max_val_since_write_[idx] - range_max;
180  if (diff > outside_range)
181  {
182  outside_range = diff;
183  }
184  }
185  if (min_val_since_write_[idx] < range_min)
186  {
187  uint32_t diff = range_min - min_val_since_write_[idx];
188  if (diff > outside_range)
189  {
190  outside_range = diff;
191  }
192  }
193  last_shared_values_[idx] = new_val;
194  }
195  if (delta > max_delta_)
196  {
197  max_delta_ = delta;
198  }
199  if (outside_range > max_outside_range_)
200  {
201  max_outside_range_ = outside_range;
202  }
203  Time stale_time = first_change_time_ - last_write_time_;
204  if (stale_time > max_stale_time_)
205  {
206  max_stale_time_ = stale_time;
207  }
208  uint32_t maxval = std::numeric_limits<uint32_t>::max();
209  // Assuming we won't rollover the time, since Time is able
210  // to store seconds since epoch, and there's no way the total stale
211  // time can exceed that.
212  if (maxval - delta < total_delta_
213  || maxval - outside_range < total_outside_range_)
214  {
215  // Refresh stats to avoid a rollover.
216  LogI(kClassName, __func__,
217  "Shared Memory averages rolling over.\n%s\n",
218  ToString().c_str());
219  total_delta_ = 0;
220  total_outside_range_ = 0;
221  total_stale_time_.Zero();
222  num_write_periods_ = 0;
223  }
224 
225  total_delta_ += delta;
226  total_outside_range_ += outside_range;
227  total_stale_time_ += stale_time;
228  }
229  last_write_time_ = now;
230  ++num_write_periods_;
231 }

234 string QueueDepthsShmStats::ToString() const
235 {
236  std::stringstream ret_ss;
237 
238  ret_ss << "Printing queue depth shared memory statistics \n";
239  ret_ss << "\tmax delta from last written: "
240  << max_delta_ << "\n";
241  ret_ss << "\tmax delta from current-previous range: "
242  << max_outside_range_ << "\n";
243  ret_ss << "\tmax stale time: "
244  << max_stale_time_.ToString() << " \n";
245 
246  double avg_delta = 0.0;
247  if (num_write_periods_ > 0)
248  {
249  avg_delta = (static_cast<double>(total_delta_) /
250  static_cast<double>(num_write_periods_));
251  }
252 
253  ret_ss << "\tavg delta from last written: "
254  << avg_delta << "\n";
255 
256  avg_delta = 0.0;
257  if (num_write_periods_ > 0)
258  {
259  avg_delta = (static_cast<double>(total_outside_range_) /
260  static_cast<double>(num_write_periods_));
261  }
262 
263  ret_ss << "\tavg delta from current-previous range: "
264  << avg_delta << "\n";
265 
266  avg_delta = 0.0;
267  if (num_write_periods_ > 0)
268  {
269  // It's possible (though very unlikely) that this cast will lose some
270  // information. However, we don't need these stats to be precise, so
271  // ignore that corner case for now.
272  avg_delta = (static_cast<double>(total_stale_time_.GetTimeInUsec()) /
273  static_cast<double>(num_write_periods_));
274  }
275 
276  ret_ss << "\tavg stale time: "
277  << avg_delta << " usec\n";
278 
279  return ret_ss.str();
280 }
