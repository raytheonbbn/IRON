70 void Zombie::ZombifyExistingPacket(Packet* pkt)
71 {
72  pkt->MakeZombie(HIGH_LATENCY_EXP);
73 
74  if (kDefaultZombieCompression)
75  {
76  LogD(kClassName, __func__,
77  "Creating compressed zombie with virtual length %zu\n",
78  pkt->GetLengthInBytes());
79  size_t virtual_length = pkt->GetLengthInBytes();
80  uint32_t virtual_size = htonl(static_cast<uint32_t>(virtual_length));
81  // Set the virtual length to the current real length.
82  pkt->set_virtual_length(virtual_length);
83  // Add it to the data portion.
84  uint8_t* buf = pkt->GetBuffer();
85 
86  size_t new_length = pkt->GetIpPayloadOffset();
87  memcpy(buf + new_length, &virtual_size, sizeof(virtual_size));
88  new_length += sizeof(virtual_size);
89 
90  pkt->SetLengthInBytes(new_length);
91  pkt->UpdateIpLen();
92 
93  // NOTE: We do not move the FEC trailer since the packet will not be
94  // passed up to the UDP proxy.
95  }
96 }

99 Packet* Zombie::CreateNewZombie(PacketPool& pkt_pool,
100  uint32_t& src_addr_nbo,
101  uint32_t& dst_addr_nbo,
102  size_t zombie_len_bytes,
103  LatencyClass lat_class)
104 {
105  if (zombie_len_bytes > kMaxZombieLenBytes
106  || zombie_len_bytes < kMinZombieLenBytes)
107  {
108  LogF(kClassName, __func__,
109  "CreateNewZombie must be called with zombie_len_bytes (%zu) between "
110  "kMinZombieLenBytes (%zu) and kMaxZombieLenBytes (%zu). "
111  "LatencyClass is %s, dst_addr is %s.\n",
112  zombie_len_bytes, kMinZombieLenBytes, kMaxZombieLenBytes,
113  Ipv4Address(dst_addr_nbo).ToString().c_str());
114  return NULL;
115  }
116  Packet* zombie = pkt_pool.Get(PACKET_NOW_TIMESTAMP);
117  zombie->InitIpPacket();
118  // MakeZombie encapsulates setting up the DSCP value, TTG, and packet type.
119  zombie->MakeZombie(lat_class);
120  struct iphdr* iphdr = zombie->GetIpHdr();
121  iphdr->id = htons(pkt_pool.GetNextIpId());
122  iphdr->protocol = kZombieProtocolNum;
123  iphdr->saddr = src_addr_nbo;
124  iphdr->daddr = dst_addr_nbo;
125 
126  size_t length = zombie->GetLengthInBytes();
127 
128  if (kDefaultZombieCompression)
129  {
130  uint32_t virtual_size = htonl(static_cast<uint32_t>(zombie_len_bytes));
131  zombie->set_virtual_length(zombie_len_bytes);
132 
133  // Add the virtual length to the packet data
134  memcpy(zombie->GetBuffer(length), &virtual_size, sizeof(virtual_size));
135  length += sizeof(virtual_size);
136  }
137  else if (zombie_len_bytes >= length)
138  {
139  // Set data to 0s to avoid any vulnerabilities from re-transmitting
140  // leftover data from the last time this packet was used.
141  // We know this will fit because zombie_len_bytes < kMaxZombieLenBytes.
142  memset(zombie->GetBuffer(length), 0, zombie_len_bytes - length);
143  length = zombie_len_bytes;
144  zombie->set_virtual_length(zombie_len_bytes);
145  }
146  else
147  {
148  // We are trying to send a zombie smaller than the minimum size of a
149  // packet when compression is disabled. This is going to cause accounting
150  // issues, because we'll look at the size of the packet (length), which
151  // may be smaller than the size of the zombie queue.
152  LogE(kClassName, __func__, "Attempted to create a zombie of size %zu, "
153  "which is smaller than the minimum packet size %zu. Returning NULL.\n",
154  zombie_len_bytes, length);
155  pkt_pool.Recycle(zombie);
156  zombie = NULL;
157  return zombie;
158  }
159  zombie->SetLengthInBytes(length);
160  zombie->UpdateIpLen();
161  zombie->UpdateIpChecksum();
162  LogD(kClassName, __func__,
163  "Created zombie with length %zu.\n", zombie->GetLengthInBytes());
164  zombie->DumpIpHdr();
165  return zombie;
166 }
