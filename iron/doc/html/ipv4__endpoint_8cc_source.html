66 Ipv4Endpoint::Ipv4Endpoint(const std::string& ep_str)
67 {
68  List<string> tokens;
69  StringUtils::Tokenize(ep_str, ":", tokens);
70 
71  if (tokens.size() != 2)
72  {
73  LogE(kClassName, __func__, "Invalid Ipv4Endpoint string provided: %s\n",
74  ep_str.c_str());
75  address_ = 0;
76  port_nbo_ = 0;
77  }
78  else
79  {
80  string token;
81  tokens.Pop(token);
82  set_address(token);
83 
84  tokens.Peek(token);
85  port_nbo_ =
86  static_cast<uint16_t>(htons(StringUtils::GetUint(token)));
87  }
88 }

127 bool Ipv4Endpoint::SetEndpoint(const string& ep_str)
128 {
129  List<string> tokens;
130  StringUtils::Tokenize(ep_str, ":", tokens);
131 
132  if (tokens.size() != 2)
133  {
134  return false;
135  }
136 
137  string addr_str;
138  tokens.Pop(addr_str);
139  string port_str;
140  tokens.Peek(port_str);
141  unsigned int port_hbo = StringUtils::GetUint(port_str);
142 
143  if ((port_hbo == UINT_MAX) || (!SetAddress(addr_str)))
144  {
145  return false;
146  }
147 
148  port_nbo_ = static_cast<uint16_t>(htons(port_hbo));
149 
150  return true;
151 }

162 {
163  struct sockaddr_in* addr = reinterpret_cast<struct sockaddr_in*>(address);
164 
165  ::memset(addr, 0, sizeof(struct sockaddr_in));
166 
167  addr->sin_family = AF_INET;
168  addr->sin_port = port_nbo_;
169  addr->sin_addr.s_addr = address_;
