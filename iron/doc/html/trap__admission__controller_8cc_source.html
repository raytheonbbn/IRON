78 bool TrapAdmissionController::CreateUtilityFn(string& utility_def,
79  uint32_t flow_id,
80  QueueDepths& queue_depths)
81 {
82  // Validate that the provided utility definition string is for the Trap
83  // utility function.
84  if (GetUtilityFnType(utility_def, flow_id) != "TRAP")
85  {
86  LogW(kClassName, __func__, "fid: %" PRIu32 ", utility definition is not "
87  "for a Trap utility function.\n", flow_id);
88  return false;
89  }
90 
91  trap_utility_ = new (std::nothrow) TrapUtility(queue_depths,
92  encoding_state_.bin_idx(),
93  encoding_state_.k_val(),
94  flow_id);
95  if (trap_utility_ == NULL)
96  {
97  LogF(kClassName, __func__, "fid: %" PRIu32 "Unable to allocate memory "
98  "for Trap utility function.\n", flow_id);
99  return false;
100  }
101 
102  ConfigureUtilityFn(trap_utility_, utility_def);
103 
104  ScheduleStepTime();
105  ScheduleCheckUtilityTime();
106 
107  return true;
108 }

111 void TrapAdmissionController::SvcEvents(Time& now)
112 {
113  // First, service the admission event.
114  SvcAdmissionEvent(now, trap_utility_);
115 
116  // Now, service the trap admission controller specific events.
117  if (restart_time_ <= now)
118  {
119  RestartTimeout(now);
120  }
121 
122  if (step_time_ <= now)
123  {
124  StepTimeout(now);
125  }
126 
127  if (check_utility_time_ <= now)
128  {
129  CheckUtilityTimeout(now);
130  }
131 }

145 void TrapAdmissionController::set_flow_state(iron::FlowState flow_state)
146 {
147  if (trap_utility_ == NULL)
148  {
149  // If we get here and don't have a utility function something is very very
150  // wrong.
151  LogF(kClassName, __func__, "Attempting to set the flow's state and there "
152  "isn't a utility function.\n");
153  return;
154  }
155 
156  Time now = Time::Now();
157 
158  if ((flow_state == iron::FLOW_OFF) || (flow_state == iron::FLOW_TRIAGED))
159  {
160  CancelScheduledEvent(next_admission_time_);
161  CancelScheduledEvent(step_time_);
162  CancelScheduledEvent(check_utility_time_);
163 
164  if (flow_state == iron::FLOW_OFF)
165  {
166  trap_utility_->ResetInertia();
167  CancelScheduledEvent(restart_time_);
168  }
169 
170  if ((flow_state == iron::FLOW_TRIAGED) &&
171  (trap_utility_->flow_state() == iron::FLOW_ON))
172  {
173  ScheduleRestartTime();
174  }
175  }
176  else if (flow_state == iron::FLOW_ON)
177  {
178  start_time_ = now;
179  next_admission_time_ = now;
180  ScheduleStepTime();
181  ScheduleCheckUtilityTime();
182  CancelScheduledEvent(restart_time_);
183  }
184 
185  trap_utility_->set_flow_state(flow_state);
186 }

211 void TrapAdmissionController::UpdateUtilityFn(std::string key_val)
212 {
213  List<string> tokens;
214  StringUtils::Tokenize(key_val, ":", tokens);
215  if (tokens.size() != 2)
216  {
217  LogE(kClassName, __func__, "Parameter %s must be of the form key:value.\n",
218  key_val.c_str());
219  return;
220  }
221 
222  string key;
223  string value;
224  tokens.Peek(key);
225  tokens.PeekBack(value);
226  if (key == "delta")
227  {
228  trap_utility_->set_delta(StringUtils::GetDouble(value));
229  LogD(kClassName, __func__, "New delta: %s.\n", value.c_str());
230  }
231  else if (key == "p")
232  {
233  trap_utility_->set_priority(StringUtils::GetDouble(value,0.0));
234  }
235  else
236  {
237  LogE(kClassName, __func__, "Update of %s not supported.\n", key.c_str());
238  }
239 }

242 void TrapAdmissionController::RestartTimeout(Time& now)
243 {
244  // Turn the flow on. All packets in the backlog are flushed because they are
245  // old.
246  trap_utility_->SetFlowOn();
247  trap_utility_->ResetInertia();
248  encoding_state_.FlushBacklog();
249 
250  // Adjust next admission time to be now. We are turning the flow back on so
251  // we can send any packets that are in the proxy queue.
252  next_admission_time_ = now;
253  SvcAdmissionEvent(now, trap_utility_);
254 
255  // Adjust the event timers.
256  ScheduleCheckUtilityTime();
257  ScheduleStepTime();
258  CancelScheduledEvent(restart_time_);
259 }

269 void TrapAdmissionController::CheckUtilityTimeout(Time& now)
270 {
271  if (trap_utility_->CheckUtility())
272  {
273  // The flow is not being properly serviced and should be triaged.
274  //
275  // - Cancel step and utility check events
276  // - Schedule the restart event
277  CancelScheduledEvent(step_time_);
278  CancelScheduledEvent(check_utility_time_);
279  ScheduleRestartTime();
280  }
281 }
