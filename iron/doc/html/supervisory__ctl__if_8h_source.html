404  bool IsThrashing(Time& now, uint32_t interval_ms)
405  {
406  if (flow_state_ == FLOW_TRIAGED)
407  {
408  return true;
409  }
410 
411  Time window = now - Time::FromMsec(1.5*interval_ms);
412  if((flow_state_ == FLOW_ON) && (last_toggle_time_ > window))
413  {
414  LogW("FlowInfo", __func__, "Thrash: %s vs %s, %u\n",
415  last_toggle_time_.ToString().c_str(), window.ToString().c_str(),
416  interval_ms);
417  return true;
418  }
419  return false;
420  }
