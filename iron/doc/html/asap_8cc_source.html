183 uint32_t ASAP::BytesToAddGivenDelay(Time delay, bool is_ls,
184  BinIndex dst_bidx) const
185 {
186  // Conversion is y = a * x^2, units of a are (bytes / ms^2)
187  // Setting a to 1 results in 10000 bytes at 100ms
188 
189  // Start with the total bytes to add based on delay
190  uint32_t delay_ms = static_cast<uint32_t>(delay.GetTimeInMsec());
191 
192  // Don't count sleep time against us
193  if (delay_ms > sleep_time_by_bin_[dst_bidx])
194  {
195  delay_ms -= sleep_time_by_bin_[dst_bidx];
196  }
197  else
198  {
199  delay_ms = 0;
200  }
201 
202  uint32_t bytes_to_add = 0;
203  uint32_t asz_delay_threshold = 0;
204  if (average_capacity_ > 0)
205  {
206  // max pkt size in bits * 1000 (sec -> ms) * 50 (constant) / cap in bps
207  asz_delay_threshold = std::min(kMaxPktSizeBits * 1000 *
208  kThresholdSlack / average_capacity_, kMinStarvationThreshMs);
209 
210  LogD(kClassName, __func__, "Starvation check %s: threshold is %"
211  PRIu32 "ms, delay is %" PRIu32 "ms"
212  PRIu32 " bytes of zombies\n",
213  bin_map_.GetIdToLog(my_bin_index_).c_str(),
214  asz_delay_threshold, delay_ms);
215 
216  if (delay_ms > asz_delay_threshold)
217  {
218  bytes_to_add =
219  static_cast<uint32_t>(kASZCoefficient * delay_ms * delay_ms);
220  }
221  }
222 
223  // Remove the bytes that we've already added since the last dequeue.
224  // This is necessary because this function is called whenever
225  // queue depths are used, but we don't want to over-count the
226  // delay experienced by the same head-of-queue packet.
227 
228  uint32_t bytes_added = delay_bytes_added_[dst_bidx];
229  if (bytes_added > bytes_to_add)
230  {
231  LogW(kClassName, __func__,
232  "Inconsistent delay to bytes conversion: bytes_added is %" PRIu32
233  " vs bytes_to_add is %" PRIu32 ": delay_ms is %" PRIu32
234  "\n",bytes_added,bytes_to_add,delay_ms);
235  bytes_to_add = 0;
236  }
237  else
238  {
239  bytes_to_add -= bytes_added;
240  }
241 
242  // Enforce a cap on the max to add. The cap should be slightly higher
243  // than the overall max minus my current gradient, and we should
244  // ensure that the cap is positive. Note that there is a special
245  // case where we have only one bin and that bin has a negative
246  // gradient. This is a "packet left behind" in the NPLB examples,
247  // and to prevent this we want to help our gradient become positive
248  // over time.
249 
250  if (is_ls)
251  {
252  if (bytes_to_add > gradient_based_ls_cap_)
253  {
254  bytes_to_add = gradient_based_ls_cap_;
255  LogD(kClassName, __func__, "LS zombie cap is %" PRIu32 "\n",
256  gradient_based_ls_cap_);
257  }
258  }
259  else
260  {
261  if (bytes_to_add > gradient_based_cap_)
262  {
263  bytes_to_add = gradient_based_cap_;
264  LogD(kClassName, __func__, "Zombie cap is %" PRIu32 "\n",
265  gradient_based_cap_);
266  }
267  }
268 
269  if (bytes_to_add > 0)
270  {
271  LogD(kClassName, __func__, "Starvation detected (threshold is %"
272  PRIu32 "ms, delay is %" PRIu32 "ms), adding %"
273  PRIu32 " bytes of zombies\n", asz_delay_threshold, delay_ms,
274  bytes_to_add);
275  }
276 
277  return bytes_to_add;
278 
279 }

282 void ASAP::AdjustQueueValuesForAntiStarvation()
283 {
284  if (!initialized_)
285  {
286  return;
287  }
288 
289  // Add zombie bytes to bins to account for the delay term.
290  // Specifically, track hold times for packets in the queue and
291  // continuously add zombies as that delay increases, according
292  // to the BytesToAddGivenDelay function. This helps prevent
293  // starvation by forcing the gradient to be larger the longer
294  // the delays become.
295 
296  // Note that this implemnetation uses separate enqueue times queues
297  // that astrictly fifo. When a packet is enqueued in the packet
298  // queue, its queueing time (for each destination in the destination
299  // vector if the packet is a multicast packet) are pushed onto
300  // the enqueue time queues, and whenever a packet is dequeued, the
301  // head of the enqueue time queue is removed/popped (for each destination
302  // in the destination vector if the packet is a multicast packet). In
303  // this way, packets can be (even partially) removed from anywhere
304  // in the packet queue, yet the time-in-queue information remains
305  // consistent with what is needed for ASAP to work correctly.
306 
307  Time now = Time::Now();
308  Packet* pkt = NULL;
309  uint32_t oldest_pkt_dst_addr =
310  bin_map_.GetViableDestAddr(my_bin_index_).address();
311  uint32_t since_last_call_ms = 0;
312 
313  // Compute the time elapsed since the last time this method was called
314  if (!time_of_last_asap_call_.IsZero())
315  {
316  Time timedelta = now - time_of_last_asap_call_;
317  since_last_call_ms = static_cast<uint32_t>(timedelta.GetTimeInMsec());
318  }
319  time_of_last_asap_call_ = now;
320 
321  Time oldest_pkt_recv_time = now;
322 
323  // Find the oldest enqueued packet in the lowest latency bin,
324  // ignoring zombies. This primarily involves looking at the head
325  // of the enqueue time queues
326 
327  bool is_ls = false;
328  for (uint8_t it = 0; it < NUM_LATENCY_DEF; ++it)
329  {
330  if (Packet::IsZombie(static_cast<LatencyClass>(it)))
331  {
332  // Ignore zombie packets.
333  continue;
334  }
335  pkt = q_mgr_.Peek(it);
336 
337  if (!pkt)
338  {
339  // Nothing in the queue (or this is a packetless queue), nothing to
340  // consider.
341  continue;
342  }
343 
344  // Note: zombie latency classes were excluded earlier in this loop
345  // There is a packet in this latency class, use it
346 
347  if (it <= LOW_LATENCY)
348  {
349  is_ls = true;
350 
351  oldest_pkt_recv_time =
352  GetOldestPktRecvTime(enqueue_time_ls_);
353  }
354  else
355  {
356  oldest_pkt_recv_time =
357  GetOldestPktRecvTime(enqueue_time_);
358  }
359 
360  // If we are here, we found a packet, so break out
361  break;
362  }
363 
364  // TODO: We've obtained the oldest packet in the lowest latency class.
365  // It's possible that this is stuck because there's no available path
366  // with low enough latency. This is a corner case that we may want to
367  // not considered starved.
368 
369  // If all queues are empty i.e., we didn't find a pkt, then
370  // the oldest pkt receive time will still be set to its initial value
371  // (now) and we don't need to add zombies.
372 
373  if (oldest_pkt_recv_time == now)
374  {
375  return;
376  }
377 
378  // Add zombie bytes based on delay.
379  // Delay is the minimum of time since last dequeue and time
380  // oldest packet has spent in the queue. This effectively
381  // is the time at the head of the queue time queue. The time
382  // of last dequeue primarily captures this; the min is needed
383  // if the packet is added after the last dequeue (so, the queue
384  // was empty).
385 
386  DstVec asap_dst_vec = 0;
387  uint32_t bytes_to_add = 0;
388  if (bin_map_.IsMcastBinIndex(my_bin_index_))
389  {
390  DstVec dst_vec = bin_map_.GetMcastDst(my_bin_index_);
391  BinIndex dst_bidx = 0;
392 
393  for (bool valid = bin_map_.GetFirstUcastBinIndex(dst_bidx);
394  valid;
395  valid = bin_map_.GetNextUcastBinIndex(dst_bidx))
396  {
397  // Only process if destination dst_bidx is in the group
398  // and is not the node this is running on
399 
400  if (bin_map_.IsBinInDstVec(dst_vec, dst_bidx) &&
401  (dst_bidx != node_bin_index_))
402  {
403  // See if there are any pkts in this queue
404  bool have_pkts = true;
405  if (is_ls)
406  {
407  if (enqueue_time_ls_[dst_bidx] != (queue<Time>*)NULL)
408  {
409  have_pkts = !enqueue_time_ls_[dst_bidx]->empty();
410  }
411  }
412  else
413  {
414  if (enqueue_time_[dst_bidx] != (queue<Time>*)NULL)
415  {
416  have_pkts = !enqueue_time_[dst_bidx]->empty();
417  }
418  }
419 
420  // Only process if there are packets enqueued for this dst
421  if (have_pkts)
422  {
423  // See if the last dequeue time is still zero, meaning
424  // that nothing has yet been dequeued for this dst. If
425  // so, set the last dequeue time to "now", set the
426  // sleep time to zero, and skip
427 
428  if (q_mgr_.last_dequeue_time(dst_bidx).IsZero())
429  {
430  q_mgr_.last_dequeue_time(dst_bidx) = now;
431  sleep_time_by_bin_[dst_bidx] = 0;
432  continue;
433  }
434 
435  Time delay = now - q_mgr_.last_dequeue_time(dst_bidx);
436 
437  // Update sleep time for this bin.
438 
439  if (since_last_call_ms > kThresholdSleepTimeMs)
440  {
441  sleep_time_by_bin_[dst_bidx] += since_last_call_ms;
442  LogD(kClassName, __func__, "Update sleeptime: delta % " PRIu32
443  ", new time %" PRIu32 "\n", since_last_call_ms,
444  sleep_time_by_bin_[dst_bidx]);
445  }
446 
447  bytes_to_add = BytesToAddGivenDelay(delay, is_ls, dst_bidx);
448  if (bytes_to_add > 0)
449  {
450  asap_dst_vec = bin_map_.AddBinToDstVec(0, dst_bidx);
451  q_mgr_.AddNewZombie(oldest_pkt_dst_addr, bytes_to_add,
452  is_ls ? HIGH_LATENCY_NPLB_LS :
453  HIGH_LATENCY_NPLB,
454  asap_dst_vec);
455  delay_bytes_added_[dst_bidx] += bytes_to_add;
456  }
457  }
458  }
459  }
460  }
461  else
462  {
463  // This is essentially identical to the above multicast code,
464  // except that we only consider one destination bin index
465  // which is identically my_bin_index_
466 
467  if (since_last_call_ms > kThresholdSleepTimeMs)
468  {
469  sleep_time_by_bin_[my_bin_index_] += since_last_call_ms;
470  LogD(kClassName, __func__, "Update sleeptime: delta % " PRIu32
471  ", new time %" PRIu32 "\n", since_last_call_ms,
472  sleep_time_by_bin_[my_bin_index_]);
473  }
474 
475  Time delay = now - Time::Max(oldest_pkt_recv_time,
476  q_mgr_.last_dequeue_time(my_bin_index_));
477 
478  bytes_to_add = BytesToAddGivenDelay(delay, is_ls, my_bin_index_);
479  if (bytes_to_add > 0)
480  {
481  q_mgr_.AddNewZombie(oldest_pkt_dst_addr, bytes_to_add,
482  is_ls ? HIGH_LATENCY_NPLB_LS : HIGH_LATENCY_NPLB,
483  asap_dst_vec);
484  delay_bytes_added_[my_bin_index_] += bytes_to_add;
485  }
486  }
487 }

490 Time ASAP::GetOldestPktRecvTime
491  (BinIndexableArray< std::queue<Time>* >& enq_time)
492 {
493  Time oldest = Time::Now();
494 
495  if (bin_map_.IsMcastBinIndex(my_bin_index_))
496  {
497  BinIndex dst_bidx = 0;
498  for (bool valid = bin_map_.GetFirstUcastBinIndex(dst_bidx);
499  valid;
500  valid = bin_map_.GetNextUcastBinIndex(dst_bidx))
501  {
502  if (enq_time[dst_bidx] &&
503  (!enq_time[dst_bidx]->empty()))
504  {
505  if (oldest > enq_time[dst_bidx]->front())
506  {
507  oldest = enq_time[dst_bidx]->front();
508  }
509  }
510  }
511  }
512  else
513  {
514  if (enq_time[my_bin_index_] &&
515  (!enq_time[my_bin_index_]->empty()))
516  {
517  oldest = enq_time[my_bin_index_]->front();
518  }
519  }
520 
521  return (oldest);
522 }

526 void ASAP::ProcessCapacityUpdate(uint32_t pc_num, double capacity_bps)
527 {
528  if (pc_num >= kMaxPathCtrls)
529  {
530  LogW(kClassName, __func__,
531  "Path controller number %" PRIu32 "out of bounds.\n", pc_num);
532  return;
533  }
534  capacity_estimates_[pc_num] = static_cast<uint64_t>(capacity_bps);
535  // Subtract 1 bin for ourself
536  uint32_t num_bins = (bin_map_.GetNumUcastBinIds() +
537  bin_map_.GetNumMcastIds() - 1);
538  uint64_t average_cap = 0;
539  for (uint8_t it = 0; it < kMaxPathCtrls; ++it)
540  {
541  average_cap += capacity_estimates_[it];
542  }
543  if (num_bins != 0)
544  {
545  average_cap /= num_bins;
546  }
547  average_capacity_ = average_cap;
548  LogD(kClassName, __func__,
549  "Capacity within BinQueueMgr updated on pc %" PRIu32 " to %.1f, average "
550  "is now %" PRIu64 " over %" PRIu32 " bins.\n", pc_num, capacity_bps,
551  average_capacity_, num_bins);
552 }

589 void ASAP::PushEnqueueTime ( BinIndexableArray< queue<Time>* >& enq_time,
590  DstVec dests)
591 {
592  Time now = Time::Now();
593 
594  if (q_mgr_.is_multicast())
595  {
596  LogD(kClassName, __func__,
597  "PushEnqueueTime called for multicast bin %s\n",
598  bin_map_.GetIdToLog(my_bin_index_,true).c_str());
599 
600  BinIndex dst_bidx = 0;
601  for (bool valid = bin_map_.GetFirstUcastBinIndex(dst_bidx);
602  valid;
603  valid = bin_map_.GetNextUcastBinIndex(dst_bidx))
604  {
605  if (bin_map_.IsBinInDstVec(dests, dst_bidx))
606  {
607  // If we haven't yet allocated the queue, do so now
608  if (enq_time[dst_bidx] == (queue<Time>*)NULL)
609  {
610  enq_time[dst_bidx] = new queue<Time>;
611  }
612 
613  enq_time[dst_bidx]->push(now);
614 
615  if (q_mgr_.non_zombie_queue_depth_bytes(dst_bidx) == 0)
616  {
617  q_mgr_.last_dequeue_time(dst_bidx) = now;
618  }
619  }
620  }
621  }
622  else
623  {
624  LogD(kClassName, __func__,
625  "PushEnqueueTime called for unicast bin %s\n",
626  bin_map_.GetIdToLog(my_bin_index_,true).c_str());
627 
628  // If we haven't yet allocated the queue, do so now
629  if (enq_time[my_bin_index_] == (queue<Time>*)NULL)
630  {
631  enq_time[my_bin_index_] = new queue<Time>;
632  }
633 
634  enq_time[my_bin_index_]->push(now);
635 
636  if (q_mgr_.non_zombie_queue_depth_bytes(my_bin_index_) == 0)
637  {
638  q_mgr_.last_dequeue_time(my_bin_index_) = now;
639  }
640  }

648  DstVec dests = dq_info.dst_vec;

664  LogF(kClassName, __func__,
665  "OnDequeue called for unsupported packet type %s\n",
666  LatencyClass_Name[lat].c_str());
667  }
668  }
669 
670  ResetASAPTracking(dests);
671 }

673 void ASAP::PopEnqueueTime ( BinIndexableArray< queue<Time>* >& enq_time,
674  DstVec dests, LatencyClass asaplat,
675  LatencyClass zlrlat)
676 {
677  Time now = Time::Now();
678 
679  if (q_mgr_.is_multicast())
680  {
681  LogD(kClassName, __func__,
682  "PopEnqueueTime called for multicast bin %s\n",
683  bin_map_.GetIdToLog(my_bin_index_,true).c_str());
